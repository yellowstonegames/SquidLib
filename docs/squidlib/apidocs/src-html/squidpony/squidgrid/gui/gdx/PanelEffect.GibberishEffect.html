017/**
018 * Various special effects that can be applied to a {@link SquidPanel} or {@link SparseLayers} as an
019 * {@link com.badlogic.gdx.scenes.scene2d.Action}. The PanelEffect class is abstract and has implementations as static
020 * inner classes, such as {@link ExplosionEffect}. Each PanelEffect specifically affects one {@link IPackedColorPanel}
021 * (an interface that both SquidPanel and SparseLayers implement), which can be a layer in a SquidLayers object or a
022 * SquidPanel/SparseLayers on its own. By adding the PanelEffect to any actor using
023 * {@link com.badlogic.gdx.scenes.scene2d.Actor#addAction(Action)} when the Actor has its act() method called after the
024 * normal map parts of the panel have been placed, the PanelEffect will advance and change what chars/colors are in
025 * the panel as specified in its implementation of the {@link #update(float)} method. Typically the Actor you add this
026 * to is the SquidPanel or SparseLayers this affects, but it could also be a SquidLayers. Most PanelEffect
027 * implementations should allow most configuration to be set in their constructors.
028 * <br>
029 * Created by Tommy Ettinger on 5/24/2017.
030 */
031@Beta
032public abstract class PanelEffect extends TemporalAction{
033    public IPackedColorPanel target;
034    public GreasedRegion validCells;
035
036    protected PanelEffect(IPackedColorPanel targeting)
037    {
038        target = targeting;
039        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
040    }
041    protected PanelEffect(IPackedColorPanel targeting, float duration)
042    {
043        target = targeting;
044        setDuration(duration);
045        validCells = new GreasedRegion(targeting.gridWidth(), targeting.gridHeight()).allOn();
046    }
047    protected PanelEffect(IPackedColorPanel targeting, float duration, GreasedRegion valid)
048    {
049        target = targeting;
050        setDuration(duration);
051        validCells = valid;
052    }

065         * The default explosion colors are normal for (non-chemical, non-electrical) fire and smoke, going from orange
066         * at the start to yellow, very light yellow, and then back to a different orange before going to smoke and
067         * fading out to translucent and then transparent by the end.
068         * <br>
069         * If you want to change the colors the explosion uses, you can either pass a List of Color (or SColor or other
070         * subclasses) to the constructor or change this array directly. The float items assigned to this should be the
071         * result of calling {@link Color#toFloatBits()} or possibly the result of mixing multiple existing floats with
072         * {@link SColor#lerpFloatColors(float, float, float)}; other floats that can be directly used by libGDX, that
073         * is, packed as ABGR floats (usually the docs will call this a packed float), can also be used.
074         */

126         * @param targeting the IPackedColorPanel to affect
127         * @param duration the duration of this PanelEffect in seconds, as a float

135            this.center = center;
136            this.radius = radius;

138            validCells.writeDoublesInto(resMap, 0.0);

256        /**
257         * Called each frame.
258         *
259         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
260         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
261         */
262        @Override
263        protected void update(float percent) {
264            int len = affected.size();
265            Coord c;
266            float f, color, light;
267            int idx, seed = System.identityHashCode(this);
268            for (int i = 0; i < len; i++) {
269                c = affected.get(i);
270                if((light = (float) lightMap[c.x][c.y]) <= 0.0)// || 0.6 * (lightMap[c.x][c.y] + percent) < 0.25)
271                    continue;
272                f = (float)SeededNoise.noise(c.x * 1.5, c.y * 1.5, percent * 5, seed)
273                        * 0.17f + percent * 1.2f;
274                if(f < 0f || 0.5 * light + f < 0.4)
275                    continue;
276                idx = (int) (f * colors.length);
277                if(idx >= colors.length - 1)
278                    color = SColor.lerpFloatColors(colors[colors.length-1], NumberTools.setSelectedByte(colors[colors.length-1], 3, (byte)0), (Math.min(0.99f, f) * colors.length) % 1f);
279                else
280                    color = SColor.lerpFloatColors(colors[idx], colors[idx+1], (f * colors.length) % 1f);
281                target.blend(c.x, c.y, color, SColor.alphaOfFloatF(color) * light * 0.25f + 0.75f);
282            }
283        }

286         * Meant for electrical bursts, this will affect character foregrounds in a GibberishEffect. This should look
287         * like sparks in GibberishEffect if the chars in {@link GibberishEffect#choices} are selected in a way that
288         * fits that theme.
289         */
290        public ExplosionEffect useElectricColors()
291        {
292            colors[0] = SColor.floatGet(0xCCCCFFEE); // SColor.PERIWINKLE

294            colors[2] = SColor.floatGet(0xCC99CCFF); // SColor.MEDIUM_LAVENDER_MAGENTA
295            colors[3] = SColor.floatGet(0xC8A2C8EE); // SColor.LILAC
296            colors[4] = SColor.floatGet(0xBF00FFDD); // SColor.ELECTRIC_PURPLE
297            colors[5] = SColor.floatGet(0x6022EEBB); // SColor.ELECTRIC_INDIGO
298            colors[6] = SColor.floatGet(0x4B008277); // SColor.INDIGO
299            return this;

990         * @param percent The percentage of completion for this action, growing from 0 to 1 over the duration. If
991         *                {@link #setReverse(boolean) reversed}, this will shrink from 1 to 0.
992         */
993        @Override
994        protected void update(float percent) {
995            affected.clear();
996            Radius.inCircle(centers[0].x, centers[0].y, radius, false, validCells.width, validCells.height, affected);
997            int len = affected.size();
998            Coord c;
999            float f, light;
1000            Coord prev = centers[0];
1001            for (int i = 0; i < 16; i++) {
1002                centers[i] = centers[i].interpolate(end, adjust(percent, i));
1003            }
1004            validCells.not();
1005            if(!prev.equals(centers[0]))
1006            {
1007                FOV.reuseFOV(resMap, lightMaps[0], centers[0].x, centers[0].y, radius + 0.5);
1008                validCells.writeDoublesInto(lightMaps[0], 0.0);
1009            }
1010            for (int i = 1; i < 16; i++) {
1011                if(!centers[i-1].equals(centers[i]))
1012                {
1013                    FOV.reuseFOV(resMap, lightMaps[i], centers[i].x, centers[i].y, radius + 0.5);
1014                    validCells.writeDoublesInto(lightMaps[i], 0.0);
1015                }
1016                else
1017                {
1018                    lightMaps[i] = lightMaps[i-1];
1019                }
1020            }
1021            validCells.not();
1022            for (int i = 0; i < len; i++) {
1023                c = affected.get(i);
1024                for (int j = 0; j < 16; j++) {
1025                    if ((light = (float) lightMaps[j][c.x][c.y]) <= 0f)
1026                        continue;
1027                    target.blend(c.x, c.y, color,  light * 0.0625f);
1028                }
1029            }
1030        }
1031    }

1035        private final float value = 2, power = 3;
1036        @Override

1038            if (a <= 0.5f) return (1 - ((float)Math.pow(value, -power * (a * 2)) - 0.125f) * 1.1428572f) * 0.5f;
1039            return (1 + (float) Math.pow(value, power * (a * 2 - 2)) - 0.25f) * 0.5714286f;
