024/** PNG-8 encoder with compression. An instance can be reused to encode multiple PNGs with minimal allocation.
025 * You can configure the target palette and how this can dither colors via the {@link #palette} field, which is a
026 * {@link PaletteReducer} object that is allowed to be null and can be reused. The methods
027 * {@link PaletteReducer#exact(Color[])} or {@link PaletteReducer#analyze(Pixmap)} can be used to make the target
028 * palette match a specific set of colors or the colors in an existing image. You can use
029 * {@link PaletteReducer#setDitherStrength(float)} to reduce (or increase) dither strength; the dithering algorithm used
030 * here is a modified version of the algorithm presented in "Simple gradient-based error-diffusion method" by Xaingyu Y.
031 * Hu in the Journal of Electronic Imaging, 2016. This algorithm uses pseudo-randomly-generated noise (it is
032 * deterministic, and is seeded using the color information) to adjust Floyd-Steinberg dithering. It yields
033 * surprisingly non-random-looking dithers, but still manages to break up artificial patterns most of the time.
034 * Note that much of the time you will want to use {@link #writePrecisely(FileHandle, Pixmap, boolean)} instead of
035 * {@link #write(FileHandle, Pixmap, boolean, boolean)}, since writePrecisely will attempt to reproduce the exact colors
036 * if there are 256 colors or less in the Pixmap, and will automatically change to calling write() if there are more
037 * than 256 colors.

302    /**
303     * Attempts to write the given Pixmap exactly as a PNG-8 image to output; this attempt will only succeed if there
304     * are no more than 256 colors in the Pixmap (treating all partially transparent colors as fully transparent).
305     * If the attempt fails, this falls back to calling {@link #write(OutputStream, Pixmap, boolean, boolean)}, which
306     * can dither the image to use no more than 255 colors (plus fully transparent) based on ditherFallback and will
307     * always analyze the Pixmap to get an accurate-enough palette, using the given threshold for analysis (which is
308     * typically between 1 and 1000, and most often near 200-400). All other write() methods in this class will
309     * reduce the color depth somewhat, but as long as the color count stays at 256 or less, this will keep the
310     * non-alpha components of colors exactly.
311     * @param output an OutputStream that will not be closed
312     * @param pixmap a Pixmap to write to the given output stream
313     * @param ditherFallback if the Pixmap contains too many colors, this determines whether it will dither the output
314     * @param threshold the analysis threshold to use if there are too many colors (min 0, practical max is over 100000)
315     * @throws IOException if OutputStream things fail for any reason
316     */
317    public void writePrecisely(OutputStream output, Pixmap pixmap, boolean ditherFallback, int threshold) throws IOException {
318        IntIntMap colorToIndex = new IntIntMap(256);
319        colorToIndex.put(0, 0);
320        int color;
321        int hasTransparent = 0;
322        final int w = pixmap.getWidth(), h = pixmap.getHeight();
323        for (int y = 0; y < h; y++) {
324            int py = flipY ? (h - y - 1) : y;
325            for (int px = 0; px < w; px++) {
326                color = pixmap.getPixel(px, py);
327                if((color & 0xFE) != 0xFE) {
328                    if(hasTransparent == 0 && colorToIndex.size >= 256)                     
329                    {
330                        write(output, pixmap, true, ditherFallback, threshold);
331                        return;
332                    }
333                    hasTransparent = 1;
334                }
335                else if(!colorToIndex.containsKey(color))
336                {
337                    colorToIndex.put(color, colorToIndex.size & 255);
338                    if(colorToIndex.size == 257 && hasTransparent == 0)
339                    {
340                        colorToIndex.remove(0, 0);
341                    }
342                    if(colorToIndex.size > 256)
343                    {
344                        write(output, pixmap, true, ditherFallback, threshold);
345                        return;
346                    }
347                }
348            }
349        }
350        int[] paletteArray = new int[colorToIndex.size];
351        for(IntIntMap.Entry ent : colorToIndex)
352        {
353            paletteArray[ent.value] = ent.key;
354        }
355        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
356        DataOutputStream dataOutput = new DataOutputStream(output);
357        dataOutput.write(SIGNATURE);
358
359        buffer.writeInt(IHDR);
360        buffer.writeInt(pixmap.getWidth());
361        buffer.writeInt(pixmap.getHeight());
362        buffer.writeByte(8); // 8 bits per component.
363        buffer.writeByte(COLOR_INDEXED);
364        buffer.writeByte(COMPRESSION_DEFLATE);
365        buffer.writeByte(FILTER_NONE);
366        buffer.writeByte(INTERLACE_NONE);
367        buffer.endChunk(dataOutput);
368
369        buffer.writeInt(PLTE);
370        for (int i = 0; i < paletteArray.length; i++) {
371            int p = paletteArray[i];
372            buffer.write(p>>>24);
373            buffer.write(p>>>16);
374            buffer.write(p>>>8);
375        }
376        buffer.endChunk(dataOutput);
377
378        if(hasTransparent == 1) {
379            buffer.writeInt(TRNS);
380            buffer.write(0);
381            buffer.endChunk(dataOutput);
382        }
383        buffer.writeInt(IDAT);
384        deflater.reset();
385
386        int lineLen = pixmap.getWidth();
387        byte[] lineOut, curLine, prevLine;
388        if (lineOutBytes == null) {
389            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;
390            curLine = (curLineBytes = new ByteArray(lineLen)).items;
391            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;
392        } else {
393            lineOut = lineOutBytes.ensureCapacity(lineLen);
394            curLine = curLineBytes.ensureCapacity(lineLen);
395            prevLine = prevLineBytes.ensureCapacity(lineLen);
396            for (int i = 0, n = lastLineLen; i < n; i++)
397            {
398                prevLine[i] = 0;
399            }
400        }
401
402        lastLineLen = lineLen;
403
404        for (int y = 0; y < h; y++) {
405            int py = flipY ? (h - y - 1) : y;
406            for (int px = 0; px < w; px++) {
407                color = pixmap.getPixel(px, py);
408                curLine[px] = (byte) colorToIndex.get(color, 0);
409            }
410
411            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
412
413            //Paeth
414            for (int x = 1; x < lineLen; x++) {
415                int a = curLine[x - 1] & 0xff;
416                int b = prevLine[x] & 0xff;
417                int c = prevLine[x - 1] & 0xff;
418                int p = a + b - c;
419                int pa = p - a;
420                if (pa < 0) pa = -pa;
421                int pb = p - b;
422                if (pb < 0) pb = -pb;
423                int pc = p - c;
424                if (pc < 0) pc = -pc;
425                if (pa <= pb && pa <= pc)
426                    c = a;
427                else if (pb <= pc)
428                    c = b;
429                lineOut[x] = (byte)(curLine[x] - c);
430            }
431
432            deflaterOutput.write(PAETH);
433            deflaterOutput.write(lineOut, 0, lineLen);
434
435            byte[] temp = curLine;
436            curLine = prevLine;
437            prevLine = temp;
438        }
439        deflaterOutput.finish();
440        buffer.endChunk(dataOutput);
441
442        buffer.writeInt(IEND);
443        buffer.endChunk(dataOutput);
444
445        output.flush();
446
447    }

449    private void writeSolid (OutputStream output, Pixmap pixmap) throws IOException{
450        final int[] paletteArray = palette.paletteArray;
451        final byte[] paletteMapping = palette.paletteMapping;
452
453        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
454        DataOutputStream dataOutput = new DataOutputStream(output);
455        dataOutput.write(SIGNATURE);
456
457        buffer.writeInt(IHDR);
458        buffer.writeInt(pixmap.getWidth());
459        buffer.writeInt(pixmap.getHeight());
460        buffer.writeByte(8); // 8 bits per component.
461        buffer.writeByte(COLOR_INDEXED);
462        buffer.writeByte(COMPRESSION_DEFLATE);
463        buffer.writeByte(FILTER_NONE);
464        buffer.writeByte(INTERLACE_NONE);
465        buffer.endChunk(dataOutput);
466
467        buffer.writeInt(PLTE);
468        for (int i = 0; i < paletteArray.length; i++) {
469            int p = paletteArray[i];
470            buffer.write(p>>>24);
471            buffer.write(p>>>16);
472            buffer.write(p>>>8);
473        }
474        buffer.endChunk(dataOutput);
475
476        boolean hasTransparent = false;
477        if(paletteArray[0] == 0) {
478            hasTransparent = true;
479            buffer.writeInt(TRNS);
480            buffer.write(0);
481            buffer.endChunk(dataOutput);
482        }
483        buffer.writeInt(IDAT);
484        deflater.reset();
485
486        int lineLen = pixmap.getWidth();
487        byte[] lineOut, curLine, prevLine;
488        if (lineOutBytes == null) {
489            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;
490            curLine = (curLineBytes = new ByteArray(lineLen)).items;
491            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;
492        } else {
493            lineOut = lineOutBytes.ensureCapacity(lineLen);
494            curLine = curLineBytes.ensureCapacity(lineLen);
495            prevLine = prevLineBytes.ensureCapacity(lineLen);
496            for (int i = 0, n = lastLineLen; i < n; i++)
497            {
498                prevLine[i] = 0;
499            }
500        }
501
502        lastLineLen = lineLen;
503
504        int color;
505        final int w = pixmap.getWidth(), h = pixmap.getHeight();
506        for (int y = 0; y < h; y++) {
507            int py = flipY ? (h - y - 1) : y;
508            for (int px = 0; px < w; px++) {
509                color = pixmap.getPixel(px, py);
510                if ((color & 0x80) == 0 && hasTransparent)
511                    curLine[px] = 0;
512                else {
513                    int rr = ((color >>> 24)       );
514                    int gg = ((color >>> 16) & 0xFF);
515                    int bb = ((color >>> 8)  & 0xFF);
516                    curLine[px] = paletteMapping[((rr << 7) & 0x7C00)
517                                    | ((gg << 2) & 0x3E0)
518                                    | ((bb >>> 3))];
519                }
520            }
521
522            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
523
524            //Paeth
525            for (int x = 1; x < lineLen; x++) {
526                int a = curLine[x - 1] & 0xff;
527                int b = prevLine[x] & 0xff;
528                int c = prevLine[x - 1] & 0xff;
529                int p = a + b - c;
530                int pa = p - a;
531                if (pa < 0) pa = -pa;
532                int pb = p - b;
533                if (pb < 0) pb = -pb;
534                int pc = p - c;
535                if (pc < 0) pc = -pc;
536                if (pa <= pb && pa <= pc)
537                    c = a;
538                else if (pb <= pc)
539                    c = b;
540                lineOut[x] = (byte)(curLine[x] - c);
541            }
542
543            deflaterOutput.write(PAETH);
544            deflaterOutput.write(lineOut, 0, lineLen);
545
546            byte[] temp = curLine;
547            curLine = prevLine;
548            prevLine = temp;
549        }
550        deflaterOutput.finish();
551        buffer.endChunk(dataOutput);
552
553        buffer.writeInt(IEND);
554        buffer.endChunk(dataOutput);
555
556        output.flush();
557    }

559    private void writeDithered (OutputStream output, Pixmap pixmap) throws IOException{
560        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
561        final int[] paletteArray = palette.paletteArray;
562        final byte[] paletteMapping = palette.paletteMapping;
563
564
565        DataOutputStream dataOutput = new DataOutputStream(output);
566        dataOutput.write(SIGNATURE);
567
568        buffer.writeInt(IHDR);
569        buffer.writeInt(pixmap.getWidth());
570        buffer.writeInt(pixmap.getHeight());
571        buffer.writeByte(8); // 8 bits per component.
572        buffer.writeByte(COLOR_INDEXED);
573        buffer.writeByte(COMPRESSION_DEFLATE);
574        buffer.writeByte(FILTER_NONE);
575        buffer.writeByte(INTERLACE_NONE);
576        buffer.endChunk(dataOutput);
577
578        buffer.writeInt(PLTE);
579        for (int i = 0; i < paletteArray.length; i++) {
580            int p = paletteArray[i];
581            buffer.write(p>>>24);
582            buffer.write(p>>>16);
583            buffer.write(p>>>8);
584        }
585        buffer.endChunk(dataOutput);
586
587        boolean hasTransparent = false;
588        if(paletteArray[0] == 0) {
589            hasTransparent = true;
590            buffer.writeInt(TRNS);
591            buffer.write(0);
592            buffer.endChunk(dataOutput);
593        }
594        buffer.writeInt(IDAT);
595        deflater.reset();
596
597        final int w = pixmap.getWidth(), h = pixmap.getHeight();
598        byte[] lineOut, curLine, prevLine;
599        byte[] curErrorRed, nextErrorRed, curErrorGreen, nextErrorGreen, curErrorBlue, nextErrorBlue;
600        if (lineOutBytes == null) {
601            lineOut = (lineOutBytes = new ByteArray(w)).items;
602            curLine = (curLineBytes = new ByteArray(w)).items;
603            prevLine = (prevLineBytes = new ByteArray(w)).items;
604        } else {
605            lineOut = lineOutBytes.ensureCapacity(w);
606            curLine = curLineBytes.ensureCapacity(w);
607            prevLine = prevLineBytes.ensureCapacity(w);
608            for (int i = 0, n = lastLineLen; i < n; i++)
609            {
610                prevLine[i] = 0;
611            }
612        }
613        if(palette.curErrorRedBytes == null)
614        {
615            curErrorRed = (palette.curErrorRedBytes = new ByteArray(w)).items;
616            nextErrorRed = (palette.nextErrorRedBytes = new ByteArray(w)).items;
617            curErrorGreen = (palette.curErrorGreenBytes = new ByteArray(w)).items;
618            nextErrorGreen = (palette.nextErrorGreenBytes = new ByteArray(w)).items;
619            curErrorBlue = (palette.curErrorBlueBytes = new ByteArray(w)).items;
620            nextErrorBlue = (palette.nextErrorBlueBytes = new ByteArray(w)).items;
621        } else {
622            curErrorRed = palette.curErrorRedBytes.ensureCapacity(w);
623            nextErrorRed = palette.nextErrorRedBytes.ensureCapacity(w);
624            curErrorGreen = palette.curErrorGreenBytes.ensureCapacity(w);
625            nextErrorGreen = palette.nextErrorGreenBytes.ensureCapacity(w);
626            curErrorBlue = palette.curErrorBlueBytes.ensureCapacity(w);
627            nextErrorBlue = palette.nextErrorBlueBytes.ensureCapacity(w);
628            for (int i = 0; i < w; i++) {
629                nextErrorRed[i] = 0;
630                nextErrorGreen[i] = 0;
631                nextErrorBlue[i] = 0;
632            }
633
634        }
635
636
637        lastLineLen = w;
638
639        int color, used, rdiff, gdiff, bdiff, state = 0xFEEDBEEF;
640        byte er, eg, eb, paletteIndex;
641        float xi1, xi2, w1 = palette.ditherStrength * 0.125f, w3 = w1 * 3f, w5 = w1 * 5f, w7 = w1 * 7f;
642        for (int y = 0; y < h; y++) {
643            int py = flipY ? (h - y - 1) : y;
644            int ny = flipY ? (h - y - 2) : y + 1;
645            for (int i = 0; i < w; i++) {
646                curErrorRed[i] = nextErrorRed[i];
647                curErrorGreen[i] = nextErrorGreen[i];
648                curErrorBlue[i] = nextErrorBlue[i];
649                nextErrorRed[i] = 0;
650                nextErrorGreen[i] = 0;
651                nextErrorBlue[i] = 0;
652            }
653            for (int px = 0; px < w; px++) {
654                color = pixmap.getPixel(px, py) & 0xF8F8F880;
655                if ((color & 0x80) == 0 && hasTransparent)
656                    curLine[px] = 0;
657                else {
658                    er = curErrorRed[px];
659                    eg = curErrorGreen[px];
660                    eb = curErrorBlue[px];
661                    color |= (color >>> 5 & 0x07070700) | 0xFE;
662                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er), 0, 0xFF);
663                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg), 0, 0xFF);
664                    int bb = MathUtils.clamp(((color >>> 8)  & 0xFF) + (eb), 0, 0xFF);
665                    curLine[px] = paletteIndex =
666                            paletteMapping[((rr << 7) & 0x7C00)
667                                    | ((gg << 2) & 0x3E0)
668                                    | ((bb >>> 3))];
669                    used = paletteArray[paletteIndex & 0xFF];
670                    rdiff = (color>>>24)-    (used>>>24);
671                    gdiff = (color>>>16&255)-(used>>>16&255);
672                    bdiff = (color>>>8&255)- (used>>>8&255);
673                    state += (color + 0x41C64E6D) ^ color >>> 7;
674                    state = (state << 21 | state >>> 11);
675                    xi1 = randomXi(state);
676                    state ^= (state << 15 | state >>> 17) + 0x9E3779B9;
677                    xi2 = randomXi(state);
678                    if(px < w - 1)
679                    {
680                        curErrorRed[px+1]   += rdiff * w7 * (1f + xi1);
681                        curErrorGreen[px+1] += gdiff * w7 * (1f + xi1);
682                        curErrorBlue[px+1]  += bdiff * w7 * (1f + xi1);
683                    }
684                    if(ny < h)
685                    {
686                        if(px > 0)
687                        {
688                            nextErrorRed[px-1]   += rdiff * w3 * (1f + xi2);
689                            nextErrorGreen[px-1] += gdiff * w3 * (1f + xi2);
690                            nextErrorBlue[px-1]  += bdiff * w3 * (1f + xi2);
691                        }
692                        if(px < w - 1)
693                        {
694                            nextErrorRed[px+1]   += rdiff * w1 * (1f - xi2);
695                            nextErrorGreen[px+1] += gdiff * w1 * (1f - xi2);
696                            nextErrorBlue[px+1]  += bdiff * w1 * (1f - xi2);
697                        }
698                        nextErrorRed[px]   += rdiff * w5 * (1f - xi1);
699                        nextErrorGreen[px] += gdiff * w5 * (1f - xi1);
700                        nextErrorBlue[px]  += bdiff * w5 * (1f - xi1);
701                    }
702                }
703            }
704
705            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
706
707            //Paeth
708            for (int x = 1; x < w; x++) {
709                int a = curLine[x - 1] & 0xff;
710                int b = prevLine[x] & 0xff;
711                int c = prevLine[x - 1] & 0xff;
712                int p = a + b - c;
713                int pa = p - a;
714                if (pa < 0) pa = -pa;
715                int pb = p - b;
716                if (pb < 0) pb = -pb;
717                int pc = p - c;
718                if (pc < 0) pc = -pc;
719                if (pa <= pb && pa <= pc)
720                    c = a;
721                else if (pb <= pc)
722                    c = b;
723                lineOut[x] = (byte)(curLine[x] - c);
724            }
725
726            deflaterOutput.write(PAETH);
727            deflaterOutput.write(lineOut, 0, w);
728
729            byte[] temp = curLine;
730            curLine = prevLine;
731            prevLine = temp;
732        }
733        deflaterOutput.finish();
734        buffer.endChunk(dataOutput);
735
736        buffer.writeInt(IEND);
737        buffer.endChunk(dataOutput);
738
739        output.flush();
740    }

747    static class ChunkBuffer extends DataOutputStream {
748        final ByteArrayOutputStream buffer;
749        final CRC32 crc;
750
751        ChunkBuffer (int initialSize) {
752            this(new ByteArrayOutputStream(initialSize), new CRC32());
753        }
754
755        private ChunkBuffer (ByteArrayOutputStream buffer, CRC32 crc) {
756            super(new CheckedOutputStream(buffer, crc));
757            this.buffer = buffer;
758            this.crc = crc;
759        }
760
761        public void endChunk (DataOutputStream target) throws IOException {
762            flush();
763            target.writeInt(buffer.size() - 4);
764            buffer.writeTo(target);
765            target.writeInt((int)crc.getValue());
766            buffer.reset();
767            crc.reset();
768        }
769    }
