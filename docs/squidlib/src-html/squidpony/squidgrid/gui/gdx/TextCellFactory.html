255     * Sets this factory to use the provided font.
256     *
257     * This is a way to complete a needed step; the font must be set before initializing, which can be done by a few
258     * methods in this class.
259     *
260     * This should be called with an argument such as "Rogue-Zodiac-6x12.fnt", that is, it should have the .fnt
261     * extension as opposed to the .png that accompanies such a bitmap font. The bitmap font should be either in the
262     * internal folder that libGDX knows about, which means it is in the assets folder of your project usually, or it
263     * can be on the classpath, which mostly applies to these resources bundled with SquidLib:
264     * <ul>
265     *     <li>DefaultResources.squareName = "Zodiac-Square-12x12.fnt"</li>
266     *     <li>DefaultResources.narrowName = "Rogue-Zodiac-6x12.fnt"</li>
267     *     <li>DefaultResources.unicodeName = "Mandrill-6x16.fnt"</li>
268     *     <li>DefaultResources.smoothName = "Inconsolata-LGC-8x18.fnt"</li>
269     *     <li>DefaultResources.squareNameLarge = "Zodiac-Square-24x24.fnt"</li>
270     *     <li>DefaultResources.narrowNameLarge = "Rogue-Zodiac-12x24.fnt"</li>
271     *     <li>DefaultResources.unicodeNameLarge = "Mandrill-12x32.fnt"</li>
272     *     <li>DefaultResources.smoothNameLarge = "Inconsolata-LGC-12x24.fnt"</li>
273     *     <li>DefaultResources.narrowNameExtraLarge = "Rogue-Zodiac-18x36.fnt"</li>
274     *     <li>There is also a sequence of resized versions of Inconsolata LGC, altered to fit in a square area. These
275     *     don't have names in DefaultResources; you should use the overload of font() that takes a BitmapFont if you
276     *     want to use the multiple, increasingly-resized versions.</li>
277     * </ul>
278     * "Rogue-Zodiac-12x24.fnt", which is easily accessed by the field DefaultResources.narrowNameLarge , can also
279     * be set using TextCellFactory.defaultNarrowFont() instead of font(). "Zodiac-Square-12x12.fnt", also accessible
280     * as DefaultResources.squareName , can be set using TextCellFactory.defaultSquareFont() instead of font().
281     * "Inconsolata-LGC-12x24.fnt", also accessible as DefaultResources.smoothNameLarge , can be set using
282     * TextCellFactory.defaultFont() instead of font(). All three of these alternatives will cache the BitmapFont if
283     * the same one is requested later, but this font() method will not.
284     * <br>
285     * See https://github.com/libgdx/libgdx/wiki/Hiero for some ways to create a bitmap font this can use. Several fonts
286     * in this list were created using Hiero (not Hiero4), and several were created with AngelCode's BMFont tool.
287     *
288     * @param fontpath the path to the font to use
289     * @return this factory for method chaining
290     */
291    public TextCellFactory font(String fontpath) {
292        if (assetManager == null) {
293            if (Gdx.files.internal(fontpath).exists())
294                bmpFont = new BitmapFont(Gdx.files.internal(fontpath));
295            else if (Gdx.files.classpath(fontpath).exists())
296                bmpFont = new BitmapFont(Gdx.files.classpath(fontpath));
297            else
298                bmpFont = DefaultResources.getIncludedFont();
299        }
300        else {
301            assetManager.load(new AssetDescriptor<>(fontpath, BitmapFont.class));
302                        /*
303                         * We're using the AssetManager not be asynchronous, but to avoid
304                         * loading a file twice (because that takes some time (tens of
305                         * milliseconds)). Hence this KISS code to avoid having to handle a
306                         * not-yet-loaded font:
307                         */
308            assetManager.finishLoading();
309            bmpFont = assetManager.get(fontpath, BitmapFont.class);
310        }
311        return this;
312    }

359    /**
360     * Sets the font to a distance field font with the given String path to a .fnt file and String path to a texture.
361     * Distance field fonts should scale cleanly to multiple resolutions without artifacts. Does not use AssetManager
362     * since you shouldn't need to reload the font if it scales with one image. You need to configure the shader to use
363     * distance field fonts unless a class already does this for you (SquidLayers handles shader configuration
364     * internally, for example). TextCellFactory has a method, configureShader(Batch), that does this and should be
365     * called while that Batch has begun rendering, typically in an override of some containing Scene2D Group's
366     * draw(Batch, float) method.
367     * <br>
368     * At least two distance field fonts are included in SquidLib; one is square, one is narrow, and they can both be
369     * accessed using either the predefined TextCellFactory objects in DefaultResources, accessible with
370     * getStretchableFont() for narrow or getStretchableSquareFont() for square, or the setter methods in this class,
371     * defaultDistanceFieldFont() for square and defaultNarrowDistanceFieldFont() for narrow.
372     * <br>
373     * To create distance field fonts that work well with monospace layout is... time-consuming and error-prone, though
374     * not especially difficult for most fonts. The process is documented as well as we can, given how differently all
375     * fonts are made, in a file not included in the distribution JAR but present on GitHub:
376     * https://github.com/SquidPony/SquidLib/blob/master/squidlib/etc/making-distance-field-fonts.txt
377     * @param fontPath the path to a .fnt bitmap font file with distance field effects applied, which requires a complex
378     *                 process to create.
379     * @param texturePath the path to the texture used by the bitmap font
380     * @return this factory for method chaining
381     */
382    public TextCellFactory fontDistanceField(String fontPath, String texturePath) {
383        Texture tex;
384        if (Gdx.files.internal(texturePath).exists()) {
385            Gdx.app.debug("font", "Using internal font texture at " + texturePath);
386            tex = new Texture(Gdx.files.internal(texturePath), true);
387            tex.setFilter(Texture.TextureFilter.Linear, Texture.TextureFilter.Linear);
388        } else if (Gdx.files.classpath(texturePath).exists()) {
389            Gdx.app.debug("font", "Using classpath font texture at " + texturePath);
390            tex = new Texture(Gdx.files.classpath(texturePath), true);
391            tex.setFilter(Texture.TextureFilter.Linear, Texture.TextureFilter.Linear);
392        } else {
393            bmpFont = DefaultResources.getIncludedFont();
394            Gdx.app.error("TextCellFactory", "Could not find font file: " + texturePath + ", using defaults");
395            return this;
396        }
397        if (Gdx.files.internal(fontPath).exists()) {
398            Gdx.app.debug("font", "Using internal font at " + fontPath);
399            bmpFont = new BitmapFont(Gdx.files.internal(fontPath), new TextureRegion(tex), false);
400            distanceField = true;
401        } else if (Gdx.files.classpath(fontPath).exists()) {
402            Gdx.app.debug("font", "Using classpath font at " + fontPath);
403            bmpFont = new BitmapFont(Gdx.files.classpath(fontPath), new TextureRegion(tex), false);
404            distanceField = true;
405        } else {
406            bmpFont = DefaultResources.getIncludedFont();
407            Gdx.app.error("TextCellFactory", "Could not find font file: " + fontPath + ", using defaults");
408        }
409        //bmpFont.getData().padBottom = bmpFont.getDescent();
410        distanceFieldScaleX = bmpFont.getSpaceWidth() - 1f;
411        distanceFieldScaleY = bmpFont.getLineHeight() - 1f;
412        return this;
413    }

793     * Returns true if the given character will fit inside the current cell
794     * dimensions with the current font.
795     *
796     * ISO Control characters, non-printing characters and invalid unicode
797     * characters are all considered by definition to fit.
798     *
799     * @param codepoint the codepoint of the char in question
800     * @return true if the char will fit, or if it is non-printing in some way; false otherwise
801     */
802    public boolean willFit(int codepoint) {
803        if (!initialized) {
804            throw new IllegalStateException("This factory has not yet been initialized!");
805        }
806
807        if (!Character.isValidCodePoint(codepoint) ||
808                (codepoint <= 0x001F) || (codepoint >= 0x007F && codepoint <= 0x009F)) // same as isIsoControl
809        {
810            return true;
811        }
812
813        return fitting.contains(String.valueOf(Character.toChars(codepoint)));
814    }
