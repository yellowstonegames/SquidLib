607    /**
608     * Place a char c into the foreground, with a foreground color specified by an index into the default palette, a
609     * background color specified in the same way, and a lightness variation for the background (0 is no change, 100 is
610     * very bright, -100 is very dark, anything past -150 or 150 will make the background almost fully black or white).
611     *
612     * @param x                   in grid cells.
613     * @param y                   in grid cells.
614     * @param c                   a character to be drawn in the foreground
615     * @param foregroundIndex     int index into the default palette for the char being drawn
616     * @param backgroundIndex     int index into the default palette for the background
617     * @param backgroundLightness int between -255 and 255 , lower numbers are darker, higher lighter.
618     */
619    public SquidLayers put(int x, int y, char c, int foregroundIndex, int backgroundIndex, int backgroundLightness) {
620        backgroundLightness = clamp(backgroundLightness, -255, 255);
621        foregroundPanel.put(x, y, c, foregroundIndex, palette);
622        values[x][y] = true;
623        lightnesses[x][y] = 256 + clamp(backgroundLightness, -255, 255);
624        backgroundPanel.put(x, y, palette.get(backgroundIndex), lightnesses[x][y] / 512f);
625        return this;
626    }
