164    /**
165     * Builds a panel with the given grid size and all other parameters determined by the factory. Even if sprite images
166     * are being used, a TextCellFactory is still needed to perform sizing and other utility functions. Importantly,
167     * this constructor takes a 2D char array argument that can be sized differently than the displayed area. The
168     * displayed area is gridWidth by gridHeight in cells, but the actualMap argument can be much larger, and only a
169     * portion will be displayed at a time. This requires some special work with the Camera and Viewports to get working
170     * correctly; in the squidlib module's examples, EverythingDemo may be a good place to see how this can be done.
171     * You can pass null for actualMap, which will simply create a char array to use internally that is exactly
172     * gridWidth by gridHeight, in cells.
173     * <br>
174     * If the TextCellFactory has not yet been initialized, then it will be sized at 12x12 px per cell. If it is null
175     * then a default one will be created and initialized. The xOffset and yOffset arguments are measured in pixels or
176     * whatever sub-cell unit of measure your game uses (world coordinates, in libGDX parlance), and change where the
177     * SquidPanel starts drawing by simply adding to the initial x and y coordinates. 0 and 0 are usually fine.
178     *
179     * @param gridWidth the number of cells horizontally
180     * @param gridHeight the number of cells vertically
181     * @param factory the factory to use for cell rendering
182     * @param center the color center to use. Can be {@code null}, which will use a default
183     * @param xOffset the x offset to start rendering at, in pixels (or some other sub-cell measurement your game uses)
184     * @param yOffset the y offset to start rendering at, in pixels (or some other sub-cell measurement your game uses)
185     * @param actualMap will often be a different size than gridWidth by gridHeight, which enables camera scrolling
186     */
187    public SquidPanel(int gridWidth, int gridHeight, TextCellFactory factory, IColorCenter<Color> center,
188                      float xOffset, float yOffset, char[][] actualMap) {
189        this.gridWidth = gridWidth;
190        this.gridHeight = gridHeight;
191        if(center == null)
192            scc = DefaultResources.getSCC();
193        else
194            scc = center;
195        if (factory == null) {
196            textFactory = new TextCellFactory();
197        }
198        else
199            textFactory = factory;
200        if (!textFactory.initialized()) {
201            textFactory.initByFont();
202        }
203
204        cellWidth = MathUtils.round(textFactory.actualCellWidth);
205        cellHeight = MathUtils.round(textFactory.actualCellHeight);
206
207        if(actualMap == null || actualMap.length <= 0)
208            contents = ArrayTools.fill(' ', gridWidth, gridHeight);
209        else
210            contents = actualMap;
211        colors = ArrayTools.fill(scc.filter(Color.CLEAR).toFloatBits(), contents.length, contents[0].length);
212
213        int w = gridWidth * cellWidth;
214        int h = gridHeight * cellHeight;
215        this.xOffset = xOffset;
216        this.yOffset = yOffset;
217        setSize(w, h);
218        animatedEntities = new OrderedSet<>();
219        autoActors = new OrderedSet<>();
220    }

382    /**
383     * Puts the given string in the chosen direction, with the first character shown (not necessarily the first in the
384     * string) at the given offset. If you use {@link Direction#LEFT}, then this effectively reverses the String and
385     * prints it with the last character of the String at the minimum-x position, which is the same position that the
386     * first character would be if you printed normally or if you gave this RIGHT (x is xOffset, y is yOffset). Giving
387     * UP acts similarly to LEFT, but has the last character at the minimum-y position and has the first character below
388     * it. The diagonals act as you would expect, combining the behavior of one of UP or DOWN with one of LEFT or RIGHT.
389     * <br>
390     * Does not word wrap. Characters that are not renderable (due to being at negative offsets or offsets greater than
391     * the grid size) will not be shown but will not cause any malfunctions.
392     *
393     * @param xOffset the x coordinate of the first character
394     * @param yOffset the y coordinate of the first character
395     * @param string the characters to be displayed
396     * @param foreground the color to draw the characters
397     * @param direction the direction the text should be written in, such as {@link Direction#RIGHT} for normal layout
398     */
399    public void put(int xOffset, int yOffset, String string, Color foreground, Direction direction) {
400        float enc = scc.filter(foreground).toFloatBits();
401        switch (direction)
402        {
403            case DOWN:
404                for (int i = 0; i < string.length(); i++) {
405                    put(xOffset, yOffset + i, string.charAt(i), enc);
406                }
407                break;
408            case UP:
409                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
410                    put(xOffset, yOffset + p, string.charAt(i), enc);
411                }
412                break;
413            case LEFT:
414                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
415                    put(xOffset + p, yOffset, string.charAt(i), enc);
416                }
417                break;
418            case DOWN_RIGHT:
419                for (int i = 0; i < string.length(); i++) {
420                    put(xOffset + i, yOffset + i, string.charAt(i), enc);
421                }
422                break;
423            case UP_RIGHT:
424                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
425                    put(xOffset + i, yOffset + p, string.charAt(i), enc);
426                }
427                break;
428            case UP_LEFT:
429                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
430                    put(xOffset + p, yOffset + p, string.charAt(i), enc);
431                }
432                break;
433            case DOWN_LEFT:
434                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
435                    put(xOffset + p, yOffset + i, string.charAt(i), enc);
436                }
437                break;
438            default:
439                for (int i = 0; i < string.length(); i++) {
440                    put(xOffset + i, yOffset, string.charAt(i), enc);
441                }
442        }
443    }

2341     * Given a "snapshot" from {@link #getSnapshot(int, int, int, int)}, this assigns the chars and colors in this
2342     * SquidPanel from the position given by putX,putY (inclusive) to putX+limitWidth,putY+limitHeight (exclusive) so
2343     * they have the values stored in the snapshot. If limitWidth or limitHeight is negative, this uses the full width
2344     * and height of the snapshot (stopping early if it would extend past the gridWidth or gridHeight of this
2345     * SquidPanel).
2346     * @param snapshot a StringBuilder in a special format as produced by {@link #getSnapshot(int, int, int, int)}
2347     * @param putX where to start placing the data from the snapshot, x position
2348     * @param putY where to start placing the data from the snapshot, y position
2349     * @param limitWidth if negative, uses all of snapshot's width as possible, otherwise restricts the width allowed
2350     * @param limitHeight if negative, uses all of snapshot's height as possible, otherwise restricts the height allowed
2351     * @return this after setting, for chaining
2352     */
2353    public SquidPanel setFromSnapshot(StringBuilder snapshot, int putX, int putY, int limitWidth, int limitHeight)
2354    {
2355        if(putX >= gridWidth || putY >= gridHeight || snapshot == null || snapshot.length() < 4) return this;
2356        if(putX < 0) putX = 0;
2357        if(putY < 0) putY = 0;
2358        int start = snapshot.indexOf(":")+1, width = StringKit.intFromDec(snapshot),
2359                height = StringKit.intFromDec(snapshot, snapshot.indexOf("x") + 1, start),
2360                run = start;
2361        if(limitWidth < 0)
2362            limitWidth = Math.min(width, gridWidth - putX);
2363        else
2364            limitWidth = Math.min(limitWidth, Math.min(width, gridWidth - putX));
2365
2366        if(limitHeight < 0)
2367            limitHeight = Math.min(height, gridHeight - putY);
2368        else
2369            limitHeight = Math.min(limitHeight, Math.min(height, gridHeight - putY));
2370        for (int x = putX, i = 0; i < limitWidth; x++, i++, run += height) {
2371            snapshot.getChars(run, run + limitHeight, contents[x], putY);
2372        }
2373        run = start + width * height;
2374        for (int x = putX, i = 0; i < limitWidth; x++, i++) {
2375            for (int y = putY, j = 0; j < limitHeight; y++, j++) {
2376                colors[x][y] = SColor.charsToFloat(snapshot, run);
