025 * You can configure the target palette and how this can dither colors via the {@link #palette} field, which is a
026 * {@link PaletteReducer} object that is allowed to be null and can be reused. The methods
027 * {@link PaletteReducer#exact(Color[])} or {@link PaletteReducer#analyze(Pixmap)} can be used to make the target
028 * palette match a specific set of colors or the colors in an existing image. You can use
029 * {@link PaletteReducer#setDitherStrength(float)} to reduce (or maybe increase) dither strength; the dithering
030 * algorithm used here is a modified version of the algorithm presented in "Simple gradient-based error-diffusion
031 * method" by Xaingyu Y. Hu in the Journal of Electronic Imaging, 2016. This algorithm uses pseudo-randomly-generated

195     * Writes the given Pixmap to the requested FileHandle, optionally computing an 8-bit palette from the most common

197     * will use 256 colors in its palette exactly with no transparent entry, but if there are more than 256 colors or

207        OutputStream output = file.write(false);
208        try {
209            write(output, pixmap, computePalette);
210        } finally {
211            StreamUtils.closeQuietly(output);
212        }
213    }

269    private void writeSolid (OutputStream output, Pixmap pixmap, boolean computePalette) throws IOException{
270        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
271        if(palette == null)
272        {
273            palette = new PaletteReducer(pixmap);
274        }
275        else if(computePalette)
276        {
277            palette.analyze(pixmap);
278        }
279        final int[] paletteArray = palette.paletteArray;
280        final byte[] paletteMapping = palette.paletteMapping;
281
282
283        DataOutputStream dataOutput = new DataOutputStream(output);
284        dataOutput.write(SIGNATURE);
285
286        buffer.writeInt(IHDR);
287        buffer.writeInt(pixmap.getWidth());
288        buffer.writeInt(pixmap.getHeight());
289        buffer.writeByte(8); // 8 bits per component.
290        buffer.writeByte(COLOR_INDEXED);
291        buffer.writeByte(COMPRESSION_DEFLATE);
292        buffer.writeByte(FILTER_NONE);
293        buffer.writeByte(INTERLACE_NONE);
294        buffer.endChunk(dataOutput);
295
296        buffer.writeInt(PLTE);
297        for (int i = 0; i < paletteArray.length; i++) {
298            int p = paletteArray[i];
299            buffer.write(p>>>24);
300            buffer.write(p>>>16);
301            buffer.write(p>>>8);
302        }
303        buffer.endChunk(dataOutput);
304
305        boolean hasTransparent = false;
306        if(paletteArray[0] == 0) {
307            hasTransparent = true;
308            buffer.writeInt(TRNS);
309            buffer.write(0);
310            buffer.endChunk(dataOutput);
311        }
312        buffer.writeInt(IDAT);
313        deflater.reset();
314
315        int lineLen = pixmap.getWidth();
316        byte[] lineOut, curLine, prevLine;
317        if (lineOutBytes == null) {
318            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;
319            curLine = (curLineBytes = new ByteArray(lineLen)).items;
320            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;
321        } else {
322            lineOut = lineOutBytes.ensureCapacity(lineLen);
323            curLine = curLineBytes.ensureCapacity(lineLen);
324            prevLine = prevLineBytes.ensureCapacity(lineLen);
325            for (int i = 0, n = lastLineLen; i < n; i++)
326            {
327                prevLine[i] = 0;
328            }
329        }
330
331        lastLineLen = lineLen;
332
333        ByteBuffer pixels = pixmap.getPixels();
334        int oldPosition = pixels.position(), color;
335        final int w = pixmap.getWidth();
336        for (int y = 0, h = pixmap.getHeight(); y < h; y++) {
337            int py = flipY ? (h - y - 1) : y;
338            for (int px = 0; px < w; px++) {
339                color = pixmap.getPixel(px, py);
340                if ((color & 0x80) == 0 && hasTransparent)
341                    curLine[px] = 0;
342                else {
343                    int rr = ((color >>> 24)       );
344                    int gg = ((color >>> 16) & 0xFF);
345                    int bb = ((color >>> 8)  & 0xFF);
346                    curLine[px] = paletteMapping[((rr << 7) & 0x7C00)
347                                    | ((gg << 2) & 0x3E0)
348                                    | ((bb >>> 3))];
349                }
350            }
351
352            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
353
354            //Paeth
355            for (int x = 1; x < lineLen; x++) {
356                int a = curLine[x - 1] & 0xff;
357                int b = prevLine[x] & 0xff;
358                int c = prevLine[x - 1] & 0xff;
359                int p = a + b - c;
360                int pa = p - a;
361                if (pa < 0) pa = -pa;
362                int pb = p - b;
363                if (pb < 0) pb = -pb;
364                int pc = p - c;
365                if (pc < 0) pc = -pc;
366                if (pa <= pb && pa <= pc)
367                    c = a;
368                else if (pb <= pc)
369                    c = b;
370                lineOut[x] = (byte)(curLine[x] - c);
371            }
372
373            deflaterOutput.write(PAETH);
374            deflaterOutput.write(lineOut, 0, lineLen);
375
376            byte[] temp = curLine;
377            curLine = prevLine;
378            prevLine = temp;
379        }
380        pixels.position(oldPosition);
381        deflaterOutput.finish();
382        buffer.endChunk(dataOutput);
383
384        buffer.writeInt(IEND);
385        buffer.endChunk(dataOutput);
386
387        output.flush();
388    }

390    private void writeDithered (OutputStream output, Pixmap pixmap, boolean computePalette) throws IOException{
391        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
392        if(palette == null)
393        {
394            palette = new PaletteReducer(pixmap);
395        }
396        else if(computePalette)
397        {
398            palette.analyze(pixmap);
399        }
400        final int[] paletteArray = palette.paletteArray;
401        final byte[] paletteMapping = palette.paletteMapping;
402
403
404        DataOutputStream dataOutput = new DataOutputStream(output);
405        dataOutput.write(SIGNATURE);
406
407        buffer.writeInt(IHDR);
408        buffer.writeInt(pixmap.getWidth());
409        buffer.writeInt(pixmap.getHeight());
410        buffer.writeByte(8); // 8 bits per component.
411        buffer.writeByte(COLOR_INDEXED);
412        buffer.writeByte(COMPRESSION_DEFLATE);
413        buffer.writeByte(FILTER_NONE);
414        buffer.writeByte(INTERLACE_NONE);
415        buffer.endChunk(dataOutput);
416
417        buffer.writeInt(PLTE);
418        for (int i = 0; i < paletteArray.length; i++) {
419            int p = paletteArray[i];
420            buffer.write(p>>>24);
421            buffer.write(p>>>16);
422            buffer.write(p>>>8);
423        }
424        buffer.endChunk(dataOutput);
425
426        boolean hasTransparent = false;
427        if(paletteArray[0] == 0) {
428            hasTransparent = true;
429            buffer.writeInt(TRNS);
430            buffer.write(0);
431            buffer.endChunk(dataOutput);
432        }
433        buffer.writeInt(IDAT);
434        deflater.reset();
435
436        final int w = pixmap.getWidth(), h = pixmap.getHeight();
437        byte[] lineOut, curLine, prevLine;
438        byte[] curErrorRed, nextErrorRed, curErrorGreen, nextErrorGreen, curErrorBlue, nextErrorBlue;
439        if (lineOutBytes == null) {
440            lineOut = (lineOutBytes = new ByteArray(w)).items;
441            curLine = (curLineBytes = new ByteArray(w)).items;
442            prevLine = (prevLineBytes = new ByteArray(w)).items;
443        } else {
444            lineOut = lineOutBytes.ensureCapacity(w);
445            curLine = curLineBytes.ensureCapacity(w);
446            prevLine = prevLineBytes.ensureCapacity(w);
447            for (int i = 0, n = lastLineLen; i < n; i++)
448            {
449                prevLine[i] = 0;
450            }
451        }
452        if(palette.curErrorRedBytes == null)
453        {
454            curErrorRed = (palette.curErrorRedBytes = new ByteArray(w)).items;
455            nextErrorRed = (palette.nextErrorRedBytes = new ByteArray(w)).items;
456            curErrorGreen = (palette.curErrorGreenBytes = new ByteArray(w)).items;
457            nextErrorGreen = (palette.nextErrorGreenBytes = new ByteArray(w)).items;
458            curErrorBlue = (palette.curErrorBlueBytes = new ByteArray(w)).items;
459            nextErrorBlue = (palette.nextErrorBlueBytes = new ByteArray(w)).items;
460        } else {
461            curErrorRed = palette.curErrorRedBytes.ensureCapacity(w);
462            nextErrorRed = palette.nextErrorRedBytes.ensureCapacity(w);
463            curErrorGreen = palette.curErrorGreenBytes.ensureCapacity(w);
464            nextErrorGreen = palette.nextErrorGreenBytes.ensureCapacity(w);
465            curErrorBlue = palette.curErrorBlueBytes.ensureCapacity(w);
466            nextErrorBlue = palette.nextErrorBlueBytes.ensureCapacity(w);
467            for (int i = 0; i < w; i++) {
468                nextErrorRed[i] = 0;
469                nextErrorGreen[i] = 0;
470                nextErrorBlue[i] = 0;
471            }
472
473        }
474
475
476        lastLineLen = w;
477
478        int color, used, rdiff, gdiff, bdiff, state = 0xFEEDBEEF;
479        byte er, eg, eb, paletteIndex;
480        float xi1, xi2, w1 = palette.ditherStrength * 0.125f, w3 = w1 * 3f, w5 = w1 * 5f, w7 = w1 * 7f;
481        for (int y = 0; y < h; y++) {
482            int py = flipY ? (h - y - 1) : y;
483            int ny = flipY ? (h - y - 2) : y + 1;
484            for (int i = 0; i < w; i++) {
485                curErrorRed[i] = nextErrorRed[i];
486                curErrorGreen[i] = nextErrorGreen[i];
487                curErrorBlue[i] = nextErrorBlue[i];
488                nextErrorRed[i] = 0;
489                nextErrorGreen[i] = 0;
490                nextErrorBlue[i] = 0;
491            }
492            for (int px = 0; px < w; px++) {
493                color = pixmap.getPixel(px, py) & 0xF8F8F880;
494                if ((color & 0x80) == 0 && hasTransparent)
495                    curLine[px] = 0;
496                else {
497                    er = curErrorRed[px];
498                    eg = curErrorGreen[px];
499                    eb = curErrorBlue[px];
500                    color |= (color >>> 5 & 0x07070700) | 0xFE;
501                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er), 0, 0xFF);
502                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg), 0, 0xFF);
503                    int bb = MathUtils.clamp(((color >>> 8)  & 0xFF) + (eb), 0, 0xFF);
504                    curLine[px] = paletteIndex =
505                            paletteMapping[((rr << 7) & 0x7C00)
506                                    | ((gg << 2) & 0x3E0)
507                                    | ((bb >>> 3))];
508                    used = paletteArray[paletteIndex & 0xFF];
509                    rdiff = (color>>>24)-    (used>>>24);
510                    gdiff = (color>>>16&255)-(used>>>16&255);
511                    bdiff = (color>>>8&255)- (used>>>8&255);
512                    state += (color + 0x41C64E6D) ^ color >>> 7;
513                    state = (state << 21 | state >>> 11);
514                    xi1 = randomXi(state);
515                    state = (state << 15 | state >>> 17) ^ 0x9E3779B9;
516                    xi2 = randomXi(state);
517                    if(px < w - 1)
518                    {
519                        curErrorRed[px+1]   += rdiff * w7 * (1f + xi1);
520                        curErrorGreen[px+1] += gdiff * w7 * (1f + xi1);
521                        curErrorBlue[px+1]  += bdiff * w7 * (1f + xi1);
522                    }
523                    if(ny < h)
524                    {
525                        if(px > 0)
526                        {
527                            nextErrorRed[px-1]   += rdiff * w3 * (1f + xi2);
528                            nextErrorGreen[px-1] += gdiff * w3 * (1f + xi2);
529                            nextErrorBlue[px-1]  += bdiff * w3 * (1f + xi2);
530                        }
531                        if(px < w - 1)
532                        {
533                            nextErrorRed[px+1]   += rdiff * w1 * (1f - xi2);
534                            nextErrorGreen[px+1] += gdiff * w1 * (1f - xi2);
535                            nextErrorBlue[px+1]  += bdiff * w1 * (1f - xi2);
536                        }
537                        nextErrorRed[px]   += rdiff * w5 * (1f - xi1);
538                        nextErrorGreen[px] += gdiff * w5 * (1f - xi1);
539                        nextErrorBlue[px]  += bdiff * w5 * (1f - xi1);
540                    }
541                }
542            }
543
544            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
545
546            //Paeth
547            for (int x = 1; x < w; x++) {
548                int a = curLine[x - 1] & 0xff;
549                int b = prevLine[x] & 0xff;
550                int c = prevLine[x - 1] & 0xff;
551                int p = a + b - c;
552                int pa = p - a;
553                if (pa < 0) pa = -pa;
554                int pb = p - b;
555                if (pb < 0) pb = -pb;
556                int pc = p - c;
557                if (pc < 0) pc = -pc;
558                if (pa <= pb && pa <= pc)
559                    c = a;
560                else if (pb <= pc)
561                    c = b;
562                lineOut[x] = (byte)(curLine[x] - c);
563            }
564
565            deflaterOutput.write(PAETH);
566            deflaterOutput.write(lineOut, 0, w);
567
568            byte[] temp = curLine;
569            curLine = prevLine;
570            prevLine = temp;
571        }
572        deflaterOutput.finish();
573        buffer.endChunk(dataOutput);
574
575        buffer.writeInt(IEND);
576        buffer.endChunk(dataOutput);
577
578        output.flush();
579    }

586    static class ChunkBuffer extends DataOutputStream {
587        final ByteArrayOutputStream buffer;
588        final CRC32 crc;
589
590        ChunkBuffer (int initialSize) {
591            this(new ByteArrayOutputStream(initialSize), new CRC32());
592        }
593
594        private ChunkBuffer (ByteArrayOutputStream buffer, CRC32 crc) {
595            super(new CheckedOutputStream(buffer, crc));
596            this.buffer = buffer;
597            this.crc = crc;
598        }
599
600        public void endChunk (DataOutputStream target) throws IOException {
601            flush();
602            target.writeInt(buffer.size() - 4);
603            buffer.writeTo(target);
604            target.writeInt((int)crc.getValue());
605            buffer.reset();
606            crc.reset();
607        }
