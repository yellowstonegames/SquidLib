016/**
017 * Displays text and images in a grid pattern, like SquidPanel, but will automatically render certain chars as images.
018 * Supports basic animations, such as sliding, wiggling, or fading the contents of a cell.
019 * <br>
020 * Grid width and height settings are in terms of number of cells. Cell width and height are in terms of number of
021 * pixels (when there is no stretching taking place due to viewport or window size).
022 * <br>
023 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
024 * @author Tommy Ettinger
025 */
026public class ImageSquidPanel extends SquidPanel {

103    /**
104     * Builds a panel with the given grid size and all other parameters determined by the factory. Even if sprite images
105     * are being used, a TextCellFactory is still needed to perform sizing and other utility functions. Importantly,
106     * this constructor takes a 2D char array argument that can be sized differently than the displayed area. The
107     * displayed area is gridWidth by gridHeight in cells, but the actualMap argument can be much larger, and only a
108     * portion will be displayed at a time. This requires some special work with the Camera and Viewports to get working
109     * correctly; in the squidlib module's examples, EverythingDemo may be a good place to see how this can be done.
110     * You can pass null for actualMap, which will simply create a char array to use internally that is exactly
111     * gridWidth by gridHeight, in cells.
112     * <br>
113     * If the TextCellFactory has not yet been initialized, then it will be sized at 12x12 px per cell. If it is null
114     * then a default one will be created and initialized. The xOffset and yOffset arguments are measured in pixels or
115     * whatever sub-cell unit of measure your game uses (world coordinates, in libGDX parlance), and change where the
116     * SquidPanel starts drawing by simply adding to the initial x and y coordinates. 0 and 0 are usually fine.
117     *
118     * @param gridWidth the number of cells horizontally
119     * @param gridHeight the number of cells vertically
120     * @param factory the factory to use for cell rendering
121     * @param center
122     *                  The color center to use. Can be {@code null}, but then must be set later on with
123     *          {@link #setColorCenter(IColorCenter)}.
124     * @param xOffset the x offset to start rendering at, in pixels (or some other sub-cell measurement your game uses)
125     * @param yOffset the y offset to start rendering at, in pixels (or some other sub-cell measurement your game uses)
126     * @param actualMap will often be a different size than gridWidth by gridHeight, which enables camera scrolling
127     */
128    public ImageSquidPanel(int gridWidth, int gridHeight, TextCellFactory factory, IColorCenter<Color> center,
129                           float xOffset, float yOffset, char[][] actualMap) {
130        super(gridWidth, gridHeight, factory, center, xOffset, yOffset, actualMap);
131        imageMap = new IntMap<>(128);
132    }

134    @Override
135    public void draw (Batch batch, float parentAlpha) {
136        textFactory.configureShader(batch);
137        int inc = onlyRenderEven ? 2 : 1, widthInc = inc * cellWidth, heightInc = inc * cellHeight;
138        TextureRegion tr;
139        float screenX = xOffset - (gridOffsetX <= 0 ? 0 : cellWidth) + getX(),
140                screenY_base = 1f + yOffset + (gridOffsetY <= 0 ? 0 : cellHeight) + gridHeight * cellHeight + getY(), screenY;
141        char c;
142        for (int x = Math.max(0, gridOffsetX-1), xx = (gridOffsetX <= 0) ? 0 : -1; xx <= gridWidth && x < contents.length; x += inc, xx += inc, screenX += widthInc) {
143            screenY = screenY_base;
144            for (int y = Math.max(0, gridOffsetY-1), yy = (gridOffsetY <= 0) ? 0 : -1; yy <= gridHeight && y < contents[x].length; y += inc, yy += inc, screenY -= heightInc) {
145                c = contents[x][y];
146                tr = imageMap.get(c);
147                if(tr == null)
148                    textFactory.draw(batch, c, colors[x][y], screenX, screenY);
149                else
150                    textFactory.draw(batch, tr, colors[x][y], screenX, screenY);
151            }
152        }
153        super.draw(batch, parentAlpha);
154        int len = animatedEntities.size();
155        for (int i = 0; i < len; i++) {
156
157            animatedEntities.getAt(i).actor.act(Gdx.graphics.getDeltaTime());
158        }
159        len = autoActors.size();
160        Actor a;
161        for (int i = 0; i < len; i++) {
162            a = autoActors.getAt(i);
163            if(a == null) continue;
164            drawActor(batch, parentAlpha, a);
165            a.act(Gdx.graphics.getDeltaTime());
166        }
167    }

180    /**
181     * Draws one AnimatedEntity, specifically the Actor it contains. Batch must be between start() and end()
182     * @param batch Must have start() called already but not stop() yet during this frame.
183     * @param parentAlpha This can be assumed to be 1.0f if you don't know it
184     * @param ac The Actor to draw; the position to draw ac is modified and reset based on some fields of this object
185     */
186    public void drawActor(Batch batch, float parentAlpha, Actor ac)
187    {
188        float prevX = ac.getX(), prevY = ac.getY();
189        ac.setPosition(prevX - gridOffsetX * cellWidth, prevY + gridOffsetY * cellHeight);
190        ac.draw(batch, parentAlpha);
191        ac.setPosition(prevX, prevY);
192    }

667    /**
668     * Gets a "snapshot" of the data represented by this ImageSquidPanel; stores the dimensions, the char data, and the
669     * color data in a way that can be set back to a SquidPanel or ImageSquidPanel using
670     * {@link #setFromSnapshot(String, int, int, int, int)} or its overload that takes a StringBuilder. The actual
671     * contents of the returned StringBuilder are unlikely to be legible in any way if read as text, and are meant to be
672     * concise and stable across versions. This overload allows the first x and y position used to be specified, as well
673     * as the width and height to use (the actual width and height stored may be different if this SquidPanel's
674     * gridWidth and/or gridHeight are smaller than the width and/or height given).
675     * <br>
676     * NOTE: For this version, the mapping of chars to images is not stored in the snapshot, allowing alternate mappings
677     * to be used, such as while graphics are being updated frequently. This also allows the snapshot to be read in from
678     * both normal SquidPanels and ImageSquidPanels.
679     * @param startX the first x position to use in the snapshot, inclusive
680     * @param startY the first y position to use in the snapshot, inclusive
681     * @param width how wide the snapshot area should be; x positions from startX to startX + width - 1 will be used
682     * @param height how tall the snapshot area should be; y positions from startY to startY + height - 1 will be used
683     * @return a StringBuilder representation of this SquidPanel's data that can be passed later to {@link #setFromSnapshot(StringBuilder, int, int, int, int)} or converted to String and passed to its overload
684     */
685    public StringBuilder getSnapshot(int startX, int startY, int width, int height) {
686        width = Math.min(gridWidth - startX, width);
687        height = Math.min(gridHeight - startY, height);
688        StringBuilder sb = new StringBuilder(width * height * 9 + 12);
689        sb.append(width).append('x').append(height).append(':');
690        for (int x = startX, i = 0; i < width; x++, i++) {
691            sb.append(contents[x], startY, height);
692        }
693        char[] reuse = new char[8];
694        for (int x = startX, i = 0; i < width; x++, i++) {
695            for (int y = startY, j = 0; j < height; y++, j++) {
696                sb.append(SColor.floatToChars(reuse, colors[x][y]));
697            }
698        }
699        return sb;
700    }

727     * ImageSquidPanel from the position given by putX,putY (inclusive) to putX+limitWidth,putY+limitHeight (exclusive)
728     * so they have the values stored in the snapshot. If limitWidth or limitHeight is negative, this uses the full
729     * width and height of the snapshot (stopping early if it would extend past the gridWidth or gridHeight of this
730     * ImageSquidPanel).
731     * @param snapshot a StringBuilder in a special format as produced by {@link #getSnapshot(int, int, int, int)}
732     * @param putX where to start placing the data from the snapshot, x position
733     * @param putY where to start placing the data from the snapshot, y position
734     * @param limitWidth if negative, uses all of snapshot's width as possible, otherwise restricts the width allowed
735     * @param limitHeight if negative, uses all of snapshot's height as possible, otherwise restricts the height allowed
736     * @return this after setting, for chaining
737     */
738    public ImageSquidPanel setFromSnapshot(StringBuilder snapshot, int putX, int putY, int limitWidth, int limitHeight)
739    {
740        if(putX >= gridWidth || putY >= gridHeight || snapshot == null || snapshot.length() < 4) return this;
741        if(putX < 0) putX = 0;
742        if(putY < 0) putY = 0;
743        int start = snapshot.indexOf(":")+1, width = StringKit.intFromDec(snapshot),
744                height = StringKit.intFromDec(snapshot, snapshot.indexOf("x") + 1, start),
745                run = start;
746        if(limitWidth < 0)
747            limitWidth = Math.min(width, gridWidth - putX);
748        else
749            limitWidth = Math.min(limitWidth, Math.min(width, gridWidth - putX));
750
751        if(limitHeight < 0)
752            limitHeight = Math.min(height, gridHeight - putY);
753        else
754            limitHeight = Math.min(limitHeight, Math.min(height, gridHeight - putY));
755        for (int x = putX, i = 0; i < limitWidth; x++, i++, run += height) {
756            snapshot.getChars(run, run + limitHeight, contents[x], putY);
757        }
758        run = start + width * height;
759        for (int x = putX, i = 0; i < limitWidth; x++, i++) {
760            for (int y = putY, j = 0; j < limitHeight; y++, j++) {
761                colors[x][y] = SColor.charsToFloat(snapshot, run);

763            }

876    /**
877     * Meant for taking easy-to-write chars and generating chars that can map to images, while still allowing the
878     * original easy-to-write char to be used as its own non-image char. Given a char that should be from the
879     * "relatively common" parts of Unicode (essentially, anything with a codepoint before 0x1000, or with a small risk
880     * of collision, anything before 0x8000), this returns a character from the "Private Use Area" of Unicode that will
881     * probably be unique for most kinds of input. 4096 possible chars can be returned by this method.
882     *
883     * As an example, you may want to map the char 'g' to an image of a goblin, but still sometimes show the actual 'g'
884     * for some other monster or a variant on goblins you don't have an image for. You could call
885     * {@code char goblinChar = ImageSquidPanel.getUnusedChar('g');} and then register goblinChar in the image mapping
886     * with {@code myImageSquidPanel.setImageSwap(goblinChar, goblinImage);}, which would allow you to draw 'g' directly
887     * by placing a 'g' with put(), or draw goblinImage by placing either goblinChar or
888     * {@code ImageSquidPanel.getUnusedChar('g')} with put().
889     *
890     * @param initial a char that ideally should be from the earlier parts of Unicode (before 0x1000 is ideal)
891     * @return a char from Unicode's private use area that is unlikely to be found otherwise or accidentally
892     */
893    public static char getUnusedChar(final char initial)
894    {
895        return (char)(((initial & 0x7000) >>> 3 ^ initial) & 0x0FFF | 0xE000);
896    }
