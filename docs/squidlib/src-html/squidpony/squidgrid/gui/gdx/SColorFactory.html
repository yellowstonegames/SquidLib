034    /**
035     * Constructs a new SColorFactory with an empty cache.
036     * <br>
037     * Implementation note: SColorFactory initially used static methods and was not intended to
038     * be constructed by users, but for compatibility with Android (where static instances may
039     * linger into another run of an application), it was reworked to require an object to be
040     * constructed and for that object to contain the cache, rather than a static instance. Code
041     * written for earlier versions of SquidLib may use static methods on SColorFactory; most
042     * code written in SquidLib 3.0.0 beta 2 and newer prefers the IColorCenter implementations
043     * such as SquidColorCenter when SColor features such as color names aren't required. Since
044     * beta 4, SquidColorCenter implements most of SColorFactory's API, and also supports an
045     * IFilter for automatic changes to requested colors.
046     */
047    public SColorFactory() {
048
049        nameLookup = new ObjectMap<>();
050        valueLookup = new IntMap<>();
051        rng = DefaultResources.getGuiRandom();
052        colorBag = new IntMap<>();
053        palettes = new ObjectMap<>();
054        floor = 1;

083    /**
084     * Returns the SColor whose value matches the one passed in. If no SColor
085     * constant matches that value, then a cached or new SColor is returned that
086     * matches the provided value.
087     *
088     * This method constructs a list of the SColor constants the first time it
089     * is called.
090     *
091     * @param argb an int encoding 256 * 256 * 256 * alpha + 256 * 256 * red + 256 * green + blue
092     * @return the SColor with value argb
093     */
094    public SColor colorForValue(int argb) {
095        if (valueLookup.size == 0) {
096            int len = SColor.FULL_PALETTE.length;
097            SColor sc;
098            for (int i = 0; i < len; i++) {
099                sc = SColor.FULL_PALETTE[i];
100                valueLookup.put(Color.argb8888(sc), sc);
101            }
102        }
103
104        return valueLookup.containsKey(argb) ? valueLookup.get(argb) : asSColor(argb);
105    }

116    /**
117     * Utility method to blend the two colors by the amount passed in as the
118     * coefficient.
119     *
120     * @param a
121     * @param b
122     * @param coef
123     * @return
124     */
125    @SuppressWarnings("unused")
126        private int blend(int a, int b, double coef) {
127        coef = MathUtils.clamp(coef, 0, 1);
128        return (int) (a + (b - a) * coef);
129    }
130    /**
131     * Utility method to blend the two colors by the amount passed in as the
132     * coefficient.
133     *
134     * @param a
135     * @param b
136     * @param coef
137     * @return
138     */
139    private float blend(float a, float b, double coef) {
140        float cf = MathUtils.clamp((float)coef, 0, 1);
141        return (a + (b - a) * cf);
142    }

144    /**
145     * Returns an SColor that is the given distance from the first color to the
146     * second color.
147     *
148     * @param color1 The first color
149     * @param color2 The second color
150     * @param coef The percent towards the second color, as 0.0 to 1.0
151     * @return
152     */
153    public SColor blend(SColor color1, SColor color2, double coef) {
154        return asSColor(blend(color1.a, color2.a, coef),
155                blend(color1.r, color2.r, coef),
156                blend(color1.g, color2.g, coef),
157                blend(color1.b, color2.b, coef));
158    }

240    /**
241     * Returns the cached color that matches the desired rgb value.
242     *
243     * If the color is not already in the cache, it is created and added to the
244     * cache.
245     *
246     * This method does not check to see if the value is already available as a
247     * SColor constant. If such functionality is desired then please use
248     * colorForValue(int rgb) instead.
249     *
250     * @param argb
251     * @return
252     */
253    public SColor asSColor(int argb) {
254        int working = argb;
255        if (floor != 1) {//need to convert to floored values
256            int a = (argb >>> 24) & 0xff;
257            a -= a % floor;
258            int r = (argb >> 16) & 0xff;
259            r -= r % floor;
260            int g = (argb >> 8) & 0xff;
261            g -= g % floor;
262            int b = (argb) & 0xff;
263            b -= b % floor;
264
265            //put back together
266            working = ((a & 0xFF) << 24)
267                    | ((r & 0xFF) << 16)
268                    | ((g & 0xFF) << 8)
269                    | (b & 0xFF);
270        }
271
272        if (colorBag.containsKey(working)) {
273            return colorBag.get(working);
274        } else {
275            SColor color = new SColor(working);
276            colorBag.put(working, color);
277            return color;
278        }
279    }
280    /**
281     * Returns the cached color that matches the desired rgb value.
282     *
283     * If the color is not already in the cache, it is created and added to the
284     * cache.
285     *
286     * This method does not check to see if the value is already available as a
287     * SColor constant. If such functionality is desired then please use
288     * colorForValue(int rgb) instead.
289     *
290     * @param a alpha, between 0.0 and 1.0 inclusive
291     * @param r red, between 0.0 and 1.0 inclusive
292     * @param g green, between 0.0 and 1.0 inclusive
293     * @param b blue, between 0.0 and 1.0 inclusive
294     * @return an SColor with an unspecified name and the given a, r, g, and b values
295     */
296    public SColor asSColor(float a, float r, float g, float b) {
297        int working = 0;
298        int aa = MathUtils.round(255 * a);
299        aa -= aa % floor;
300        int rr = MathUtils.round(255 * r);
301        rr -= rr % floor;
302        int gg = MathUtils.round(255 * g);
303        gg -= gg % floor;
304        int bb = MathUtils.round(255 * b);
305        bb -= bb % floor;
306
307        //put back together
308        working = ((aa & 0xFF) << 24)
309                | ((rr & 0xFF) << 16)
310                | ((gg & 0xFF) << 8)
311                | (bb & 0xFF);
312
313
314        if (colorBag.containsKey(working)) {
315            return colorBag.get(working);
316        } else {
317            SColor color = new SColor(working);
318            colorBag.put(working, color);
319            return color;
320        }
321    }

335    /**
336     * Returns an SColor with the given values, with those values clamped
337     * between 0 and 255.
338     *
339     * @param a alpha, between 0 and 255 inclusive
340     * @param r red, between 0 and 255 inclusive
341     * @param g green, between 0 and 255 inclusive
342     * @param b blue, between 0 and 255 inclusive
343     * @return an SColor with an unspecified name and the given a, r, g, and b values
344     */
345    public SColor asSColor(int a, int r, int g, int b) {
346        a = Math.max(Math.min(a, 255), 0);
347        r = Math.max(Math.min(r, 255), 0);
348        g = Math.max(Math.min(g, 255), 0);
349        b = Math.max(Math.min(b, 255), 0);
350        return asSColor((a << 24) | (r << 16) | (g << 8) | b);
351    }

430     * Returns an SColor that is the fully desaturated (greyscale) version of
431     * the provided color.
432     *
433     * @param color
434     * @return
435     */
436    public SColor desaturated(SColor color) {
437        int r = MathUtils.round(color.r * 255);
438        int g = MathUtils.round(color.g * 255);
439        int b = MathUtils.round(color.b * 255);
440
441        int average = (int) (r * 0.299 + g * 0.587 + b * 0.114);
442
443        return asSColor(average, average, average);
444    }

459    /**
460     * Returns a list of colors starting at the first color and moving to the
461     * second color. The end point colors are included in the list.
462     *
463     * @param color1 starting color
464     * @param color2 ending color
465     * @return an ArrayList of SColor going from color1 to color2; the length should be no greater than 256.
466     */
467    public ArrayList<SColor> asGradient(SColor color1, SColor color2) {
468        String name = paletteNamer(color1, color2);
469        if (palettes.containsKey(name)) {
470            return palettes.get(name);
471        }
472
473        //get the gradient
474        Queue<Coord3D> gradient = Bresenham.line3D(scolorToCoord3D(color1), scolorToCoord3D(color2));
475        ArrayList<SColor> ret = new ArrayList<>();
476        for (Coord3D coord : gradient) {
477            ret.add(coord3DToSColor(coord));
478        }
479
480        palettes.put(name, ret);
481        return ret;
482    }

506    /**
507     * Returns the SColor that is the provided percent towards the end of the
508     * palette. Bounds are checked so as long as there is at least one color in
509     * the palette, values below 0 will return the first element and values
510     * above 1 will return the last element;
511     *
512     * If there is no palette keyed to the provided name, null is returned.
513     *
514     * @param name
515     * @param percent
516     * @return
517     */
518    public SColor fromPalette(String name, float percent) {
519        ArrayList<SColor> list = palettes.get(name);
520        if (list == null) {
521            return null;
522        }
523
524        int index = Math.round(list.size() * percent);//find the index that's the given percent into the gradient
525        index = Math.min(index, list.size() - 1);
526        index = Math.max(index, 0);
527        return list.get(index);
528    }

567    /**
568     * Places the palette into the cache, along with each of the member colors.
569     *
570     * @param name
571     * @param palette
572     */
573    public void addPalette(String name, ArrayList<SColor> palette) {
574        ArrayList<SColor> temp = new ArrayList<>();
575
576        //make sure all the colors in the palette are also in the general color cache
577        for (SColor sc : palette) {
578            temp.add(asSColor(sc.a, sc.r, sc.g, sc.b));
579        }
580
581        palettes.put(name, temp);
582    }

584    /**
585     * Converts the provided color into a three dimensional coordinate point for
586     * use in the Bresenham algorithms.
587     *
588     * @param color
589     * @return
590     */
591    private Coord3D scolorToCoord3D(SColor color) {
592        return new Coord3D(MathUtils.floor(color.r * 255), MathUtils.floor(color.g * 255), MathUtils.floor(color.b * 255));
593    }
594
595    /**
596     * Converts the provided three dimensional coordinate into a color for use
597     * in the Bresenham algorithms.
598     *
599     * @param coord
600     * @return
601     */
602    private SColor coord3DToSColor(Coord3D coord) {
603        return asSColor(coord.x, coord.y, coord.z);
604    }
605
606    private String paletteNamer(SColor color1, SColor color2) {
607        return color1.getName() + " to " + color2.getName();
608    }
