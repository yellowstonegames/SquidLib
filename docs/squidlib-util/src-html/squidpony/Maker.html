023    /**
024     * Makes a LinkedHashMap (LHM) with key and value types inferred from the types of k0 and v0, and considers all
025     * parameters key-value pairs, casting the Objects at positions 0, 2, 4... etc. to K and the objects at positions
026     * 1, 3, 5... etc. to V. If rest has an odd-number length, then it discards the last item. If any pair of items in
027     * rest cannot be cast to the correct type of K or V, then this inserts nothing for that pair and logs information
028     * on the problematic pair to the static Maker.issueLog field.
029     * @param k0 the first key; used to infer the types of other keys if generic parameters aren't specified.
030     * @param v0 the first value; used to infer the types of other values if generic parameters aren't specified.
031     * @param rest an array or vararg of keys and values in pairs; should contain alternating K, V, K, V... elements
032     * @param <K> the type of keys in the returned LinkedHashMap; if not specified, will be inferred from k0
033     * @param <V> the type of values in the returned LinkedHashMap; if not specified, will be inferred from v0
034     * @return a freshly-made LinkedHashMap with K keys and V values, using k0, v0, and the contents of rest to fill it
035     */
036    @SuppressWarnings("unchecked")
037    public static <K, V> LinkedHashMap<K, V> makeLHM(K k0, V v0, Object... rest)
038    {
039        if(rest == null || rest.length == 0)
040        {
041            LinkedHashMap<K, V> lhm = new LinkedHashMap<>(2);
042            lhm.put(k0, v0);
043            return lhm;
044        }
045        LinkedHashMap<K, V> lhm = new LinkedHashMap<>(1 + (rest.length / 2));
046        lhm.put(k0, v0);
047
048        for (int i = 0; i < rest.length - 1; i+=2) {
049            try {
050                lhm.put((K) rest[i], (V) rest[i + 1]);
051            }catch (ClassCastException cce) {
052                issueLog.append("makeLHM call had a casting problem with pair at rest[");
053                issueLog.append(i);
054                issueLog.append("] and/or rest[");
055                issueLog.append(i + 1);
056                issueLog.append("], with contents: ");
057                issueLog.append(rest[i]);
058                issueLog.append(", ");
059                issueLog.append(rest[i+1]);
060                issueLog.append(".\n\nException messages:\n");
061                issueLog.append(cce);
062                String msg = cce.getMessage();
063                if (msg != null) {
064                    issueLog.append('\n');
065                    issueLog.append(msg);
066                }
067                issueLog.append('\n');
068            }
069        }
070        return lhm;
071    }

087    /**
088     * Makes an ArrayList of T given an array or vararg of T elements.
089     * @param elements an array or vararg of T
090     * @param <T> just about any non-primitive type
091     * @return a newly-allocated ArrayList containing all of elements, in order
092     */
093    @SuppressWarnings("unchecked")
094    public static <T> ArrayList<T> makeList(T... elements) {
095        if(elements == null) return null;
096        ArrayList<T> list = new ArrayList<>(elements.length);
097        Collections.addAll(list, elements);
098        return list;
099    }

311    /**
312     * Makes an EnumOrderedMap (EOM) with key and value types inferred from the types of k0 and v0, and considers all
313     * remaining parameters key-value pairs, casting the Objects at positions 0, 2, 4... etc. to K and the objects at

316     * information on the problematic pair to the static Maker.issueLog field.
317     *
318     * @param k0 the first key, which must be an Enum; used to infer the types of other keys if generic parameters aren't specified.
319     * @param v0 the first value; used to infer the types of other values if generic parameters aren't specified.
320     * @param rest an array or vararg of keys and values in pairs; should contain alternating K, V, K, V... elements
