001package squidpony;
002
003import regexodus.Matcher;
004import regexodus.Pattern;
005import squidpony.squidmath.IntVLA;
006
007import java.util.ArrayList;
008import java.util.Iterator;
009
010/**
011 * A simple format parser for String-based configuration or data files where JSON is overkill.
012 * Supports only one type, String, but allows each String to have arbitrary nested levels of
013 * String children as if in sub-lists. You can interpret the Strings however you want, and
014 * quoting each String isn't necessary if they are just one word ("bare words" are allowed).
015 * <br>
016 * Inspired strongly by STOB, http://stobml.org/ , but no code is shared and the format is
017 * slightly different. The main differences are that ObText supports nested block comments
018 * using the syntax /[delimiter/comment here/delimiter]/, and that it uses Python-like
019 * "heredoc" syntax for raw strings surrounded by triple-apostrophes '''like this''' with
020 * optional initial and final newlines in the raw string ignored. We use square brackets in
021 * place of curly braces to mark children associated with a string.
022 */
023public class ObText implements Iterable<String>{
024    public static final Pattern pattern = Pattern.compile(
025            "(?>'''(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?({=s}[\\d\\D]*?)(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?''')" +
026            "|(?>'''({=s}[\\d\\D]*?)''')" +
027            "|(?>({=q}[\"'])({=s}[\\d\\D]*?)(?<!\\\\){\\q})" +
028            "|({=comment}(?>//|^#!)(?>\\V*))" +
029            "|({=block}/\\[({=q}\\S*)/(?:[\\d\\D]*?)/{\\q}\\]/)" +
030            "|({=s}[^\\s\\[\\]\"'#\\\\]+)" +
031            "|({=o}\\[)" +
032            "|({=c}\\])"
033
034    );
035    public static final Matcher m = pattern.matcher();
036
037    protected final ArrayList<String> strings = new ArrayList<String>(64);
038    protected final IntVLA neighbors = new IntVLA(64);
039    private final IntVLA nesting = new IntVLA(16);
040    public ObText()
041    {
042
043    }
044    public ObText(CharSequence text)
045    {
046        parse(text);
047    }
048    public ObText parse(CharSequence text)
049    {
050        m.setTarget(text);
051        nesting.clear();
052        int t = -1;
053        while (m.find()) {
054            if (m.isCaptured("s")) {
055                strings.add(m.group("s"));
056                neighbors.add(1);
057            }
058            else if(m.isCaptured("o"))
059            {
060                nesting.add(neighbors.size - 1);
061            }
062            else if(m.isCaptured("c"))
063            {
064                neighbors.incr(t = nesting.pop(), neighbors.size - t - 1);
065                if(t < neighbors.size - 1)
066                    neighbors.set(neighbors.size-1, 0);
067            }
068        }
069        return this;
070    }
071
072    /**
073     * Returns an iterator over elements of type {@code T}.
074     *
075     * @return an Iterator.
076     */
077    @Override
078    public ItemIterator iterator() {
079        return new ItemIterator();
080    }
081
082    /**
083     * A one-way iterator through this ObText's String items; note that this can be
084     * instructed to descend into child sequences with {@link #child()}, which should
085     * only be called after checking that a child exists with {@link #hasChild()}.
086     * {@link #remove()} is not supported, but {@link #next()} and {@link #hasNext()}
087     * are, of course.
088     */
089    public class ItemIterator implements Iterator<String>
090    {
091        ItemIterator()
092        {
093        }
094        ItemIterator(int i)
095        {
096            index = i % neighbors.size;
097            current = index - 1;
098        }
099        int index = 0, current = -1;
100        /**
101         * Returns {@code true} if the iteration has more elements.
102         * (In other words, returns {@code true} if {@link #next} would
103         * return an element rather than throwing an exception.)
104         *
105         * @return {@code true} if the iteration has more elements
106         */
107        @Override
108        public boolean hasNext() {
109            return index < neighbors.size && (current < 0 || neighbors.get(current) > 0);
110        }
111        /**
112         * Returns {@code true} if the ObText.Item has any child elements.
113         * (In other words, returns {@code true} if {@link #child} would
114         * return an element rather than throwing an exception.)
115         *
116         * @return {@code true} if this has any children
117         */
118        public boolean hasChild() {
119            return index < neighbors.size - 1 && ((current < 0 && neighbors.get(0) > 1) || neighbors.get(current) > 1);
120        }
121
122        /**
123         * Returns the next element in the iteration.
124         *
125         * @return the next element in the iteration
126         * @throws java.util.NoSuchElementException if the iteration has no more elements
127         */
128        @Override
129        public String next() {
130            if(current < 0)
131            {
132                if(strings.isEmpty())
133                    throw new java.util.NoSuchElementException("No more sibling items in ObText object");
134                current = 0;
135                index = neighbors.get(0);
136                return strings.get(0);
137            }
138            int i;
139            if(index >= neighbors.size || (i = neighbors.get(current)) <= 0)
140                throw new java.util.NoSuchElementException("No more sibling items in ObText object");
141            index = i + (current = index);
142            return strings.get(current);
143        }
144        /**
145         * Returns the first child of this ObText.Item and descends into the sequence of child elements.
146         *
147         * @return the first child of this item
148         * @throws java.util.NoSuchElementException if the iteration has no children
149         */
150        public String child() {
151            if(current < 0)
152            {
153                if(neighbors.size <= 0 || neighbors.get(0) == 1)
154                    throw new java.util.NoSuchElementException("No more sibling items in ObText object");
155                current = 0;
156                index = 1;
157                return strings.get(1);
158            }
159            if(current >= neighbors.size - 2 || neighbors.get(current) == 1)
160                throw new java.util.NoSuchElementException("No more sibling items in ObText object");
161            index = ++current + 1;
162            return strings.get(current);
163        }
164
165        @Override
166        public void remove() {
167            throw new UnsupportedOperationException("remove() not supported");
168
169        }
170    }
171
172}
