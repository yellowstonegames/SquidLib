014/**
015 * A simple format parser for String-based configuration or data files where JSON is overkill.
016 * Supports only one type, String, but allows each String to have arbitrary nested levels of
017 * String children as if in sub-lists. You can interpret the Strings however you want, and
018 * quoting each String isn't necessary if they are just one word ("bare words" are allowed).
019 * <br>
020 * The main way of using this is to get an ObText.ItemIterator value using {@link #iterator()},
021 * which acts as a normal Iterator over the top-level Strings (not children of anything), but
022 * to call its {@link ItemIterator#hasChild()} method when you expect potential child elements,
023 * then {@link ItemIterator#children()} to get another ItemIterator over the child elements if
024 * you want to explore deeper.
025 * <br>
026 * This implements Collection of String but is (mostly) unmodifiable; you can call
027 * {@link #parse(CharSequence)} to append the results of parsing more formatted text, or call
028 * {@link #clear()} to remove all data. {@link #add(Object)} and {@link #remove(Object)} are
029 * not implemented and throw exceptions. A quirk of how this implements Collection is that it
030 * only considers the top-level Strings to be part of the Collection for length and for
031 * {@link #contains(Object)}, and will ignore child strings unless you access them via
032 * {@link ItemIterator#children()} on an item that has at least one child.
033 * <br>
034 * Format example:
035 * <pre>
036 * hello world
037 * 'how are you today?' [just great thanks]
038 * hooray!
039 *
040 * complexity?
041 * [it is possible [yes this is a good example]
042 * 'escapes like \[\'\] all work'
043 * ]
044 *
045 * comments are allowed // like this
046 * comments can have different forms # like this
047 * // block comments like in c are allowed
048 * / * but because this example is in javadoc, this example is not actually a comment * /
049 * // remove the spaces between each slash and asterisk to make the last line a comment.
050 * /[delimit/or block comments with delimiters/delimit]/
051 *
052 * '''
053 * raw strings (heredocs) look like this normally.
054 *     they permit characters without escapes, ]][][[ \/\/\ ,
055 *     except for triple quotes.
056 *     they keep newlines and indentation intact,
057 * except for up to one newline ignored adjacent to each triple quote.
058 * '''
059 *
060 * [[different[
061 * if you may need triple quotes
062 *     in the raw string, use a different syntax that allows delimiters.
063 * here, the delimiter is '''different''', just to be different.]different]]
064 * </pre>
065 * <br>
066 * Inspired strongly by STOB, http://stobml.org/ , but no code is shared and the format is
067 * slightly different. The main differences are that ObText supports nested block comments
068 * using the syntax {@code /[delimiter/contents/delimiter]/} where delimiter may be empty
069 * but must match on both sides, and contents is the body of the comment. ObText uses Python-
070 * like "heredoc" syntax for raw strings surrounded by triple-apostrophes '''like so''' with
071 * optional initial and final newlines in the raw string ignored. An alternate raw string
072 * syntax is present that allows delimiters, using the syntax
073 * {@code [[delimiter[contents]delimiter]]}, where again delimiter may be empty and contents
074 * is the body of the raw string. We use square brackets in place of STOB's curly braces to
075 * mark children associated with a string.
076 */
077@Beta
078public class ObText extends AbstractCollection<String>{
079    public static final Pattern pattern = Pattern.compile(
080            "(?>'''(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?({=s}.*?)(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?''')" +
081            "|(?>\\[\\[({=q}[^\\[\\]]*)\\[(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?({=s}.*?)(?:[\n\u000C\f\r\u0085\u2028\u2029]|\r\n)?\\]{\\q}\\]\\])" +
082            "|(?>({=q}[\"'])({=s}.*?)(?<!\\\\){\\q})" +
083            "|(?>(?>//|#)(?>\\V*))" +
084            "|(?>/\\*(?:.*?)\\*/)" +
085            "|(?>/\\[({=q}\\S*)/(?:.*?)/{\\q}\\]/)" +
086            "|({=s}[^\\s\\[\\]\"'#\\\\]+)" +
087            "|({=o}\\[)" +
088            "|({=c}\\])", REFlags.DOTALL | REFlags.UNICODE
089    );
090    protected static final Pattern illegalBareWord = Pattern.compile("[\\s\\[\\]\"'#\\\\]|(?:/[/\\*])"),
091            needsRaw = Pattern.compile("(?<!\\\\)[\\[\\]]|\\\\$");
092    protected static final Matcher m = pattern.matcher();
093    protected static final Matcher bare = illegalBareWord.matcher(), raw = needsRaw.matcher();
094
095    protected final ArrayList<String> strings = new ArrayList<String>(64);
096    protected final IntVLA neighbors = new IntVLA(64);
097    private final IntVLA nesting = new IntVLA(16);
098    protected int length = 0;
099    public ObText()
100    {
101
102    }
103    public ObText(CharSequence text)
104    {
105        parse(text);
106    }

108    /**
109     * Parses the given text (a String or other CharSequence) and appends it into this ObText.
110     * @param text a CharSequence (such as a String) using ObText formatting, as described in this class' JavaDocs
111     * @return this ObText object after appending the parsed text, for chaining
112     */
113    public ObText parse(CharSequence text)
114    {
115        m.setTarget(text);
116        nesting.clear();
117        int t = -1;
118        while (m.find()) {
119            if (m.isCaptured("s")) {
120                strings.add(m.group("s"));
121                neighbors.add(1);
122                if(nesting.isEmpty()) length++;
123            }
124            else if(m.isCaptured("o"))
125            {
126                nesting.add(neighbors.size - 1);
127            }
128            else if(m.isCaptured("c"))
129            {
130                neighbors.incr(t = nesting.pop(), neighbors.size - t - 1);
131                if(t < neighbors.size - 1)
132                    neighbors.set(neighbors.size-1, 0);
133            }
134        }
135        return this;
136    }
137
138    @Override
139    public void clear() {
140        strings.clear();
141        neighbors.clear();
142        length = 0;
143    }
144
145    @Override
146    public int size() {
147        return length;
148    }

156    public ItemIterator iterator() {
157        return new ItemIterator();

160    /**
161     * A one-way iterator through this ObText's String items. Note that this can get
162     * an iterator into a child sequence with {@link #children()}, which should
163     * only be called after checking that a child exists with {@link #hasChild()}.
164     * {@link #remove()} is not supported, but {@link #next()} and {@link #hasNext()}
165     * are, of course.
166     */
167    public class ItemIterator implements Iterator<String>
168    {
169        ItemIterator()
170        {
171        }
172        ItemIterator(int i)
173        {
174            current = i % neighbors.size;
175            index = current;
176        }
177        int index = 0, current = -1;
178        /**
179         * Returns {@code true} if the iteration has more elements.
180         * (In other words, returns {@code true} if {@link #next} would
181         * return an element rather than throwing an exception.)
182         *
183         * @return {@code true} if the iteration has more elements
184         */
185        @Override
186        public boolean hasNext() {
187            return index < neighbors.size && (current < 0 || neighbors.get(current) > 0);
188        }
189        /**
190         * Returns {@code true} if the ObText.Item has any child elements.
191         * (In other words, returns {@code true} if {@link #children} would
192         * return an ItemIterator rather than throwing an exception.)
193         *
194         * @return {@code true} if this has any children
195         */
196        public boolean hasChild() {
197            return index < neighbors.size - 1 && ((current < 0 && neighbors.get(0) > 1) || neighbors.get(current) > 1);
198        }
199
200        /**
201         * Returns the next element in the iteration.
202         *
203         * @return the next element in the iteration
204         * @throws java.util.NoSuchElementException if the iteration has no more elements
205         */
206        @Override
207        public String next() {
208            if(current < 0)
209            {
210                if(strings.isEmpty())
211                    throw new java.util.NoSuchElementException("No more sibling items in ObText object");
212                current = 0;
213                index = neighbors.get(0);
214                return strings.get(0);
215            }
216            if(index >= neighbors.size || neighbors.get(current) <= 0)
217                throw new java.util.NoSuchElementException("No more sibling items in ObText object");
218            index = neighbors.get(index) + (current = index);
219            return strings.get(current);
220        }
221        /**
222         * Returns the first child of this ObText.Item and descends into the sequence of child elements.
223         *
224         * @return the first child of this item
225         * @throws java.util.NoSuchElementException if the iteration has no children
226         */
227        public ItemIterator children() {
228            if(current < 0)
229            {
230                if(neighbors.size <= 0 || neighbors.get(0) == 1)
231                    throw new java.util.NoSuchElementException("No current children in ObText object");
232                return new ItemIterator(1);
233            }
234            if(current >= neighbors.size - 2 || neighbors.get(current) == 1)
235                throw new java.util.NoSuchElementException("No current children in ObText object");
236            return new ItemIterator(current+1);
237        }
238
239        @Override
240        public void remove() {
241            throw new UnsupportedOperationException("remove() not supported");
242
243        }
244    }

246    // Used to generate randomized delimiters using up to 9 non-English letters.
247    // call while assigning your state with randomChars(state += 0x9E3779B97F4A7C15L, myChars)
248    // that assumes you have a 9-element char[] called myChars
249    // as long as z/state is deterministic (i.e. based on a hash), this should be too
250    private static void randomChars(long z, char[] mut)
251    {
252        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
253        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
254        z ^= (z >>> 31);
255
256        mut[0] = letters[(int)(128 + (z & 127))];
257        mut[1] = letters[(int)(128 + (z >>> 7 & 127))];
258        mut[2] = letters[(int)(128 + (z >>> 14 & 127))];
259        mut[3] = letters[(int)(128 + (z >>> 21 & 127))];
260        mut[4] = letters[(int)(128 + (z >>> 28 & 127))];
261        mut[5] = letters[(int)(128 + (z >>> 35 & 127))];
262        mut[6] = letters[(int)(128 + (z >>> 42 & 127))];
263        mut[7] = letters[(int)(128 + (z >>> 49 & 127))];
264        mut[8] = letters[(int)(128 + (z >>> 56 & 127))];
265    }

267    public static void appendQuoted(StringBuilder sb, String text)
268    {
269        if(text == null || text.isEmpty()) {
270            sb.append("''");
271            return;
272        }
273        bare.setTarget(text);
274        if(!bare.find())
275            sb.append(text);
276        else
277        {
278            raw.setTarget(text);
279            if(raw.find()) {
280
281                if (text.contains("'''")) {
282                    long state = CrossHash.Wisp.hash64(text);
283                    char[] myChars = new char[9];
284                    int count;
285                    do {
286                        randomChars(state += 0x9E3779B97F4A7C15L, myChars);
287                        count = StringKit.containsPart(text, myChars, "]", "]]");
288                    } while (count == 12);
289                    sb.append("[[").append(myChars, 0, count).append("[\n").append(text).append("\n]")
290                            .append(myChars, 0, count).append("]]");
291                } else {
292                    sb.append("'''\n").append(text).append("\n'''");
293                }
294            }
295            else if(!text.contains("'"))
296            {
297                sb.append('\'').append(text).append('\'');
298            }
299            else
300            {
301                if(text.contains("\""))
302                {
303                    if(text.contains("'''"))
304                    {
305                        long state = CrossHash.Wisp.hash64(text);
306                        char[] myChars = new char[9];
307                        int count;
308                        do
309                        {
310                            randomChars(state += 0x9E3779B97F4A7C15L, myChars);
311                            count = StringKit.containsPart(text, myChars);
312                        }while(count == 9);
313                        sb.append("[[").append(myChars, 0, count).append("[\n").append(text).append("\n]")
314                                .append(myChars, 0, count).append("]]");
315                    }
316                    else
317                    {
318                        sb.append("'''\n").append(text).append("\n'''");
319                    }
320                }
321                else
322                {
323                    sb.append('"').append(text).append('"');
324                }
325            }
326        }
327    }

329    @Override
330    public String toString() {
331        return "ObText object: [[[[\n" + serializeToString() + "\n]]]]";
332    }
333
334    @Override
335    public boolean equals(Object o) {
336        if (this == o) return true;
337        if (o == null || getClass() != o.getClass()) return false;
338
339        ObText o2 = (ObText) o;
340
341        if (!strings.equals(o2.strings)) return false;
342        return neighbors.equals(o2.neighbors);
343    }
344
345    @Override
346    public int hashCode() {
347        return CrossHash.Wisp.hash(strings) + CrossHash.Wisp.hash(neighbors.items);
348    }

350    public String serializeToString()
351    {
352        StringBuilder sb = new StringBuilder(100);
353        iterate(sb, iterator());
354        return sb.toString();
355    }

357    /**
358     * Deserializes an ObText that was serialized by {@link #serializeToString()} or {@link #toString()}, and will
359     * ignore the prefix and suffix that toString appends for readability (these are "ObText object: [[[[ " and " ]]]]",
360     * for reference). This is otherwise the same as calling the constructor {@link #ObText(CharSequence)}.
361     * @param data a String that is usually produced by serializeToString or toString on an ObText
362     * @return a new ObText produced by parsing data (disregarding any prefix or suffix from toString() )
363     */
364    public static ObText deserializeFromString(String data)
365    {
366        if(data.startsWith("ObText object: [[[[\n"))
367        {
368            return new ObText(data.substring(20, data.length() - 5));
369        }
370        return new ObText(data);
371    }
372
373    private static void iterate(StringBuilder sb, ObText.ItemIterator it)
374    {
375        while (it.hasNext()) {
376            appendQuoted(sb, it.next());
377            sb.append('\n');
378            if (it.hasChild()) {
379                sb.append("[\n");
380                iterate(sb, it.children());
381                sb.append("]\n");
382            }
383        }
384    }
