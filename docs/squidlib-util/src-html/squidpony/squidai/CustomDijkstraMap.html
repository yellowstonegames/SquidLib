013/**
014 * An alternative to AStarSearch when you want to fully explore a search space, or when you want a gradient
015 * floodfill, with customizable rules for what is considered adjacent. This can be used for games where
016 * rotation matters (and possibly costs movement), for games with thin walls (where a wall between cells
017 * prevents travel between those two cells even if the wall doesn't occupy a walkable cell), for games where
018 * the edges between cells may have some requisite to travel across, like a vertical amount that must be
019 * hopped up or down between cells, and for games that have portals between distant cells on the same map.
020 * <br>
021 * As a bit of introduction, the article http://www.roguebasin.com/index.php?title=Dijkstra_Maps_Visualized can
022 * provide some useful information on how these work and how to visualize the information they can produce, while
023 * http://www.roguebasin.com/index.php?title=The_Incredible_Power_of_Dijkstra_Maps is an inspiring list of the
024 * various features Dijkstra Maps can enable.
025 * <br>
026 * If you can't remember how to spell this, just remember: Does It Just Know Stuff? That's Really Awesome!
027 * Created by Tommy Ettinger on 4/4/2015.
028 */

230     * Used to initialize or re-initialize a CustomDijkstraMap that needs a new PhysicalMap because it either wasn't given
231     * one when it was constructed, or because the contents of the terrain have changed permanently (not if a
232     * creature moved; for that you pass the positions of creatures that block paths to scan() or findPath() ).
233     *
234     * @param level
235     * @return
236     */
237    public CustomDijkstraMap initialize(final double[] level) {
238        width = adjacency.width;
239        height = adjacency.height;
240        int len = level.length;
241        gradientMap = new double[len];
242        physicalMap = new double[len];
243        costMap = new int[len];
244        System.arraycopy(level, 0, gradientMap, 0, len);
245        System.arraycopy(level, 0, physicalMap, 0, len);
246        for (int i = 0; i < len; i++) {
247            costMap[i] = (gradientMap[i] > FLOOR) ? '#' : '.';
248        }
249        adjacency.costRules.putAndMoveToFirst('#', WALL);
250
251        neighbors = adjacency.neighborMaps();
252        heuristics = new double[adjacency.directions.length];
253        for (int i = 0; i < heuristics.length; i++) {
254            heuristics[i] = adjacency.measurement.heuristic(adjacency.directions[i]);
255        }
256        initialized = true;
257        return this;
258    }

272    /**
273     * Used to initialize or re-initialize a CustomDijkstraMap that needs a new PhysicalMap because it either wasn't given
274     * one when it was constructed, or because the contents of the terrain have changed permanently (not if a
275     * creature moved; for that you pass the positions of creatures that block paths to scan() or findPath() ). This
276     * initialize() method allows you to specify an alternate wall char other than the default character, '#' .
277     *
278     * @param level
279     * @param alternateWall
280     * @return
281     */
282    public CustomDijkstraMap initialize(final char[][] level, char alternateWall) {
283        width = level.length;
284        height = level[0].length;
285        int rot = adjacency.rotations, len = width*height*rot, dex;
286        gradientMap = new double[len];
287        physicalMap = new double[len];
288        costMap = new int[len];
289        IntDoubleOrderedMap cst = adjacency.costRules;
290        cst.putAndMoveToFirst(alternateWall, WALL);
291        int c;
292        for (int x = 0; x < width; x++) {
293            for (int y = 0; y < height; y++) {
294                c = level[x][y];
295                double t = (c == alternateWall) ? WALL : FLOOR;
296                for (int r = 0; r < rot; r++) {
297                    dex = adjacency.composite(x, y, r, 0);
298                    gradientMap[dex] = t;
299                    physicalMap[dex] = t;
300                    costMap[dex] = c;
301                }
302            }
303        }
304        neighbors = adjacency.neighborMaps();
305        heuristics = new double[adjacency.directions.length];
306        for (int i = 0; i < heuristics.length; i++) {
307            heuristics[i] = adjacency.measurement.heuristic(adjacency.directions[i]);
308        }
309        initialized = true;
310        return this;
311    }

313    /**
314     * Used to initialize the entry cost modifiers for games that require variable costs to enter squares. This expects
315     * a char[][] of the same exact dimensions as the 2D array that was used to previously initialize() this
316     * CustomDijkstraMap, treating the '#' char as a wall (impassable) and anything else as having a normal cost to enter.
317     * The costs can be accessed later by using costMap directly (which will have a valid value when this does not
318     * throw an exception), or by calling setCost().
319     *
320     * @param level a 2D char array that uses '#' for walls
321     * @return this CustomDijkstraMap for chaining.
322     */
323    public CustomDijkstraMap initializeCost(final char[][] level) {
324        if (!initialized) throw new IllegalStateException("CustomDijkstraMap must be initialized first!");
325        int rot = adjacency.rotations;
326        int c;
327        for (int x = 0; x < width; x++) {
328            for (int y = 0; y < height; y++) {
329                c = level[x][y];
330                for (int r = 0; r < rot; r++) {
331                    costMap[adjacency.composite(x, y, r, 0)] = c;
332                }
333            }
334        }
335        return this;
336    }

338    /**
339     * Used to initialize the entry cost modifiers for games that require variable costs to enter squares. This expects
340     * an int[] with length equal to the length of any inner array of neighbors (a field that is given a value during
341     * initialize() by this object's Adjacency value), using the int corresponding to a location as the tile type to
342     * look up for that location, as a key in {@link Adjacency#costRules}, even if an int isn't what this class would
343     * assign normally -- although, walls and other impassable values should be given '#' (which can be put in an int
344     * array) or the value of alternateWall, if this was given one, as a value. The tiles can be accessed later by using
345     * costMap directly (which will have a valid value when this does not throw an exception), or by calling setCost().
346     * <p/>
347     * This method should be slightly more efficient than the other initializeCost methods.
348     *
349     * @param tiles an int array that already has tile types that {@link #adjacency} can find values for
350     * @return this CustomDijkstraMap for chaining.
351     */
352    public CustomDijkstraMap initializeCost(final int[] tiles) {
353        if (!initialized)
354            throw new IllegalStateException("CustomDijkstraMap must be initialized first!");
355        if(tiles.length != gradientMap.length)
356            throw new IllegalArgumentException("costs.length must equal gradientMap.length");
357        costMap = new int[tiles.length];
358        System.arraycopy(tiles, 0, costMap, 0, tiles.length);
359        return this;
360    }

363     * Gets the appropriate DijkstraMap.Measurement to pass to a constructor if you already have a Radius.
364     * Matches SQUARE or CUBE to CHEBYSHEV, DIAMOND or OCTAHEDRON to MANHATTAN, and CIRCLE or SPHERE to EUCLIDEAN.
365     *
366     * @param radius the Radius to find the corresponding Measurement for
367     * @return a DijkstraMap.Measurement that matches radius; SQUARE to CHEBYSHEV, DIAMOND to MANHATTAN, etc.
368     */
369    public static Measurement findMeasurement(Radius radius) {
370        if (radius.equals2D(Radius.SQUARE))
371            return Measurement.CHEBYSHEV;
372        else if (radius.equals2D(Radius.DIAMOND))
373            return Measurement.MANHATTAN;
374        else
375            return Measurement.EUCLIDEAN;
376    }

430     * Marks a cell's type for pathfinding cost as tile (it still will look up the tile in the
431     * {@link Adjacency#costRules} field of {@link #adjacency} when it tries to move through one), unless the cell is a
432     * wall or unreachable area (then it always sets the cost to a value that should have the same cost as a wall).
433     * The normal usage of this is something like {@code setCost(position, '.')} for maps without rotation (this sets
434     * the cost of moving into the cell position to the cost of entering a floor marked with '.'; thos is looked up in
435     * the Adjacency's cost rules and those can be set with {@link Adjacency#addCostRule(char, double)}). If the map has
436     * rotation, {@code setCost(position, '.' | 0x10000)} will change the cost to turn while standing on a tile to the
437     * cost of turning on a '.' floor, though this is just one way Adjacency can be implemented (it's how
438     * RotationAdjacency works).
439     *
440     * @param pt the encoded position/rotation/height to set the cost for
441     * @param tile typically a char such as '.' for floors, but if this uses rotation, turns on that tile are different
442     */
443    public void setCost(int pt, int tile) {
444        if (!initialized || !adjacency.validate(pt)) return;
445        if (physicalMap[pt] > FLOOR) {
446            costMap[pt] = adjacency.costRules.firstIntKey();
447            return;
448        }
449        costMap[pt] = tile;
450    }

522    /**
523     * Recalculate the CustomDijkstra map and return it. Cells that were marked as goals with setGoal will have
524     * a value of 0, the cells adjacent to goals will have a value of 1, and cells progressively further
525     * from goals will have a value equal to the distance from the nearest goal. The exceptions are walls,
526     * which will have a value defined by the WALL constant in this class, and areas that the scan was
527     * unable to reach, which will have a value defined by the DARK constant in this class (typically,
528     * these areas should not be used to place NPCs or items and should be filled with walls). This uses the
529     * current measurement.
530     *
531     * @param usable how much of impassable to actually use; should usually be equal to impassable.length, but can be
532     *               anything if impassable is null (then, it is ignored). This exists to differentiate this method from
533     *               the overload that takes an IntVLA when that argument is null, but also to give some flexibility.
534     * @param impassable An array of int keys (encoded by an Adjacency, usually) representing the locations of enemies
535     * or other moving obstacles to a path that cannot be moved through; this can be null (meaning no obstacles).
536     * @return An int array using the dimensions of what this knows about the physical map.
537     */
538    public double[] scan(int usable, int[] impassable) {
539        if(impassable == null)
540            scanInternal(-1, null, -1);
541        else
542            scanInternal(-1, impassable, Math.min(usable, impassable.length));
543        int maxLength = gradientMap.length;
544        double[] gradientClone = new double[maxLength];
545        for (int l = 0; l < maxLength; l++) {
546            if (gradientMap[l] == FLOOR) {
547                gradientMap[l] = DARK;
548            }
549        }
550        System.arraycopy(gradientMap, 0, gradientClone, 0, maxLength);
551        return gradientClone;
552    }
553    /**
554     * Recalculate the CustomDijkstra map and return it. Cells that were marked as goals with setGoal will have
555     * a value of 0, the cells adjacent to goals will have a value of 1, and cells progressively further
556     * from goals will have a value equal to the distance from the nearest goal. The exceptions are walls,
557     * which will have a value defined by the WALL constant in this class, and areas that the scan was
558     * unable to reach, which will have a value defined by the DARK constant in this class (typically,
559     * these areas should not be used to place NPCs or items and should be filled with walls). This uses the
560     * current measurement.
561     *
562     * @param impassable An array of int keys (encoded by an Adjacency, usually) representing the locations of enemies
563     * or other moving obstacles to a path that cannot be moved through; this can be null (meaning no obstacles).
564     * @return An int array using the dimensions of what this knows about the physical map.
565     */
566    public double[] scan(IntVLA impassable) {
567        if(impassable == null)
568            scanInternal(-1, null, -1);
569        else
570            scanInternal(-1, impassable.items, impassable.size);
571        int maxLength = gradientMap.length;
