013/**
014 * An AOE type that has a center Coord only and only affects that single Coord. Useful if you need an AOE implementation
015 * for something that does not actually affect an area.
016 * This will produce doubles for its {@link #findArea()} method which are equal to 1.0.
017 *
018 * This class doesn't use any other SquidLib class to create its area of effect.
019 * Created by Tommy Ettinger on 7/13/2015.
020 */
021public class PointAOE implements AOE, Serializable {
022    private static final long serialVersionUID = 2L;
023    private Coord center, origin = null;
024    private int mapWidth, mapHeight;
025    private Reach reach = new Reach(1, 1, Radius.SQUARE, AimLimit.FREE);
026
027    public PointAOE(Coord center)
028    {
029        this.center = center;
030    }
031    public PointAOE(Coord center, int minRange, int maxRange)
032    {
033        this.center = center;
034        reach.minDistance = minRange;
035        reach.maxDistance = maxRange;
036    }

043    public void setCenter(Coord center) {
044
045        if (center.isWithin(mapWidth, mapHeight) &&
046                AreaUtils.verifyReach(reach, origin, center))

048            this.center = center;

057    @Override
058    public boolean mayContainTarget(Collection<Coord> targets) {
059        for (Coord p : targets)
060        {
061            if(center.x == p.x && center.y == p.y)
062                return true;
063        }
064        return false;
065    }

067    @Override
068    public OrderedMap<Coord, ArrayList<Coord>> idealLocations(Collection<Coord> targets, Collection<Coord> requiredExclusions) {
069        if(targets == null)
070            return new OrderedMap<>();
071
072        int totalTargets = targets.size();
073        OrderedMap<Coord, ArrayList<Coord>> bestPoints = new OrderedMap<>(totalTargets);
074
075        if(totalTargets == 0)
076            return bestPoints;
077
078
079        double dist = 0.0;
080        for(Coord p : targets) {
081            if (AreaUtils.verifyReach(reach, origin, p)) {
082
083                dist = reach.metric.radius(origin.x, origin.y, p.x, p.y);
084                if (dist <= reach.maxDistance && dist >= reach.minDistance) {
085                    ArrayList<Coord> ap = new ArrayList<>();
086                    ap.add(p);
087                    bestPoints.put(p, ap);
088                }
089            }
090        }
091        return bestPoints;
092    }
093

095    @Override
096    public OrderedMap<Coord, ArrayList<Coord>> idealLocations(Collection<Coord> priorityTargets, Collection<Coord> lesserTargets, Collection<Coord> requiredExclusions) {
097        if(priorityTargets == null)
098            return idealLocations(lesserTargets, requiredExclusions);
099        if(requiredExclusions == null) requiredExclusions = new OrderedSet<>();
100
101        int totalTargets = priorityTargets.size() + lesserTargets.size();
102        OrderedMap<Coord, ArrayList<Coord>> bestPoints = new OrderedMap<>(totalTargets * 4);
103
104        if(totalTargets == 0)
105            return bestPoints;
106
107        double dist = 0.0;
108
109        for(Coord p : priorityTargets) {
110            if (AreaUtils.verifyReach(reach, origin, p)) {
111
112                dist = reach.metric.radius(origin.x, origin.y, p.x, p.y);
113                if (dist <= reach.maxDistance && dist >= reach.minDistance) {
114                    ArrayList<Coord> ap = new ArrayList<>();
115                    ap.add(p);
116                    ap.add(p);
117                    ap.add(p);
118                    ap.add(p);
119                    bestPoints.put(p, ap);
120                }
121            }
122        }
123        if(bestPoints.isEmpty()) {
124            for (Coord p : lesserTargets) {
125                if (AreaUtils.verifyReach(reach, origin, p)) {
126
127                    dist = reach.metric.radius(origin.x, origin.y, p.x, p.y);
128                    if (dist <= reach.maxDistance && dist >= reach.minDistance) {
129                        ArrayList<Coord> ap = new ArrayList<>();
130                        ap.add(p);
131                        bestPoints.put(p, ap);
132                    }
133                }
134            }
135        }
136        return bestPoints;
137
138    }

225    @Override
226    public OrderedMap<Coord, Double> findArea() {
227        OrderedMap<Coord, Double> ret = new OrderedMap<>(1);
228        ret.put(Coord.get(center.x, center.y), 1.0);
229        return ret;
230    }
