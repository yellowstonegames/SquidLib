016/**
017 * An AOE type that has an origin, a radius, an angle, and a span; it will blast from the origin to a length equal to
018 * radius along the angle (in degrees), moving somewhat around corners/obstacles, and also spread a total of span
019 * degrees around the angle (a span of 90 will affect a full quadrant, centered on angle). You can specify the
020 * RadiusType to Radius.DIAMOND for Manhattan distance, RADIUS.SQUARE for Chebyshev, or RADIUS.CIRCLE for Euclidean.
021 *
022 * RADIUS.CIRCLE (Euclidean measurement) will produce the most real-looking cones. This will produce doubles for its
023 * {@link #findArea()} method which are greater than 0.0 and less than or equal to 1.0.
024 *
025 * This class uses {@link FOV} to create its area of effect.
026 * Created by Tommy Ettinger on 7/13/2015.
027 */

191    @Override
192    public boolean mayContainTarget(Collection<Coord> targets) {
193        for (Coord p : targets) {
194            if (radiusType.radius(origin.x, origin.y, p.x, p.y) <= radius) {
195                double d = (angle - NumberTools.atan2_(p.y - origin.y, p.x - origin.x) * 360.0);
196                if(d > 180)
197                    d = 360 - d;
198                if(d < span * 0.5)
199                    return true;
200            }
201        }
202        return false;
203    }

205    @Override
206    public OrderedMap<Coord, ArrayList<Coord>> idealLocations(Collection<Coord> targets, Collection<Coord> requiredExclusions) {
207        if(targets == null)
208            return new OrderedMap<>();
209        if(requiredExclusions == null) requiredExclusions = new OrderedSet<>();
210
211        //requiredExclusions.remove(origin);
212        int totalTargets = targets.size();
213        OrderedMap<Coord, ArrayList<Coord>> bestPoints = new OrderedMap<>(totalTargets * 8);
214
215        if(totalTargets == 0)
216            return bestPoints;
217
218        Coord[] ts = targets.toArray(new Coord[targets.size()]);
219        Coord[] exs = requiredExclusions.toArray(new Coord[requiredExclusions.size()]);
220        Coord t;
221
222        double[][][] compositeMap = new double[totalTargets][dungeon.length][dungeon[0].length];
223        double tAngle; //, tStartAngle, tEndAngle;
224
225
226        char[][] dungeonCopy = new char[dungeon.length][dungeon[0].length];
227        double[][] tmpfov = new double[dungeon.length][dungeon[0].length];
228        Coord tempPt;
229
230        for (int i = 0; i < exs.length; i++) {
231            t = exs[i];
232//            tRadius = radiusType.radius(origin.x, origin.y, t.x, t.y);
233            tAngle = NumberTools.atan2_(t.y - origin.y, t.x - origin.x) * 360.0;
234//            tStartAngle = Math.abs((tAngle - span / 2.0) % 360.0);
235//            tEndAngle = Math.abs((tAngle + span / 2.0) % 360.0);
236            tmpfov = fov.calculateFOV(map, origin.x, origin.y, radius, radiusType, tAngle, span);
237            for (int x = 0; x < dungeon.length; x++) {
238                for (int y = 0; y < dungeon[x].length; y++) {
239                    tempPt = Coord.get(x, y);
240                    dungeonCopy[x][y] = !(origin.x == x && origin.y == y) && (tmpfov[x][y] > 0.0 || !AreaUtils.verifyLimit(reach.limit, origin, tempPt)) ? '!' : dungeon[x][y];
241                }
242            }
243        }
244        
245        DijkstraMap.Measurement dmm = DijkstraMap.Measurement.MANHATTAN;
246        if(radiusType == Radius.SQUARE || radiusType == Radius.CUBE) dmm = DijkstraMap.Measurement.CHEBYSHEV;
247        else if(radiusType == Radius.CIRCLE || radiusType == Radius.SPHERE) dmm = DijkstraMap.Measurement.EUCLIDEAN;
248        DijkstraMap dm = new DijkstraMap(dungeon, dmm);
249        for (int i = 0; i < ts.length; ++i) {
250            dm.initialize(dungeon);
251            t = ts[i];
252//            tRadius = radiusType.radius(origin.x, origin.y, t.x, t.y);
253            tAngle = NumberTools.atan2_(t.y - origin.y, t.x - origin.x) * 360.0;
254//            tStartAngle = Math.abs((tAngle - span / 2.0) % 360.0);
255//            tEndAngle = Math.abs((tAngle + span / 2.0) % 360.0);
256
257            tmpfov = fov.calculateFOV(map, origin.x, origin.y, radius, radiusType, tAngle, span);
258
259
260            for (int x = 0; x < dungeon.length; x++) {
261                for (int y = 0; y < dungeon[x].length; y++) {
262                    if (tmpfov[x][y] > 0.0)
263                    {
264                        compositeMap[i][x][y] = dm.physicalMap[x][y];
265                    }
266                    else compositeMap[i][x][y] = DijkstraMap.WALL;
267                }
268            }
269            if(compositeMap[i][t.x][t.y] > DijkstraMap.FLOOR)
270            {
271                for (int x = 0; x < dungeon.length; x++) {
272                    Arrays.fill(compositeMap[i][x], 99999.0);
273                }
274                continue;
275            }
276
277
278            dm.initialize(compositeMap[i]);
279            dm.setGoal(t);
280            dm.scan(null, null, true);
281            for (int x = 0; x < dungeon.length; x++) { 
282                for (int y = 0; y < dungeon[0].length; y++) { 
283                    if(dm.gradientMap[x][y] < DijkstraMap.FLOOR && dungeonCopy[x][y] != '!')
284                        compositeMap[i][x][y] = dm.gradientMap[x][y];
285                    else
286                        compositeMap[i][x][y] = 99999.0;
287                }
288            }
289        }
290        double bestQuality = 99999 * ts.length;
291        double[][] qualityMap = new double[dungeon.length][dungeon[0].length];
292        for (int x = 0; x < qualityMap.length; x++) {
293            for (int y = 0; y < qualityMap[x].length; y++) {
294                qualityMap[x][y] = 0.0;
295                long bits = 0;
296                for (int i = 0; i < ts.length; ++i) {
297                    qualityMap[x][y] += compositeMap[i][x][y];
298                    if(compositeMap[i][x][y] < 99999.0 && i < 63)
299                        bits |= 1 << i;
300                }
301                if(qualityMap[x][y] < bestQuality)
302                {
303                    ArrayList<Coord> ap = new ArrayList<>();
304
305                    for (int i = 0; i < ts.length && i < 63; ++i) {
306                        if((bits & (1 << i)) != 0)
307                            ap.add(ts[i]);
308                    }
309                    if(ap.size() > 0) {
310                        bestQuality = qualityMap[x][y];
311                        bestPoints.clear();
312                        bestPoints.put(Coord.get(x, y), ap);
313                    }
314                }
315                else if(qualityMap[x][y] == bestQuality) {
316                    ArrayList<Coord> ap = new ArrayList<>();
317
318                    for (int i = 0; i < ts.length && i < 63; ++i) {
319                        if ((bits & (1 << i)) != 0)
320                            ap.add(ts[i]);
321                    }
322                    if (ap.size() > 0) {
323                        bestPoints.put(Coord.get(x, y), ap);
324                    }
325                }
326
327            }
328        }
329
330        return bestPoints;
331    }

333    @Override
334    public OrderedMap<Coord, ArrayList<Coord>> idealLocations(Collection<Coord> priorityTargets, Collection<Coord> lesserTargets, Collection<Coord> requiredExclusions) {
335        if(priorityTargets == null)
336            return idealLocations(lesserTargets, requiredExclusions);
337        if(requiredExclusions == null) requiredExclusions = new OrderedSet<>();
338
339        //requiredExclusions.remove(origin);
340        int totalTargets = priorityTargets.size() + lesserTargets.size();
341        OrderedMap<Coord, ArrayList<Coord>> bestPoints = new OrderedMap<>(totalTargets * 8);
342
343        if(totalTargets == 0)
344            return bestPoints;
345
346        Coord[] pts = priorityTargets.toArray(new Coord[priorityTargets.size()]);
347        Coord[] lts = lesserTargets.toArray(new Coord[lesserTargets.size()]);
348        Coord[] exs = requiredExclusions.toArray(new Coord[requiredExclusions.size()]);
349        Coord t;
350
351        double[][][] compositeMap = new double[totalTargets][dungeon.length][dungeon[0].length];
352        double tAngle; //, tStartAngle, tEndAngle;
353
354        char[][] dungeonCopy = new char[dungeon.length][dungeon[0].length],
355                dungeonPriorities = new char[dungeon.length][dungeon[0].length];
356        for (int i = 0; i < dungeon.length; i++) {
357            System.arraycopy(dungeon[i], 0, dungeonCopy[i], 0, dungeon[i].length);
358            Arrays.fill(dungeonPriorities[i], '#');
359        }
360        double[][] tmpfov;
361        Coord tempPt = Coord.get(0, 0);
362        for (int i = 0; i < exs.length; ++i) {
363            t = exs[i];
364
365
366            tAngle = NumberTools.atan2_(t.y - origin.y, t.x - origin.x) * 360.0;
367            //            tStartAngle = Math.abs((tAngle - span / 2.0) % 360.0);
368//            tEndAngle = Math.abs((tAngle + span / 2.0) % 360.0);
369            tmpfov = fov.calculateFOV(map, origin.x, origin.y, radius, radiusType, tAngle, span);
370            for (int x = 0; x < dungeon.length; x++) {
371                for (int y = 0; y < dungeon[x].length; y++) {
372                    tempPt = Coord.get(x, y);
373                    dungeonCopy[x][y] = (tmpfov[x][y] > 0.0 || !AreaUtils.verifyLimit(reach.limit, origin, tempPt)) ? '!' : dungeonCopy[x][y];
374                }
375            }
376        }
377
378        t = pts[0];
379
380        DijkstraMap.Measurement dmm = DijkstraMap.Measurement.MANHATTAN;
381        if(radiusType == Radius.SQUARE || radiusType == Radius.CUBE) dmm = DijkstraMap.Measurement.CHEBYSHEV;
382        else if(radiusType == Radius.CIRCLE || radiusType == Radius.SPHERE) dmm = DijkstraMap.Measurement.EUCLIDEAN;
383
384        for (int i = 0; i < pts.length; ++i) {
385            DijkstraMap dm = new DijkstraMap(dungeon, dmm);
386            t = pts[i];
387            tAngle = NumberTools.atan2_(t.y - origin.y, t.x - origin.x) * 360.0;
388//            tStartAngle = Math.abs((tAngle - span / 2.0) % 360.0);
389//            tEndAngle = Math.abs((tAngle + span / 2.0) % 360.0);
390
391            tmpfov = fov.calculateFOV(map, origin.x, origin.y, radius, radiusType, tAngle, span);
392
393            for (int x = 0; x < dungeon.length; x++) {
394                for (int y = 0; y < dungeon[x].length; y++) {
395                    if (tmpfov[x][y] > 0.0){
396                        compositeMap[i][x][y] = dm.physicalMap[x][y];
397                        dungeonPriorities[x][y] = dungeon[x][y];
398                    }
399                    else compositeMap[i][x][y] = DijkstraMap.WALL;
400                }
401            }
402            if(compositeMap[i][pts[i].x][pts[i].y] > DijkstraMap.FLOOR)
403            {
404                for (int x = 0; x < dungeon.length; x++) {
405                    Arrays.fill(compositeMap[i][x], 399999.0);
406                }
407                continue;
408            }
409
410
411
412            dm.initialize(compositeMap[i]);
413            dm.setGoal(t);
414            dm.scan(null, null, true);
415            for (int x = 0; x < dungeon.length; x++) {
416                for (int y = 0; y < dungeon[x].length; y++) {
417                    compositeMap[i][x][y] = (dm.gradientMap[x][y] < DijkstraMap.FLOOR  && dungeonCopy[x][y] != '!') ? dm.gradientMap[x][y] : 399999.0;
418                }
419            }
420        }
421
422        t = lts[0];
423
424        for (int i = pts.length; i < totalTargets; ++i) {
425            DijkstraMap dm = new DijkstraMap(dungeon, dmm);
426            t = lts[i - pts.length];
427
428            tAngle = NumberTools.atan2_(t.y - origin.y, t.x - origin.x) * 360.0;
429            //            tStartAngle = Math.abs((tAngle - span / 2.0) % 360.0);
430//            tEndAngle = Math.abs((tAngle + span / 2.0) % 360.0);
431
432            tmpfov = fov.calculateFOV(map, origin.x, origin.y, radius, radiusType, tAngle, span);
433
434            for (int x = 0; x < dungeon.length; x++) {
435                for (int y = 0; y < dungeon[x].length; y++) {
436                    if (tmpfov[x][y] > 0.0){
437                        compositeMap[i][x][y] = dm.physicalMap[x][y];
438                    }
439                    else compositeMap[i][x][y] = DijkstraMap.WALL;
440                }
441            }
442            if(compositeMap[i][lts[i - pts.length].x][lts[i - pts.length].y] > DijkstraMap.FLOOR)
443            {
444                for (int x = 0; x < dungeon.length; x++)
445                {
446                    Arrays.fill(compositeMap[i][x], 99999.0);
447                }
448                continue;
449            }
450
451
452            dm.initialize(compositeMap[i]);
453            dm.setGoal(t);
454            dm.scan(null, null, true);
455            for (int x = 0; x < dungeon.length; x++) {
456                for (int y = 0; y < dungeon[x].length; y++) {
457                    compositeMap[i][x][y] = (dm.gradientMap[x][y] < DijkstraMap.FLOOR && dungeonCopy[x][y] != '!' && dungeonPriorities[x][y] != '#') ? dm.gradientMap[x][y] : 99999.0;
458                }
459            }
460        }
461        double bestQuality = 99999 * lts.length + 399999 * pts.length;
462        double[][] qualityMap = new double[dungeon.length][dungeon[0].length];
463        for (int x = 0; x < qualityMap.length; x++) {
464            for (int y = 0; y < qualityMap[x].length; y++) {
465                qualityMap[x][y] = 0.0;
466                long pbits = 0, lbits = 0;
467                for (int i = 0; i < pts.length; ++i) {
468                    qualityMap[x][y] += compositeMap[i][x][y];
469                    if(compositeMap[i][x][y] < 399999.0 && i < 63)
470                        pbits |= 1 << i;
471                }
472                for (int i = pts.length; i < totalTargets; ++i) {
473                    qualityMap[x][y] += compositeMap[i][x][y];
474                    if(compositeMap[i][x][y] < 99999.0 && i < 63)
475                        lbits |= 1 << i;
476                }
477                if(qualityMap[x][y] < bestQuality)
478                {
479                    ArrayList<Coord> ap = new ArrayList<>();
480
481                    for (int i = 0; i < pts.length && i < 63; ++i) {
482                        if((pbits & (1 << i)) != 0)
483                            ap.add(pts[i]);
484                    }
485                    for (int i = pts.length; i < totalTargets && i < 63; ++i) {
486                        if((lbits & (1 << i)) != 0)
487                            ap.add(lts[i - pts.length]);
488                    }
489
490                    if(ap.size() > 0) {
491                        bestQuality = qualityMap[x][y];
492                        bestPoints.clear();
493                        bestPoints.put(Coord.get(x, y), ap);
494                    }
495
496                }
497                else if(qualityMap[x][y] == bestQuality) {
498                    ArrayList<Coord> ap = new ArrayList<>();
499
500                    for (int i = 0; i < pts.length && i < 63; ++i) {
501                        if ((pbits & (1 << i)) != 0) {
502                            ap.add(pts[i]);
503                            ap.add(pts[i]);
504                            ap.add(pts[i]);
505                            ap.add(pts[i]);
506                        }
507                    }
508                    for (int i = pts.length; i < totalTargets && i < 63; ++i) {
509                        if ((lbits & (1 << i)) != 0)
510                            ap.add(lts[i - pts.length]);
511                    }
512                    if (ap.size() > 0) {
513                        bestPoints.put(Coord.get(x, y), ap);
514                    }
515                }
516            }
517        }
518
519        return bestPoints;
520    }
