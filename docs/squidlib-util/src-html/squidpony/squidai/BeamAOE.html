019/**
020 * Beam Area of Effect that affects an slightly expanded (Elias) line from a given origin Coord out to a given length,
021 * plus an optional radius of cells around the path of the line, while respecting obstacles in its path and possibly
022 * stopping if obstructed. There are several ways to specify the BeamAOE's direction and length, including specifying
023 * an endpoint or specifying an angle in degrees and a distance to the end of the line (without the radius included).
024 * You can specify the RadiusType to Radius.DIAMOND for Manhattan distance, RADIUS.SQUARE for Chebyshev, or
025 * RADIUS.CIRCLE for Euclidean.
026 *
027 * You may want the LineAOE class instead of this. LineAOE travels point-to-point and does not restrict length, while
028 * BeamAOE travels a specific length (and may have a radius, like LineAOE) but then stops only after the travel down the
029 * length and radius has reached its end. This difference is relevant if a game has effects that have a definite
030 * area measured in a rectangle or elongated pillbox shape, such as a "20-foot-wide bolt of lightning, 100 feet long."
031 * BeamAOE is more suitable for that effect, while LineAOE may be more suitable for things like focused lasers that
032 * pass through small (likely fleshy) obstacles but stop after hitting the aimed-at target.
033 *
034 * BeamAOE will strike a small area behind the user and in the opposite direction of the target if the radius is
035 * greater than 0. This behavior may be altered in a future version.
036 *
037 * This will produce doubles for its findArea() method which are equal to 1.0.
038 *
039 * This class uses squidpony.squidmath.Elias and squidpony.squidai.DijkstraMap to create its area of effect.
040 * Created by Tommy Ettinger on 7/14/2015.
041 */

158    private double[][] initDijkstra()
159    {
160        los.isReachable(dungeon, origin.x, origin.y, end.x, end.y, rt);
161        Queue<Coord> lit = los.getLastPath();
162
163        dijkstra.initialize(dungeon);
164        for(Coord p : lit)
165        {
166            dijkstra.setGoal(p);
167        }
168        if(radius == 0)
169            return dijkstra.gradientMap;
170        return dijkstra.partialScan(radius, null);
171    }

299    public boolean mayContainTarget(Set<Coord> targets) {
300        for (Coord p : targets)
301        {
302            if(rt.radius(origin.x, origin.y, p.x, p.y) + rt.radius(end.x, end.y, p.x, p.y) -
303                    rt.radius(origin.x, origin.y, end.x, end.y) <= 3.0 + radius)
304                return true;
305        }

309    @Override
310    public OrderedMap<Coord, ArrayList<Coord>> idealLocations(Set<Coord> targets, Set<Coord> requiredExclusions) {
311        if(targets == null)
312            return new OrderedMap<>();
313        if(requiredExclusions == null) requiredExclusions = new OrderedSet<>();
314
315        //requiredExclusions.remove(origin);
316
317        int totalTargets = targets.size();
318        OrderedMap<Coord, ArrayList<Coord>> bestPoints = new OrderedMap<>(totalTargets * 8);
319
320        if(totalTargets == 0)
321            return bestPoints;
322
323        Coord[] ts = targets.toArray(new Coord[targets.size()]);
324        Coord[] exs = requiredExclusions.toArray(new Coord[requiredExclusions.size()]);
325        Coord t;
326
327        double[][][] compositeMap = new double[ts.length][dungeon.length][dungeon[0].length];
328
329        char[][] dungeonCopy = new char[dungeon.length][dungeon[0].length];
330        for (int i = 0; i < dungeon.length; i++) {
331            System.arraycopy(dungeon[i], 0, dungeonCopy[i], 0, dungeon[i].length);
332        }
333        DijkstraMap dt = new DijkstraMap(dungeon, dijkstra.measurement);
334        double[][] resMap = DungeonUtility.generateResistances(dungeon);
335        Coord tempPt = Coord.get(0, 0);
336        for (int i = 0; i < exs.length; ++i) {
337            t = rt.extend(origin, exs[i], length, false, dungeon.length, dungeon[0].length);
338            dt.resetMap();
339            dt.clearGoals();
340            los.isReachable(resMap, origin.x, origin.y, t.x, t.y, rt);
341            Queue<Coord> lit = los.getLastPath();
342
343
344            for(Coord p : lit)
345            {
346                dt.setGoal(p);
347            }
348            if(radius > 0)
349                dt.partialScan(radius, null);
350
351            for (int x = 0; x < dungeon.length; x++) {
352                for (int y = 0; y < dungeon[x].length; y++) {
353                    tempPt = Coord.get(x, y);
354                    dungeonCopy[x][y] = (dt.gradientMap[x][y] < DijkstraMap.FLOOR || !AreaUtils.verifyReach(reach, origin, tempPt)) ? '!' : dungeonCopy[x][y];
355                }
356            }
357        }
358
359        //t = rt.extend(origin, ts[0], length, false, dungeon.length, dungeon[0].length);
360
361        for (int i = 0; i < ts.length; ++i) {
362            DijkstraMap dm = new DijkstraMap(dungeon, dijkstra.measurement);
363
364            t = rt.extend(origin, ts[i], length, false, dungeon.length, dungeon[0].length);
365            dt.resetMap();
366            dt.clearGoals();
367            los.isReachable(resMap, origin.x, origin.y, t.x, t.y, rt);
368            Queue<Coord> lit = los.getLastPath();
369
370            for(Coord p : lit)
371            {
372                dt.setGoal(p);
373            }
374            if(radius > 0)
375                dt.partialScan(radius, null);
376
377
378            double dist = 0.0;
379            for (int x = 0; x < dungeon.length; x++) {
380                for (int y = 0; y < dungeon[x].length; y++) {
381                    if (dt.gradientMap[x][y] < DijkstraMap.FLOOR){
382                        dist = reach.metric.radius(origin.x, origin.y, x, y);
383                        if(dist <= reach.maxDistance + radius && dist >= reach.minDistance - radius)
384                            compositeMap[i][x][y] = dm.physicalMap[x][y];
385                        else
386                            compositeMap[i][x][y] = DijkstraMap.WALL;
387                    }
388                    else compositeMap[i][x][y] = DijkstraMap.WALL;
389                }
390            }
391            if(compositeMap[i][ts[i].x][ts[i].y] > DijkstraMap.FLOOR)
392            {
393                for (int x = 0; x < dungeon.length; x++) {
394                    Arrays.fill(compositeMap[i][x], 99999.0);
395                }
396                continue;
397            }
398
399
400            dm.initialize(compositeMap[i]);
401            dm.setGoal(ts[i]);
402            dm.scan(null);
403            for (int x = 0; x < dungeon.length; x++) {
404                for (int y = 0; y < dungeon[x].length; y++) {
405                    compositeMap[i][x][y] = (dm.gradientMap[x][y] < DijkstraMap.FLOOR  && dungeonCopy[x][y] != '!') ? dm.gradientMap[x][y] : 99999.0;
406                }
407            }
408        }
409        double bestQuality = 99999 * ts.length;
410        double[][] qualityMap = new double[dungeon.length][dungeon[0].length];
411        for (int x = 0; x < qualityMap.length; x++) {
412            for (int y = 0; y < qualityMap[x].length; y++) {
413                qualityMap[x][y] = 0.0;
414                long bits = 0;
415                for (int i = 0; i < ts.length; ++i) {
416                    qualityMap[x][y] += compositeMap[i][x][y];
417                    if(compositeMap[i][x][y] < 99999.0 && i < 63)
418                        bits |= 1 << i;
419                }
420                if(qualityMap[x][y] < bestQuality)
421                {
422                    ArrayList<Coord> ap = new ArrayList<>();
423
424                    for (int i = 0; i < ts.length && i < 63; ++i) {
425                        if((bits & (1 << i)) != 0)
426                            ap.add(ts[i]);
427                    }
428
429                    if(ap.size() > 0) {
430                        bestQuality = qualityMap[x][y];
431                        bestPoints.clear();
432                        bestPoints.put(Coord.get(x, y), ap);
433                    }
434                }
435                else if(qualityMap[x][y] == bestQuality)
436                {
437                    ArrayList<Coord> ap = new ArrayList<>();
438
439                    for (int i = 0; i < ts.length && i < 63; ++i) {
440                        if((bits & (1 << i)) != 0)
441                            ap.add(ts[i]);
442                    }
443
444                    if (ap.size() > 0) {
445                        bestPoints.put(Coord.get(x, y), ap);
446                    }
447                }
448            }
449        }
450
451        return bestPoints;
452    }
