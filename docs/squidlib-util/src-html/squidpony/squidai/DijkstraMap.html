011/**
012 * An alternative to AStarSearch when you want to fully explore a search space, or when you want a gradient floodfill.
013 * It's currently significantly faster that AStarSearch, and also supports pathfinding to the nearest of multiple
014 * goals, which is not possible with AStarSearch. This last feature enables a whole host of others, like pathfinding
015 * for creatures that can attack targets between a specified minimum and maximum distance, and there's also the
016 * standard uses of Dijkstra Maps such as finding ideal paths to run away.
017 * As a bit of introduction, the article http://www.roguebasin.com/index.php?title=Dijkstra_Maps_Visualized can
018 * provide some useful information on how these work and how to visualize the information they can produce, while
019 * http://www.roguebasin.com/index.php?title=The_Incredible_Power_of_Dijkstra_Maps is an inspiring list of the
020 * various features Dijkstra Maps can enable.
021 * <br>
022 * If you can't remember how to spell this, just remember: Does It Just Know Stuff? That's Really Awesome!
023 * Created by Tommy Ettinger on 4/4/2015.
024 */

028    private static final double root2 = Math.sqrt(2.0);
029    /**
030     * The type of heuristic to use.
031     */
032    public enum Measurement {
033
034        /**
035         * The distance it takes when only the four primary directions can be
036         * moved in. The default.
037         */
038        MANHATTAN,
039        /**
040         * The distance it takes when diagonal movement costs the same as
041         * cardinal movement.
042         */
043        CHEBYSHEV,
044        /**
045         * The distance it takes as the crow flies. This will NOT affect movement cost when calculating a path,
046         * only the preferred squares to travel to (resulting in drastically more reasonable-looking paths).
047         */
048        EUCLIDEAN;
049
050        public double heuristic(Direction target) {
051            switch (this) {
052                case CHEBYSHEV:
053                    return 1.0;
054                case EUCLIDEAN:
055                    switch (target) {
056                        case DOWN_LEFT:
057                        case DOWN_RIGHT:
058                        case UP_LEFT:
059                        case UP_RIGHT:
060                            return root2;
061                        default:
062                            return 1.0;
063                    }
064            }
065            return 1.0;
066        }
067
068        public int directionCount() {
069            switch (this) {
070                case MANHATTAN: return 4;
071                default: return 8;
072            }
073        }
074    }

079     * the length of paths, though it will change the DijkstraMap's gradientMap to have many non-integer values, and
080     * that in turn will make paths this finds much more realistic and smooth (favoring orthogonal directions unless a
081     * diagonal one is a better option).
082     */
083    public Measurement measurement = Measurement.MANHATTAN;

306     * Used to initialize or re-initialize a DijkstraMap that needs a new PhysicalMap because it either wasn't given
307     * one when it was constructed, or because the contents of the terrain have changed permanently (not if a
308     * creature moved; for that you pass the positions of creatures that block paths to scan() or findPath() ).

310     * @param level
311     * @return
312     */
313    public DijkstraMap initialize(final double[][] level) {
314        width = level.length;
315        height = level[0].length;
316        gradientMap = new double[width][height];
317        safetyMap = new double[width][height];
318        physicalMap = new double[width][height];
319        costMap = new double[width][height];
320        targetMap = new Coord[width][height];
321        for (int x = 0; x < width; x++) {
322            System.arraycopy(level[x], 0, gradientMap[x], 0, height);
323            System.arraycopy(level[x], 0, physicalMap[x], 0, height);
324            Arrays.fill(costMap[x], 1.0);
325        }
326        initialized = true;
327        return this;
328    }

464            return DijkstraMap.Measurement.MANHATTAN;

538    /**
539     * Marks a cell as a goal for pathfinding, unless the cell is a wall or unreachable area (then it does nothing).
540     *
541     * @param x
542     * @param y
543     */
544    public void setGoal(int x, int y) {
545        if (!initialized || x < 0 || x >= width || y < 0 || y >= height) return;
546        if (physicalMap[x][y] > FLOOR) {
547            return;
548        }
549
550        goals.put(Coord.pureEncode(x, y), GOAL);
551    }

556     * @param pt

558    public void setGoal(Coord pt) {
559        if (!initialized || !pt.isWithin(width, height)) return;

562        }

566    /**
567     * Marks many cells as goals for pathfinding, ignoring cells in walls or unreachable areas. More efficient than
568     * a loop that calls {@link #setGoal(Coord)} over and over, since this doesn't need to do a bounds check. The
569     * GreasedRegion passed to this should have the same width and height as this DijkstraMap.
570     *
571     * @param pts a GreasedRegion containing "on" cells to treat as goals; should have the same width and height as this
572     */
573    public void setGoals(GreasedRegion pts) {
574        if (!initialized || pts.width > width || pts.height > height) return;
575        int[] enc = new GreasedRegion(physicalMap, FLOOR).and(pts).asEncoded();
576        double[] gls = new double[enc.length];
577        Arrays.fill(gls, GOAL);
578        goals.putAll(enc, gls);
579    }

765    /**
766     * Recalculate the Dijkstra map and return it. Cells that were marked as goals with setGoal will have
767     * a value of 0, the cells adjacent to goals will have a value of 1, and cells progressively further
768     * from goals will have a value equal to the distance from the nearest goal. The exceptions are walls,
769     * which will have a value defined by the WALL constant in this class, and areas that the scan was
770     * unable to reach, which will have a value defined by the DARK constant in this class (typically,
771     * these areas should not be used to place NPCs or items and should be filled with walls). This uses the
772     * current measurement.
773     *
774     * @param impassable A Set of Position keys representing the locations of enemies or other moving obstacles to a
775     *                   path that cannot be moved through; this can be null if there are no such obstacles.
776     * @return A 2D double[width][height] using the width and height of what this knows about the physical map.
777     */
778    public double[][] scan(Collection<Coord> impassable) {
779        if (!initialized) return null;
780        if (impassable == null)
781            impassable = Collections.emptySet();
782        for (Coord pt : impassable) {
783            closed.put(pt.encode(), WALL);
784        }
785        Coord dec, adj, cen;
786        int enc;
787
788        for (IntDoubleOrderedMap.MapEntry entry : goals.mapEntrySet()) {
789            //if (closed.containsKey(entry.getIntKey()))
790            //    continue;
791            //    closed.remove(entry.getIntKey());
792            dec = Coord.decode(entry.getIntKey());
793            gradientMap[dec.x][dec.y] = entry.getDoubleValue();
794        }
795        double currentLowest = 999000;
796        IntDoubleOrderedMap lowest = new IntDoubleOrderedMap();
797
798        for (int y = 0; y < height; y++) {
799            for (int x = 0; x < width; x++) {
800                if (gradientMap[x][y] > FLOOR && !goals.containsKey(Coord.pureEncode(x, y)))
801                    closed.put(Coord.pureEncode(x, y), physicalMap[x][y]);
802                else if (gradientMap[x][y] < currentLowest) {
803                    currentLowest = gradientMap[x][y];
804                    lowest.clear();
805                    lowest.put(Coord.pureEncode(x, y), currentLowest);
806                } else if (gradientMap[x][y] == currentLowest) {
807                    lowest.put(Coord.pureEncode(x, y), currentLowest);
808                }
809            }
810        }
811        int numAssigned = lowest.size();
812        mappedCount = goals.size();
813        open.putAll(lowest);
814        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
815        while (numAssigned > 0) {
816//            ++iter;
817            numAssigned = 0;
818
819            for (IntDoubleOrderedMap.MapEntry cell : open.mapEntrySet()) {
820                cen = Coord.decode(cell.getIntKey());
821                for (int d = 0; d < dirs.length; d++) {
822                    adj = cen.translate(dirs[d].deltaX, dirs[d].deltaY);
823                    if (adj.x < 0 || adj.y < 0 || width <= adj.x || height <= adj.y)
824                        /* Outside the map */
825                        continue;
826                    if(d >= 4 && blockingRequirement > 0) // diagonal
827                    {
828                        if((gradientMap[cen.x + dirs[d].deltaX][cen.y] > FLOOR ? 1 : 0)
829                                + (gradientMap[cen.x][cen.y + dirs[d].deltaY] > FLOOR ? 1 : 0)
830                                >= blockingRequirement)
831                        {
832                            continue;
833                        }
834                    }
835                    enc = adj.encode();
836                    double h = measurement.heuristic(dirs[d]);
837                    if (!closed.containsKey(enc) && !open.containsKey(enc) && gradientMap[cen.x][cen.y] + h * costMap[adj.x][adj.y] < gradientMap[adj.x][adj.y]) {
838                        setFresh(adj, cell.getDoubleValue() + h * costMap[adj.x][adj.y]);
839                        ++numAssigned;
840                        ++mappedCount;
841                    }
842                }
843            }
844//            closed.putAll(open);
845            open.clear();
846            open.putAll(fresh);
847            fresh.clear();
848        }
849        closed.clear();
850        open.clear();
851
852        double[][] gradientClone = new double[width][height];
853        for (int x = 0; x < width; x++) {
854            for (int y = 0; y < height; y++) {
855                if (gradientMap[x][y] == FLOOR) {
856                    gradientMap[x][y] = DARK;
857                }
858            }
859            System.arraycopy(gradientMap[x], 0, gradientClone[x], 0, height);
860        }
861
862        return gradientClone;
863    }

1057    /**
1058     * If you have a target or group of targets you want to pathfind to without scanning the full map, this can be good.
1059     * It may find sub-optimal paths in the presence of costs to move into cells. It is useful when you want to move in
1060     * a straight line to a known nearby goal.
1061     *
1062     * @param start   your starting location
1063     * @param targets an array or vararg of Coords to pathfind to the nearest of
1064     * @return an ArrayList of Coord that goes from a cell adjacent to start and goes to one of the targets. Copy of path.
1065     */
1066    public ArrayList<Coord> findShortcutPath(Coord start, Coord... targets) {
1067        if (targets.length == 0) {
1068            cutShort = true;
1069            path.clear();
1070            return new ArrayList<>(path);
1071        }
1072        Coord currentPos = findNearest(start, targets);
1073        while (true) {
1074            if (frustration > 500) {
1075                path.clear();
1076                break;
1077            }
1078            double best = gradientMap[currentPos.x][currentPos.y];
1079            final Direction[] dirs = appendDirToShuffle(rng);
1080            int choice = rng.nextInt(measurement.directionCount() + 1);
1081
1082            for (int d = 0; d < measurement.directionCount() + 1; d++) {
1083                Coord pt = Coord.get(currentPos.x + dirs[d].deltaX, currentPos.y + dirs[d].deltaY);
1084                if(!pt.isWithin(width, height))
1085                    continue;
1086                if (gradientMap[pt.x][pt.y] < best) {
1087                    if (dirs[choice] == Direction.NONE || !path.contains(pt)) {
1088                        best = gradientMap[pt.x][pt.y];
1089                        choice = d;
1090                    }
1091                }
1092            }
1093
1094            if (best >= gradientMap[currentPos.x][currentPos.y] || physicalMap[currentPos.x + dirs[choice].deltaX][currentPos.y + dirs[choice].deltaY] > FLOOR) {
1095                cutShort = true;
1096                frustration = 0;
1097                return new ArrayList<>(path);
1098            }
1099            currentPos = currentPos.translate(dirs[choice].deltaX, dirs[choice].deltaY);
1100            if (gradientMap[currentPos.x][currentPos.y] == 0)
1101                break;
1102            path.add(currentPos);
1103            frustration++;
1104        }
1105        frustration = 0;
1106        cutShort = false;
1107        Collections.reverse(path);
1108        return new ArrayList<>(path);

1191     * Recalculate the Dijkstra map for a creature that is potentially larger than 1x1 cell and return it. The value of
1192     * a cell in the returned Dijkstra map assumes that a creature is square, with a side length equal to the passed
1193     * size, that its minimum-x, minimum-y cell is the starting cell, and that any cell with a distance number
1194     * represents the distance for the creature's minimum-x, minimum-y cell to reach it. Cells that cannot be entered
1195     * by the minimum-x, minimum-y cell because of sizing (such as a floor cell next to a maximum-x and/or maximum-y
1196     * wall if size is &gt; 1) will be marked as DARK. Cells that were marked as goals with setGoal will have
1197     * a value of 0, the cells adjacent to goals will have a value of 1, and cells progressively further
1198     * from goals will have a value equal to the distance from the nearest goal. The exceptions are walls,
1199     * which will have a value defined by the WALL constant in this class, and areas that the scan was
1200     * unable to reach, which will have a value defined by the DARK constant in this class. (typically,
1201     * these areas should not be used to place NPCs or items and should be filled with walls). This uses the
1202     * current measurement.
1203     *
1204     * @param impassable A Set of Position keys representing the locations of enemies or other moving obstacles to a
1205     *                   path that cannot be moved through; this can be null if there are no such obstacles.
1206     * @param size       The length of one side of a square creature using this to find a path, i.e. 2 for a 2x2 cell
1207     *                   creature. Non-square creatures are not supported because turning is really hard.
1208     * @return A 2D double[width][height] using the width and height of what this knows about the physical map.
1209     */
1210    public double[][] scan(Collection<Coord> impassable, int size) {
1211        if (!initialized) return null;
1212        if (impassable == null)
1213            impassable = new OrderedSet<>();

1215        for (Coord pt : impassable) {
1216            blocking.put(pt.encode(), WALL);
1217            for (int x = 0; x < size; x++) {
1218                for (int y = 0; y < size; y++) {
1219                    if (x + y != 0 && gradientMap[pt.x - x][pt.y - y] <= FLOOR)
1220                        blocking.put(Coord.pureEncode(pt.x - x, pt.y - y), DARK);
1221                }
1222            }
1223        }
1224        closed.putAll(blocking);
1225        Coord dec, cen, adj;
1226        int enc;
1227        for (IntDoubleOrderedMap.MapEntry entry : goals.mapEntrySet()) {
1228            //if (closed.containsKey(entry.getIntKey()))
1229            //    closed.remove(entry.getIntKey());
1230            dec = Coord.decode(entry.getIntKey());
1231            gradientMap[dec.x][dec.y] = entry.getDoubleValue();
1232        }
1233        mappedCount = goals.size();
1234        double currentLowest = 999000;
1235        IntDoubleOrderedMap lowest = new IntDoubleOrderedMap();
1236        int temp, p;
1237        for (int y = 0; y < height; y++) {
1238            I_AM_BECOME_DEATH_DESTROYER_OF_WORLDS:
1239            for (int x = 0; x < width; x++) {
1240                p = Coord.pureEncode(x, y);
1241                if (gradientMap[x][y] > FLOOR && !goals.containsKey(p)) {
1242                    closed.put(p, physicalMap[x][y]);
1243                    if (gradientMap[x][y] == WALL) {
1244                        for (int i = 0; i < size; i++) {
1245                            if (x - i < 0)
1246                                continue;
1247                            for (int j = 0; j < size; j++) {
1248                                temp = Coord.pureEncode(x - i, y - j);
1249                                if (y - j < 0 || closed.containsKey(temp))
1250                                    continue;
1251                                if (gradientMap[x - i][y - j] <= FLOOR && !goals.containsKey(temp))
1252                                    closed.put(temp, DARK);
1253                            }
1254                        }
1255                    }
1256                } else if (gradientMap[x][y] < currentLowest && !closed.containsKey(p)) {
1257                    for (int i = 0; i < size; i++) {
1258                        if (x + i >= width)
1259                            continue I_AM_BECOME_DEATH_DESTROYER_OF_WORLDS;
1260                        for (int j = 0; j < size; j++) {
1261                            temp = Coord.pureEncode(x + i, y + j);
1262                            if (y + j >= height || closed.containsKey(temp))
1263                                continue I_AM_BECOME_DEATH_DESTROYER_OF_WORLDS;
1264                        }
1265                    }
1266
1267                    currentLowest = gradientMap[x][y];
1268                    lowest.clear();
1269                    lowest.put(Coord.pureEncode(x, y), currentLowest);
1270
1271                } else if (gradientMap[x][y] == currentLowest && !closed.containsKey(p)) {
1272                    if (!closed.containsKey(p)) {
1273                        for (int i = 0; i < size; i++) {
1274                            if (x + i >= width)
1275                                continue I_AM_BECOME_DEATH_DESTROYER_OF_WORLDS;
1276                            for (int j = 0; j < size; j++) {
1277                                temp = Coord.pureEncode(x + i, y + j);
1278                                if (y + j >= height || closed.containsKey(temp))
1279                                    continue I_AM_BECOME_DEATH_DESTROYER_OF_WORLDS;
1280                            }
1281                        }
1282                        lowest.put(p, currentLowest);
1283                    }
1284                }
1285            }
1286        }
1287        int numAssigned = lowest.size();
1288        open.putAll(lowest);
1289        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
1290        while (numAssigned > 0) {
1291            numAssigned = 0;
1292            for (IntDoubleOrderedMap.MapEntry cell : open.mapEntrySet()) {
1293                cen = Coord.decode(cell.getIntKey());
1294                for (int d = 0; d < dirs.length; d++) {
1295                    adj = cen.translate(dirs[d].deltaX, dirs[d].deltaY);
1296                    if (adj.x < 0 || adj.y < 0 || width <= adj.x || height <= adj.y)
1297                        /* Outside the map */
1298                        continue;
1299                    enc = adj.encode();
1300                    double h = measurement.heuristic(dirs[d]);
1301                    if (!closed.containsKey(enc) && !open.containsKey(enc) && gradientMap[cen.x][cen.y] + h * costMap[adj.x][adj.y] < gradientMap[adj.x][adj.y]) {
1302                        setFresh(adj, cell.getDoubleValue() + h * costMap[adj.x][adj.y]);
1303                        ++numAssigned;
1304                        ++mappedCount;
1305                    }
1306                }
1307            }
1308//            closed.putAll(open);
1309            open = new IntDoubleOrderedMap(fresh);
1310            fresh.clear();
1311        }
1312        closed.clear();
1313        open.clear();
1314
1315
1316        double[][] gradientClone = new double[width][height];
1317        for (int x = 0; x < width; x++) {
1318            for (int y = 0; y < height; y++) {
1319                if (gradientMap[x][y] == FLOOR) {
1320                    gradientMap[x][y] = DARK;
1321                }
1322            }
1323            System.arraycopy(gradientMap[x], 0, gradientClone[x], 0, height);
1324        }
1325        return gradientClone;

3668     * If this is 1, having one orthogonal obstacle adjacent to both the current cell and the cell the pathfinder is
3669     * trying to diagonally enter will block diagonal moves. This generally blocks movement around corners, the "hard
3670     * corner" rule used in some games.
3671     * <br>
3672     * If this is 2, having two orthogonal obstacles adjacent to both the current cell and the cell the pathfinder is
3673     * trying to diagonally enter will block diagonal moves. As an example, if there is a wall to the north and a wall
3674     * to the east, then the pathfinder won't be able to move northeast even if there is a floor there.
3675     * @param blockingRequirement the desired level of blocking required to stop a diagonal move
3676     */
3677    public void setBlockingRequirement(int blockingRequirement) {
3678        this.blockingRequirement = blockingRequirement > 2 ? 2 : blockingRequirement < 0 ? 0 : blockingRequirement;
3679    }
