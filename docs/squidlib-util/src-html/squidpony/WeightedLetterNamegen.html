274    private static final char[] vowels = {'a', 'e', 'i', 'o'};//not using y because it looks strange as a vowel in names
275    private static final int LAST_LETTER_CANDIDATES_MAX = 52;
276
277    private StatefulRNG rng;
278    private String[] names;
279    private int consonantLimit;
280    private ArrayList<Integer> sizes;
281
282    private HashMap<Character, HashMap<Character, ProbabilityTable<Character>>> letters;
283    private ArrayList<Character> firstLetterSamples;
284    private ArrayList<Character> lastLetterSamples;
285    private DamerauLevenshteinAlgorithm dla = new DamerauLevenshteinAlgorithm(1, 1, 1, 1);

287    /**
288     * Creates the generator by seeding the provided list of names.
289     *

296    /**
297     * Creates the generator by seeding the provided list of names.
298     *
299     * @param names          an array of Strings that are typical names to be emulated
300     * @param consonantLimit the maximum allowed consonants in a row
301     */
302    public WeightedLetterNamegen(String[] names, int consonantLimit) {
303        this(names, consonantLimit, new StatefulRNG());
304    }

306    /**
307     * Creates the generator by seeding the provided list of names. The given RNG will be used for
308     * all random decisions this has to make, so if it has the same state (and RandomnessSource) on
309     * different runs through the program, it will produce the same names reliably.
310     *
311     * @param names          an array of Strings that are typical names to be emulated
312     * @param consonantLimit the maximum allowed consonants in a row
313     * @param rng            the source of randomness to be used
314     */
315    public WeightedLetterNamegen(String[] names, int consonantLimit, StatefulRNG rng) {
316        this.names = names;
317        this.consonantLimit = consonantLimit;
318        this.rng = rng;
319        init();
320    }
321
322    /**
323     * Initialization, statistically measures letter likelihood.
324     */
325    private void init() {
326        sizes = new ArrayList<>();
327        letters = new HashMap<>();
328        firstLetterSamples = new ArrayList<>();
329        lastLetterSamples = new ArrayList<>();
330
331        for (int i = 0; i < names.length - 1; i++) {
332            String name = names[i];
333            if (name == null || name.length() < 1) {
334                continue;
335            }
336
337            // (1) Insert size
338            sizes.add(name.length());
339
340            // (2) Grab first letter
341            firstLetterSamples.add(name.charAt(0));
342
343            // (3) Grab last letter
344            lastLetterSamples.add(name.charAt(name.length() - 1));
345
346            // (4) Process all letters
347            for (int n = 0; n < name.length() - 1; n++) {
348                char letter = name.charAt(n);
349                char nextLetter = name.charAt(n + 1);
350
351                // Create letter if it doesn't exist
352                HashMap<Character, ProbabilityTable<Character>> wl = letters.get(letter);
353                if (wl == null) {
354                    wl = new HashMap<>();
355                    letters.put(letter, wl);
356                }
357                ProbabilityTable<Character> wlg = wl.get(letter);
358                if (wlg == null) {
359                    wlg = new ProbabilityTable<>((StatefulRandomness) rng.getRandomness());
360                    wl.put(letter, wlg);
361                }
362                wlg.add(nextLetter, 1);
363
364                // If letter was uppercase (beginning of name), also add a lowercase entry
365                if (Category.Lu.contains(letter)) {
366                    letter = Character.toLowerCase(letter);
367
368                    wlg = wl.get(letter);
369                    if (wlg == null) {
370                        wlg = new ProbabilityTable<>((StatefulRandomness) rng.getRandomness());
371                        wl.put(letter, wlg);
372                    }
373                    wlg.add(nextLetter, 1);
374                }
375            }
376        }
377    }

379    private StringBuilder generateInner(StringBuilder name) {
380        for (int runs = 0; runs < LAST_LETTER_CANDIDATES_MAX; runs++) {
381            name.setLength(0);
382            // Pick size
383            int size = rng.getRandomElement(sizes);
384
385            // Pick first letter
386            char latest = rng.getRandomElement(firstLetterSamples);
387            name.append(latest);
388
389            for (int i = 1; i < size - 2; i++) {
390                name.append(latest = getRandomNextLetter(latest));
391            }
392
393            // Attempt to find a last letter
394            for (int lastLetterFits = 0; lastLetterFits < LAST_LETTER_CANDIDATES_MAX; lastLetterFits++) {
395                char lastLetter = rng.getRandomElement(lastLetterSamples);
396                char intermediateLetterCandidate = getIntermediateLetter(latest, lastLetter);
397
398                // Only attach last letter if the candidate is valid (if no candidate, the antepenultimate letter always occurs at the end)
399                if (Category.L.contains(intermediateLetterCandidate)) {
400                    name.append(intermediateLetterCandidate).append(lastLetter);
401                    break;
402                }
403            }
404
405            // Check that the word has no triple letter sequences, and that the Levenshtein distance is kosher
406            if (validateGrouping(name) && checkLevenshtein(name)) {
407                return name;
408            }
409        }
410        name.setLength(0);
411        return name.append(rng.getRandomElement(names));
412    }
413    /**
414     * Gets one random String name.
415     *
416     * @return a single random String name
417     */
418
419    public String generate() {
420        return generateInner(new StringBuilder(32)).toString();
421    }

423    /**
424     * Gets an ArrayList of random String names, sized to match amountToGenerate.
425     * @param amountToGenerate how many String items to include in the returned ArrayList
426     * @return an ArrayList of random String names
427     */
428    public ArrayList<String> generateList(int amountToGenerate) {
429        ArrayList<String> result = new ArrayList<>();
430
431        StringBuilder name = new StringBuilder(32);
432        for (int i = 0; i < amountToGenerate; i++) {
433            result.add(generateInner(name).toString());
434        }
435
436        return result;
437    }

450    /**
451     * Searches for the best fit letter between the letter before and the letter
452     * after (non-random). Used to determine penultimate letters in names.
453     *
454     * @param   letterBefore    The letter before the desired letter.
455     * @param   letterAfter     The letter after the desired letter.
456     * @return  The best fit letter between the provided letters.
457     */
458    private char getIntermediateLetter(char letterBefore, char letterAfter) {
459        if (Category.L.contains(letterBefore) && Category.L.contains(letterAfter)) {
460            // First grab all letters that come after the 'letterBefore'
461            HashMap<Character, ProbabilityTable<Character>> wl = letters.get(letterBefore);
462            if (wl == null) {
463                return getRandomNextLetter(letterBefore);
464            }
465            Set<Character> letterCandidates = wl.get(letterBefore).items();
466
467            char bestFitLetter = '\'';
468            int bestFitScore = 0;
469
470            // Step through candidates, and return best scoring letter
471            for (char letter : letterCandidates) {
472                wl = letters.get(letter);
473                if (wl == null) {
474                    continue;
475                }
476                ProbabilityTable<Character> weightedLetterGroup = wl.get(letterBefore);
477                if (weightedLetterGroup != null) {
478                    int letterCounter = weightedLetterGroup.weight(letterAfter);
479                    if (letterCounter > bestFitScore) {
480                        bestFitLetter = letter;
481                        bestFitScore = letterCounter;
482                    }
483                }
484            }
485
486            return bestFitLetter;
487        } else {
488            return '-';
489        }
490    }

492    /**
493     * Checks that no three letters happen in succession.
494     *
495     * @param   name    The name CharSequence
496     * @return  True if no triple letter sequence is found.
497     */
498    private boolean validateGrouping(CharSequence name) {
499        for (int i = 2; i < name.length(); i++) {
500            if (name.charAt(i) == name.charAt(i - 1) && name.charAt(i) == name.charAt(i - 2)) {
501                return false;
502            }
503        }
504        int consonants = 0;
505        for (int i = 0; i < name.length(); i++) {
506            if (isVowel(name.charAt(i))) {
507                consonants = 0;
508            } else {
509                if (++consonants > consonantLimit) {
510                    return false;
511                }
512            }
513        }
514        return true;
515    }
516
517    private boolean isVowel(char c) {
518        switch(c)
519        {
520            case 'a':
521            case 'e':
522            case 'i':
523            case 'o':
524            case 'u':
525                return true;
526            default:
527                return false;
528        }
529    }

531    /**
532     * Checks that the Damerau-Levenshtein distance of this name is within a
533     * given bias from a name on the master list.
534     *
535     * @param   name    The name string.
536     * @return  True if a name is found that is within the bias.
537     */
538    private boolean checkLevenshtein(CharSequence name) {
539        int levenshteinBias = name.length() / 2;
540
541        for (String name1 : names) {
542            int levenshteinDistance = dla.execute(name, name1);
543            if (levenshteinDistance <= levenshteinBias) {
544                return true;
545            }
546        }
547
548        return false;
549    }
550
551    private char getRandomNextLetter(char letter) {
552        if (letters.containsKey(letter)) {
553            return letters.get(letter).get(letter).random();
554        } else {
555            return vowels[rng.nextIntHasty(4)];
556        }
557    }
