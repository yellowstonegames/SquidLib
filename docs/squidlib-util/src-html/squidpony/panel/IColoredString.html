262                /**
263                 * @param one
264                 * @param two 
265                 * @return Whether {@code one} represents the same content as
266                 *         {@code two}.
267                 */
268                /*
269                 * Method could be smarter, i.e. return true more often, by doing some
270                 * normalization. It is unnecessary if you only create instances of
271                 * IColoredString.Impl.
272                 */
273                public static <T> boolean equals(IColoredString<T> one, IColoredString<T> two) {
274                        if (one == two)
275                                return true;
276
277                        final Iterator<IColoredString.Bucket<T>> oneIt = one.iterator();
278                        final Iterator<IColoredString.Bucket<T>> twoIt = two.iterator();
279                        while (true) {
280                                if (oneIt.hasNext()) {
281                                        if (twoIt.hasNext()) {
282                                                final Bucket<T> oneb = oneIt.next();
283                                                final Bucket<T> twob = twoIt.next();
284                                                if (!equals(oneb.getText(), twob.getText()))
285                                                        return false;
286                                                if (!equals(oneb.getColor(), twob.getColor()))
287                                                        return false;
288                                                continue;
289                                        } else
290                                                /* 'this' not terminated, but 'other' is. */
291                                                return false;
292                                } else {
293                                        if (twoIt.hasNext())
294                                                /* 'this' terminated, but not 'other'. */
295                                                return false;
296                                        else
297                                                /* Both terminated */
298                                                break;
299                                }
300
301                        }
302                        return true;
303                }

320                @Override
321                public void append(String text, T color) {
322                        if (text == null || text.isEmpty())
323                                return;
324
325                        if (fragments.isEmpty())
326                                fragments.add(new Bucket<>(text, color));
327                        else {
328                                final Bucket<T> last = fragments.getLast();
329                                if (equals(last.color, color)) {
330                                        /* Append to the last bucket, to avoid extending the list */
331                                        final Bucket<T> novel = last.append(text);
332                                        fragments.removeLast();
333                                        fragments.addLast(novel);
334                                } else
335                                        fragments.add(new Bucket<>(text, color));
336                        }
337                }

357                @Override
358                public void replaceColor(/* @Nullable */ T old, /* @Nullable */ T new_) {
359                        if (equals(old, new_))
360                                /* Nothing to do */
361                                return;
362
363                        final ListIterator<Bucket<T>> it = fragments.listIterator();
364                        while (it.hasNext()) {
365                                final Bucket<T> bucket = it.next();
366                                if (equals(bucket.color, old)) {
367                                        /* Replace */
368                                        it.remove();
369                                        it.add(new Bucket<T>(bucket.getText(), new_));
370                                }
371                                /* else leave untouched */
372                        }
373                }

391                @Override
392                public void setLength(int len) {
393                        int l = 0;
394                        final ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
395                        while (it.hasNext()) {
396                                final IColoredString.Bucket<T> next = it.next();
397                                final String ftext = next.text;
398                                final int flen = ftext.length();
399                                final int nextl = l + flen;
400                                if (nextl < len) {
401                                        /* Nothing to do */
402                                        l += flen;
403                                        continue;
404                                } else if (nextl == len) {
405                                        /* Delete all next fragments */
406                                        while (it.hasNext()) {
407                                                it.next();
408                                                it.remove();
409                                        }
410                                        /* We'll exit the outer loop right away */
411                                } else {
412                                        assert len < nextl;
413                                        /* Trim this fragment */
414                                        final IColoredString.Bucket<T> trimmed = next.setLength(len - l);
415                                        /* Replace this fragment */
416                                        it.remove();
417                                        it.add(trimmed);
418                                        /* Delete all next fragments */
419                                        while (it.hasNext()) {
420                                                it.next();
421                                                it.remove();
422                                        }
423                                        /* We'll exit the outer loop right away */
424                                }
425                        }
426                }

428                @Override
429                public List<IColoredString<T>> wrap(int width)
430                {
431                        // the one special case that wouldn't involve buf
432                        if (width <= 0) {
433                                /* Really, you should not rely on this behavior */
434                                System.err.println("Cannot wrap string in empty display");
435                                final List<IColoredString<T>> result = new LinkedList<>();
436                                result.add(this);
437                                return result;
438                        }
439                        // delegate to the buf-appending overload
440                        return wrap(width, new ArrayList<IColoredString<T>>());
441                }

443                @Override
444                public List<IColoredString<T>> wrap(int width, List<IColoredString<T>> buf) {
445                        if (width <= 0) {
446                                /* Really, you should not rely on this behavior */
447                                System.err.println("Cannot wrap string in empty display");
448                                final List<IColoredString<T>> result = new LinkedList<>();
449                                result.add(this);
450                                return result;
451                        }
452                        if (isEmpty()) {
453                                /*
454                                 * Catch this case early on, as empty lines are eaten below (see
455                                 * code after the while). Checking emptiness is cheap anyway.
456                                 */
457                                buf.add(this);
458                                return buf;
459                        }
460
461                        IColoredString<T> current = create();
462                        int curlen = 0;
463                        final Iterator<Bucket<T>> it = iterator();
464                        while (it.hasNext()) {
465                                final Bucket<T> next = it.next();
466                                final String bucket = next.getText();
467                                final String[] split = StringKit.split(bucket," ");
468                                final T color = next.color;
469                                for (int i = 0; i < split.length; i++) {
470                                        // This section was needed when using String.split() above, but not for
471                                        // StringKit.split(), which keeps leading and trailing delimiters.
472//                                      if (i == split.length - 1 && bucket.endsWith(" "))
473//                                              /*
474//                                               * Do not lose trailing space that got eaten by
475//                                               * 'bucket.split'.
476//                                               */
477//                                              split[i] = split[i] + " ";
478                                        final String chunk = split[i];
479                                        final int chunklen = chunk.length();
480                                        final boolean addLeadingSpace = 0 < curlen && 0 < i;
481                                        if (curlen + chunklen + (addLeadingSpace ? 1 : 0) <= width) {
482                                                if (addLeadingSpace) {
483                                                        /*
484                                                         * Do not forget space on which chunk got split. If
485                                                         * the space is offscreen, it's harmless, hence not
486                                                         * checking it.
487                                                         */
488                                                        current.append(' ', null);
489                                                        curlen++;
490                                                }
491
492                                                /* Can add it */
493                                                current.append(chunk, color);
494                                                /* Extend size */
495                                                curlen += chunklen;
496                                        } else {
497                                                /* Need to wrap */
498                                                /* Flush content so far */
499                                                if (!current.isEmpty())
500                                                        buf.add(current);
501                                                /*
502                                                 * else: line was prepared, but did not contain anything
503                                                 */
504                                                if (chunklen <= width) {
505                                                        current = create();
506                                                        current.append(chunk, color);
507                                                        /* Reinit size */
508                                                        curlen = chunklen;
509                                                } else {
510                                                        /*
511                                                         * This word is too long. Adding it and preparing a
512                                                         * new line immediately.
513                                                         */
514                                                        /* Add */
515                                                        buf.add(new Impl<>(chunk, color));
516                                                        /* Prepare for next rolls */
517                                                        current = create();
518                                                        /* Reinit size */
519                                                        curlen = 0;
520                                                }
521                                        }
522                                }
523                        }
524
525                        if (!current.isEmpty()) {
526                                /* Flush rest */
527                                buf.add(current);
528                        }
529
530                        return buf;
531                }

533                @Override
534                /*
535                 * smelC: not the cutest result (we should add spaces both from the left
536                 * and the right, instead of just from the left), but better than
537                 * nothing.
538                 */
539                public IColoredString<T> justify(int width) {
540                        int length = length();
541
542                        if (width <= length)
543                                /*
544                                 * If width==length, we're good. If width<length, we cannot
545                                 * adjust
546                                 */
547                                return this;
548
549                        int totalDiff = width - length;
550                        assert 0 < totalDiff;
551
552                        if (width <= totalDiff * 3)
553                                /* Too much of a difference, it would look very weird. */
554                                return this;
555
556                        final IColoredString.Impl<T> result = create();
557
558                        ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
559                        final int nbb = fragments.size();
560                        final int[] bucketToNbSpaces = new int[nbb];
561                        /* The number of buckets that can contribute to justifying */
562                        int totalNbSpaces = 0;
563                        /* The index of the last bucket that has spaces */
564                        int lastHopeIndex = -1;
565                        {
566                                int i = 0;
567                                while (it.hasNext()) {
568                                        final Bucket<T> next = it.next();
569                                        final int nbs = nbSpaces(next.getText());
570                                        totalNbSpaces += nbs;
571                                        bucketToNbSpaces[i] = nbs;
572                                        i++;
573                                }
574
575                                if (totalNbSpaces == 0)
576                                        /* Cannot do anything */
577                                        return this;
578
579                                for (int j = bucketToNbSpaces.length - 1; 0 <= j; j--) {
580                                        if (0 < bucketToNbSpaces[j]) {
581                                                lastHopeIndex = j;
582                                                break;
583                                        }
584                                }
585                                /* Holds because we ruled out 'totalNbSpaces == 0' before */
586                                assert 0 <= lastHopeIndex;
587                        }
588
589                        // we know totalNbSpaces cannot be 0 from prior checks, so division is OK
590                        int toAddPerSpace = (totalDiff / totalNbSpaces);
591                        int totalRest = totalDiff - (toAddPerSpace * totalNbSpaces);
592                        assert 0 <= totalRest;
593
594                        int bidx = -1;
595
596                        it = fragments.listIterator();
597
598                        while (it.hasNext() && 0 < totalDiff) {
599                                bidx++;
600                                final Bucket<T> next = it.next();
601                                final String bucket = next.getText();
602                                final int blength = bucket.length();
603                                final int localNbSpaces = bucketToNbSpaces[bidx];
604                                if (localNbSpaces == 0) {
605                                        /* Cannot change it */
606                                        result.append(next);
607                                        continue;
608                                }
609                                int localDiff = localNbSpaces * toAddPerSpace;
610                                assert localDiff <= totalDiff;
611                                int nb = localDiff / localNbSpaces;
612                                int localRest = localDiff - (nb * localNbSpaces);
613                                if (localRest == 0 && 0 < totalRest) {
614                                        /*
615                                         * Take one for the group. This avoids flushing all spaces
616                                         * needed in the 'last hope' cases below.
617                                         */
618                                        localRest = 1;
619                                }
620                                assert 0 <= localRest;
621                                assert localRest <= totalRest;
622                                String novel = "";
623                                int eatenSpaces = 1;
624                                for (int i = 0; i < blength; i++) {
625                                        final char c = bucket.charAt(i);
626                                        novel += c;
627                                        if (c == ' ' && (0 < localDiff || 0 < totalDiff || 0 < localRest || 0 < totalRest)) {
628                                                /* Can (and should) add an extra space */
629                                                for (int j = 0; j < nb && 0 < localDiff; j++) {
630                                                        novel += " ";
631                                                        localDiff--;
632                                                        totalDiff--;
633                                                }
634                                                if (0 < localRest || 0 < totalRest) {
635                                                        if (eatenSpaces == localNbSpaces) {
636                                                                /* I'm the last hope for this bucket */
637                                                                for (int j = 0; j < localRest; j++) {
638                                                                        novel += " ";
639                                                                        localRest--;
640                                                                        totalRest--;
641                                                                }
642                                                                if (bidx == lastHopeIndex) {
643                                                                        /* I'm the last hope globally */
644                                                                        while (0 < totalRest) {
645                                                                                novel += " ";
646                                                                                totalRest--;
647                                                                        }
648                                                                }
649                                                        } else {
650                                                                if (0 < localRest && 0 < totalRest) {
651                                                                        /* Not the last hope: take one only */
652                                                                        novel += " ";
653                                                                        localRest--;
654                                                                        totalRest--;
655                                                                }
656                                                        }
657                                                }
658                                                eatenSpaces++;
659                                        }
660                                }
661                                /* I did my job */
662                                assert localRest == 0;
663                                /* If I was the hope, I did my job */
664                                assert bidx != lastHopeIndex || totalRest == 0;
665                                result.append(novel, next.getColor());
666                        }
667
668                        while (it.hasNext())
669                                result.append(it.next());
670
671                        return result;
672                }

703                @Override
704                public T colorAt(int index) {
705                        final ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
706                        int now = 0;
707                        while (it.hasNext()) {
708                                final IColoredString.Bucket<T> next = it.next();
709                                final String ftext = next.text;
710                                final int flen = ftext.length();
711                                final int nextl = now + flen;
712                                if (index < nextl)
713                                        return next.color;
714                                now += flen;
715                        }
716                        throw new NoSuchElementException("Color at index " + index + " in " + this);
717                }

825        class Bucket<T> {
826
827                protected final String text;
828                protected final/* @Nullable */T color;
829
830                public Bucket(String text, /* @Nullable */T color) {
831                        this.text = text == null ? "" : text;
832                        this.color = color;
833                }
834
835                /**
836                 * @param text
837                 * @return An instance whose text is {@code this.text + text}. Color is
838                 *         unchanged.
839                 */
840                public Bucket<T> append(String text) {
841                        if (text == null || text.isEmpty())
842                                /* Let's save an allocation */
843                                return this;
844                        else
845                                return new Bucket<>(this.text + text, color);
846                }

854                }
