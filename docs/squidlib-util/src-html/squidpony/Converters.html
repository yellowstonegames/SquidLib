026    public static <K> StringConvert<OrderedSet<K>> convertOrderedSet(final StringConvert<K> convert) {
027        CharSequence[] types = StringConvert.asArray("OrderedSet", convert.name);
028        StringConvert found = StringConvert.lookup(types);
029        if (found != null)
030            return found; // in this case we've already created a StringConvert for this type combination
031
032        return new StringConvert<OrderedSet<K>>(types) {
033            @Override
034            public String stringify(OrderedSet<K> item) {
035                StringBuilder sb = new StringBuilder(100);
036                K k;
037                for (int i = 0; i < item.size(); ) {
038                    k = item.getAt(i);
039                    if (item == k)
040                        return "";
041                    ObText.appendQuoted(sb, convert.stringify(k));
042                    if (++i < item.size())
043                        sb.append(' ');
044                }
045                return sb.toString();
046            }
047
048            @Override
049            public OrderedSet<K> restore(String text) {
050                ObText.ContentMatcher m = makeMatcher(text);
051                OrderedSet<K> d = new OrderedSet<>();
052                while (m.find()) {
053                    if (m.hasMatch()) {
054                        d.add(convert.restore(m.getMatch()));
055                    }
056                }
057                return d;
058            }
059        };
060    }

070    public static <K, V> StringConvert<OrderedMap<K, V>> convertOrderedMap(final StringConvert<K> convertK, final StringConvert<V> convertV) {
071        CharSequence[] types = StringConvert.asArray("OrderedMap", convertK.name, convertV.name);
072        StringConvert found = StringConvert.lookup(types);
073        if (found != null)
074            return found; // in this case we've already created a StringConvert for this type combination
075
076        return new StringConvert<OrderedMap<K, V>>(types) {
077            @Override
078            public String stringify(OrderedMap<K, V> item) {
079                StringBuilder sb = new StringBuilder(100);
080                K k;
081                V v;
082                for (int i = 0; i < item.size(); ) {
083                    k = item.keyAt(i);
084                    if (k == item)
085                        return "";
086                    appendQuoted(sb, convertK.stringify(k));
087                    sb.append(' ');
088                    v = item.getAt(i);
089                    if (v == item)
090                        return "";
091                    appendQuoted(sb, convertV.stringify(v));
092                    if (++i < item.size())
093                        sb.append('\n');
094                }
095                return sb.toString();
096            }
097
098            @Override
099            public OrderedMap<K, V> restore(String text) {
100                ObText.ContentMatcher m = makeMatcher(text);
101                OrderedMap<K, V> d = new OrderedMap<>();
102                String t;
103                while (m.find()) {
104                    if (m.hasMatch()) {
105                        t = m.getMatch();
106                        if (m.find() && m.hasMatch()) {
107                            d.put(convertK.restore(t), convertV.restore(m.getMatch()));
108                        }
109                    }
110                }
111                return d;
112            }
113        };
114    }

180    public static final StringConvert<Coord[]> convertArrayCoord = new StringConvert<Coord[]>("Coord[]") {
181        @Override
182        public String stringify(Coord[] item) {
183            int len = item.length;
184            StringBuilder sb = new StringBuilder(len * 5);
185            for (int i = 0; i < len; ) {
186                sb.append(item[i].x).append(',').append(item[i].y);
187                if (++i < len)
188                    sb.append(';');
189            }
190            return sb.toString();
191        }
192
193        @Override
194        public Coord[] restore(String text) {
195            Coord[] coords = new Coord[StringKit.count(text, ';') + 1];
196            int start = -1, end = text.indexOf(',');
197            for (int i = 0; i < coords.length; i++) {
198                coords[i] = Coord.get(StringKit.intFromDec(text, start + 1, end),
199                        StringKit.intFromDec(text, end + 1, (start = text.indexOf(';', end + 1))));
200                end = text.indexOf(',', start + 1);
201            }
202            return coords;
203        }
204    };

364    public static final StringConvert<boolean[]> convertArrayBoolean = new StringConvert<boolean[]>("boolean[]") {
365        @Override
366        public String stringify(boolean[] item) {
367            return StringKit.join(",", item);
368        }
369
370        @Override
371        public boolean[] restore(String text) {
372            int amount = StringKit.count(text, ",");
373            if (amount <= 0) return new boolean[]{"true".equals(text)};
374            boolean[] splat = new boolean[amount + 1];
375            int dl = 1, idx = -dl, idx2;
376            for (int i = 0; i < amount; i++) {
377                splat[i] = "true".equals(StringKit.safeSubstring(text, idx + dl, idx = text.indexOf(',', idx + dl)));
378            }
379            if ((idx2 = text.indexOf(',', idx + dl)) < 0) {
380                splat[amount] = "true".equals(StringKit.safeSubstring(text, idx + dl, text.length()));
381            } else {
382                splat[amount] = "true".equals(StringKit.safeSubstring(text, idx + dl, idx2));
383            }
384            return splat;
385        }
386    };

547    public static final StringConvert<char[][]> convertArrayChar2D = new StringConvert<char[][]>("char[][]") {
548        @Override
549        public String stringify(char[][] item) {
550            int len, l2, sum;
551            if (item == null) return "N"; // N for null
552            if((len = item.length) <= 0) return "R0|0|";
553            sum = l2 = item[0].length;
554            char regular = 'R'; // R for rectangular
555            for (int i = 1; i < len; i++) {
556                if(item[i] == null)
557                {
558                    regular = 'J'; // J for jagged
559                }
560                else if(l2 != (l2 = item[i].length))
561                {
562                    regular = 'J';  // J for jagged
563                    sum += l2;
564                }
565            }
566            StringBuilder sb;
567            if(regular == 'R')
568            {
569                sb = new StringBuilder(len * l2 + 15);
570                sb.append('R').append(len).append('|').append(l2).append('|');
571                for (int i = 0; i < len; i++) {
572                    sb.append(item[i]);
573                }
574            }
575            else
576            {
577                sb = new StringBuilder(len * 7 + sum + 8);
578                sb.append('J').append(len).append('|');
579                for (int i = 0; i < len; i++) {
580                    if(item[i] == null)
581                        sb.append("-|");
582                    else
583                        sb.append(item[i].length).append('|').append(item[i]);
584                }
585            }
586            return sb.toString();
587        }
588
589        @Override
590        public char[][] restore(String text) {
591            if(text == null || text.length() <= 1) return null;
592            if(text.charAt(0) == 'R')
593            {
594                int width, height, start = 1, end = text.indexOf('|');
595                width = StringKit.intFromDec(text, start, end);
596                start = end+1;
597                end = text.indexOf('|', start);
598                height = StringKit.intFromDec(text, start, end);
599                start = end+1;
600                char[][] val = new char[width][height];
601                for (int i = 0; i < width; i++) {
602                    text.getChars(start, start += height, val[i], 0);
603                }
604                return val;
605            }
606            else
607            {
608                int width, current, start = 1, end = text.indexOf('|');
609                width = StringKit.intFromDec(text, start, end);
610                start = end + 1;
611                char[][] val = new char[width][];
612                for (int i = 0; i < width; i++) {
613                    end = text.indexOf('|', start);
614                    if (text.charAt(start) == '-') {
615                        val[i] = null;
616                        start = end + 1;
617                    } else {
618                        current = StringKit.intFromDec(text, start, end);
619                        start = end + 1;
620                        val[i] = new char[current];
621                        text.getChars(start, start += current, val[i], 0);
622                    }
623                }
624                return val;
625            }
626        }
