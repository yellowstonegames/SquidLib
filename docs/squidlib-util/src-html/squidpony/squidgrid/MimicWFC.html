013/**
014 * A port of WaveFunctionCollapse by ExUtumno/mxgmn; takes a single sample of a grid to imitate and produces one or more
015 * grids of requested sizes that have a similar layout of cells to the sample. Samples are given as {@code int[][]}
016 * where an int is usually an index into an array, list, {@link squidpony.squidmath.Arrangement}, or some similar
017 * indexed collection of items (such as char values or colors) that would be used instead of an int directly. The
018 * original WaveFunctionCollapse code, <a href="https://github.com/mxgmn/WaveFunctionCollapse">here</a>, used colors in
019 * bitmap images, but this uses 2D int arrays that can stand as substitutes for colors or chars.
020 * <br>
021 * Created by Tommy Ettinger on 3/28/2018. Port of https://github.com/mxgmn/WaveFunctionCollapse
022 */

050    public MimicWFC(int[][] itemGrid, int order, int width, int height, boolean periodicInput, boolean periodicOutput, int symmetry, int ground)
051    {
052        FMX = width;
053        FMY = height;
054
055        this.order = order;
056        periodic = periodicOutput;
057
058        int SMX = itemGrid.length, SMY = itemGrid[0].length;
059        //colors = new List<Color>();
060        choices = new Arrangement<>(SMX * SMY);
061        int[][] sample = new int[SMX][SMY];
062        for (int y = 0; y < SMY; y++) {
063            for (int x = 0; x < SMX; x++)
064            {
065                int color = itemGrid[x][y];
066                int i = choices.addOrIndex(color);
067                sample[x][y] = i;
068            }
069        }
070
071        int C = choices.size();
072        long W = MathExtras.raiseToPower(C, order * order);
073
074
075
076//        Dictionary<long, int> weights = new Dictionary<long, int>();
077//        List<long> ordering = new List<long>();
078        OrderedMap<int[], Integer> weights = new OrderedMap<>(CrossHash.intHasher);
079
080        for (int y = 0; y < (periodicInput ? SMY : SMY - order + 1); y++) {
081            for (int x = 0; x < (periodicInput ? SMX : SMX - order + 1); x++) {
082                int[][] ps = new int[8][];
083
084                ps[0] = patternFromSample(x, y, sample, SMX, SMY);
085                ps[1] = reflect(ps[0]);
086                ps[2] = rotate(ps[0]);
087                ps[3] = reflect(ps[2]);
088                ps[4] = rotate(ps[2]);
089                ps[5] = reflect(ps[4]);
090                ps[6] = rotate(ps[4]);
091                ps[7] = reflect(ps[6]);
092
093                for (int k = 0; k < symmetry; k++) {
094                    int[] ind = ps[k];
095                    Integer wt = weights.get(ind);
096                    if (wt != null) weights.put(ind, wt + 1);
097                    else {
098                        weights.put(ind, 1);
099                    }
100                }
101            }
102        }
103
104        totalOptions = weights.size();
105        this.ground = (ground + totalOptions) % totalOptions;
106        patterns = new int[totalOptions][];
107        baseWeights = new double[totalOptions];
108
109        for (int w = 0; w < totalOptions; w++) {
110            patterns[w] = weights.keyAt(w);
111            baseWeights[w] = weights.getAt(w);
112        }
113        
114
115        propagator = new int[4][][];
116        IntVLA list = new IntVLA(totalOptions);
117        for (int d = 0; d < 4; d++)
118        {
119            propagator[d] = new int[totalOptions][];
120            for (int t = 0; t < totalOptions; t++)
121            {
122                list.clear();
123                for (int t2 = 0; t2 < totalOptions; t2++) if (agrees(patterns[t], patterns[t2], DX[d], DY[d])) list.add(t2);
124                propagator[d][t] = list.toArray();
125            }
126        }
127    }

129    private long index(byte[] p, long C)
130    {
131        long result = 0, power = 1;
132        for (int i = 0; i < p.length; i++)
133        {
134            result += p[p.length - 1 - i] * power;
135            power *= C;
136        }
137        return result;
138    }
139
140    private byte[] patternFromIndex(long ind, long power, long C)
141    {
142        long residue = ind;
143        byte[] result = new byte[order * order];
144
145        for (int i = 0; i < result.length; i++)
146        {
147            power /= C;
148            int count = 0;
149
150            while (residue >= power)
151            {
152                residue -= power;
153                count++;
154            }
155
156            result[i] = (byte)count;
157        }
158
159        return result;
160    }

173    private int[] patternFromSample(int x, int y, int[][] sample, int SMX, int SMY) {
174        int[] result = new int[order * order];
175        for (int dy = 0; dy < order; dy++) {
176            for (int dx = 0; dx < order; dx++) {
177                result[dx + dy * order] = sample[(x + dx) % SMX][(y + dy) % SMY];
178            }
179        }
180        return result;
181    }
182    private int[] rotate(int[] p)
183    {
184        int[] result = new int[order * order];
185        for (int y = 0; y < order; y++) {
186            for (int x = 0; x < order; x++){
187                result[x + y * order] = p[order - 1 - y + x * order];
188            }
189        }
190        return result;
191    }
192    private int[] reflect(int[] p)
193    {
194        int[] result = new int[order * order];
195        for (int y = 0; y < order; y++) {
196            for (int x = 0; x < order; x++){
197                result[x + y * order] = p[order - 1 - x + y * order];
198            }
199        }
200        return result;
201    }
202    private boolean agrees(int[] p1, int[] p2, int dx, int dy)
203    {
204        int xmin = dx < 0 ? 0 : dx, xmax = dx < 0 ? dx + order : order,
205                ymin = dy < 0 ? 0 : dy, ymax = dy < 0 ? dy + order : order;
206        for (int y = ymin; y < ymax; y++) {
207            for (int x = xmin; x < xmax; x++) {
208                if (p1[x + order * y] != p2[x - dx + order * (y - dy)])
209                    return false;
210            }
211        }
212        return true;
213    }

215    private void init()
216    {
217        wave = new boolean[FMX * FMY][];
218        compatible = new int[wave.length][][];
219        for (int i = 0; i < wave.length; i++)
220        {
221            wave[i] = new boolean[totalOptions];
222            compatible[i] = new int[totalOptions][];
223            for (int t = 0; t < totalOptions; t++) compatible[i][t] = new int[4];
224        }
225
226        weightLogWeights = new double[totalOptions];
227        sumOfWeights = 0;
228        sumOfWeightLogWeights = 0;
229
230        for (int t = 0; t < totalOptions; t++)
231        {
232            weightLogWeights[t] = baseWeights[t] * Math.log(baseWeights[t]);
233            sumOfWeights += baseWeights[t];
234            sumOfWeightLogWeights += weightLogWeights[t];
235        }
236
237        startingEntropy = Math.log(sumOfWeights) - sumOfWeightLogWeights / sumOfWeights;
238
239        sumsOfOnes = new int[FMX * FMY];
240        sumsOfWeights = new double[FMX * FMY];
241        sumsOfWeightLogWeights = new double[FMX * FMY];
242        entropies = new double[FMX * FMY];
243
244        stack = new int[wave.length * totalOptions << 1];
245        stacksize = 0;
246    }
247
248    private Boolean observe()
249    {
250        double min = 1E+3;
251        int argmin = -1;
252
253        for (int i = 0; i < wave.length; i++)
254        {
255            if (onBoundary(i % FMX, i / FMX)) continue;
256
257            int amount = sumsOfOnes[i];
258            if (amount == 0) return false;
259
260            double entropy = entropies[i];
261            if (amount > 1 && entropy <= min)
262            {
263                double noise = 1E-6 * random.nextDouble();
264                if (entropy + noise < min)
265                {
266                    min = entropy + noise;
267                    argmin = i;
268                }
269            }
270        }
271
272        if (argmin == -1)
273        {
274            observed = new int[FMX * FMY];
275            for (int i = 0; i < wave.length; i++) {
276                for (int t = 0; t < totalOptions; t++) {
277                    if (wave[i][t]) { 
278                        observed[i] = t;
279                        break;
280                    }
281                }
282            }
283            return true;
284        }
285
286        double[] distribution = new double[totalOptions];
287        double sum = 0.0, x = 0.0;
288        for (int t = 0; t < totalOptions; t++)
289        {
290            sum += (distribution[t] = wave[argmin][t] ? baseWeights[t] : 0);
291        }
292        int r = 0;
293        sum = random.nextDouble(sum);
294        for (; r < totalOptions; r++) {
295            if((x += distribution[r]) > sum)
296                break;
297        }
298
299        boolean[] w = wave[argmin];
300        for (int t = 0; t < totalOptions; t++){
301            if (w[t] != (t == r))
302                ban(argmin, t);
303        }
304
305        return null;
306    }

308    private void propagate()
309    {
310        while (stacksize > 0)
311        {
312            int i1 = stack[stacksize - 2], e2 = stack[stacksize - 1];
313            stacksize -= 2;
314            int x1 = i1 % FMX, y1 = i1 / FMX;
315
316            for (int d = 0; d < 4; d++)
317            {
318                int dx = DX[d], dy = DY[d];
319                int x2 = x1 + dx, y2 = y1 + dy;
320                if (onBoundary(x2, y2)) continue;
321
322                if (x2 < 0) x2 += FMX;
323                else if (x2 >= FMX) x2 -= FMX;
324                if (y2 < 0) y2 += FMY;
325                else if (y2 >= FMY) y2 -= FMY;
326
327                int i2 = x2 + y2 * FMX;
328                int[] p = propagator[d][e2];
329                int[][] compat = compatible[i2];
330
331                for (int l = 0; l < p.length; l++)
332                {
333                    int t2 = p[l];
334                    int[] comp = compat[t2];
335
336                    comp[d]--;
337                    if (comp[d] == 0) ban(i2, t2);
338                }
339            }
340        }
341    }

343    public boolean run(long seed, int limit)
344    {
345        if (wave == null) init();
346
347        clear();
348        random = new RNG(seed);
349
350        for (int l = 0; l < limit || limit == 0; l++)
351        {
352            Boolean result = observe();
353            if (result != null) return result;
354            propagate();
355        }
356
357        return true;
358    }

360    public boolean run(IRNG rng, int limit)
361    {
362        if (wave == null) init();
363
364        clear();
365        random = rng;
366
367        for (int l = 0; l < limit || limit == 0; l++)
368        {
369            Boolean result = observe();
370            if (result != null) return result;
371            propagate();
372        }
373
374        return true;
375    }

377    private void ban(int i, int t)
378    {
379        wave[i][t] = false;
380
381        int[] comp = compatible[i][t];
382        for (int d = 0; d < 4; d++) comp[d] = 0;
383        stack[stacksize++] = i;
384        stack[stacksize++] = t;
385
386        double sum = sumsOfWeights[i];
387        entropies[i] += sumsOfWeightLogWeights[i] / sum - Math.log(sum);
388
389        sumsOfOnes[i] -= 1;
390        sumsOfWeights[i] -= baseWeights[t];
391        sumsOfWeightLogWeights[i] -= weightLogWeights[t];
392
393        sum = sumsOfWeights[i];
394        entropies[i] -= sumsOfWeightLogWeights[i] / sum - Math.log(sum);
395    }

398    private boolean onBoundary(int x, int y) {
399        return !periodic && (x + order > FMX || y + order > FMY || x < 0 || y < 0);
400    }

402    public int[][] result()
403    {
404        int[][] result = new int[FMX][FMY];
405
406        if (observed != null)
407        {
408            for (int y = 0; y < FMY; y++)
409            {
410                int dy = y < FMY - order + 1 ? 0 : order - 1;
411                for (int x = 0; x < FMX; x++)
412                {
413                    int dx = x < FMX - order + 1 ? 0 : order - 1;
414                    result[x][y] = choices.keyAt(patterns[observed[x - dx + (y - dy) * FMX]][dx + dy * order]);
415                }
416            }
417        }
418        return result;
419    }
420
421    private void clear()
422    {
423        for (int i = 0; i < wave.length; i++)
424        {
425            for (int t = 0; t < totalOptions; t++)
426            {
427                wave[i][t] = true;
428                for (int d = 0; d < 4; d++) compatible[i][t][d] = propagator[OPPOSITE[d]][t].length;
429            }
430
431            sumsOfOnes[i] = baseWeights.length;
432            sumsOfWeights[i] = sumOfWeights;
433            sumsOfWeightLogWeights[i] = sumOfWeightLogWeights;
434            entropies[i] = startingEntropy;
435        }
436
437
438        if (ground != 0)
439        {
440            for (int x = 0; x < FMX; x++)
441            {
442                for (int t = 0; t < totalOptions; t++) if (t != ground) ban(x + (FMY - 1) * FMX, t);
443                for (int y = 0; y < FMY - 1; y++) ban(x + y * FMX, ground);
444            }
445
446            propagate();
447        }
448    }
449    private static int[] DX = { -1, 0, 1, 0 };
450    private static int[] DY = { 0, 1, 0, -1 };
451    private static int[] OPPOSITE = { 2, 3, 0, 1 };
