011/**
012 * This class provides methods for calculating Field of View in grids. Field of
013 * View (FOV) algorithms determine how much area surrounding a point can be
014 * seen. They return a two dimensional array of doubles, representing the amount
015 * of view (typically sight, but perhaps sound, smell, etc.) which the origin
016 * has of each cell.
017 * <br>
018 * The input resistanceMap is considered the percent of opacity. This resistance
019 * is on top of the resistance applied from the light spreading out. You can
020 * obtain a resistance map easily with the DungeonUtility.generateResistances()
021 * method, which uses defaults for common chars used in SquidLib, but you may
022 * also want to create a resistance map manually if a given char means something
023 * very different in your game. This is easy enough to do by looping over all the
024 * x,y positions in your char[][] map and running a switch statement on each char,
025 * assigning a double to the same x,y position in a double[][]. The value should
026 * be between 0.0 (unblocked) for things light passes through, 1.0 (blocked) for
027 * things light can't pass at all, and possibly other values if you have
028 * translucent obstacles.
029 * <br>
030 * The returned light map is considered the percent of light in the cells.
031 * <br>
032 * Not all implementations are required to provide percentage levels of light.
033 * In such cases the returned values will be 0 for no light and 1.0 for fully
034 * lit. Implementations that return this way note so in their documentation.
035 * Currently, all implementations do provide percentage levels.
036 * <br>
037 * All solvers perform bounds checking so solid borders in the map are not
038 * required.
039 * <br>
040 * Static methods are provided to add together FOV maps in the simple way
041 * (disregarding visibility of distant FOV from a given cell), or the more
042 * practical way for roguelikes (where a cell needs to be within line-of-sight
043 * in the first place for a distant light to illuminate it). The second method
044 * relies on an LOS map, which is essentially the same as a very-high-radius
045 * FOV map and can be easily obtained with calculateLOSMap().
046 * <br>
047 * If you want to iterate through cells that are visible in a double[][] returned
048 * by FOV, you can pass that double[][] to the constructor for Region, and you
049 * can use the Region as a reliably-ordered List of Coord (among other things).
050 * The order Region iterates in is somewhat strange, and doesn't, for example,
051 * start at the center of an FOV map, but it will be the same every time you
052 * create a Region with the same FOV map (or the same visible Coords).
053 *
054 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
055 */
056public class FOV implements Serializable {

159    /**
160     * Calculates the Field Of View for the provided map from the given x, y
161     * coordinates. Returns a light map where the values represent a percentage
162     * of fully lit.
163     *
164     * The starting point for the calculation is considered to be at the center
165     * of the origin cell. Radius determinations are determined by the provided
166     * RadiusStrategy.
167     *
168     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
169     * @param startX the horizontal component of the starting location
170     * @param startY the vertical component of the starting location
171     * @param radius the distance the light will extend to
172     * @param radiusTechnique provides a means to calculate the radius as desired
173     * @return the computed light grid
174     */
175    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius, Radius radiusTechnique) {
176        double rad = Math.max(1, radius);
177        double decay = 1.0 / rad;
178
179        int width = resistanceMap.length;
180        int height = resistanceMap[0].length;
181
182        initializeLightMap(width, height);
183        light[startX][startY] = 1;//make the starting space full power
184
185        switch (type) {
186            case RIPPLE:
187            case RIPPLE_LOOSE:
188            case RIPPLE_TIGHT:
189            case RIPPLE_VERY_LOOSE:
190                initializeNearLight(width, height);
191                doRippleFOV(light, rippleValue(type), startX, startY, startX, startY, decay, rad, resistanceMap, nearLight, radiusTechnique);
192                break;
193            case SHADOW:
194                for (Direction d : Direction.DIAGONALS) {
195                    shadowCast(1, 1.0, 0.0, 0, d.deltaX, d.deltaY, 0, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
196                    shadowCast(1, 1.0, 0.0, d.deltaX, 0, 0, d.deltaY, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
197                }
198                break;
199        }
200        return light;
201    }

203        /**
204     * Calculates the Field Of View for the provided map from the given x, y
205     * coordinates. Returns a light map where the values represent a percentage
206     * of fully lit.
207     *
208     * The starting point for the calculation is considered to be at the center
209     * of the origin cell. Radius determinations are determined by the provided
210     * RadiusStrategy. A conical section of FOV is lit by this method if
211     * span is greater than 0.
212     *
213     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
214     * @param startX the horizontal component of the starting location
215     * @param startY the vertical component of the starting location
216     * @param radius the distance the light will extend to
217     * @param radiusTechnique provides a means to calculate the radius as desired
218     * @param angle the angle in degrees that will be the center of the FOV cone, 0 points right
219     * @param span the angle in degrees that measures the full arc contained in the FOV cone
220     * @return the computed light grid
221     */
222    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius,
223                                   Radius radiusTechnique, double angle, double span) {
224
225        radius = Math.max(1, radius);
226
227        double decay = 1.0 / (radius + 1);
228
229        angle = ((angle >= 360.0 || angle < 0.0)
230                ? MathExtras.remainder(angle, 360.0) : angle) * 0.002777777777777778;
231        span = span * 0.002777777777777778;
232        int width = resistanceMap.length;
233        int height = resistanceMap[0].length;
234
235        initializeLightMap(width, height);
236        light[startX][startY] = 1;//make the starting space full power
237
238        switch (type) {
239            case RIPPLE:
240            case RIPPLE_LOOSE:
241            case RIPPLE_TIGHT:
242            case RIPPLE_VERY_LOOSE:
243                initializeNearLight(width, height);
244                doRippleFOV(light, rippleValue(type), startX, startY, startX, startY, decay, radius, resistanceMap, nearLight, radiusTechnique, angle, span);
245                break;
246            case SHADOW:
247                light = shadowCastLimited(1, 1.0, 0.0, 0, 1, 1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
248                light = shadowCastLimited(1, 1.0, 0.0, 1, 0, 0, 1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
249
250                light = shadowCastLimited(1, 1.0, 0.0, 0, -1, 1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
251                light = shadowCastLimited(1, 1.0, 0.0, -1, 0, 0, 1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
252
253                light = shadowCastLimited(1, 1.0, 0.0, 0, -1, -1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
254                light = shadowCastLimited(1, 1.0, 0.0, -1, 0, 0, -1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
255
256                light = shadowCastLimited(1, 1.0, 0.0, 0, 1, -1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
257                light = shadowCastLimited(1, 1.0, 0.0, 1, 0, 0, -1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
258                break;
259        }
260
261        return light;
262    }

446    /**
447     * Reuses the existing light 2D array and fills it with a straight-line bouncing path of light that reflects its way
448     * through the given resistanceMap from startX, startY until it uses up the given distance. The angle the path
449     * takes is given in degrees, and the angle used can change as obstacles are hit (reflecting backwards if it hits a
450     * corner pointing directly into or away from its path). This can be used something like an LOS method, but because
451     * the path can be traveled back over, an array or Queue becomes somewhat more complex, and the decreasing numbers
452     * for a straight line that stack may make more sense for how this could be used (especially with visual effects).
453     * This currently allows the path to pass through single-cell wall-like obstacles without changing direction, e.g.
454     * it passes through pillars, but will bounce if it hits a bigger wall.
455     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
456     * @param light the grid of cells to assign to; may have existing values, and 0.0 is used to mean "unlit"
457     * @param startX the horizontal component of the starting location
458     * @param startY the vertical component of the starting location
459     * @param distance the distance the light will extend to
460     * @param angle in degrees, the angle to start the path traveling in
461     * @return the given light parameter, after modifications
462     */
463    public static double[][] bouncingLine(double[][] resistanceMap, double[][] light, int startX, int startY, double distance, double angle)
464    {
465        double rad = Math.max(1, distance);
466        double decay = 1.0 / rad;
467        ArrayTools.fill(light, 0);
468        light[startX][startY] = 1;//make the starting space full power
469        angle = Math.toRadians((angle > 360.0 || angle < 0.0)
470                ? MathExtras.remainder(angle, 360.0) : angle);
471        float s = (float) NumberTools.sin(angle),
472                c = (float) NumberTools.cos(angle);
473        double deteriorate = 1.0;
474        int dx, dy, width = resistanceMap.length, height = resistanceMap[0].length;
475        for (int d = 1; d <= rad; ) {
476            dx = startX + Math.round(c * d);
477            if(dx < 0 || dx > width)
478                break;
479            dy = startY + Math.round(s * d);
480            if(dy < 0 || dy > height)
481                break;
482            deteriorate -= decay;
483            //check if it's within the lightable area and light if needed
484            if (deteriorate > 0.0) {
485                light[dx][dy] = Math.min(light[dx][dy] + deteriorate, 1.0);
486                if (resistanceMap[dx][dy] >= 1 && deteriorate > decay)
487                {
488                    startX = dx;
489                    startY = dy;
490                    d = 1;
491                    double flipX = resistanceMap[startX + Math.round(-c * d)][dy],
492                            flipY = resistanceMap[dx][startY + Math.round(-s * d)];
493                    if(flipX >= 1.0)
494                        s = -s;
495                    if(flipY >= 1.0)
496                        c = -c;
497                }
498                else ++d;
499
500            }
501            else break;
502        }
503
504        return light;
505    }

569        final ArrayDeque<Coord> dq = new ArrayDeque<>();
570        int width = lightMap.length;
571        int height = lightMap[0].length;
572        dq.offer(Coord.get(x, y));
573        while (!dq.isEmpty()) {
574            Coord p = dq.removeFirst();
575            if (lightMap[p.x][p.y] <= 0 || indirect[p.x][p.y]) {
576                continue;//no light to spread
577            }
578
579            for (Direction dir : Direction.OUTWARDS) {
580                int x2 = p.x + dir.deltaX;
581                int y2 = p.y + dir.deltaY;
582                if (x2 < 0 || x2 >= width || y2 < 0 || y2 >= height //out of bounds
583                        || radiusStrategy.radius(startx, starty, x2, y2) >= radius + 1) {//+1 to cover starting tile
584                    continue;
585                }
586
587                double surroundingLight = nearRippleLight(x2, y2, ripple, startx, starty, decay, lightMap, map, indirect, radiusStrategy);
588                if (lightMap[x2][y2] < surroundingLight) {
589                    lightMap[x2][y2] = surroundingLight;
590                    if (map[x2][y2] < 1) {//make sure it's not a wall
591                        dq.offer(Coord.get(x2, y2));//redo neighbors since this one's light changed
592                    }
593                }
594            }
595        }

705                                         double radius, int startx, int starty, double decay, double[][] lightMap,
706                                         double[][] map, Radius radiusStrategy, boolean binary) {
707        double newStart = 0;
708        if (start < end) {
709            return lightMap;
710        }
711        int width = lightMap.length;
712        int height = lightMap[0].length;
713
714        boolean blocked = false;
715        for (int distance = row; distance <= radius && distance < width + height && !blocked; distance++) {
716            int deltaY = -distance;
717            for (int deltaX = -distance; deltaX <= 0; deltaX++) {
718                int currentX = startx + deltaX * xx + deltaY * xy;
719                int currentY = starty + deltaX * yx + deltaY * yy;
720                double leftSlope = (deltaX - 0.5f) / (deltaY + 0.5f);
721                double rightSlope = (deltaX + 0.5f) / (deltaY - 0.5f);
722
723                if (!(currentX >= 0 && currentY >= 0 && currentX < width && currentY < height) || start < rightSlope) {
724                    continue;
725                } else if (end > leftSlope) {
726                    break;
727                }
728                double deltaRadius = radiusStrategy.radius(deltaX, deltaY);
729                //check if it's within the lightable area and light if needed
730                if (deltaRadius <= radius) {
731                    if (binary)
732                        lightMap[currentX][currentY] = 1.0;
733                    else {
734                        double bright = 1 - decay * deltaRadius;
735                        lightMap[currentX][currentY] = bright;
736                    }
737                }
738
739                if (blocked) { //previous cell was a blocking one
740                    if (map[currentX][currentY] >= 1) {//hit a wall
741                        newStart = rightSlope;
742                    } else {
743                        blocked = false;
744                        start = newStart;
745                    }
746                } else {
747                    if (map[currentX][currentY] >= 1 && distance < radius) {//hit a wall within sight line
748                        blocked = true;
749                        lightMap = shadowCast(distance + 1, start, leftSlope, xx, xy, yx, yy, radius, startx, starty, decay, lightMap, map, radiusStrategy);
750                        newStart = rightSlope;
751                    }
752                }
753            }
754        }
755        return lightMap;
756    }

810    private static final double[] directionRanges = new double[8];
811    /**
812     * Calculates the Field Of View for the provided map from the given x, y
813     * coordinates, lighting with the view "pointed at" the given {@code angle} in degrees,
814     * extending to different ranges based on the direction the light is traveling.
815     * The direction ranges are {@code forward}, {@code sideForward}, {@code side},
816     * {@code sideBack}, and {@code back}; all are multiplied by {@code radius}.
817     * Assigns to, and returns, a light map where the values represent a percentage of fully
818     * lit. The values in light are cleared before this is run, because prior state can make
819     * this give incorrect results. You can use {@link #addFOVsInto(double[][], double[][]...)}
820     * if you want to mix FOV results, which works as an alternative to using the prior light state.
821     * <br>
822     * The starting point for the calculation is considered to be at the center
823     * of the origin cell. Radius determinations are determined by the provided
824     * RadiusStrategy. If all direction ranges are the same, this acts like
825     * {@link #reuseFOV(double[][], double[][], int, int, double, Radius)}; otherwise
826     * may produce conical shapes (potentially more than one, or overlapping ones).
827     *
828     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
829     * @param light the grid of cells to assign to; may have existing values, and 0.0 is used to mean "unlit"
830     * @param startX the horizontal component of the starting location
831     * @param startY the vertical component of the starting location
832     * @param radius the distance the light will extend to (roughly); direction ranges will be multiplied by this
833     * @param radiusTechnique provides a means to shape the FOV by changing distance calculation (circle, square, etc.)
834     * @param angle the angle in degrees that will be the center of the FOV cone, 0 points right
835     * @param forward the range to extend when the light is within 22.5 degrees of angle; will be interpolated with sideForward
836     * @param sideForward the range to extend when the light is between 22.5 and 67.5 degrees of angle; will be interpolated with forward or side
837     * @param side the range to extend when the light is between 67.5 and 112.5 degrees of angle; will be interpolated with sideForward or sideBack
838     * @param sideBack the range to extend when the light is between 112.5 and 157.5 degrees of angle; will be interpolated with side or back
839     * @param back the range to extend when the light is more than 157.5 degrees away from angle; will be interpolated with sideBack
840     * @return the computed light grid (the same as {@code light})
841     */
842    public static double[][] reuseFOV(double[][] resistanceMap, double[][] light, int startX, int startY,
843                                      double radius, Radius radiusTechnique, double angle,
844                                      double forward, double sideForward, double side, double sideBack, double back) {
845        directionRanges[0] = forward * radius;
846        directionRanges[7] = directionRanges[1] = sideForward * radius;
847        directionRanges[6] = directionRanges[2] = side * radius;
848        directionRanges[5] = directionRanges[3] = sideBack * radius;
849        directionRanges[4] = back * radius;
850
851        radius = Math.max(1, radius);
852        ArrayTools.fill(light, 0);
853        light[startX][startY] = 1;//make the starting space full power
854        angle = ((angle >= 360.0 || angle < 0.0)
855                ? MathExtras.remainder(angle, 360.0) : angle) * 0.002777777777777778;
856
857
858        light = shadowCastPersonalized(1, 1.0, 0.0, 0, 1, 1, 0,   radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
859        light = shadowCastPersonalized(1, 1.0, 0.0, 1, 0, 0, 1,   radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
860        light = shadowCastPersonalized(1, 1.0, 0.0, 0, -1, 1, 0,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
861        light = shadowCastPersonalized(1, 1.0, 0.0, -1, 0, 0, 1,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
862        light = shadowCastPersonalized(1, 1.0, 0.0, 0, -1, -1, 0, radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
863        light = shadowCastPersonalized(1, 1.0, 0.0, -1, 0, 0, -1, radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
864        light = shadowCastPersonalized(1, 1.0, 0.0, 0, 1, -1, 0,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
865        light = shadowCastPersonalized(1, 1.0, 0.0, 1, 0, 0, -1,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
866        return light;
867    }

1000        Iterator<double[][]> it = maps.iterator();
1001        if(!it.hasNext())
1002            return new double[0][0];
1003        double[][] map = ArrayTools.copy(it.next()), t;
1004        while (it.hasNext())
1005        {
1006            t = it.next();
1007            for (int x = 0; x < map.length && x < t.length; x++) {
1008                for (int y = 0; y < map[x].length && y < t[x].length; y++) {
1009                    map[x][y] += t[x][y];
1010                }
1011            }
1012        }
1013        for (int x = 0; x < map.length; x++) {
1014            for (int y = 0; y < map[x].length; y++) {
1015                if(map[x][y] > 1.0) map[x][y] = 1.0;
1016            }
1017        }
1018        return map;
1019    }

1123    /**
1124     * Calculates what cells are visible from (startX,startY) using the given resistanceMap; this can be given to
1125     * mixVisibleFOVs() to limit extra light sources to those visible from the starting point. Just like calling
1126     * calculateFOV(), this creates a new double[][]; there doesn't appear to be a way to work with Ripple FOV and avoid
1127     * needing an empty double[][] every time, since it uses previously-placed light to determine how it should spread.
1128     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
1129     * @param startX the center of the LOS map; typically the player's x-position
1130     * @param startY the center of the LOS map; typically the player's y-position
1131     * @return an LOS map with the given starting point
1132     */
1133    public double[][] calculateLOSMap(double[][] resistanceMap, int startX, int startY)
1134    {
1135        if(resistanceMap == null || resistanceMap.length <= 0  || resistanceMap[0].length <= 0)
1136            return new double[0][0];
1137
1138        int width = resistanceMap.length;
1139        int height = resistanceMap[0].length;
1140        double rad = width + height;
1141        double decay = 1.0 / rad;
1142
1143        initializeLightMap(width, height);
1144        light[startX][startY] = 1;//make the starting space full power
1145
1146        switch (type) {
1147            case RIPPLE:
1148            case RIPPLE_LOOSE:
1149            case RIPPLE_TIGHT:
1150            case RIPPLE_VERY_LOOSE:
1151                initializeNearLight(width, height);
1152                doRippleFOV(light, rippleValue(type), startX, startY, startX, startY, decay, rad, resistanceMap, nearLight, Radius.SQUARE);
1153                for (int x = 0; x < width; x++) {
1154                    for (int y = 0; y < height; y++) {
1155                        if(light[x][y] > 0.0001)
1156                            light[x][y] = 1.0;
1157                    }
1158                }
1159                break;
1160            case SHADOW:
1161                for (Direction d : Direction.DIAGONALS) {
1162                    shadowCast(1, 1.0, 0.0, 0, d.deltaX, d.deltaY, 0, rad, startX, startY, decay, light, resistanceMap, Radius.SQUARE, true);
1163                    shadowCast(1, 1.0, 0.0, d.deltaX, 0, 0, d.deltaY, rad, startX, startY, decay, light, resistanceMap, Radius.SQUARE, true);
1164                }
1165                break;
1166        }
1167        return light;
1168    }
