001package squidpony.squidgrid.zone;
002
003import squidpony.squidgrid.zone.Zone.Skeleton;
004import squidpony.squidmath.Coord;
005import squidpony.squidmath.OrderedSet;
006
007import java.util.ArrayList;
008import java.util.Collection;
009import java.util.List;
010
011/**
012 * A zone defined by a {@link List}.
013 * 
014 * @author smelC
015 */
016public class ListZone extends Skeleton {
017
018        protected final List<Coord> coords;
019
020        private static final long serialVersionUID = 1166468942544595692L;
021
022        public ListZone(List<Coord> coords) {
023                this.coords = coords;
024        }
025
026        public ListZone(Collection<Coord> coordCollection)
027        {
028                this.coords = new ArrayList<>(coordCollection);
029        }
030        @Override
031        public boolean isEmpty() {
032                return coords.isEmpty();
033        }
034
035        @Override
036        public int size() {
037                return coords.size();
038        }
039
040        @Override
041        public boolean contains(Coord c) {
042                return coords.contains(c);
043        }
044
045        @Override
046        public boolean contains(int x, int y) {
047                return coords.contains(Coord.get(x, y));
048        }
049
050        @Override
051        public List<Coord> getAll() {
052                return coords;
053        }
054
055        @Override
056        public ListZone expand(int distance)
057        {
058                if(distance <= 0)
059                        return new ListZone(coords);
060                OrderedSet<Coord> distinct = new OrderedSet<>(coords);
061                for (Coord c : coords) {
062                        distinct.add(c.translate(1, 0));
063                        distinct.add(c.translate(-1, 0));
064                        distinct.add(c.translate(0, 1));
065                        distinct.add(c.translate(0, -1));
066                }
067                ArrayList<Coord> next = new ArrayList<>(distinct);
068                for (int i = 1; i < distance; i++) {
069                        distinct.clear();
070                        distinct.addAll(next);
071                        for (Coord c : next) {
072                                distinct.add(c.translate(1, 0));
073                                distinct.add(c.translate(-1, 0));
074                                distinct.add(c.translate(0, 1));
075                                distinct.add(c.translate(0, -1));
076                        }
077                        next.clear();
078                        next.addAll(distinct);
079                }
080                return new ListZone(next);
081        }
082
083        @Override
084        public String toString() {
085                return coords.toString();
086        }
087}
