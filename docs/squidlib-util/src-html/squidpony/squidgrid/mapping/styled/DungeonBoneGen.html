158    private Tile chooseTile(Tile[] list, int numlist, int[] y_positions, int[] x_positions) {
159        int a = c_color[y_positions[0]][x_positions[0]];
160        int b = c_color[y_positions[1]][x_positions[1]];
161        int c = c_color[y_positions[2]][x_positions[2]];
162        int d = c_color[y_positions[3]][x_positions[3]];
163        int e = c_color[y_positions[4]][x_positions[4]];
164        int f = c_color[y_positions[5]][x_positions[5]];
165        int i, n, match = Integer.MAX_VALUE, pass;
166        for (pass = 0; pass < 2; ++pass) {
167            n = 0;
168            // pass #1:
169            //   count number of variants that match this partial set of constraints
170            // pass #2:
171            //   stop on randomly selected match
172            for (i = 0; i < numlist; ++i) {
173                Tile tile = list[i];
174                if ((a < 0 || a == tile.a_constraint) &&
175                        (b < 0 || b == tile.b_constraint) &&
176                        (c < 0 || c == tile.c_constraint) &&
177                        (d < 0 || d == tile.d_constraint) &&
178                        (e < 0 || e == tile.e_constraint) &&
179                        (f < 0 || f == tile.f_constraint)) {
180                    n += 1;
181                    if (n > match) {
182                        // use list[i]
183                        // update constraints to reflect what we placed
184                        c_color[y_positions[0]][x_positions[0]] = tile.a_constraint;
185                        c_color[y_positions[1]][x_positions[1]] = tile.b_constraint;
186                        c_color[y_positions[2]][x_positions[2]] = tile.c_constraint;
187                        c_color[y_positions[3]][x_positions[3]] = tile.d_constraint;
188                        c_color[y_positions[4]][x_positions[4]] = tile.e_constraint;
189                        c_color[y_positions[5]][x_positions[5]] = tile.f_constraint;
190                        return tile;
191                    }
192                }
193            }
194            if (n == 0) {
195                return null;
196            }
197            match = rng.nextInt(n);
198        }
199        return null;
200    }

203        int a, b, c, d, e, f;
204        if (upright) {
205            a = h_color[y_positions[0]][x_positions[0]];

209            e = v_color[y_positions[4]][x_positions[4]];

212            a = h_color[y_positions[0]][x_positions[0]];
213            b = h_color[y_positions[1]][x_positions[1]];
214            c = v_color[y_positions[2]][x_positions[2]];
215            d = v_color[y_positions[3]][x_positions[3]];
216            e = h_color[y_positions[4]][x_positions[4]];
217            f = h_color[y_positions[5]][x_positions[5]];
218        }

224            // pass #2:
225            //   stop on randomly selected match
226            for (i = 0; i < numlist; ++i) {
227                Tile tile = list[i];
228                if ((a < 0 || a == tile.a_constraint) &&

234                    n += 1;
235                    if (n > match) {
236                        // use list[i]
237                        // update constraints to reflect what we placed
238                        if (upright) {
239                            h_color[y_positions[0]][x_positions[0]] = tile.a_constraint;

245                        } else {
246                            h_color[y_positions[0]][x_positions[0]] = tile.a_constraint;
247                            h_color[y_positions[1]][x_positions[1]] = tile.b_constraint;
248                            v_color[y_positions[2]][x_positions[2]] = tile.c_constraint;

254                    }
255                }
256            }
257            if (n == 0) {
258                return null;
259            }
260            match = rng.nextInt(n);
261        }
262        return null;
263    }

280    /**
281     * Changes the outer edge of a char[][] to the wall char, '#'.
282     *
283     * @param map A char[][] that stores map data.
284     * @return
285     */
286    public static char[][] wallWrap(char[][] map) {
287        int upperY = map[0].length - 1;
288        int upperX = map.length - 1;
289        for (int i = 0; i < map.length; i++) {
290            map[i][0] = '#';
291            map[i][upperY] = '#';
292        }
293        for (int i = 0; i < map[0].length; i++) {
294            map[0][i] = '#';
295            map[upperX][i] = '#';
296        }
297        return map;
298    }

300    /**
301     * Changes the outer edge of this dungeon to the wall char, '#'.
302     *
303     * @return The modified dungeon, a char[][].
304     */
305    public char[][] wallWrap() {
306        int upperY = high - 1;
307        int upperX = wide - 1;
308        for (int i = 0; i < wide; i++) {
309            dungeon[i][0] = '#';
310            dungeon[i][upperY] = '#';
311        }
312        for (int i = 0; i < high; i++) {
313            dungeon[0][i] = '#';
314            dungeon[upperX][i] = '#';
315        }
316        return dungeon;
317    }

319    private boolean matchingAdjacent(int y, int x)
320    {
321        return c_color[y][x] == c_color[y + 1][x + 1];
322    }
323
324    private int changeColor(int old_color, int num_options) {
325
326        int offset = 1 + rng.nextInt(num_options - 1);
327        return (old_color + offset) % num_options;
328    }

330    /**
331     * Generate a dungeon given a Tileset.
332     * If you have your own Tileset gained by parsing your own JSON, use
333     * this to generate a dungeon using it. Consider using DungeonBoneGen.wallWrap
334     * to surround the edges with walls. Assigns the returned result to a member
335     * of this class, 'dungeon'.
336     *
337     * @param ts A Tileset; if you don't have one of these available, use a TilesetType enum instead to select a predefined one.
338     * @param h  Height of the dungeon to generate in chars.
339     * @param w  Width of the dungeon to generate in chars.
340     * @return A row-major char[][] with h rows and w columns; it will be filled with '#' for walls and '.' for floors.
341     */
342    public char[][] generate(Tileset ts, int w, int h) {
343        wide = Math.max(1, w);
344        high = Math.max(1, h);
345        region.resizeAndEmpty(wide, high);
346        workingRegion.resizeAndEmpty(wide, high);
347        int sidelen = ts.config.short_side_length;
348        int xmax = (wide / sidelen) + 6;
349        int ymax = (high / sidelen) + 6;
350        if (xmax > 1006) {
351            return null;
352        }
353        if (ymax > 1006) {
354            return null;
355        }
356        if (ts.config.is_corner) {
357            c_color = new int[ymax][xmax];
358            int i = 0, j = 0, ypos = -1 * sidelen;
359            int[] cc = ts.config.num_colors;
360
361            for (j = 0; j < ymax; ++j) {
362                for (i = 0; i < xmax; ++i) {
363                    c_color[j][i] = rng.nextInt(cc[(i - j + 1) & 3]); // select from cc based on corner type
364                }
365            }
366
367            // Repetition reduction
368            // now go back through and make sure we don't have adjacent 3x2 vertices that are identical,
369            // to avoid really obvious repetition (which happens easily with extreme weights)
370            for (j = 0; j < ymax - 3; ++j) {
371                for (i = 0; i < xmax - 3; ++i) {
372                    int p = (i - j + 1) & 3; // corner type
373                    if (i + 3 >= 1006) {
374                        return null;
375                    }
376                    if (j + 3 >= 1006) {
377                        return null;
378                    }
379                    if (matchingAdjacent(j, i) && matchingAdjacent(j + 1, i) && matchingAdjacent(j + 2, i)
380                            && matchingAdjacent(j, i + 1) && matchingAdjacent(j + 1, i + 1) && matchingAdjacent(j + 2, i + 1)) {
381                        p = ((i + 1) - (j + 1) + 1) & 3;
382                        if (cc[p] > 1)
383                            c_color[j + 1][i + 1] = changeColor(c_color[j + 1][i + 1], cc[p]);
384                    }
385
386                    if (matchingAdjacent(j, i) && matchingAdjacent(j, i + 1) && matchingAdjacent(j, i + 2)
387                            && matchingAdjacent(j + 1, i) && matchingAdjacent(j + 1, i + 1) && matchingAdjacent(j + 1, i + 2)) {
388                        p = ((i + 2) - (j + 1) + 1) & 3;
389                        if (cc[p] > 1)
390                            c_color[j + 1][i + 2] = changeColor(c_color[j + 1][i + 2], cc[p]);
391                    }
392                }
393            }
394
395
396            for (j = -1; ypos < high; ++j) {
397                // a general herringbone row consists of:
398                //    horizontal left block, the bottom of a previous vertical, the top of a new vertical
399                int phase = j & 3;
400                // displace horizontally according to pattern
401                if (phase == 0) {
402                    i = 0;
403                } else {
404                    i = phase - 4;
405                }
406                for (; ; i += 4) {
407                    int xpos = i * sidelen;
408                    if (xpos >= wide)
409                        break;
410                    // horizontal left-block
411                    if (xpos + sidelen * 2 >= 0 && ypos >= 0) {
412                        Tile t = chooseTile(
413                                ts.h_tiles, ts.h_tiles.length,
414                                new int[]{j + 2, j + 2, j + 2, j + 3, j + 3, j + 3},
415                                new int[]{i + 2, i + 3, i + 4, i + 2, i + 3, i + 4});
416
417                        if (t == null)
418                            return null;
419
420                        //trans_output = insert(trans_output, t.data, ypos, xpos);
421                        region.or(workingRegion.refill(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
422                    }
423                    xpos += sidelen * 2;
424                    // now we're at the end of a previous vertical one
425                    xpos += sidelen;
426                    // now we're at the start of a new vertical one
427                    if (xpos < wide) {
428                        Tile t = chooseTile(
429                                ts.v_tiles, ts.v_tiles.length,
430                                new int[]{j + 2, j + 3, j + 4, j + 2, j + 3, j + 4},
431                                new int[]{i + 5, i + 5, i + 5, i + 6, i + 6, i + 6});
432
433                        if (t == null)
434                            return null;
435                        //trans_output = insert(trans_output, t.data, ypos, xpos);
436                        region.or(workingRegion.refill(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
437                    }
438                }
439                ypos += sidelen;
440            }
441        } else {
442            int i, j, ypos;
443            v_color = new int[ymax][xmax];
444            h_color = new int[ymax][xmax];
445            for (int yy = 0; yy < ymax; yy++) {
446                for (int xx = 0; xx < xmax; xx++) {
447                    v_color[yy][xx] = -1;
448                    h_color[yy][xx] = -1;
449                }
450            }
451
452            ypos = -1 * sidelen;
453            for (j = -1; ypos < high; ++j) {
454                // a general herringbone row consists of:
455                //    horizontal left block, the bottom of a previous vertical, the top of a new vertical
456                int phase = j & 3;
457                // displace horizontally according to pattern
458                if (phase == 0) {
459                    i = 0;
460                } else {
461                    i = phase - 4;
462                }
463                for (; ; i += 4) {
464                    int xpos = i * sidelen;
465                    if (xpos >= wide)
466                        break;
467                    // horizontal left-block
468                    if (xpos + sidelen * 2 >= 0 && ypos >= 0) {
469                        Tile t = chooseTile(
470                                ts.h_tiles, ts.h_tiles.length, false,
471                                new int[]{j + 2, j + 2, j + 2, j + 2, j + 3, j + 3},
472                                new int[]{i + 2, i + 3, i + 2, i + 4, i + 2, i + 3});
473
474                        if (t == null)
475                            return null;
476                        //trans_output = insert(trans_output, t.data, ypos, xpos);
477                        region.or(workingRegion.refill(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
478                    }
479                    xpos += sidelen * 2;
480                    // now we're at the end of a previous vertical one
481                    xpos += sidelen;
482                    // now we're at the start of a new vertical one
483                    if (xpos < wide) {
484                        Tile t = chooseTile(
485                                ts.v_tiles, ts.v_tiles.length, true,
486                                new int[]{j + 2, j + 2, j + 2, j + 3, j + 3, j + 4},
487                                new int[]{i + 5, i + 5, i + 6, i + 5, i + 6, i + 5});
488
489                        if (t == null)
490                            return null;
491                        //trans_output = insert(trans_output, t.data, ypos, xpos);
492                        region.or(workingRegion.refill(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
493                    }
494                }
495                ypos += sidelen;
496            }
497        }
498        /*for (int x = 0; x < w; x++) {
499            for (int y = 0; y < h; y++) {
500                output[x][y] = trans_output[y][x];
501            }
502        }*/
503        dungeon = region.toChars();
504        return dungeon;
505    }

507    /**
508     * Provides a string representation of the latest generated dungeon.
509     *
510     * @return
511     */
512    @Override
513        public String toString() {
514        char[][] trans = new char[high][wide];
515        for (int x = 0; x < wide; x++) {
516            for (int y = 0; y < high; y++) {
517                trans[y][x] = dungeon[x][y];
518            }
519        }
520        StringBuffer sb = new StringBuffer();
521        for (int row = 0; row < high; row++) {
522            sb.append(trans[row]);
523            sb.append('\n');
524        }
525        return sb.toString();
526    }
