001package squidpony.squidgrid.mapping;
002
003import squidpony.Maker;
004import squidpony.squidgrid.Direction;
005import squidpony.squidmath.IRNG;
006import squidpony.squidmath.RNG;
007
008import java.util.*;
009
010/**
011 * Recursively divided maze. Creates only walls and passages.
012 *
013 * This dungeon generator is based on a port of the rot.js version.
014 *
015 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
016 */
017public class DividedMazeGenerator {
018
019    private class DividedMazeRoom {
020
021        private int left, top, right, bottom;
022
023        public DividedMazeRoom(int left, int top, int right, int bottom) {
024            this.left = left;
025            this.top = top;
026            this.right = right;
027            this.bottom = bottom;
028        }
029    }
030
031    private int width, height;
032    private boolean[][] map;
033    private IRNG rng;
034
035    /**
036     * Sets up the generator to make mazes the given width and height. The mazes
037     * have a solid wall border.
038     *
039     * @param width
040     * @param height
041     */
042    public DividedMazeGenerator(int width, int height) {
043        this.width = width;
044        this.height = height;
045        rng = new RNG();
046    }
047
048    /**
049     * Sets up the generator to make mazes the given width and height. The mazes
050     * have a solid wall border.
051     *
052     * @param width in cells
053     * @param height in cells
054     * @param rng the random number generator to use
055     */
056    public DividedMazeGenerator(int width, int height, IRNG rng) {
057        this.width = width;
058        this.height = height;
059        this.rng = rng;
060    }
061
062    /**
063     * Builds a maze. True values represent walls.
064     *
065     * @return
066     */
067    public boolean[][] create() {
068        map = new boolean[width][height];
069
070        for (int x = 0; x < width; x++) {
071            for (int y = 0; y < height; y++) {
072                map[x][y] = x == 0 || y == 0 || x + 1 == width || y + 1 == height;
073            }
074        }
075
076        process();
077
078        return map;
079    }
080
081    private void process() {
082        ArrayDeque<DividedMazeRoom> stack = new ArrayDeque<>();
083        stack.offer(new DividedMazeRoom(1, 1, width - 2, height - 2));
084        while (!stack.isEmpty()) {
085            DividedMazeRoom room = stack.removeFirst();
086            ArrayList<Integer> availX = new ArrayList<>(),
087                               availY = new ArrayList<>();
088
089            for (int x = room.left + 1; x < room.right; x++) {
090                boolean top = map[x][room.top - 1];
091                boolean bottom = map[x][room.bottom + 1];
092                if (top && bottom && x % 2 == 0) {
093                    availX.add(x);
094                }
095            }
096
097            for (int y = room.top + 1; y < room.bottom; y++) {
098                boolean left = map[room.left - 1][y];
099                boolean right = map[room.right + 1][y];
100                if (left && right && y % 2 == 0) {
101                    availY.add(y);
102                }
103            }
104
105            if (availX.isEmpty() || availY.isEmpty()) {
106                continue;
107            }
108
109            int x2 = rng.getRandomElement(availX);
110            int y2 = rng.getRandomElement(availY);
111
112            map[x2][y2] = true;
113
114            for (Direction dir : Direction.CARDINALS) {
115                switch (dir) {
116                    case LEFT:
117                        for (int x = room.left; x < x2; x++) {
118                            map[x][y2] = true;
119                        }
120                        break;
121                    case RIGHT:
122                        for (int x = x2 + 1; x <= room.right; x++) {
123                            map[x][y2] = true;
124                        }
125                        break;
126                    case UP:
127                        for (int y = room.top; y < y2; y++) {
128                            map[x2][y] = true;
129                        }
130                        break;
131                    case DOWN:
132                        for (int y = y2 + 1; y <= room.bottom; y++) {
133                            map[x2][y] = true;
134                        }
135                        break;
136                    case NONE:
137                        break;
138                                case DOWN_LEFT:
139                                case DOWN_RIGHT:
140                                case UP_LEFT:
141                                case UP_RIGHT:
142                                        throw new IllegalStateException("There should only be cardinal directions here");
143                }
144            }
145
146            ArrayList<Direction> dirs = Maker.makeList(Direction.CARDINALS);
147            dirs.remove(rng.getRandomElement(dirs));
148
149            for (Direction dir : dirs) {
150                switch (dir) {
151                    case LEFT:
152                        map[rng.between(room.left, x2)][y2] = false;
153                        break;
154                    case RIGHT:
155                        map[rng.between(x2 + 1, room.right + 1)][y2] = false;
156                        break;
157                    case UP:
158                        map[x2][rng.between(room.top, y2)] = false;
159                        break;
160                    case DOWN:
161                        map[x2][rng.between(y2 + 1, room.bottom + 1)] = false;
162                        break;
163                    case NONE:
164                        break;
165                                case DOWN_LEFT:
166                                case DOWN_RIGHT:
167                                case UP_LEFT:
168                                case UP_RIGHT:
169                                        throw new IllegalStateException("There should only be cardinal directions here");
170                }
171            }
172
173            stack.offer(new DividedMazeRoom(room.left, room.top, x2 - 1, y2 - 1));
174            stack.offer(new DividedMazeRoom(x2 + 1, room.top, room.right, y2 - 1));
175            stack.offer(new DividedMazeRoom(room.left, y2 + 1, x2 - 1, room.bottom));
176            stack.offer(new DividedMazeRoom(x2 + 1, y2 + 1, room.right, room.bottom));
177        }
178    }
179
180}
