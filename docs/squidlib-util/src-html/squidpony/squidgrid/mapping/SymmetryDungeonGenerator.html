010/**
011 * A variant on {@link MixedGenerator} that creates bi-radially symmetric maps (basically a yin-yang shape). Useful for

019    public static OrderedMap<Coord, List<Coord>> removeSomeOverlap(int width, int height, List<Coord> sequence)
020    {
021        List<Coord> s2 = new ArrayList<>(sequence.size());
022        for(Coord c : sequence)
023        {
024            if(c.x * 1.0 / width + c.y * 1.0 / height <= 1.0)
025                s2.add(c);
026        }
027        return listToMap(s2);
028    }
029    public static OrderedMap<Coord, List<Coord>> removeSomeOverlap(int width, int height, Map<Coord, List<Coord>> connections) {
030        OrderedMap<Coord, List<Coord>> om2 = new OrderedMap<>(connections.size());
031        Set<Coord> keyset = connections.keySet(), newkeys = new OrderedSet<>(connections.size());
032        for (Coord c : keyset) {
033            if (c.x * 1.0 / width + c.y * 1.0 / height <= 1.0) {
034                newkeys.add(c);
035            }
036        }
037        Coord[] keys = newkeys.toArray(new Coord[newkeys.size()]);
038        for (int i = 0; i < keys.length; i++) {
039            Coord c = keys[i];
040            if (c.x * 1.0 / width + c.y * 1.0 / height <= 1.0) {
041                List<Coord> cs = new ArrayList<>(4);
042                for (Coord c2 : connections.get(c)) {
043                    if (c2.x * 1.0 / width + c2.y * 1.0 / height <= 1.0) {
044                        cs.add(c2);
045                    } else if (keys[(i + 1) % keys.length].x * 1.0 / width +
046                            keys[(i + 1) % keys.length].y * 1.0 / height <= 1.0) {
047                        cs.add(keys[(i + 1) % keys.length]);
048                    }
049
050                }
051                om2.put(c, cs);
052            }
053        }
054        return om2;
055    }

058     * This version of the constructor uses Poisson Disk sampling to generate the points it will draw caves and
059     * corridors between, ensuring a minimum distance between points, but it does not ensure that paths between points
060     * will avoid overlapping with rooms or other paths. You call the different carver-adding methods to affect what the
061     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
062     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.

064     * @param width  the width of the final map in cells
065     * @param height the height of the final map in cells
066     * @param rng    an RNG object to use for random choices; this make a lot of random choices.
067     * @see PoissonDisk used to ensure spacing for the points.
068     */

109     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
110     * This version of the constructor uses a LinkedHashMap with Coord keys and Coord array values to determine a
111     * branching path for the dungeon to take; each key will connect once to each of the Coords in its value, and you
112     * usually don't want to connect in both directions. You call the different carver-adding methods to affect what the
113     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
114     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
115     *
116     * @param width              the width of the final map in cells
117     * @param height             the height of the final map in cells
118     * @param rng                an RNG object to use for random choices; this make a lot of random choices.
119     * @param connections        a Map of Coord keys to arrays of Coord to connect to next; shouldn't connect both ways
120     * @param roomSizeMultiplier a float multiplier that will be applied to each room's width and height
121     * @see SerpentMapGenerator a class that uses this technique
122     */
123    public SymmetryDungeonGenerator(int width, int height, RNG rng, OrderedMap<Coord, List<Coord>> connections, float roomSizeMultiplier) {
124        super(width, height, rng, crossConnect(width, height, connections), roomSizeMultiplier);
125    }

127    protected static OrderedMap<Coord, List<Coord>> listToMap(List<Coord> sequence)
128    {
129        OrderedMap<Coord, List<Coord>> conns = new OrderedMap<>(sequence.size() - 1);
130        for (int i = 0; i < sequence.size() - 1; i++) {
131            Coord c1 = sequence.get(i), c2 = sequence.get(i+1);
132            List<Coord> cs = new ArrayList<>(1);
133            cs.add(c2);
134            conns.put(c1, cs);
135        }
136        return conns;
137    }
138
139    protected static OrderedMap<Coord, List<Coord>> crossConnect(int width, int height, Map<Coord, List<Coord>> connections)
140    {
141        OrderedMap<Coord, List<Coord>> conns = new OrderedMap<>(connections.size());
142        for(Map.Entry<Coord, List<Coord>> entry : connections.entrySet())
143        {
144            conns.put(entry.getKey(), new ArrayList<>(entry.getValue()));
145        }
146        double lowest1 = 9999, lowest2 = 9999, lowest3 = 9999, lowest4 = 9999;
147        Coord l1 = null, l2 = null, l3 = null, l4 = null, r1 = null, r2 = null, r3 = null, r4 = null;
148        for(List<Coord> left : connections.values())
149        {
150            for(List<Coord> right : connections.values())
151            {
152                for(Coord lc : left)
153                {
154                    for(Coord rc : right)
155                    {
156                        Coord rc2 = Coord.get(width - 1 - rc.x, height - 1 - rc.y);
157                        double dist = lc.distance(rc2);
158                        if(dist < 0.001)
159                            continue;
160                        if(dist < lowest1)
161                        {
162                            lowest1 = dist;
163                            l1 = lc;
164                            r1 = rc2;
165                        }
166                        else if(dist < lowest2 && !lc.equals(l1) && !rc2.equals(r1))
167                        {
168                            lowest2 = dist;
169                            l2 = lc;
170                            r2 = rc2;
171                        }
172                        else if(dist < lowest3
173                                && !lc.equals(l1) && !rc2.equals(r1) && !lc.equals(l2) && !rc2.equals(r2))
174                        {
175                            lowest3 = dist;
176                            l3 = lc;
177                            r3 = rc2;
178                        }
179                        else if(dist < lowest4
180                                && !lc.equals(l1) && !rc2.equals(r1)
181                                && !lc.equals(l2) && !rc2.equals(r2)
182                                && !lc.equals(l3) && !rc2.equals(r3))
183                        {
184                            lowest4 = dist;
185                            l4 = lc;
186                            r4 = rc2;
187                        }
188                    }
189                }
190            }
191        }
192        if(l1 != null && r1 != null)
193        {
194            if(conns.containsKey(l1))
195            {
196                conns.get(l1).add(r1);
197            }
198            else if(conns.containsKey(r1))
199            {
200                conns.get(r1).add(l1);
201            }
202        }
203        if(l2 != null && r2 != null)
204        {
205            if(conns.containsKey(l2))
206            {
207                conns.get(l2).add(r2);
208            }
209            else if(conns.containsKey(r2))
210            {
211                conns.get(r2).add(l2);
212            }
213        }
214        if(l3 != null && r3 != null)
215        {
216            if(conns.containsKey(l3))
217            {
218                conns.get(l3).add(r3);
219            }
220            else if(conns.containsKey(r3))
221            {
222                conns.get(r3).add(l3);
223            }
224        }
225        if(l4 != null && r4 != null)
226        {
227            if(conns.containsKey(l4))
228            {
229                conns.get(l4).add(r4);
230            }
231            else if(conns.containsKey(r4))
232            {
233                conns.get(r4).add(l4);
234            }
235        }
236        return conns;
237    }
