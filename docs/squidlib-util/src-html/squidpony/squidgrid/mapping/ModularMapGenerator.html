009/**
010 * Generator for maps of high-tech areas like space stations or starships, with repeated modules laid out in random ways.
011 * Different from traditional fantasy dungeon generation in that it should seem generally less chaotic in how it's laid
012 * out, and repeated elements with minor tweaks should be especially common. May also be useful in fantasy games for
013 * regimented areas built by well-organized military forces.
014 * <br>
015 * Preview: https://gist.github.com/tommyettinger/c711f8fc83fa9919245d88092444bf7f
016 * Created by Tommy Ettinger on 4/2/2016.
017 */

033    private void putModule(short[] module) {
034        char[][] unp = CoordPacker.unpackChar(module, '.', '#');
035        MapModule mm = new MapModule(ArrayTools.insert(unp,
036                ArrayTools.fill('#', unp.length + 2, unp[0].length + 2), 1, 1));
037        //short[] b = CoordPacker.rectangle(1 + mm.max.x, 1 + mm.max.y);
038        //modules.put(b, mm);
039        //inverseModules.put(CoordPacker.negatePacked(b), mm);
040        ArrayList<MapModule> mms = modules.get(mm.category);
041        if (mms == null) {
042            mms = new ArrayList<>(16);
043            mms.add(mm);
044            modules.put(mm.category, mms);
045        } else
046            mms.add(mm);
047    }
048
049    private void putRectangle(int width, int height, float multiplier) {
050        putModule(CoordPacker.rectangle(Math.round(width * multiplier), Math.round(height * multiplier)));
051    }
052
053    private void putCircle(int radius, float multiplier) {
054        putModule(CoordPacker.circle(Coord.get(Math.round(radius * multiplier + 1), Math.round(radius * multiplier + 1)),
055                Math.round(radius * multiplier),
056                Math.round((radius + 1) * 2 * multiplier + 1), Math.round((radius + 1) * 2 * multiplier + 1)));
057    }

059    private void initModules() {
060        layout = new RegionMap<>(64);
061        //modules = new RegionMap<>(64);
062        //inverseModules = new RegionMap<>(64);
063        modules = new OrderedMap<>(64);
064        for (int i = 1; i <= 64; i <<= 1) {
065            ArrayList<MapModule> mms = new ArrayList<>(16);
066            modules.put(i, mms);
067        }
068        displacement = new OrderedMap<>(64);
069        float multiplier = 1;//(float) Math.sqrt(Math.max(1f, Math.min(width, height) / 24f));
070        putRectangle(2, 2, multiplier);
071        putRectangle(3, 3, multiplier);
072        putRectangle(4, 4, multiplier);
073        putRectangle(4, 2, multiplier);
074        putRectangle(2, 4, multiplier);
075        putRectangle(6, 6, multiplier);
076        putRectangle(6, 3, multiplier);
077        putRectangle(3, 6, multiplier);
078        putCircle(2, multiplier);
079
080        putRectangle(8, 8, multiplier);
081        putRectangle(6, 12, multiplier);
082        putRectangle(12, 6, multiplier);
083        putCircle(4, multiplier);
084
085        putRectangle(14, 14, multiplier);
086        putRectangle(9, 18, multiplier);
087        putRectangle(18, 9, multiplier);
088        putRectangle(14, 18, multiplier);
089        putRectangle(18, 14, multiplier);
090        putCircle(7, multiplier);
091    }

100    /**
101     * Make a ModularMapGenerator with the given height and width; the RNG used for generating a dungeon and
102     * adding features will be a StatefulRNG (backed by LightRNG) using a random seed.
103     *
104     * @param width  The width of the dungeon in cells
105     * @param height The height of the dungeon in cells
106     */
107    public ModularMapGenerator(int width, int height) {
108        this(width, height, new StatefulRNG());
109    }

111    /**
112     * Make a ModularMapGenerator with the given height, width, and RNG. Use this if you want to seed the RNG.
113     *
114     * @param width  The width of the dungeon in cells
115     * @param height The height of the dungeon in cells
116     * @param rng    The RNG to use for all purposes in this class; if it is a StatefulRNG, then it will be used as-is,
117     *               but if it is not a StatefulRNG, a new StatefulRNG will be used, randomly seeded by this parameter
118     */
119    public ModularMapGenerator(int width, int height, RNG rng) {
120        CoordPacker.init();
121        this.rng = (rng instanceof StatefulRNG) ? (StatefulRNG) rng : new StatefulRNG(rng.nextLong());
122        utility = new DungeonUtility(this.rng);
123        rebuildSeed = this.rng.getState();
124        this.height = height;
125        this.width = width;
126        map = new char[width][height];
127        environment = new int[width][height];
128        for (int x = 0; x < this.width; x++) {
129            Arrays.fill(map[x], '#');
130        }
131        initModules();
132    }

181    public char[][] generate() {
182        MapModule mm, mm2;
183        int xPos, yPos, categorySize = 32, alteredSize = (categorySize * 3) >>> 1, bits = 5, ctr;
184        Coord[][] grid = new Coord[1][1];
185        // find biggest category and drop down as many modules as we can fit
186        for (; categorySize >= 4; categorySize >>= 1, alteredSize = (categorySize * 3) >>> 1, bits--) {
187            if (width / alteredSize <= 0 || height / alteredSize <= 0)
188                continue;
189            grid = new Coord[width / alteredSize][height / alteredSize];
190            ctr = 0;
191            for (int xLimit = alteredSize - 1, x = 0; xLimit < width; xLimit += alteredSize, x += alteredSize) {
192                for (int yLimit = alteredSize - 1, y = 0; yLimit < height; yLimit += alteredSize, y += alteredSize) {
193                    if (layout.allAt(x + alteredSize / 2, y + alteredSize / 2).isEmpty()) // && (bits <= 3 || rng.nextInt(5) < bits)
194                    {
195                        if (rng.between(2, grid.length * grid[0].length + 3) == ctr++)
196                            continue;
197                        mm = rng.getRandomElement(modules.get(categorySize));
198                        if (mm == null) break;
199                        mm = mm.rotate(rng.nextInt(4));
200                        xPos = rng.nextInt(3) << (bits - 2);
201                        yPos = rng.nextInt(3) << (bits - 2);
202                        for (int px = 0; px <= mm.max.x; px++) {
203                            System.arraycopy(mm.map[px], 0, map[px + x + xPos], y + yPos, mm.max.y + 1);
204                            System.arraycopy(mm.environment[px], 0, environment[px + x + xPos], y + yPos, mm.max.y + 1);
205                        }
206                        layout.put(CoordPacker.rectangle(x + xPos, y + yPos, categorySize, categorySize), mm);
207                        displacement.put(Coord.get(x + xPos, y + yPos), mm);
208                        grid[x / alteredSize][y / alteredSize] = Coord.get(x + xPos, y + yPos);
209                    }
210                }
211            }
212            if (!layout.isEmpty())
213                break;
214        }
215        Coord a, b;
216        int gw = grid.length;
217        if (gw > 0) {
218            int gh = grid[0].length;
219            for (int w = 0; w < gw; w++) {
220                for (int h = 0; h < gh; h++) {
221                    a = grid[w][h];
222                    if (a == null)
223                        continue;
224                    int connectors = rng.nextInt(16) | rng.nextInt(16);
225                    if ((connectors & 1) == 1 && w > 0 && grid[w - 1][h] != null) {
226                        b = grid[w - 1][h];
227                        connectLeftRight(displacement.get(b), b.x, b.y, displacement.get(a), a.x, a.y);
228                    }
229                    if ((connectors & 2) == 2 && w < gw - 1 && grid[w + 1][h] != null) {
230                        b = grid[w + 1][h];
231                        connectLeftRight(displacement.get(a), a.x, a.y, displacement.get(b), b.x, b.y);
232                    }
233                    if ((connectors & 4) == 4 && h > 0 && grid[w][h - 1] != null) {
234                        b = grid[w][h - 1];
235                        connectTopBottom(displacement.get(b), b.x, b.y, displacement.get(a), a.x, a.y);
236                    }
237                    if ((connectors & 8) == 8 && h < gh - 1 && grid[w][h + 1] != null) {
238                        b = grid[w][h + 1];
239                        connectTopBottom(displacement.get(a), a.x, a.y, displacement.get(b), b.x, b.y);
240                    }
241                }
242            }
243        }
244        Coord begin;
245        short[] packed;
246        for (Map.Entry<Coord, MapModule> dmm : displacement.entrySet()) {
247            begin = dmm.getKey();
248            mm = dmm.getValue();
249            //int newCat = mm.category;
250            //if(newCat >= 16) newCat >>>= 1;
251            //if(newCat >= 8) newCat >>>= 1;
252            //mm2 = rng.getRandomElement(modules.get(newCat));
253            int shiftsX = begin.x - (mm.category * 3 / 2) * ((begin.x * 2) / (3 * mm.category)),
254                    shiftsY = begin.y - (mm.category * 3 / 2) * ((begin.y * 2) / (3 * mm.category)),
255                    leftSize = Integer.highestOneBit(shiftsX),
256                    rightSize = Integer.highestOneBit((mm.category >>> 1) - shiftsX),
257                    topSize = Integer.highestOneBit(shiftsY),
258                    bottomSize = Integer.highestOneBit((mm.category >>> 1) - shiftsY);
259            if (leftSize >= 4 && !mm.leftDoors.isEmpty()) {
260                mm2 = rng.getRandomElement(modules.get(leftSize));
261                if (mm2 == null) continue;
262                if (mm2.rightDoors.isEmpty()) {
263                    if (!mm2.topDoors.isEmpty())
264                        mm2 = mm2.rotate(1);
265                    else if (!mm2.leftDoors.isEmpty())
266                        mm2 = mm2.flip(true, false);
267                    else if (!mm2.bottomDoors.isEmpty())
268                        mm2 = mm2.rotate(3);
269                    else continue;
270                }
271                for (int i = 0; i < 4; i++) {
272                    packed = CoordPacker.rectangle(begin.x - shiftsX, begin.y + i * mm.category / 4, leftSize, leftSize);
273                    if (layout.containsRegion(packed))
274                        continue;
275                    for (int px = 0; px <= mm2.max.x; px++) {
276                        System.arraycopy(mm2.map[px], 0, map[px + begin.x - shiftsX], begin.y + i * mm.category / 4, mm2.max.y + 1);
277                        System.arraycopy(mm2.environment[px], 0, environment[px + begin.x - shiftsX], begin.y + i * mm.category / 4, mm2.max.y + 1);
278                    }
279                    layout.put(packed, mm2);
280                    connectLeftRight(mm2, begin.x - shiftsX, begin.y + i * mm.category / 4, mm, begin.x, begin.y);
281                }
282            }
283            if (rightSize >= 4 && !mm.rightDoors.isEmpty()) {
284                mm2 = rng.getRandomElement(modules.get(rightSize));
285                if (mm2 == null) continue;
286                if (mm2.leftDoors.isEmpty()) {
287                    if (!mm2.bottomDoors.isEmpty())
288                        mm2 = mm2.rotate(1);
289                    else if (!mm2.rightDoors.isEmpty())
290                        mm2 = mm2.flip(true, false);
291                    else if (!mm2.topDoors.isEmpty())
292                        mm2 = mm2.rotate(3);
293                    else continue;
294                }
295                for (int i = 0; i < 4; i++) {
296                    packed = CoordPacker.rectangle(begin.x + mm.category, begin.y + i * mm.category / 4, rightSize, rightSize);
297                    if (layout.containsRegion(packed))
298                        continue;
299                    for (int px = 0; px <= mm2.max.x; px++) {
300                        System.arraycopy(mm2.map[px], 0, map[px + begin.x + mm.category], begin.y + i * mm.category / 4, mm2.max.y + 1);
301                        System.arraycopy(mm2.environment[px], 0, environment[px + begin.x + mm.category], begin.y + i * mm.category / 4, mm2.max.y + 1);
302                    }
303                    layout.put(packed, mm2);
304                    connectLeftRight(mm, begin.x, begin.y, mm2, begin.x + mm.category, begin.y + i * mm.category / 4);
305                }
306            }
307            if (topSize >= 4 && !mm.topDoors.isEmpty()) {
308                mm2 = rng.getRandomElement(modules.get(topSize));
309                if (mm2 == null) continue;
310                if (mm2.bottomDoors.isEmpty()) {
311                    if (!mm2.leftDoors.isEmpty())
312                        mm2 = mm2.rotate(3);
313                    else if (!mm2.topDoors.isEmpty())
314                        mm2 = mm2.flip(false, true);
315                    else if (!mm2.rightDoors.isEmpty())
316                        mm2 = mm2.rotate(1);
317                    else continue;
318                }
319                for (int i = 0; i < 4; i++) {
320                    packed = CoordPacker.rectangle(begin.x + i * mm.category / 4, begin.y - shiftsY, topSize, topSize);
321                    if (layout.containsRegion(packed))
322                        continue;
323                    for (int px = 0; px <= mm2.max.x; px++) {
324                        System.arraycopy(mm2.map[px], 0, map[px + begin.x + i * mm.category / 4], begin.y - shiftsY, mm2.max.y + 1);
325                        System.arraycopy(mm2.environment[px], 0, environment[px + begin.x + i * mm.category / 4], begin.y - shiftsY, mm2.max.y + 1);
326                    }
327                    layout.put(packed, mm2);
328                    connectTopBottom(mm2, begin.x + i * mm.category / 4, begin.y - shiftsY, mm, begin.x, begin.y);
329                }
330            }
331            if (bottomSize >= 4 && !mm.bottomDoors.isEmpty()) {
332                mm2 = rng.getRandomElement(modules.get(bottomSize));
333                if (mm2 == null) continue;
334                if (mm2.topDoors.isEmpty()) {
335                    if (!mm2.rightDoors.isEmpty())
336                        mm2 = mm2.rotate(1);
337                    else if (!mm2.topDoors.isEmpty())
338                        mm2 = mm2.flip(false, true);
339                    else if (!mm2.leftDoors.isEmpty())
340                        mm2 = mm2.rotate(3);
341                    else continue;
342                }
343                for (int i = 0; i < 4; i++) {
344                    packed = CoordPacker.rectangle(begin.x + i * mm.category / 4, begin.y + mm.category, bottomSize, bottomSize);
345                    if (layout.containsRegion(packed))
346                        continue;
347                    for (int px = 0; px <= mm2.max.x; px++) {
348                        System.arraycopy(mm2.map[px], 0, map[px + begin.x + i * mm.category / 4], begin.y + mm.category, mm2.max.y + 1);
349                        System.arraycopy(mm2.environment[px], 0, environment[px + begin.x + i * mm.category / 4], begin.y + mm.category, mm2.max.y + 1);
350                    }
351                    layout.put(packed, mm2);
352                    connectTopBottom(mm, begin.x, begin.y, mm2, begin.x + i * mm.category / 4, begin.y + mm.category);
353                }
354            }
355        }
356        return map;
357    }

412    private void connectLeftRight(MapModule left, int leftX, int leftY, MapModule right, int rightX, int rightY) {
413        if (left.rightDoors == null || left.rightDoors.isEmpty()
414                || right.leftDoors == null || right.leftDoors.isEmpty())
415            return;
416        List<Coord> line = new ArrayList<>(1), temp;
417        int best = 1024;
418        Coord tl, tr, twl, twr, wl = null, wr = null;
419        for (Coord l : left.rightDoors) {
420            tl = twl = l.translate(leftX, leftY);
421            if (tl.x > 0 && tl.x < width - 1 && map[tl.x - 1][tl.y] != '#')
422                tl = Coord.get(tl.x + 1, tl.y);
423            else if (tl.x > 0 && tl.x < width - 1 && map[tl.x + 1][tl.y] != '#')
424                tl = Coord.get(tl.x - 1, tl.y);
425            else if (tl.y > 0 && tl.y < height - 1 && map[tl.x][tl.y - 1] != '#')
426                tl = Coord.get(tl.x, tl.y + 1);
427            else if (tl.y > 0 && tl.y < height - 1 && map[tl.x][tl.y + 1] != '#')
428                tl = Coord.get(tl.x, tl.y - 1);
429            else
430                continue;
431
432            for (Coord r : right.leftDoors) {
433                tr = twr = r.translate(rightX, rightY);
434                if (tr.x > 0 && tr.x < width - 1 && map[tr.x - 1][tr.y] != '#')
435                    tr = Coord.get(tr.x + 1, tr.y);
436                else if (tr.x > 0 && tr.x < width - 1 && map[tr.x + 1][tr.y] != '#')
437                    tr = Coord.get(tr.x - 1, tr.y);
438                else if (tr.y > 0 && tr.y < height - 1 && map[tr.x][tr.y - 1] != '#')
439                    tr = Coord.get(tr.x, tr.y + 1);
440                else if (tr.y > 0 && tr.y < height - 1 && map[tr.x][tr.y + 1] != '#')
441                    tr = Coord.get(tr.x, tr.y - 1);
442                else
443                    continue;
444                temp = OrthoLine.line(tl, tr);
445                if (temp.size() < best) {
446                    line = temp;
447                    best = line.size();
448                    wl = twl;
449                    wr = twr;
450                }
451            }
452        }
453        temp = new ArrayList<>(line.size());
454        for (Coord c : line) {
455            if (map[c.x][c.y] == '#') {
456                map[c.x][c.y] = '.';
457                environment[c.x][c.y] = DungeonUtility.CORRIDOR_FLOOR;
458                temp.add(c);
459            }
460        }
461        if (wl != null && map[wl.x][wl.y] == '#') {
462            //if(line.isEmpty())
463            map[wl.x][wl.y] = '.';
464            environment[wl.x][wl.y] = DungeonUtility.ROOM_FLOOR;
465            //else
466            //    map[wl.x][wl.y] = '+';
467        }
468        if (wr != null && map[wr.x][wr.y] == '#') {
469            //if(line.isEmpty())
470            map[wr.x][wr.y] = '.';
471            environment[wr.x][wr.y] = DungeonUtility.ROOM_FLOOR;
472            //else
473            //    map[wr.x][wr.y] = '+';
474        }
475        layout.put(CoordPacker.packSeveral(temp), null);
476
477    }

479    private void connectTopBottom(MapModule top, int topX, int topY, MapModule bottom, int bottomX, int bottomY) {
480        if (top.bottomDoors == null || top.bottomDoors.isEmpty()
481                || bottom.topDoors == null || bottom.topDoors.isEmpty())
482            return;
483        List<Coord> line = new ArrayList<>(1), temp;
484        int best = 1024;
485        Coord tt, tb, twt, twb, wt = null, wb = null;
486        for (Coord l : top.bottomDoors) {
487            tt = twt = l.translate(topX, topY);
488            if (tt.y > 0 && tt.y < height - 1 && map[tt.x][tt.y - 1] != '#')
489                tt = Coord.get(tt.x, tt.y + 1);
490            else if (tt.y > 0 && tt.y < height - 1 && map[tt.x][tt.y + 1] != '#')
491                tt = Coord.get(tt.x, tt.y - 1);
492            else if (tt.x > 0 && tt.x < width - 1 && map[tt.x - 1][tt.y] != '#')
493                tt = Coord.get(tt.x + 1, tt.y);
494            else if (tt.x > 0 && tt.x < width - 1 && map[tt.x + 1][tt.y] != '#')
495                tt = Coord.get(tt.x - 1, tt.y);
496            else
497                continue;
498
499            for (Coord r : bottom.topDoors) {
500                tb = twb = r.translate(bottomX, bottomY);
501                if (tb.y > 0 && tb.y < height - 1 && map[tb.x][tb.y - 1] != '#')
502                    tb = Coord.get(tb.x, tb.y + 1);
503                else if (tb.y > 0 && tb.y < height - 1 && map[tb.x][tb.y + 1] != '#')
504                    tb = Coord.get(tb.x, tb.y - 1);
505                else if (tb.x > 0 && tb.x < width - 1 && map[tb.x - 1][tb.y] != '#')
506                    tb = Coord.get(tb.x + 1, tb.y);
507                else if (tb.x > 0 && tb.x < width - 1 && map[tb.x + 1][tb.y] != '#')
508                    tb = Coord.get(tb.x - 1, tb.y);
509                else
510                    continue;
511                temp = OrthoLine.line(tt, tb);
512                if (temp.size() < best) {
513                    line = temp;
514                    best = line.size();
515                    wt = twt;
516                    wb = twb;
517                }
518            }
519        }
520        temp = new ArrayList<>(line.size());
521        for (Coord c : line) {
522            if (map[c.x][c.y] == '#') {
523                map[c.x][c.y] = '.';
524                environment[c.x][c.y] = DungeonUtility.CORRIDOR_FLOOR;
525                temp.add(c);
526            }
527        }
528        if (wt != null && map[wt.x][wt.y] == '#') {
