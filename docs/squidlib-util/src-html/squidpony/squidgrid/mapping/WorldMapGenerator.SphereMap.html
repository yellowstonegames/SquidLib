012/**
013 * Can be used to generate world maps with a wide variety of data, starting with height, temperature and moisture.
014 * From there, you can determine biome information in as much detail as your game needs, with a default implementation
015 * available that assigns a single biome to each cell based on heat/moisture. The maps this produces are valid
016 * for spherical or toroidal world projections, and will wrap from edge to opposite edge seamlessly thanks to a
017 * technique from the Accidental Noise Library ( https://www.gamedev.net/blog/33/entry-2138456-seamless-noise/ ) that
018 * involves getting a 2D slice of 4D Simplex noise. Because of how Simplex noise works, this also allows extremely high
019 * zoom levels as long as certain parameters are within reason. You can access the height map with the
020 * {@link #heightData} field, the heat map with the {@link #heatData} field, the moisture map with the
021 * {@link #moistureData} field, and a special map that stores ints representing the codes for various ranges of
022 * elevation (0 to 8 inclusive, with 0 the deepest ocean and 8 the highest mountains) with {@link #heightCodeData}. The
023 * last map should be noted as being the simplest way to find what is land and what is water; any height code 4 or
024 * greater is land, and any height code 3 or less is water. This can produce rivers, and keeps that information in a
025 * GreasedRegion (alongside a GreasedRegion containing lake positions) instead of in the other map data. This class does
026 * not use Coord at all, but if you want maps with width and/or height greater than 256, and you want to use the river
027 * or lake data as a Collection of Coord, then you should call {@link Coord#expandPoolTo(int, int)}
028 * with your width and height so the Coords remain safely pooled. If you're fine with keeping rivers and lakes as
029 * GreasedRegions and not requesting Coord values from them, then you don't need to do anything with Coord. Certain
030 * parts of this class are not necessary to generate. For example, you can enable river generation if you want by
031 * setting {@link #generateRivers} to true, but it doesn't produce very good results right now.
032 * <br>
033 * The main trade-off this makes to obtain better quality is reduced speed; generating a 512x512 map on a circa-2016
034 * laptop (i7-6700HQ processor at 2.6 GHz) takes about 1 second (about 1.15 seconds for an un-zoomed map, 0.95 or so
035 * seconds to increase zoom at double resolution). If you don't need a 512x512 map, this takes commensurately less time
036 * to generate less grid cells, with 64x64 maps generating faster than they can be accurately seen on the same hardware.
037 */

089     * Arc sine approximation with fairly low error while still being faster than {@link NumberTools#sin(double)}.
090     * This formula is number 201 in <a href=">http://www.fastcode.dk/fastcodeproject/articles/index.htm">Dennis
091     * Kjaer Christensen's unfinished math work on arc sine approximation</a>. This method is about 40 times faster
092     * than {@link Math#asin(double)}.
093     * @param a an input to the inverse sine function, from -1 to 1 inclusive (error is higher approaching -1 or 1)
094     * @return an output from the inverse sine function, from -PI/2 to PI/2 inclusive.
095     */
096    protected static double asin(double a) {
097        return (a * (1.0 + (a *= a) * (-0.141514171442891431 + a * -0.719110791477959357))) /
098                (1.0 + a * (-0.439110389941411144 + a * -0.471306172023844527));
099    }

221                usedWidth = (width >> zoom), usedHeight = (height >> zoom), landMod, coolMod, state);

234    /**
235     * Halves the resolution of the map and doubles the area it covers repeatedly, halving {@code zoomAmount} times; the
236     * 2D arrays this uses keep their sizes. This version of zoomOut allows you to specify where the zoom should be
237     * centered, using the current coordinates (if the map size is 256x256, then coordinates should be between 0 and
238     * 255, and will refer to the currently used area and not necessarily the full world size).
239     * <br>
240     * Only has an effect if you have previously zoomed in using {@link #zoomIn(int, int, int)} or its overload.
241     * @param zoomCenterX the center X position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
242     * @param zoomCenterY the center Y position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
243     */
244    public void zoomOut(int zoomAmount, int zoomCenterX, int zoomCenterY)
245    {
246        zoomAmount = Math.min(zoom, zoomAmount);
247        if(zoomAmount == 0) return;
248        if(zoomAmount < 0) {
249            zoomIn(-zoomAmount, zoomCenterX, zoomCenterY);
250            return;
251        }
252        if(zoom > 0)
253        {
254            if(seed != cachedState)
255            {
256                generate(rng.nextLong());
257            }
258            zoomStartX = Math.min(Math.max(
259                    (zoomStartX + (zoomCenterX - (width >> 1))) >> zoomAmount,
260                    width >> 1), (width << zoom - zoomAmount) - (width >> 1));
261            zoomStartY = Math.min(Math.max(
262                    (zoomStartY + (zoomCenterY - (height >> 1))) >> zoomAmount,
263                    height >> 1), (height << zoom - zoomAmount) - (height >> 1));
264//            System.out.printf("zoomOut, zoomStartX: %d, zoomStartY: %d\n", zoomStartX, zoomStartY);
265            zoom -= zoomAmount;
266            startCacheX.pop();
267            startCacheY.pop();
268            startCacheX.add(Math.min(Math.max(startCacheX.pop() + (zoomCenterX >> zoom + 1) - (width >> zoom + 2),
269                    0), width - (width >> zoom)));
270            startCacheY.add(Math.min(Math.max(startCacheY.pop() + (zoomCenterY >> zoom + 1) - (height >> zoom + 2),
271                    0), height - (height >> zoom)));
272//            zoomStartX = Math.min(Math.max((zoomStartX >> 1) + (zoomCenterX >> zoom + 1) - (width >> zoom + 2),
273//                    0), width - (width >> zoom));
274//            zoomStartY = Math.min(Math.max((zoomStartY >> 1) + (zoomCenterY >> zoom + 1) - (height >> zoom + 2),
275//                    0), height - (height >> zoom));
276            regenerate(startX = (zoomStartX >> zoom) - (width >> zoom + 1), startY = (zoomStartY >> zoom) - (height >> zoom + 1),
277                    //startCacheX.peek(), startCacheY.peek(),
278                    usedWidth = width >> zoom,  usedHeight = height >> zoom,
279                    landModifier, coolingModifier, cachedState);
280            rng.setState(cachedState);
281        }

297     * Doubles the resolution of the map and halves the area it covers repeatedly, doubling {@code zoomAmount} times;
298     * the 2D arrays this uses keep their sizes. This version of zoomIn allows you to specify where the zoom should be
299     * centered, using the current coordinates (if the map size is 256x256, then coordinates should be between 0 and
300     * 255, and will refer to the currently used area and not necessarily the full world size).
301     * <br>
302     * Although there is no technical restriction on maximum zoom, zooming in more than 5 times (64x scale or greater)
303     * will make the map appear somewhat less realistic due to rounded shapes appearing more bubble-like and less like a
304     * normal landscape.
305     * @param zoomCenterX the center X position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
306     * @param zoomCenterY the center Y position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
307     */
308    public void zoomIn(int zoomAmount, int zoomCenterX, int zoomCenterY)
309    {
310        if(zoomAmount == 0) return;
311        if(zoomAmount < 0)
312        {
313            zoomOut(-zoomAmount, zoomCenterX, zoomCenterY);
314            return;
315        }
316        if(seed != cachedState)
317        {
318            generate(rng.nextLong());
319        }
320        zoomStartX = Math.min(Math.max(
321                (zoomStartX + zoomCenterX - (width >> 1) << zoomAmount),
322                width >> 1), (width << zoom + zoomAmount) - (width >> 1));
323//        int oldZoomY = zoomStartY;
324        zoomStartY = Math.min(Math.max(
325                (zoomStartY + zoomCenterY - (height >> 1) << zoomAmount),
326                height >> 1), (height << zoom + zoomAmount) - (height >> 1));
327//        System.out.printf("zoomIn, zoomStartX: %d, zoomStartY: %d, oldZoomY: %d, unedited: %d, upperCap: %d\n", zoomStartX, zoomStartY,
328//                oldZoomY, (oldZoomY + zoomCenterY - (height >> 1) << zoomAmount), (height << zoom + zoomAmount) - (height >> 1));
329        zoom += zoomAmount;
330        if(startCacheX.isEmpty())
331        {
332            startCacheX.add(0);
333            startCacheY.add(0);
334        }
335        else {
336            startCacheX.add(Math.min(Math.max(startCacheX.peek() + (zoomCenterX >> zoom - 1) - (width >> zoom + 1),
337                    0), width - (width >> zoom)));
338            startCacheY.add(Math.min(Math.max(startCacheY.peek() + (zoomCenterY >> zoom - 1) - (height >> zoom + 1),
339                    0), height - (height >> zoom)));
340        }
341        regenerate(startX = (zoomStartX >> zoom) - (width >> 1 + zoom), startY = (zoomStartY >> zoom) - (height >> 1 + zoom),
342                //startCacheX.peek(), startCacheY.peek(),
343                usedWidth = width >> zoom, usedHeight = height >> zoom,
344                landModifier, coolingModifier, cachedState);
345        rng.setState(cachedState);
346    }

834                return;
835            if(heatCodeData == null || (heatCodeData.length != world.width || heatCodeData[0].length != world.height))

848                        biomeCodeData[x][y] = 60;
849                        continue;
850                    }
