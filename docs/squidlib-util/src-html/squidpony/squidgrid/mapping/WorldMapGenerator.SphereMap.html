009/**
010 * Can be used to generate world maps with a wide variety of data, starting with height, temperature and moisture.
011 * From there, you can determine biome information in as much detail as your game needs, with a default implementation
012 * available that assigns a single biome to each cell based on heat/moisture. The maps this produces are valid
013 * for spherical or toroidal world projections, and will wrap from edge to opposite edge seamlessly thanks to a
014 * technique from the Accidental Noise Library ( https://www.gamedev.net/blog/33/entry-2138456-seamless-noise/ ) that
015 * involves getting a 2D slice of 4D Simplex noise. Because of how Simplex noise works, this also allows extremely high
016 * zoom levels as long as certain parameters are within reason. You can access the height map with the
017 * {@link #heightData} field, the heat map with the {@link #heatData} field, the moisture map with the
018 * {@link #moistureData} field, and a special map that stores ints representing the codes for various ranges of
019 * elevation (0 to 8 inclusive, with 0 the deepest ocean and 8 the highest mountains) with {@link #heightCodeData}. The
020 * last map should be noted as being the simplest way to find what is land and what is water; any height code 4 or
021 * greater is land, and any height code 3 or less is water. This can produce rivers, and keeps that information in a
022 * GreasedRegion (alongside a GreasedRegion containing lake positions) instead of in the other map data. This class does
023 * not use Coord at all, but if you want maps with width and/or height greater than 256, and you want to use the river
024 * or lake data as a Collection of Coord, then you should call {@link Coord#expandPoolTo(int, int)}
025 * with your width and height so the Coords remain safely pooled. If you're fine with keeping rivers and lakes as
026 * GreasedRegions and not requesting Coord values from them, then you don't need to do anything with Coord. Certain
027 * parts of this class are not necessary to generate, just in case you want river-less maps or something similar;
028 * setting {@link #generateRivers} to false will disable river generation (it defaults to true).
029 * <br>
030 * The main trade-off this makes to obtain better quality is reduced speed; generating a 512x512 map on a circa-2016
031 * laptop (i7-6700HQ processor at 2.6 GHz) takes about 1 second (about 1.15 seconds for an un-zoomed map, 0.95 or so
032 * seconds to increase zoom at double resolution). If you don't need a 512x512 map, this takes commensurately less time
033 * to generate less grid cells, with 64x64 maps generating faster than they can be accurately seen on the same hardware.
034 * River positions are produced using a different method, and do not involve the Simplex noise parts other than using
035 * the height map to determine flow. Zooming with rivers is tricky, and generally requires starting from the outermost
036 * zoom level and progressively enlarging and adding detail to all rivers as zoom increases on specified points.
037 */

151     * Generates a world using the specified RNG state as a long, with specific water and cooling modifiers that affect
152     * the land-water ratio and the average temperature, respectively.
153     * The worlds this produces will always have width and height as specified in the constructor (default 256x256).
154     * You can call {@link #zoomIn(int, int, int)} to double the resolution and center on the specified area, but the width
155     * and height of the 2D arrays this changed, such as {@link #heightData} and {@link #moistureData} will be the same.
156     * @param waterMod should be between 0.85 and 1.2; a random value will be used if this is negative
157     * @param coolMod should be between 0.85 and 1.4; a random value will be used if this is negative
158     * @param state the state to give this generator's RNG; if the same as the last call, this will reuse data
159     */
160    public void generate(double waterMod, double coolMod, long state)
161    {
162        if(cachedState != state || waterMod != waterModifier || coolMod != coolingModifier)
163        {
164            seed = state;
165            zoom = 0;
166            startCacheX.clear();
167            startCacheY.clear();
168            startCacheX.add(0);
169            startCacheY.add(0);
170            zoomStartX = width >> 1;
171            zoomStartY = height >> 1;
172
173        }
174        //System.out.printf("generate, zoomStartX: %d, zoomStartY: %d\n", zoomStartX, zoomStartY);
175
176        regenerate((zoomStartX >> zoom) - (width >> 1 + zoom), (zoomStartY >> zoom) - (height >> 1 + zoom),
177                //startCacheX.peek(), startCacheY.peek(),
178                (width >> zoom), (height >> zoom), waterMod, coolMod, state);
179    }

182     * Halves the resolution of the map and doubles the area it covers; the 2D arrays this uses keep their sizes. This
183     * version of zoomOut always zooms out from the center of the currently used area.
184     * <br>
185     * Only has an effect if you have previously zoomed in using {@link #zoomIn(int, int, int)} or its overload.
186     */
187    public void zoomOut()
188    {
189        zoomOut(1, width >> 1, height >> 1);
190    }
191    /**
192     * Halves the resolution of the map and doubles the area it covers repeatedly, halving {@code zoomAmount} times; the
193     * 2D arrays this uses keep their sizes. This version of zoomOut allows you to specify where the zoom should be
194     * centered, using the current coordinates (if the map size is 256x256, then coordinates should be between 0 and
195     * 255, and will refer to the currently used area and not necessarily the full world size).
196     * <br>
197     * Only has an effect if you have previously zoomed in using {@link #zoomIn(int, int, int)} or its overload.
198     * @param zoomCenterX the center X position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
199     * @param zoomCenterY the center Y position to zoom out from; if too close to an edge, this will stop moving before it would extend past an edge
200     */
201    public void zoomOut(int zoomAmount, int zoomCenterX, int zoomCenterY)
202    {
203        zoomAmount = Math.min(zoom, zoomAmount);
204        if(zoomAmount == 0) return;
205        if(zoomAmount < 0) {
206            zoomIn(-zoomAmount, zoomCenterX, zoomCenterY);
207            return;
208        }
209        if(zoom > 0)
210        {
211            if(seed != cachedState)
212            {
213                generate(rng.nextLong());
214            }
215            zoomStartX = Math.min(Math.max(
216                    (zoomStartX + (zoomCenterX - (width >> 1))) >> zoomAmount,
217                    width >> 1), (width << zoom - zoomAmount) - (width >> 1));
218            zoomStartY = Math.min(Math.max(
219                    (zoomStartY + (zoomCenterY - (height >> 1))) >> zoomAmount,
220                    height >> 1), (height << zoom - zoomAmount) - (height >> 1));
221//            System.out.printf("zoomOut, zoomStartX: %d, zoomStartY: %d\n", zoomStartX, zoomStartY);
222            zoom -= zoomAmount;
223            startCacheX.pop();
224            startCacheY.pop();
225            startCacheX.add(Math.min(Math.max(startCacheX.pop() + (zoomCenterX >> zoom + 1) - (width >> zoom + 2),
226                    0), width - (width >> zoom)));
227            startCacheY.add(Math.min(Math.max(startCacheY.pop() + (zoomCenterY >> zoom + 1) - (height >> zoom + 2),
228                    0), height - (height >> zoom)));
229//            zoomStartX = Math.min(Math.max((zoomStartX >> 1) + (zoomCenterX >> zoom + 1) - (width >> zoom + 2),
230//                    0), width - (width >> zoom));
231//            zoomStartY = Math.min(Math.max((zoomStartY >> 1) + (zoomCenterY >> zoom + 1) - (height >> zoom + 2),
232//                    0), height - (height >> zoom));
233            regenerate((zoomStartX >> zoom) - (width >> zoom + 1), (zoomStartY >> zoom) - (height >> zoom + 1),
234                    //startCacheX.peek(), startCacheY.peek(),
235                    width >> zoom, height >> zoom,
236                    waterModifier, coolingModifier, cachedState);
237            rng.setState(cachedState);
238        }
239
240    }
241    /**
242     * Doubles the resolution of the map and halves the area it covers; the 2D arrays this uses keep their sizes. This
243     * version of zoomIn always zooms in to the center of the currently used area.
244     * <br>
245     * Although there is no technical restriction on maximum zoom, zooming in more than 5 times (64x scale or greater)
246     * will make the map appear somewhat less realistic due to rounded shapes appearing more bubble-like and less like a
247     * normal landscape.
248     */
249    public void zoomIn()
250    {
251        zoomIn(1, width >> 1, height >> 1);
252    }
253    /**
254     * Doubles the resolution of the map and halves the area it covers repeatedly, doubling {@code zoomAmount} times;
255     * the 2D arrays this uses keep their sizes. This version of zoomIn allows you to specify where the zoom should be
256     * centered, using the current coordinates (if the map size is 256x256, then coordinates should be between 0 and
257     * 255, and will refer to the currently used area and not necessarily the full world size).
258     * <br>
259     * Although there is no technical restriction on maximum zoom, zooming in more than 5 times (64x scale or greater)
260     * will make the map appear somewhat less realistic due to rounded shapes appearing more bubble-like and less like a
261     * normal landscape.
262     * @param zoomCenterX the center X position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
263     * @param zoomCenterY the center Y position to zoom in to; if too close to an edge, this will stop moving before it would extend past an edge
264     */
265    public void zoomIn(int zoomAmount, int zoomCenterX, int zoomCenterY)
266    {
267        if(zoomAmount == 0) return;
268        if(zoomAmount < 0)
269        {
270            zoomOut(-zoomAmount, zoomCenterX, zoomCenterY);
271            return;
272        }
273        if(seed != cachedState)
274        {
275            generate(rng.nextLong());
276        }
277        zoomStartX = Math.min(Math.max(
278                (zoomStartX + zoomCenterX - (width >> 1) << zoomAmount),
279                width >> 1), (width << zoom + zoomAmount) - (width >> 1));
280//        int oldZoomY = zoomStartY;
281        zoomStartY = Math.min(Math.max(
282                (zoomStartY + zoomCenterY - (height >> 1) << zoomAmount),
283                height >> 1), (height << zoom + zoomAmount) - (height >> 1));
284//        System.out.printf("zoomIn, zoomStartX: %d, zoomStartY: %d, oldZoomY: %d, unedited: %d, upperCap: %d\n", zoomStartX, zoomStartY,
285//                oldZoomY, (oldZoomY + zoomCenterY - (height >> 1) << zoomAmount), (height << zoom + zoomAmount) - (height >> 1));
286        zoom += zoomAmount;
287        if(startCacheX.isEmpty())
288        {
289            startCacheX.add(0);
290            startCacheY.add(0);
291        }
292        else {
293            startCacheX.add(Math.min(Math.max(startCacheX.peek() + (zoomCenterX >> zoom - 1) - (width >> zoom + 1),
294                    0), width - (width >> zoom)));
295            startCacheY.add(Math.min(Math.max(startCacheY.peek() + (zoomCenterY >> zoom - 1) - (height >> zoom + 1),
296                    0), height - (height >> zoom)));
297        }
298        regenerate((zoomStartX >> zoom) - (width >> 1 + zoom), (zoomStartY >> zoom) - (height >> 1 + zoom),
299                //startCacheX.peek(), startCacheY.peek(),
300                width >> zoom, height >> zoom,
301                waterModifier, coolingModifier, cachedState);
302        rng.setState(cachedState);
303    }

308    public int codeHeight(final double high)
309    {
310        if(high < deepWaterUpper)
311            return 0;
312        if(high < mediumWaterUpper)
313            return 1;
314        if(high < shallowWaterUpper)
315            return 2;
316        if(high < coastalWaterUpper)
317            return 3;
318        if(high < sandUpper)
319            return 4;
320        if(high < grassUpper)
321            return 5;
322        if(high < forestUpper)
323            return 6;
324        if(high < rockUpper)
325            return 7;
326        return 8;
327    }

677         * Analyzes the last world produced by the given WorldMapGenerator and uses all of its generated information to
678         * assign biome codes for each cell (along with heat and moisture codes). After calling this, biome codes can be
679         * taken from {@link #biomeCodeData} and used as indices into {@link #biomeTable} or a custom biome table.
680         * @param world a WorldMapGenerator that should have generated at least one map; it may be at any zoom
681         */
682        public void makeBiomes(WorldMapGenerator world) {
683            if(world == null || world.width <= 0 || world.height <= 0)
684                return;
685            if(heatCodeData == null || (heatCodeData.length != world.width || heatCodeData[0].length != world.height))
686                heatCodeData = new int[world.width][world.height];
687            if(moistureCodeData == null || (moistureCodeData.length != world.width || moistureCodeData[0].length != world.height))
688                moistureCodeData = new int[world.width][world.height];
689            if(biomeCodeData == null || (biomeCodeData.length != world.width || biomeCodeData[0].length != world.height))
690                biomeCodeData = new int[world.width][world.height];
691            final double i_hot = (world.maxHeat == world.minHeat) ? 1.0 : 1.0 / (world.maxHeat - world.minHeat);
692            for (int x = 0; x < world.width; x++) {
693                for (int y = 0; y < world.height; y++) {
694                    final double hot = (world.heatData[x][y] - world.minHeat) * i_hot, moist = world.moistureData[x][y],
695                            fresh = world.freshwaterData[x][y];
696                    final int heightCode = world.heightCodeData[x][y];
697                    int hc, mc;
698                    boolean isLake = world.generateRivers && heightCode >= 4 && fresh > 0.65 && fresh + moist * 2.35 > 2.75,//world.partialLakeData.contains(x, y) && heightCode >= 4,
699                            isRiver = world.generateRivers && !isLake && heightCode >= 4 && fresh > 0.55 && fresh + moist * 2.2 > 2.15;//world.partialRiverData.contains(x, y) && heightCode >= 4;
700                    if (moist > wetterValueUpper) {
701                        mc = 5;
702                    } else if (moist > wetValueUpper) {
703                        mc = 4;
704                    } else if (moist > dryValueUpper) {
705                        mc = 3;
706                    } else if (moist > drierValueUpper) {
707                        mc = 2;
708                    } else if (moist > driestValueUpper) {
709                        mc = 1;
710                    } else {
711                        mc = 0;
712                    }
713
714                    if (hot > warmerValueUpper) {
715                        hc = 5;
716                    } else if (hot > warmValueUpper) {
717                        hc = 4;
718                    } else if (hot > coldValueUpper) {
719                        hc = 3;
720                    } else if (hot > colderValueUpper) {
721                        hc = 2;
722                    } else if (hot > coldestValueUpper) {
723                        hc = 1;
724                    } else {
725                        hc = 0;
726                    }
727
728                    heatCodeData[x][y] = hc;
729                    moistureCodeData[x][y] = mc;
730                    biomeCodeData[x][y] = isLake ? hc + 48 : (isRiver ? hc + 42 : ((heightCode == 4) ? hc + 36 : hc + mc * 6));
731                }
732            }
733        }
