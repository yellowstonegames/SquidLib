001package squidpony.squidgrid.mapping;
002
003
004import squidpony.annotation.Beta;
005import squidpony.squidgrid.Direction;
006import squidpony.squidmath.Coord;
007import squidpony.squidmath.IRNG;
008import squidpony.squidmath.RNG;
009
010import java.util.ArrayList;
011
012/**
013 *
014 *
015 * Based in part on code from http://weblog.jamisbuck.org/2011/1/27/maze-generation-growing-tree-algorithm
016 *
017 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
018 */
019@Beta
020public class GrowingTreeMazeGenerator {
021
022    private IRNG rng;
023    private int width, height;
024
025    public GrowingTreeMazeGenerator(int width, int height) {
026        this.width = width;
027        this.height = height;
028        rng = new RNG();
029    }
030    public GrowingTreeMazeGenerator(int width, int height, IRNG rng) {
031        this.width = width;
032        this.height = height;
033        this.rng = rng;
034    }
035
036    /**
037     * Builds and returns a boolean mapping of a maze using the provided chooser method object.
038     * 
039     * @param choosing the callback object for making the split decision
040     * @return 
041     */
042    public boolean[][] create(ChoosingMethod choosing) {
043        boolean[][] map = new boolean[width][height];
044        boolean[][] visited = new boolean[width][height];
045
046        int x = rng.nextInt(width / 2);
047        int y = rng.nextInt(height / 2);
048        x *= 2;
049        y *= 2;
050
051        ArrayList<Coord> deck = new ArrayList<>();
052        deck.add(Coord.get(x, y));
053
054        Direction[] dirs = Direction.CARDINALS;
055        while (!deck.isEmpty()) {
056            int i = choosing.chooseIndex(deck.size());
057            Coord p = deck.get(i);
058            dirs = rng.shuffle(dirs, new Direction[dirs.length]);
059
060            boolean foundNeighbor = false;
061            for (Direction dir : dirs) {
062                x = p.x + dir.deltaX * 2;
063                y = p.y + dir.deltaY * 2;
064                if (x >= 0 && x < width && y >= 0 && y < height) {
065                    if (!visited[x][y]) {
066                        foundNeighbor = true;
067//                        if (rng.nextBoolean()) {
068                            visited[x][y] = true;
069//                        }
070                        map[x][y] = true;
071                        map[p.x + dir.deltaX][p.y + dir.deltaY] = true;
072                        deck.add(Coord.get(x, y));
073                        break;
074                    }
075                }
076            }
077
078            if (!foundNeighbor) {
079                deck.remove(p);
080            }
081        }
082
083        return map;
084    }
085
086    public interface ChoosingMethod {
087
088        /**
089         * Given the size to choose from, will return a single value smaller than the passed in value and greater than
090         * or equal to 0. The value chosen is dependant on the individual implementation.
091         *
092         * @param size
093         * @return
094         */
095        int chooseIndex(int size);
096    }
097}
