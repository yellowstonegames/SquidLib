001package squidpony.squidgrid.mapping.locks.generators;
002
003import squidpony.squidgrid.mapping.locks.IRoomLayout;
004
005/**
006 * Interface for classes that provide methods to procedurally generate new
007 * {@link IRoomLayout}s.
008 */
009public interface ILayoutGenerator {
010
011    /**
012     * Generates a new {@link IRoomLayout}.
013     */
014    void generate();
015    
016    /**
017     * Gets the most recently generated {@link IRoomLayout}.
018     * 
019     * @return the most recently generated IRoomLayout
020     */
021    IRoomLayout getRoomLayout();
022    
023}
