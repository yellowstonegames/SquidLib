001package squidpony.squidgrid.mapping.locks.util;
002
003import squidpony.squidgrid.mapping.Rectangle;
004import squidpony.squidmath.Coord;
005
006import java.util.List;
007
008/**
009 * Created by Tommy Ettinger on 1/4/2017.
010 */
011public class Rect2I implements Rectangle {
012    public Coord topLeft;
013    public int width;
014    public int height;
015
016    public Rect2I(Coord min, int w, int h)
017    {
018        topLeft = min;
019        width = w;
020        height = h;
021    }
022
023    public static Rect2I fromExtremes(int minX, int minY, int maxX, int maxY)
024    {
025        return new Rect2I(Coord.get(minX, minY), maxX - minX, maxY - minY);
026    }
027
028    @Override
029    public Coord getBottomLeft() {
030        return topLeft;
031    }
032
033    @Override
034    public int getWidth() {
035        return width;
036    }
037
038    @Override
039    public int getHeight() {
040        return height;
041    }
042
043    public int left()
044    {
045        return topLeft.x;
046    }
047    public int top()
048    {
049        return topLeft.y;
050    }
051
052    public int right()
053    {
054        return topLeft.x + width;
055    }
056    public int bottom()
057    {
058        return topLeft.y + height;
059    }
060
061    @Override
062    public boolean isEmpty() {
063        return width > 0 && height > 0;
064    }
065
066    @Override
067    public int size() {
068        return width * height;
069    }
070
071    @Override
072    public boolean contains(int x, int y) {
073        return x >= topLeft.x && x < topLeft.x + width && y >= topLeft.y && y < topLeft.y + height;
074    }
075
076    @Override
077    public boolean contains(Coord coord) {
078        return coord.x >= topLeft.x && coord.x < topLeft.x + width
079                && coord.y >= topLeft.y && coord.y < topLeft.y + height;
080    }
081
082    @Override
083    public List<Coord> getAll() {
084        return Rectangle.Utils.cellsList(this);
085    }
086}
