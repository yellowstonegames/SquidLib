001package squidpony.squidgrid.mapping.locks;
002
003import squidpony.squidgrid.mapping.locks.util.Rect2I;
004
005import java.util.Collection;
006
007/**
008 * Represents the spatial layout of a lock-and-key puzzle and contains all
009 * {@link Symbol}s, {@link Room}s and {@link Edge}s within the puzzle.
010 */
011public interface IRoomLayout {
012
013    /**
014     * @return  the rooms within the dungeon
015     */
016    Collection<Room> getRooms();
017    /**
018     * @return the number of rooms in the dungeon
019     */
020    int roomCount();
021
022    /**
023     * @param id        the id of the room
024     * @return  the room with the given id
025     */
026    Room get(int id);
027
028    /**
029     * Adds a new room to the dungeon, overwriting any rooms already in it that
030     * have the same coordinates.
031     * 
032     * @param room  the room to add
033     */
034    void add(Room room);
035
036    /**
037     * Adds a one-way unconditional edge between the given rooms.
038     * A one-way edge may be used to travel from room1 to room2, but not room2
039     * to room1.
040     * 
041     * @param room1 the first room to link
042     * @param room2 the second room to link
043     */
044    void linkOneWay(Room room1, Room room2);
045    /**
046     * Adds a two-way unconditional edge between the given rooms.
047     * A two-way edge may be used to travel from each room to the other.
048     * 
049     * @param room1 the first room to link
050     * @param room2 the second room to link
051     */
052    void link(Room room1, Room room2);
053    /**
054     * Adds a one-way conditional edge between the given rooms.
055     * A one-way edge may be used to travel from room1 to room2, but not room2
056     * to room1.
057     * 
058     * @param room1 the first room to link
059     * @param room2 the second room to link
060     * @param cond  the condition on the edge
061     */
062    void linkOneWay(Room room1, Room room2, int cond);
063    /**
064     * Adds a two-way conditional edge between the given rooms.
065     * A two-way edge may be used to travel from each room to the other.
066     * 
067     * @param room1 the first room to link
068     * @param room2 the second room to link
069     * @param cond  the condition on the edge
070     */
071    void link(Room room1, Room room2, int cond);
072    /**
073     * Tests whether two rooms are linked.
074     * Two rooms are linked if there are any edges (in any direction) between
075     * them.
076     * 
077     * @return  true if the rooms are linked, false otherwise
078     */
079    boolean roomsAreLinked(Room room1, Room room2);
080    
081    /**
082     * @return  the room containing the START symbol
083     */
084    Room findStart();
085    /**
086     * @return  the room containing the BOSS symbol
087     */
088    Room findBoss();
089    /**
090     * @return  the room containing the GOAL symbol
091     */
092    Room findGoal();
093    /**
094     * @return  the room containing the SWITCH symbol
095     */
096    Room findSwitch();
097    
098    /**
099     * Gets the {@link Rect2I} that encloses every room within the dungeon.
100     * <p>
101     * The Bounds object has the X coordinate of the West-most room, the Y
102     * coordinate of the North-most room, the 'right' coordinate of the
103     * East-most room, and the 'bottom' coordinate of the South-most room.
104     * 
105     * @return  the rectangle enclosing every room within the dungeon
106     */
107    Rect2I getExtentBounds();
108
109}
