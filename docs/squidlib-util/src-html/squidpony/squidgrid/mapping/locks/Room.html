001package squidpony.squidgrid.mapping.locks;
002
003import squidpony.squidgrid.Direction;
004import squidpony.squidgrid.mapping.locks.generators.LayoutGenerator;
005import squidpony.squidmath.Coord;
006
007import java.util.*;
008
009/**
010 * Represents an individual space within the dungeon.
011 * <p>
012 * A Room contains:
013 * <ul>
014 * <li>an item ({@link Symbol}) that the player may (at his or her choice)
015 *      collect by passing through this Room;
016 * <li>an intensity, which is a measure of the relative difficulty of the room
017 *      and ranges from 0.0 to 1.0;
018 * <li>{@link Edge}s for each door to an adjacent Room.
019 * </ul>
020 */
021public class Room {
022
023    protected Condition precond;
024    public final int id;
025    protected Coord center;
026    protected int item;
027    protected ArrayList<Edge> edges;
028    protected double intensity;
029    protected Room parent;
030    protected List<Room> children;
031    
032    /**
033     * Creates a Room at the given coordinates, with the given parent,
034     * containing a specific item, and having a certain pre-{@link Condition}.
035     * <p>
036     * The parent of a room is the parent node of this Room in the initial
037     * tree of the dungeon during
038     * {@link LayoutGenerator#generate()}, and
039     * before
040     * {@link LayoutGenerator#graphify()}.
041     * 
042     * @param coords    the coordinates of the new room
043     * @param parent    the parent room or null if it is the root / entry room
044     * @param item      the symbol to place in the room or null if no item
045     * @param precond   the precondition of the room
046     * @see Condition
047     */
048    public Room(int id, Set<Coord> coords, Room parent, int item, Condition precond) {
049        this.id = id;
050        this.item = item;
051        this.edges = new ArrayList<Edge>();
052        this.precond = precond;
053        this.intensity = 0.0;
054        this.parent = parent;
055        this.children = new ArrayList<Room>(3);
056        // all edges initially null
057        
058        int x = 0, y = 0;
059        for (Coord xy: coords) {
060            x += xy.x; y += xy.y;
061        }
062        center = Coord.get(x/coords.size(), y/coords.size());
063    }
064    
065    public Room(int id, Coord coord, Room parent, int item, Condition precond) {
066        this(id, Collections.singleton(coord), parent, item,
067                precond);
068    }
069    
070    /**
071     * @return the intensity of the Room
072     * @see Room
073     */
074    public double getIntensity() {
075        return intensity;
076    }
077    
078    /**
079     * @param intensity the value to set the Room's intensity to
080     * @see Room
081     */
082    public void setIntensity(double intensity) {
083        this.intensity = intensity;
084    }
085
086    /**
087     * @return  the item contained in the Room, or null if there is none
088     */
089    public int getItem() {
090        return item;
091    }
092
093    /**
094     * @param item  the item to place in the Room
095     */
096    public void setItem(int item) {
097        this.item = item;
098    }
099
100    /**
101     * Gets the array of {@link Edge} slots this Room has. There is one slot
102     * for each compass {@link Direction}. Non-null slots in this array
103     * represent links between this Room and adjacent Rooms.
104     * 
105     * @return the array of Edges
106     */
107    public List<Edge> getEdges() {
108        return edges;
109    }
110    
111    /**
112     * Gets the Edge object for a link in a given direction.
113     * 
114     * @param targetRoomId the adjacent Room's id
115     * @return  the {@link Edge} for the link in the given direction, or null if
116     *          there is no link from this Room in the given direction
117     */
118    public Edge getEdge(int targetRoomId) {
119        Edge e;
120        for (int i = edges.size() - 1; i >= 0; i--) {
121            e = edges.get(i);
122            if (e.getTargetRoomId() == targetRoomId)
123                return e;
124        }
125        return null;
126    }
127    
128    public Edge setEdge(int targetRoomId, int symbol) {
129        Edge e = getEdge(targetRoomId);
130        if (e != null) {
131            e.symbol = symbol;
132        } else {
133            e = new Edge(targetRoomId, symbol);
134            edges.add(e);
135        }
136        return e;
137    }
138    
139    /**
140     * Gets the number of Rooms this Room is linked to.
141     * 
142     * @return  the number of links
143     */
144    public int linkCount() {
145        return edges.size();
146    }
147    
148    /**
149     * @return whether this room is the entry to the dungeon.
150     */
151    public boolean isStart() {
152        return Symbol.isStart(item);
153    }
154    
155    /**
156     * @return whether this room is the goal room of the dungeon.
157     */
158    public boolean isGoal() {
159        return Symbol.isGoal(item);
160    }
161
162    /**
163     * @return whether this room contains the dungeon's boss.
164     */
165    public boolean isBoss() {
166        return Symbol.isBoss(item);
167    }
168    
169    /**
170     * @return whether this room contains the dungeon's switch object.
171     */
172    public boolean isSwitch() {
173        return Symbol.isSwitch(item);
174    }
175    
176    /**
177     * @return the precondition for this Room
178     * @see Condition
179     */
180    public Condition getPrecond() {
181        return precond;
182    }
183    
184    /**
185     * @param precond   the precondition to set this Room's to
186     * @see Condition
187     */
188    public void setPrecond(Condition precond) {
189        this.precond = precond;
190    }
191
192    /**
193     * @return the parent of this Room
194     * @see Room#Room
195     */
196    public Room getParent() {
197        return parent;
198    }
199
200    /**
201     * @param parent the Room to set this Room's parent to
202     * @see Room#Room
203     */
204    public void setParent(Room parent) {
205        this.parent = parent;
206    }
207    
208    /**
209     * @return the collection of Rooms this Room is a parent of
210     * @see Room#Room
211     */
212    public Collection<Room> getChildren() {
213        return children;
214    }
215    
216    /**
217     * Registers this Room as a parent of another.
218     * Does not modify the child room's parent property.
219     * 
220     * @param child the room to parent
221     */
222    public void addChild(Room child) {
223        children.add(child);
224    }
225    
226    public Coord getCenter() {
227        return center;
228    }
229    
230    public String toString() {
231        return "Room(" + center.toString() + ")";
232    }
233    
234}
