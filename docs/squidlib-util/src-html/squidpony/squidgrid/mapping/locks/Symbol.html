001package squidpony.squidgrid.mapping.locks;
002
003/**
004 * Represents a single key or lock within the lock-and-key puzzle.
005 * <p>
006 * Each Symbol has a 'value'. Two Symbols are equivalent iff they have the same
007 * 'value'.
008 */
009public class Symbol {
010    /**
011     * Symbol map with special meanings.
012     * <p>
013     * Certain items (such as START, GOAL, BOSS) serve no purpose in the puzzle
014     * other than as markers for the client of the library to place special game
015     * objects.
016     * <p>
017     * The SWITCH_ON and SWITCH_OFF symbols do not appear in rooms, only in
018     * {@link Condition}s and {@link Edge}s.
019     * <p>
020     * The SWITCH item does not give the player the SWITCH symbol, instead the
021     * player may choose to either
022     * <ul>
023     * <li>lose the SWITCH_OFF symbol (if they have it), and gain the SWITCH_ON
024     *      symbol; or
025     * <li>lose the SWITCH_ON symbol (if they have it), and gain the SWITCH_OFF
026     *      symbol.
027     * <ul>
028     * <p>
029     */
030    public static final int
031            NOTHING = Integer.MIN_VALUE,
032            START = -1,
033            GOAL = -2,
034            BOSS = -3,
035            SWITCH_ON = -4,     // used as a condition (lock)
036            SWITCH_OFF = -5,    // used as a condition (lock)
037            SWITCH = -6;        // used as an item (key) within a room
038    
039    /**
040     * @return whether the symbol is the special START symbol
041     */
042    public static boolean isStart(final int value) {
043        return value == START;
044    }
045    
046    /**
047     * @return whether the symbol is the special GOAL symbol
048     */
049    public static boolean isGoal(final int value) {
050        return value == GOAL;
051    }
052    
053    /**
054     * @return whether the symbol is the special BOSS symbol
055     */
056    public static boolean isBoss(final int value) {
057        return value == BOSS;
058    }
059    
060    /**
061     * @return whether the symbol is the special SWITCH symbol
062     */
063    public static boolean isSwitch(final int value) {
064        return value == SWITCH;
065    }
066    
067    /**
068     * @return whether the symbol is one of the special SWITCH_{ON,OFF} symbols
069     */
070    public static boolean isSwitchState(final int value) {
071        return value == SWITCH_ON || value == SWITCH_OFF;
072    }
073    
074    public static String toString(final int value) {
075        if (value == START)
076            return "Start";
077        else if (value == GOAL)
078            return "Goal";
079        else if (value == BOSS)
080            return "Boss";
081        else if (value == SWITCH_ON)
082            return "ON";
083        else if (value == SWITCH_OFF)
084            return "OFF";
085        else if (value == SWITCH)
086            return "SW";
087        else if (value >= 0 && value < 26)
088            return Character.toString((char)((int)'A' + value));
089        else
090            return Integer.toString(value);
091
092    }
093    
094}
