001package squidpony.squidgrid.mapping.locks;
002
003/**
004 * Used to represent {@link Room}s' preconditions.
005 * <p>
006 * A Room's precondition can be considered the set of Symbols from the other
007 * Rooms that the player must have collected to be able to reach this room. For
008 * instance, if the Room is behind a locked door, the precondition for the
009 * Room includes the key for that lock.
010 * <p>
011 * In practice, since there is always a time ordering on the collection of keys,
012 * this can be implemented as a count of the number of keys the player must have
013 * (the 'keyLevel').
014 * <p>
015 * The state of the {@link RoomLayout}'s switch is also recorded in the Condition.
016 * A Room behind a link that requires the switch to be flipped into a particular
017 * state will have a precondition that includes the switch's state.
018 * <p>
019 * A Condition is 'satisfied' when the player has all the keys it requires and
020 * when the dungeon's switch is in the state that it requires.
021 * <p>
022 * A Condition x implies a Condition y if and only if y is satisfied whenever x
023 * is.
024 */
025public class Condition {
026
027    /**
028     * A type to represent the required state of a switch for the Condition to
029     * be satisfied.
030     */
031    public enum SwitchState {
032        /**
033         * The switch may be in any state.
034         */
035        EITHER,
036        /**
037         * The switch must be off.
038         */
039        OFF,
040        /**
041         * The switch must be on.
042         */
043        ON;
044        
045        /**
046         * Convert this SwitchState to a {@link Symbol}.
047         * 
048         * @return  a symbol representing the required state of the switch or
049         *          null if the switch may be in any state
050         */
051        public int toSymbol() {
052            switch (this) {
053            case OFF:
054                return Symbol.SWITCH_OFF;
055            case ON:
056                return Symbol.SWITCH_ON;
057            default:
058                return Symbol.NOTHING;
059            }
060        }
061        
062        /**
063         * Invert the required state of the switch.
064         * 
065         * @return  a SwitchState with the opposite required switch state or
066         *          this SwitchState if no particular state is required
067         */
068        public SwitchState invert() {
069            switch (this) {
070            case OFF: return ON;
071            case ON: return OFF;
072            default:
073                return this;
074            }
075        }
076    }
077
078    protected int keyLevel;
079    
080    protected SwitchState switchState;
081
082    /**
083     * Create a Condition that is always satisfied.
084     */
085    public Condition() {
086        keyLevel = 0;
087        switchState = SwitchState.EITHER;
088    }
089    
090    /**
091     * Creates a Condition that requires the player to have a particular
092     * {@link Symbol}.
093     * 
094     * @param e the symbol that the player must have for the Condition to be
095     *          satisfied
096     */
097    public Condition(int e) {
098        if (e == Symbol.SWITCH_OFF) {
099            keyLevel = 0;
100            switchState = SwitchState.OFF;
101        } else if (e == Symbol.SWITCH_ON) {
102            keyLevel = 0;
103            switchState = SwitchState.ON;
104        } else {
105            keyLevel = e+1;
106            switchState = SwitchState.EITHER;
107        }
108    }
109    
110    /**
111     * Creates a Condition from another Condition (copy it).
112     * 
113     * @param other the other Condition
114     */
115    public Condition(Condition other) {
116        keyLevel = other.keyLevel;
117        switchState = other.switchState;
118    }
119    
120    /**
121     * Creates a Condition that requires the switch to be in a particular state.
122     * 
123     * @param switchState   the required state for the switch to be in
124     */
125    public Condition(SwitchState switchState) {
126        keyLevel = 0;
127        this.switchState = switchState;
128    }
129    
130    @Override
131    public boolean equals(Object other) {
132        if (other instanceof Condition) {
133            Condition o = (Condition)other;
134            return keyLevel == o.keyLevel && switchState == o.switchState;
135        } else {
136            return super.equals(other);
137        }
138    }
139    
140    private void add(int sym) {
141        if (sym == Symbol.SWITCH_OFF) {
142            assert switchState == null;
143            switchState = SwitchState.OFF;
144        } else if (sym == Symbol.SWITCH_ON) {
145            assert switchState == null;
146            switchState = SwitchState.ON;
147        } else {
148            keyLevel = Math.max(keyLevel, sym+1);
149        }
150    }
151    private void add(Condition cond) {
152        if (switchState == SwitchState.EITHER) {
153            switchState = cond.switchState;
154        } else {
155            assert switchState == cond.switchState;
156        }
157        keyLevel = Math.max(keyLevel, cond.keyLevel);
158    }
159    
160    /**
161     * Creates a new Condition that requires this Condition to be satisfied and
162     * requires another {@link Symbol} to be obtained as well.
163     * 
164     * @param sym   the added symbol the player must have for the new Condition
165     *              to be satisfied
166     * @return      the new Condition
167     */
168    public Condition and(int sym) {
169        Condition result = new Condition(this);
170        result.add(sym);
171        return result;
172    }
173    
174    /**
175     * Creates a new Condition that requires this Condition and another
176     * Condition to both be satisfied.
177     * 
178     * @param other the other Condition that must be satisfied.
179     * @return      the new Condition
180     */
181    public Condition and(Condition other) {
182        if (other == null) return this;
183        Condition result = new Condition(this);
184        result.add(other);
185        return result;
186    }
187    
188    /**
189     * Determines whether another Condition is necessarily true if this one is.
190     * 
191     * @param other the other Condition
192     * @return  whether the other Condition is implied by this one
193     */
194    public boolean implies(Condition other) {
195        return keyLevel >= other.keyLevel &&
196                (switchState == other.switchState ||
197                other.switchState == SwitchState.EITHER);
198    }
199    /**
200     * Determines whether this Condition implies that a particular
201     * {@link Symbol} has been obtained.
202     * 
203     * @param s the Symbol
204     * @return  whether the Symbol is implied by this Condition
205     */
206    public boolean implies(int s) {
207        return implies(new Condition(s));
208    }
209    
210    /**
211     * Gets the single {@link Symbol} needed to make this Condition and another
212     * Condition identical.
213     * <p>
214     * If {@link #and}ed to both Conditions, the Conditions would then imply
215     * each other.
216     * 
217     * @param other the other Condition
218     * @return  the Symbol needed to make the Conditions identical, or null if
219     *          there is no single Symbol that would make them identical or if
220     *          they are already identical.
221     */
222    public int singleSymbolDifference(Condition other) {
223        // If the difference between this and other can be made up by obtaining
224        // a single new symbol, this returns the symbol. If multiple or no
225        // symbols are required, returns Symbol.NOTHING.
226        
227        if (this.equals(other)) return Symbol.NOTHING;
228        if (switchState == other.switchState) {
229            return Math.max(keyLevel, other.keyLevel)-1;
230        } else {
231            if (keyLevel != other.keyLevel) return Symbol.NOTHING;
232            // Multiple symbols needed        ^^^
233            if (switchState != SwitchState.EITHER &&
234                    other.switchState != SwitchState.EITHER)
235                return Symbol.NOTHING;
236            
237            SwitchState nonEither = switchState != SwitchState.EITHER
238                    ? switchState
239                    : other.switchState;
240            
241            return nonEither == SwitchState.ON
242                    ? Symbol.SWITCH_ON
243                    : Symbol.SWITCH_OFF;
244        }
245    }
246    
247    @Override
248    public String toString() {
249        String result = "";
250        if (keyLevel != 0) {
251            result += Symbol.toString(keyLevel-1);
252        }
253        if (switchState != SwitchState.EITHER) {
254            if (!"".equals(result)) result += ",";
255            result += Symbol.toString(switchState.toSymbol());
256        }
257        return result;
258    }
259    
260    /**
261     * Get the number of keys that need to have been obtained for this Condition
262     * to be satisfied.
263     * 
264     * @return the number of keys
265     */
266    public int getKeyLevel() {
267        return keyLevel;
268    }
269    
270    /**
271     * Get the state the switch is required to be in for this Condition to be
272     * satisfied.
273     */
274    public SwitchState getSwitchState() {
275        return switchState;
276    }
277    
278}
