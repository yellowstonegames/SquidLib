001package squidpony.squidgrid.mapping.locks.constraints;
002
003import squidpony.squidmath.Coord;
004
005import java.util.ArrayList;
006import java.util.Collection;
007import java.util.List;
008
009/**
010 * Constrains the coordinates where Rooms may be placed to be only those within
011 * the {@link SpaceMap}, as well as placing limitations on the number of keys
012 * and switches.
013 * 
014 * @see CountConstraints
015 * @see SpaceMap
016 */
017public class SpaceConstraints extends CountConstraints {
018
019    public static final int DEFAULT_MAX_KEYS = 4,
020            DEFAULT_MAX_SWITCHES = 1;
021    
022    protected SpaceMap spaceMap;
023    
024    public SpaceConstraints(SpaceMap spaceMap) {
025        super(spaceMap.numberSpaces(), DEFAULT_MAX_KEYS, DEFAULT_MAX_SWITCHES);
026        this.spaceMap = spaceMap;
027    }
028
029    @Override
030    protected boolean validRoomCoords(Coord c) {
031        return spaceMap.get(c);
032    }
033
034    @Override
035    public Collection<Integer> initialRooms() {
036        List<Integer> ids = new ArrayList<Integer>();
037        for (Coord xy: spaceMap.getBottomSpaces()) {
038            ids.add(getRoomId(xy));
039        }
040        return ids;
041    }
042    
043    
044
045}
