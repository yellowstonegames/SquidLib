001package squidpony.squidgrid.mapping.locks.constraints;
002
003import squidpony.squidgrid.Direction;
004import squidpony.squidgrid.mapping.locks.IRoomLayout;
005import squidpony.squidgrid.mapping.locks.generators.ILayoutGenerator;
006import squidpony.squidmath.Arrangement;
007import squidpony.squidmath.Coord;
008import squidpony.squidmath.IntVLA;
009
010import java.util.*;
011
012/**
013 * Limits the {@link ILayoutGenerator} in
014 * the <i>number</i> of keys, switches and rooms it is allowed to place.
015 * 
016 * Also restrict to a grid of 1x1 rooms.
017 * 
018 * @see ILayoutConstraints
019 */
020public class CountConstraints implements ILayoutConstraints {
021
022    protected int maxSpaces, maxKeys, maxSwitches;
023
024    protected Arrangement<Coord> roomIds;
025    protected int firstRoomId;
026    
027    public CountConstraints(int maxSpaces, int maxKeys, int maxSwitches) {
028        this.maxSpaces = maxSpaces;
029        this.maxKeys = maxKeys;
030        this.maxSwitches = maxSwitches;
031
032        roomIds = new Arrangement<Coord>();
033        Coord first = Coord.get(0,0);
034        firstRoomId = getRoomId(first);
035    }
036    
037    public int getRoomId(Coord xy) {
038        if (roomIds.containsKey(xy)) {
039            return roomIds.get(xy);
040        } else {
041            roomIds.add(xy);
042            return roomIds.size()-1;
043        }
044    }
045    
046    public Coord getRoomCoords(int id) {
047        assert roomIds.containsValue(id);
048        return roomIds.keyAt(id);
049
050    }
051    
052    @Override
053    public int getMaxRooms() {
054        return maxSpaces;
055    }
056    
057    public void setMaxSpaces(int maxSpaces) {
058        this.maxSpaces = maxSpaces;
059    }
060    
061    @Override
062    public Collection<Integer> initialRooms() {
063        return Collections.singleton(firstRoomId);
064    }
065
066    @Override
067    public int getMaxKeys() {
068        return maxKeys;
069    }
070    
071    public void setMaxKeys(int maxKeys) {
072        this.maxKeys = maxKeys;
073    }
074    
075    @Override
076    public boolean isAcceptable(IRoomLayout dungeon) {
077        return true;
078    }
079
080    @Override
081    public int getMaxSwitches() {
082        return maxSwitches;
083    }
084
085    public void setMaxSwitches(int maxSwitches) {
086        this.maxSwitches = maxSwitches;
087    }
088
089    protected boolean validRoomCoords(Coord c) {
090        return c.y <= 0;
091    }
092    
093    @Override
094    public IntVLA getAdjacentRooms(int id, int keyLevel) {
095        Coord xy = roomIds.keyAt(id);
096        IntVLA ids = new IntVLA();
097        for (Direction d: Direction.CARDINALS) {
098            Coord neighbor = xy.translate(d);
099            if (validRoomCoords(neighbor))
100                ids.add(getRoomId(neighbor));
101        }
102        return ids;
103    }
104
105    @Override
106    public Set<Coord> getCoords(int id) {
107        return new TreeSet<Coord>(Arrays.asList(getRoomCoords(id)));
108    }
109
110    @Override
111    public double edgeGraphifyProbability(int id, int nextId) {
112        return 0.2;
113    }
114
115    @Override
116    public boolean roomCanFitItem(int id, int key) {
117        return true;
118    }
119
120}
