025package squidpony.squidgrid.mapping;
026/**
027 * Added to SquidLib by Tommy Ettinger on 7/4/2018, using MIT-licensed work by Justin Kunimune from
028 * <a href="https://github.com/jkunimune15/Map-Projections/blob/9f820aba788ba0b37a1c67128a4c861d243b4a46/src/utils/NumericalAnalysis.java">his Map-Projections repo</a>.
029 * @author jkunimune
030 * @author Tommy Ettinger
031 */
032public class ProjectionTools {
033    /**
034     * Performs a definite integral using Simpson's rule and a constant step size; hard-coded to integrate a
035     * hyperellipse function.
036     * @param a The start of the integration region
037     * @param b The end of the integration region (must be greater than a)
038     * @param h The step size (must be positive)
039     * @param kappa the kappa value of the hyperellipse
040     * @return some magic stuff needed for Tobler Hyperelliptical maps
041     */
042    public static double simpsonIntegrateHyperellipse(double a, double b, double h, double kappa) {
043        double sum = 0, ik = 1/kappa;
044        for (double x = a; x < b; x += h) {
045            if (x+h > b) h = b-x;
046            sum += h/6*(Math.pow(1 - Math.pow(Math.abs(x), kappa), ik) 
047                    + 4*Math.pow(1 - Math.pow(Math.abs(x + h * 0.5), kappa), ik) 
048                    + Math.pow(1 - Math.pow(Math.abs(x + h), kappa), ik));
049        }
050        return sum;
051    }
052
053    /**
054     * Solves a simple ODE using Simpson's rule and a constant step size; hard-coded to solve a hyperelliptical map
055     * projection task.
056     * @param T The maximum time value at which to sample (must be positive)
057     * @param y the double array to fill with samples; must not be null and must have length 1 or greater
058     * @param h The internal step size (must be positive)
059     * @param alpha part of the hyperelliptical projection's parameters 
060     * @param kappa part of the hyperelliptical projection's parameters
061     * @param epsilon calculated beforehand using {@link #simpsonIntegrateHyperellipse(double, double, double, double)}
062     * @return y, after modifications
063     */
064    public static double[] simpsonODESolveHyperellipse(final double T, final double[] y, final double h, final double alpha, final double kappa, final double epsilon)
065    {
066        final int n = y.length - 1;
067        double t = 0;
068        double sum = 0;
069        for (int i = 0; i <= n; i++) {
070            while (t < i * T / n) {
071                final double tph = Math.min(t + h, i * T / n);
072                sum += (tph - t) / 6 * (Math.abs((alpha + (1-alpha)*Math.pow(1 - Math.pow(Math.abs(t), kappa), 1.0/kappa)) / (alpha + (1-alpha)*epsilon))
073                        + 4 * Math.abs((alpha + (1-alpha)*Math.pow(1 - Math.pow(Math.abs((t + tph) * 0.5), kappa), 1.0/kappa)) / (alpha + (1-alpha)*epsilon))
074                        + Math.abs((alpha + (1-alpha)*Math.pow(1 - Math.pow(Math.abs(tph), kappa), 1.0/kappa)) / (alpha + (1-alpha)*epsilon)));
075                t = tph;
076            }
077            y[i] = sum;
078        }
079        return y;
080    }
081
082    /**
083     * Part of computing a hyperellipse; takes only a y parameter corresponding to the y on a map and a kappa parameter
084     * used by Tobler's hyperelliptical projection to determine shape.
085     * @param y y on a map, usually -1.0 to 1.0
086     * @param kappa one of the Tobler parameters
087     * @return I'm guessing the actual y used after hyperelliptical distortion; not sure
088     */
089    public static double hyperellipse(double y, double kappa) {
090        return Math.pow(1 - Math.pow(Math.abs(y),kappa), 1/kappa);
091    }
092}
