016/**
017 * When you have a world map as produced by {@link WorldMapGenerator} or some other source, you may want to fill it with
018 * claims by various nations/factions, possibly procedural or possibly hand-made. This can assign contiguous areas of
019 * land to various factions, while ignoring some amount of "wild" land if desired, and keeping oceans unclaimed.
020 * The factions can be given procedural names in an atlas that is linked to the chars used by the world map.
021 * Uses {@link MultiSpill} internally to produce the semi-random nation shapes. Stores an {@link #atlas} that can be
022 * used to figure out what a char in a produced 2D char array means for its claiming nation, a {@link #briefAtlas} that
023 * will have short, identifiable parts of generated names corresponding to the same chars as in atlas, and an OrderedMap
024 * of {@link #spokenLanguages} that contains any randomly generated languages produced for nations.
025 * <a href="https://gist.github.com/tommyettinger/4a16a09bebed8e2fe8473c8ea444a2dd">Example output of a related class</a>.
026 */
027@Beta
028public class PoliticalMapper {
029    public int width;
030    public int height;
031    public StatefulRNG rng;
032    public String name;
033    public char[][] politicalMap;
034    public static final char[] letters = ArrayTools.letterSpan(256);
035    /**
036     * Maps chars, as found in the returned array from generate(), to Strings that store the full name of nations.
037     */
038    public final OrderedMap<Character, String> atlas = new OrderedMap<>(32);
039    /**
040     * Maps chars, as found in the returned array from generate(), to Strings that store the short name of nations.
041     */
042    public final OrderedMap<Character, String> briefAtlas = new OrderedMap<>(32);
043    /**
044     * Maps chars, as found in the returned array from generate(), to Strings that store the languages spoken in those
045     * nations, which could be user-assigned, unassigned, or randomly-generated.
046     */
047    public final OrderedMap<Character, List<FakeLanguageGen>> spokenLanguages = new OrderedMap<>(32);
048
049    public PoliticalMapper()
050    {
051        name = "Permadeath Planet";
052        rng = new StatefulRNG(CrossHash.hash64(name));
053    }
054    /**
055     * Constructs a SpillWorldMap using the given world name, and uses the world name as the
056     * basis for all future random generation in this object.
057     *
058     * @param worldName a String name for the world that will be used as a seed for all random generation here
059     */
060    public PoliticalMapper(String worldName) {
061        name = worldName;
062        rng = new StatefulRNG(CrossHash.hash64(name));
063    }
064    /**
065     * Constructs a SpillWorldMap using the given world name, and uses the world name as the
066     * basis for all future random generation in this object.
067     *
068     * @param random an RNG to generate the name for the world in a random language, which will also serve as a seed
069     */
070    public PoliticalMapper(RNG random) {
071        this(FakeLanguageGen.SIMPLISH.word(random, true));
072    }
073    /**
074     * Produces a political map for the land stored in the given WorldMapGenerator, with the given number
075     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
076     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
077     * land. The amount of unclaimed land is determined by the controlledFraction parameter, which will be clamped
078     * between 0.0 and 1.0, with higher numbers resulting in more land owned by factions and lower numbers meaning more
079     * wilderness. This version generates an atlas with the procedural names of all the factions and a
080     * mapping to the chars used in the output; the atlas will be in the {@link #atlas} member of this object. For every
081     * Character key in atlas, there will be a String value in atlas that is the name of the nation, and for the same
082     * key in {@link #spokenLanguages}, there will be a non-empty List of {@link FakeLanguageGen} languages (usually
083     * one, sometimes two) that should match any names generated for the nation. Ocean and Wilderness get the default
084     * FakeLanguageGen instances "ELF" and "DEMONIC", in case you need languages for those areas for some reason.
085     * @param wmg a WorldMapGenerator that has produced a map; this gets the land parts of the map to assign claims to,
086     *            including rivers and lakes as part of nations but not oceans
087     * @param factionCount the number of factions to have claiming land, cannot be negative or more than 255
088     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
089     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
090     */
091    public char[][] generate(WorldMapGenerator wmg, int factionCount, double controlledFraction) {
092        return generate(new GreasedRegion(wmg.heightCodeData, 4, 999), factionCount, controlledFraction);
093    }
094    /**
095     * Produces a political map for the land stored in the "on" cells of the given GreasedRegion, with the given number
096     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
097     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
098     * land. The amount of unclaimed land is determined by the controlledFraction parameter, which will be clamped
099     * between 0.0 and 1.0, with higher numbers resulting in more land owned by factions and lower numbers meaning more
100     * wilderness. This version generates an atlas with the procedural names of all the factions and a
101     * mapping to the chars used in the output; the atlas will be in the {@link #atlas} member of this object. For every
102     * Character key in atlas, there will be a String value in atlas that is the name of the nation, and for the same
103     * key in {@link #spokenLanguages}, there will be a non-empty List of {@link FakeLanguageGen} languages (usually
104     * one, sometimes two) that should match any names generated for the nation. Ocean and Wilderness get the default
105     * FakeLanguageGen instances "ELF" and "DEMONIC", in case you need languages for those areas for some reason.
106     * @param land a GreasedRegion that stores "on" cells for land and "off" cells for anything un-claimable, like ocean
107     * @param factionCount the number of factions to have claiming land, cannot be negative or more than 255
108     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
109     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
110     */
111    public char[][] generate(GreasedRegion land, int factionCount, double controlledFraction) {
112        factionCount &= 255;
113        width = land.width;
114        height = land.height;
115        MultiSpill spreader = new MultiSpill(new short[width][height], Spill.Measurement.MANHATTAN, rng);
116        Coord.expandPoolTo(width, height);
117        GreasedRegion map = land.copy();
118        //Coord[] centers = map.randomSeparated(0.1, rng, factionCount);
119        int controlled = (int) (map.size() * Math.max(0.0, Math.min(1.0, controlledFraction)));
120        map.randomScatter(rng, (width + height) / 25, factionCount);
121
122        spreader.initialize(land.toChars());
123        OrderedMap<Coord, Double> entries = new OrderedMap<>();
124        entries.put(Coord.get(-1, -1), 0.0);
125        for (int i = 0; i < factionCount; i++) {
126            entries.put(map.nth(i), rng.between(0.5, 1.0));
127        }
128        spreader.start(entries, controlled, null);
129        short[][] sm = spreader.spillMap;
130        politicalMap = new char[width][height];
131        for (int x = 0; x < width; x++) {
132            for (int y = 0; y < height; y++) {
133                politicalMap[x][y] = (sm[x][y] == -1) ? '~' : (sm[x][y] == 0) ? '%' : letters[(sm[x][y] - 1) & 255];
134            }
135        }
136
137        atlas.clear();
138        briefAtlas.clear();
139        spokenLanguages.clear();
140        atlas.put('~', "Ocean");
141        briefAtlas.put('~', "Ocean");
142        spokenLanguages.put('~', Collections.singletonList(FakeLanguageGen.ELF));
143        atlas.put('%', "Wilderness");
144        briefAtlas.put('%', "Wilderness");
145        spokenLanguages.put('%', Collections.singletonList(FakeLanguageGen.DEMONIC));
146
147        if (factionCount > 0) {
148            Thesaurus th = new Thesaurus(rng.nextLong());
149            th.addKnownCategories();
150            for (int i = 0; i < factionCount && i < 256; i++) {
151                atlas.put(letters[i], th.makeNationName());
152                briefAtlas.put(letters[i], th.latestGenerated);
153                if(th.randomLanguages == null || th.randomLanguages.isEmpty())
154                    spokenLanguages.put(letters[i], Collections.singletonList(FakeLanguageGen.randomLanguage(rng)));
155                else
156                    spokenLanguages.put(letters[i], new ArrayList<>(th.randomLanguages));
157            }
158        }
159        return politicalMap;
160    }
161    /**
162     * Produces a political map for the land stored in the given WorldMapGenerator, with the given number
163     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
164     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
165     * land. The amount of unclaimed land is determined by the controlledFraction parameter, which will be clamped
166     * between 0.0 and 1.0, with higher numbers resulting in more land owned by factions and lower numbers meaning more
167     * wilderness. This version uses an existing atlas and does not assign to {@link #spokenLanguages}; it does not
168     * alter the existingAtlas parameter but uses it to determine what should be in this class' {@link #atlas} field.
169     * The atlas field will always contain '~' as the first key in its ordering (with value "Ocean" if no value was
170     * already assigned in existingAtlas to that key), and '%' as the second key (with value "Wilderness" if not already
171     * assigned); later entries will be taken from existingAtlas (not duplicating '~' or '%', but using the rest).
172     * @param wmg a WorldMapGenerator that has produced a map; this gets the land parts of the map to assign claims to,
173     *            including rivers and lakes as part of nations but not oceans
174     * @param existingAtlas a Map (ideally an OrderedMap) of Character keys to be used in the 2D array, to String values
175     *                      that are the names of nations; should not have size greater than 255
176     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
177     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
178     */
179    public char[][] generate(WorldMapGenerator wmg, Map<Character, String> existingAtlas, double controlledFraction) {
180        return generate(new GreasedRegion(wmg.heightCodeData, 4, 999), existingAtlas, controlledFraction);
181    }
182    /**

184     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
185     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
186     * land. The amount of unclaimed land is determined by the controlledFraction parameter, which will be clamped
187     * between 0.0 and 1.0, with higher numbers resulting in more land owned by factions and lower numbers meaning more
188     * wilderness. This version uses an existing atlas and does not assign to {@link #spokenLanguages}; it does not
189     * alter the existingAtlas parameter but uses it to determine what should be in this class' {@link #atlas} field.
190     * The atlas field will always contain '~' as the first key in its ordering (with value "Ocean" if no value was
191     * already assigned in existingAtlas to that key), and '%' as the second key (with value "Wilderness" if not already
192     * assigned); later entries will be taken from existingAtlas (not duplicating '~' or '%', but using the rest).
193     * @param land a GreasedRegion that stores "on" cells for land and "off" cells for anything un-claimable, like ocean

195     *                      that are the names of nations; should not have size greater than 255
196     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
197     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
198     */
199    public char[][] generate(GreasedRegion land, Map<Character, String> existingAtlas, double controlledFraction) {
200        atlas.clear();
201        briefAtlas.clear();
202        atlas.putAll(existingAtlas);
203        if(atlas.getAndMoveToFirst('%') == null)
204            atlas.putAndMoveToFirst('%', "Wilderness");
205        if(atlas.getAndMoveToFirst('~') == null)
206            atlas.putAndMoveToFirst('~', "Ocean");
207        int factionCount = atlas.size() - 2;
208        briefAtlas.putAll(atlas);
209        width = land.width;
210        height = land.height;
211        MultiSpill spreader = new MultiSpill(new short[width][height], Spill.Measurement.MANHATTAN, rng);
212        Coord.expandPoolTo(width, height);
213        GreasedRegion map = land.copy();
214        //Coord[] centers = map.randomSeparated(0.1, rng, factionCount);
215        int controlled = (int) (map.size() * Math.max(0.0, Math.min(1.0, controlledFraction)));
216        map.randomScatter(rng, (width + height) / 25, factionCount);
217
218        spreader.initialize(land.toChars());
219        OrderedMap<Coord, Double> entries = new OrderedMap<>();
220        entries.put(Coord.get(-1, -1), 0.0);
221        for (int i = 0; i < factionCount; i++) {
222            entries.put(map.nth(i), rng.between(0.5, 1.0));
223        }
224        spreader.start(entries, controlled, null);
225        short[][] sm = spreader.spillMap;
226        politicalMap = new char[width][height];
227        for (int x = 0; x < width; x++) {
228            for (int y = 0; y < height; y++) {
229                politicalMap[x][y] = (sm[x][y] == -1) ? '~' : (sm[x][y] == 0) ? '%' : atlas.keyAt((sm[x][y] + 1));
230            }
231        }
232        return politicalMap;
233    }
234    /**
235     * Produces a political map for the land stored in the given WorldMapGenerator, with the given number
236     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
237     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
238     * land. The amount of unclaimed land is determined by the controlledFraction parameter, which will be clamped
239     * between 0.0 and 1.0, with higher numbers resulting in more land owned by factions and lower numbers meaning more
240     * wilderness. This version uses a "recipe for an atlas" instead of a complete atlas; this is an OrderedMap of
241     * Character keys to FakeLanguageGen values, where each key represents a faction and each value is the language to
242     * use to generate names for that faction. This does assign to {@link #spokenLanguages}, but it doesn't change the
243     * actual FakeLanguageGen objects, since they are immutable. It may add some "factions" if not present to represent
244     * oceans and unclaimed land. The atlas field will always contain '~' as the first key in its ordering (with value
245     * "Ocean" if no value was already assigned in existingAtlas to that key, or a random nation name in the language
246     * that was mapped if there is one), and '%' as the second key (with value "Wilderness" if not already assigned, or
247     * a similar random nation name if there is one); later entries will be taken from existingAtlas (not duplicating
248     * '~' or '%', but using the rest).
249     * @param wmg a WorldMapGenerator that has produced a map; this gets the land parts of the map to assign claims to,
250     *            including rivers and lakes as part of nations but not oceans
251     * @param atlasLanguages an OrderedMap of Character keys to be used in the 2D array, to FakeLanguageGen objects that
252     *                       will be used to generate names; should not have size greater than 255
253     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
254     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
255     */
256    public char[][] generate(WorldMapGenerator wmg, OrderedMap<Character, FakeLanguageGen> atlasLanguages, double controlledFraction) {
257        return generate(new GreasedRegion(wmg.heightCodeData, 4, 999), atlasLanguages, controlledFraction);
258    }
259    /**
260     * Produces a political map for the land stored in the "on" cells of the given GreasedRegion, with the given number
261     * of factions trying to take land in the world (essentially, nations). The output is a 2D char array where each
262     * letter char is tied to a different faction, while '~' is always water, and '%' is always wilderness or unclaimed
263     * land. The amount of unclaimed land is determined by the controlledFraction parameter, which will be clamped
264     * between 0.0 and 1.0, with higher numbers resulting in more land owned by factions and lower numbers meaning more
265     * wilderness. This version uses a "recipe for an atlas" instead of a complete atlas; this is an OrderedMap of
266     * Character keys to FakeLanguageGen values, where each key represents a faction and each value is the language to
267     * use to generate names for that faction. This does assign to {@link #spokenLanguages}, but it doesn't change the
268     * actual FakeLanguageGen objects, since they are immutable. It may add some "factions" if not present to represent
269     * oceans and unclaimed land. The atlas field will always contain '~' as the first key in its ordering (with value
270     * "Ocean" if no value was already assigned in existingAtlas to that key, or a random nation name in the language
271     * that was mapped if there is one), and '%' as the second key (with value "Wilderness" if not already assigned, or
272     * a similar random nation name if there is one); later entries will be taken from existingAtlas (not duplicating
273     * '~' or '%', but using the rest).
274     * @param land a GreasedRegion that stores "on" cells for land and "off" cells for anything un-claimable, like ocean
275     * @param atlasLanguages an OrderedMap of Character keys to be used in the 2D array, to FakeLanguageGen objects that
276     *                       will be used to generate names; should not have size greater than 255
277     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
278     * @return a 2D char array where letters represent the claiming faction, '~' is water, and '%' is unclaimed
279     */
280    public char[][] generate(GreasedRegion land, OrderedMap<Character, FakeLanguageGen> atlasLanguages, double controlledFraction) {
281        atlas.clear();
282        briefAtlas.clear();
283        spokenLanguages.clear();
284
285        Thesaurus th = new Thesaurus(rng.nextLong());
286        th.addKnownCategories();
287        FakeLanguageGen flg;
288        if((flg = atlasLanguages.get('~')) == null) {
289            atlas.put('~', "Ocean");
290            briefAtlas.put('~', "Ocean");
291            spokenLanguages.put('~', Collections.singletonList(FakeLanguageGen.ELF));
292        }
293        else {
294            atlas.put('~', th.makeNationName(flg));
295            briefAtlas.put('~', th.latestGenerated);
296            spokenLanguages.put('~', Collections.singletonList(flg));
297        }
298        if((flg = atlasLanguages.get('%')) == null) {
299            atlas.put('%', "Wilderness");
300            briefAtlas.put('%', "Wilderness");
301            spokenLanguages.put('%', Collections.singletonList(FakeLanguageGen.DEMONIC));
302        }
303        else {
304            atlas.put('%', th.makeNationName(flg));
305            briefAtlas.put('%', th.latestGenerated);
306            spokenLanguages.put('%', Collections.singletonList(flg));
307        }
308
309        for (int i = 0; i < atlasLanguages.size() && i < 256; i++) {
310            Character c = atlasLanguages.keyAt(i);
311            flg = atlasLanguages.getAt(i);
312            atlas.put(c, th.makeNationName(flg));
313            briefAtlas.put(c, th.latestGenerated);
314            spokenLanguages.put(c, Collections.singletonList(flg));
315        }
316        int factionCount = atlas.size() - 2;
317        width = land.width;
318        height = land.height;
319        MultiSpill spreader = new MultiSpill(new short[width][height], Spill.Measurement.MANHATTAN, rng);
320        Coord.expandPoolTo(width, height);
321        GreasedRegion map = land.copy();
322        //Coord[] centers = map.randomSeparated(0.1, rng, factionCount);
323        int controlled = (int) (map.size() * Math.max(0.0, Math.min(1.0, controlledFraction)));
324        map.randomScatter(rng, (width + height) / 25, factionCount);
325
326        spreader.initialize(land.toChars());
327        OrderedMap<Coord, Double> entries = new OrderedMap<>();
328        entries.put(Coord.get(-1, -1), 0.0);
329        for (int i = 0; i < factionCount; i++) {
330            entries.put(map.nth(i), rng.between(0.5, 1.0));
331        }
332        spreader.start(entries, controlled, null);
333        short[][] sm = spreader.spillMap;
334        politicalMap = new char[width][height];
335        for (int x = 0; x < width; x++) {
336            for (int y = 0; y < height; y++) {
337                politicalMap[x][y] = atlas.keyAt(sm[x][y] + 1);
338            }
339        }
340        return politicalMap;
341    }
