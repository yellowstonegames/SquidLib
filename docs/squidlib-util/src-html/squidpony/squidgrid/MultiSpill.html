008/**
009 * A randomized flood-fill implementation that can be used for level generation (e.g. filling ponds and lakes), for
010 * gas propagation, or for all sorts of fluid-dynamics-on-the-cheap.
011 * Created by Tommy Ettinger on 4/7/2015.

015    /**
016     * This affects how distance is measured on diagonal directions vs. orthogonal directions. MANHATTAN should form a
017     * diamond shape on a featureless map, while CHEBYSHEV and EUCLIDEAN will form a square. If you only call
018     * Spill.start() once, you should strongly prefer MANHATTAN, even if the rest of the game uses another
019     * measurement, because CHEBYSHEV and EUCLIDEAN can produce odd, gap-filled flood-fills.  Any case where you have
020     * too many gaps can be corrected to varying extent by calling start() more than once with slowly increasing values.
021     * Because start() will extend from the existing area of the Spill, holes are likely to be filled after a few calls,
022     * but if the last call to start() tries to fill too many more cells than the previous one, it can cause holes on
023     * the periphery of the Spill area.
024     */
025    public Measurement measurement = Measurement.MANHATTAN;

200    /**
201     * Used to initialize or re-initialize a Spill that needs a new PhysicalMap because it either wasn't given
202     * one when it was constructed, or because the contents of the terrain have changed permanently.
203     * @param level a short[][] that should have been the spillMap of another MultiSpill
204     * @return this for chaining
205     */
206    public MultiSpill initialize(final short[][] level) {
207        fresh = new ArrayList<>();
208        width = level.length;
209        height = level[0].length;
210        spillMap = new short[width][height];
211        anySpillMap = new GreasedRegion(level, 1, 0x7fff);
212        anyFreshMap = new GreasedRegion(width, height);
213        physicalMap = new boolean[width][height];
214        for (int y = 0; y < height; y++) {
215            for (int x = 0; x < width; x++) {
216                spillMap[x][y] = level[x][y];
217                physicalMap[x][y] = level[x][y] >= 0;
218            }
219        }
220        initialized = true;
221        return this;
222    }

225     * Used to initialize or re-initialize a Spill that needs a new PhysicalMap because it either wasn't given
226     * one when it was constructed, or because the contents of the terrain have changed permanently (not if a
227     * creature moved; for that you pass the positions of creatures that block paths to scan() or findPath() ).
228     * @param level a char[][] that should use '#' for walls and '.' for floors
229     * @return this for chaining

277     * Resets the spillMap to being empty.
278     */
279    public void resetMap() {
280        if(!initialized) return;
281        anySpillMap.clear();
282        for (int y = 0; y < height; y++) {
283            for (int x = 0; x < width; x++) {
284                spillMap[x][y] = -1;
285            }
286        }
287    }

321        if(!initialized) return;

323        anyFreshMap.insert(x, y);

326    protected void setFresh(int idx, final Coord pt) {
327        if(!initialized) return;
328        if(anyFreshMap.contains(pt.x, pt.y))
329            return;
330        fresh.get(idx).add(pt);
331        anyFreshMap.insert(pt);
332    }

334    /**
335     * Recalculate the spillMap and return the spreadPattern. The cell corresponding to a Coord in entries will be true,
336     * the cells near each of those will be true if chosen at random from all passable cells adjacent to a
337     * filled (true) cell, and all other cells will be false. This takes a total number of cells to attempt
338     * to fill (the volume parameter), which can be negative to simply fill the whole map, and will fill less if it has
339     * completely exhausted all passable cells from all sources in entries.
340     * If the measurement this Spill uses is anything other than MANHATTAN, you can expect many gaps in the first
341     * filled area.  Subsequent calls to start() with the same entry and a higher volume will expand the area
342     * of the Spill, and are likely to fill any gaps after a few subsequent calls. Increasing the volume slowly
343     * is the best way to ensure that gaps only exist on the very edge if you use a non-MANHATTAN measurement.
344     *
345     * @param entries the first cell for each spiller to spread from, which should really be passable.
346     * @param volume the total number of cells to attempt to fill; if negative will fill the whole map.
347     * @param impassable a Collection, ideally a Set or GreasedRegion, holding Coord items representing the locations of
348     *                   moving obstacles to a fill that cannot be moved through; null means no obstacles exist.
349     * @return an ArrayList of Points that this will enter, in order starting with entry at index 0, until it
350     * reaches its volume or fills its boundaries completely.
351     */
352    public ArrayList<ArrayList<Coord>> start(List<Coord> entries, int volume, Collection<Coord> impassable) {
353        if(!initialized) return null;
354        if(impassable == null)
355            impassable = Collections.emptySet();
356        if(volume < 0)
357            volume = Integer.MAX_VALUE;
358        ArrayList<Coord> spillers = new ArrayList<>(entries);
359        spreadPattern = new ArrayList<>(spillers.size());
360        fresh.clear();
361        filled = 0;
362        boolean hasFresh = false;
363        for (short i = 0; i < spillers.size(); i++) {
364            spreadPattern.add(new ArrayList<Coord>(128));
365            OrderedSet<Coord> os = new OrderedSet<>(128);
366            fresh.add(os);
367            Coord c = spillers.get(i);
368            spillMap[c.x][c.y] = i;
369            if(!impassable.contains(c)) {
370                os.add(c);
371                hasFresh = true;
372            }
373        }
374        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
375        OrderedSet<Coord> currentFresh;
376        while (hasFresh && filled < volume) {
377            hasFresh = false;
378            for (short i = 0; i < spillers.size() && filled < volume; i++) {
379                currentFresh = fresh.get(i);
380                if(currentFresh.isEmpty())
381                    continue;
382                else
383                    hasFresh = true;
384                Coord cell = currentFresh.randomItem(rng);//.toArray(new Coord[currentFresh.size()])[rng.nextInt(currentFresh.size())];
385
386                spreadPattern.get(i).add(cell);
387                spillMap[cell.x][cell.y] = i;
388                filled++;
389                anySpillMap.insert(cell.x, cell.y);
390
391                for (int d = 0; d < dirs.length; d++) {
392                    Coord adj = cell.translate(dirs[d].deltaX, dirs[d].deltaY);
393                    if(!adj.isWithin(width, height))
394                        continue;
395                    double h = heuristic(dirs[d]);
396                    if (physicalMap[adj.x][adj.y] && !anySpillMap.contains(adj.x, adj.y) && !impassable.contains(adj) && rng.nextDouble(h) <= 1.0) {
397                        setFresh(i, adj);
398                    }
399                }
400                currentFresh.remove(cell);
401                anyFreshMap.remove(cell.x, cell.y);
402            }
403        }
404        return spreadPattern;
405    }

407    /**
408     * Recalculate the spillMap and return the spreadPattern. The cell corresponding to a key in entries will be true,
409     * the cells near each of those will be true if chosen at random from all passable cells adjacent to a
410     * filled (true) cell, and all other cells will be false. This takes a total number of cells to attempt
411     * to fill (the volume parameter), which can be negative to simply fill the whole map, and will fill less if it has
412     * completely exhausted all passable cells from all sources in entries. It uses the values in entries to determine
413     * whether it should advance from a particular key in that step or not; this choice is pseudo-random. If you have
414     * some values that are at or near 1.0 and some values that are closer to 0.0, you should expect the keys for the
415     * higher values to spread further out than the keys associated with lower values.
416     * <br>
417     * If the measurement this Spill uses is anything other than MANHATTAN, you can expect many gaps in the first
418     * filled area.  Subsequent calls to start() with the same entry and a higher volume will expand the area
419     * of the Spill, and are likely to fill any gaps after a few subsequent calls. Increasing the volume slowly
420     * is the best way to ensure that gaps only exist on the very edge if you use a non-MANHATTAN measurement.
421     * <br>
422     * The intended purpose for this method is filling contiguous areas of dungeon with certain terrain features, but it
423     * has plenty of other uses as well.
424     * @param entries key: the first cell for each spiller to spread from. value: the bias toward advancing this key;
425     *                1.0 will always advance, 0.0 will never advance beyond the key, in between will randomly choose
426     * @param volume the total number of cells to attempt to fill; if negative will fill the whole map.
427     * @param impassable a Collection, ideally a Set or GreasedRegion, holding Coord items representing the locations of
428     *                   moving obstacles to a fill that cannot be moved through; null means no obstacles exist.
429     * @return an ArrayList of Points that this will enter, in order starting with entry at index 0, until it
430     * reaches its volume or fills its boundaries completely.
431     */
432    public ArrayList<ArrayList<Coord>> start(OrderedMap<Coord, Double> entries, int volume, Collection<Coord> impassable) {
433        if(!initialized || entries == null) return null;
434        if(impassable == null)
435            impassable = Collections.emptySet();
436        if(volume < 0)
437            volume = Integer.MAX_VALUE;
438        int sz = entries.size();
439        ArrayList<Coord> spillers = new ArrayList<>(entries.keySet());
440        ArrayList<Double>
441                biases = new ArrayList<>(sz);
442        spreadPattern = new ArrayList<>(sz);
443        fresh.clear();
444        filled = 0;
445        boolean hasFresh = false;
446        for (short i = 0; i < sz; i++) {
447            spreadPattern.add(new ArrayList<Coord>(128));
448            OrderedSet<Coord> os = new OrderedSet<Coord>(128);
449            fresh.add(os);
450            Coord c = spillers.get(i);
451            Double d = entries.getAt(i);
452            biases.add(d);
453            if (d <= 0.0001 || c.x < 0 || c.y < 0)
454                continue;
455            spillMap[c.x][c.y] = i;
456            if (!impassable.contains(c)) {
457                os.add(c);
458                hasFresh = true;
459            }
460        }
461
462        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
463
464        while (hasFresh && filled < volume) {
465            hasFresh = false;
466            for (short i = 0; i < spillers.size() && filled < volume; i++) {
467                OrderedSet<Coord> currentFresh = fresh.get(i);
468                if(currentFresh.isEmpty())
469                    continue;
470                else
471                    hasFresh = true;
472                if(rng.nextDouble() < biases.get(i)) {
473                    Coord cell = currentFresh.randomItem(rng);
474                    spreadPattern.get(i).add(cell);
475                    spillMap[cell.x][cell.y] = i;
476                    filled++;
477                    anySpillMap.insert(cell.x, cell.y);
478
479
480                    for (int d = 0; d < dirs.length; d++) {
481                        Coord adj = cell.translate(dirs[d].deltaX, dirs[d].deltaY);
482                        if(!adj.isWithin(width, height))
483                            continue;
484                        double h = heuristic(dirs[d]);
485                        if (physicalMap[adj.x][adj.y] && !anySpillMap.contains(adj.x, adj.y) && !impassable.contains(adj)
486                                && rng.nextDouble(h) <= 1.0) {
487                            setFresh(i, adj);
488                        }
489                    }
490                    currentFresh.remove(cell);
491                    anyFreshMap.remove(cell.x, cell.y);
492                }
493            }
494        }
495        return spreadPattern;
496    }

498    private static final double root2 = Math.sqrt(2.0);
499    private double heuristic(Direction target) {
500        switch (measurement) {
501            case MANHATTAN:
502            case CHEBYSHEV:
503                return 1.0;
504            case EUCLIDEAN:
505                switch (target) {
506                    case DOWN_LEFT:
507                    case DOWN_RIGHT:
508                    case UP_LEFT:
509                    case UP_RIGHT:
510                        return root2;
511                    default:
512                        return  1.0;
513                }
514        }
515        return 1.0;
516    }
