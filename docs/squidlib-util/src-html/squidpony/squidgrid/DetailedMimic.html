016/**
017 * Similar to MimicFill, this class can be used to imitate the style of an existing piece of data, but this works on
018 * more than just booleans; it can produce similar styles of texture (its original use in SynTex), of map, of item
019 * placement, and so on by specifying a different technique for differentiating two int values.
020 * <br>
021 * Two options are now available; the process() method allows the slow analyze() step to be computed once,
022 * before the rest of processing is potentially done many times, but the new neoProcess() method produces
023 * comparable or better results and is drastically faster even without needing analysis. This means
024 * neoProcess(), based on P.F. Harrison's algorithm in SynTex, is strongly recommended now.
025 * <br>
026 * Example results of neoProcess(), with a procedural dungeon first and the mimic versions after it:
027 * https://gist.github.com/tommyettinger/405336fb0fc74838d806c021aabe77da (you may need to click Raw and zoom
028 * out somewhat for it to render well).
029 * <br>
030 * Ported from https://github.com/mxgmn/SynTex by Tommy Ettinger on 6/9/2016.
031 */
032public class DetailedMimic {

035     * Constructor that uses an unseeded RNG and, without any instruction otherwise, assumes the ints this is asked to

044    /**
045     * Constructor that uses an unseeded RNG (effectively a random seed) and the given AestheticDifference. An example
046     * piece of code that implements an AestheticDifference is available in the docs for
047     * {@link AestheticDifference#difference(int, int)}; it is also considered a functional interface if you use Java 8
048     * or newer. You can also use the ready-made implementation {@link AestheticDifference#rgba8888} if you have int
049     * data that represents RGBA8888 colors, which can be obtained from libGDX Colors or SColors in the display module.
050     * @param diff an implementation of the AestheticDifference interface, such as {@link AestheticDifference#rgba8888};
051     *             may be null, but that forces all calls to processing methods to treat discrete as true
052     */
053    public DetailedMimic(AestheticDifference diff)
054    {
055        this(diff, new RNG());
056    }

058    /**
059     * Constructor that uses the given RNG and the given AestheticDifference. An example
060     * piece of code that implements an AestheticDifference is available in the docs for
061     * {@link AestheticDifference#difference(int, int)}; it is also considered a functional interface if you use Java 8
062     * or newer. You can also use the ready-made implementation {@link AestheticDifference#rgba8888} if you have int
063     * data that represents RGBA8888 colors, which can be obtained from libGDX Colors or SColors in the display module.
064     * @param diff an implementation of the AestheticDifference interface, such as {@link AestheticDifference#rgba8888};
065     *             may be null, but that forces all calls to processing methods to treat discrete as true
066     * @param rng an IRNG, such as an RNG, to generate random factors; may be seeded to produce reliable output
067     */
068    public DetailedMimic(AestheticDifference diff, IRNG rng)
069    {
070        random = rng;
071        difference = diff;
072        analyzed = null;
073    }

075    /**
076     * DISCOURAGED; use {@link #neoProcess(int[], int, int, int, int, int, int, boolean)} instead, which doesn't need a
077     * separate analysis step.
078     * Analyzes a sample as a 1D int array and stores the needed info to call
079     * {@link #process(int[], int, int, int, int, int, int, double, boolean)} any number of times later on without
080     * recalculating some heavy-weight information.
081     * @param sample a 1D array of ints that can be compared by the AestheticDifference this uses (or any ints if
082     *               discrete is true)
083     * @param width the width of the area in sample this should use (sample can be interpreted as different shapes)
084     * @param height the height of the area in sample this should use (sample can be interpreted as different shapes)
085     * @param detailLevel how much detail to try for; if 0 or less this does nothing, 2 works well in general
086     * @param proximity how far away to consider cells as affecting another; 3 works well
087     * @param discrete false if this can produce ints other than those in the input; true if it uses a fixed set
088     */
089    public void analyze(int[] sample, int width, int height, int detailLevel, int proximity, boolean discrete)
090    {
091        discrete |= (difference == null);
092        if(detailLevel > 0)
093            analysis(sample, width, height, detailLevel, proximity, discrete);
094    }
095    /**
096     * DISCOURAGED; use {@link #neoProcess(int[], int, int, int, int, int, int, boolean)} instead, which doesn't need a
097     * separate analysis step.
098     * Processes a sample as a 1D int array and returns a different 1D int array that mimics the input. If the last time
099     * this was called used the same sample, sampleWidth, and sampleHeight parameters, or if
100     * {@link #analyze(int[], int, int, int, int, boolean)} was called with its width equal to sampleWidth and its
101     * height equal to sampleHeight, then this doesn't need to perform as many expensive calculations.
102     * @param sample a 1D array of ints that can be compared by the AestheticDifference this uses (or any ints if
103     *               discrete is true)
104     * @param sampleWidth the width of the area in sample this should use (sample can be interpreted as different shapes)
105     * @param sampleHeight the height of the area in sample this should use (sample can be interpreted as different shapes)
106     * @param targetWidth the desired width of the output
107     * @param targetHeight the desired height of the output
108     * @param detailLevel how much detail to try for; if 0 or less this doesn't perform analysis and has somewhat lower
109     *                    quality, but 2 works well in general
110     * @param proximity how far away to consider cells as affecting another; 3 works well
111     * @param temperature a level of unpredictability in the output relative to the input; must be greater than 0
112     * @param discrete false if this can produce ints other than those in the input; true if it uses a fixed set
113     * @return a new 1D int array that can be interpreted as having targetWidth and targetHeight, and mimics sample
114     */
115    public int[] process(int[] sample, int sampleWidth, int sampleHeight, int targetWidth, int targetHeight,
116                         int detailLevel, int proximity, double temperature, boolean discrete)
117    {
118        discrete |= (difference == null);
119        if(detailLevel > 0)
120        {
121            if(analyzed == null || analyzed.length != sampleWidth || analyzed.length == 0 ||
122                    analyzed[0].length != sampleHeight)
123                analyze(sample, sampleWidth, sampleHeight, detailLevel, proximity, discrete);
124            return coherentSynthesis(sample, sampleWidth, sampleHeight, analyzed, proximity,
125                    targetWidth, targetHeight, temperature, discrete);
126        }
127        else {
128            return fullSynthesis(sample, sampleWidth, sampleHeight, proximity,
129                    targetWidth, targetHeight, temperature, discrete);
130        }
131    }

133    /**
134     * Processes a 1D int array representing 2D storage of values that can be compared by this object's
135     * AestheticDifference (or any values if that is null or discrete is true), and returns a 1D array representing data
136     * with potentially different dimensions but similar appearance to sample.
137     * @param sample a 1D array of ints that can be compared by the AestheticDifference this uses (or any ints if
138     *               discrete is true)
139     * @param sampleWidth the width of the area in sample this should use (sample can be interpreted as different shapes)
140     * @param sampleHeight the height of the area in sample this should use (sample can be interpreted as different shapes)
141     * @param targetWidth the desired width of the output
142     * @param targetHeight the desired height of the output
143     * @param detailLevel how much detail to try for; here this will always be treated as at least 1
144     * @param proximity how far away to consider cells as affecting another; 3 works well
145     * @param discrete false if this can produce ints other than those in the input; true if it uses a fixed set
146     * @return a new 1D int array that can be interpreted as having targetWidth and targetHeight, and mimics sample
147     */
148    public int[] neoProcess(int[] sample, int sampleWidth, int sampleHeight, int targetWidth, int targetHeight,
149                         int detailLevel, int proximity, boolean discrete)
150    {
151            return reSynthesis(sample, sampleWidth, sampleHeight, proximity, 20,
152                    Math.max(1, detailLevel), discrete || (difference == null), targetWidth, targetHeight);
153    }

159    private void analysis(int[] bitmap, int width, int height, int K, int N, boolean indexed)
160    {
161        int area = width * height;
162        analyzed = new int[area][];
163        OrderedSet<Integer> points = new OrderedSet<>(area);
164        for (int i = 0; i < area; i++) points.add(i);
165
166        double[] similarities = new double[area * area];
167        for (int i = 0; i < area; i++) for (int j = 0; j < area; j++)
168            similarities[i * area + j] = similarities[j * area + i] != 0 ? similarities[j * area + i] :
169                    similarity(i, bitmap, width, height, j, bitmap, width, height, N, null, indexed);
170
171        for (int i = 0; i < area; i++)
172        {
173            analyzed[i] = new int[K];
174            OrderedSet<Integer> copy = new OrderedSet<>(points);
175
176            analyzed[i][0] = i;
177            copy.remove(i);
178
179            for (int k = 1; k < K; k++)
180            {
181                double max = -10000;
182                int argmax = -1;
183
184                for(Integer p : copy)
185                {
186                    double s = similarities[i * area + p];
187                    if (s > max)
188                    {
189                        max = s;
190                        argmax = p;
191                    }
192                }
193
194                analyzed[i][k] = argmax;
195                copy.remove(argmax);
196            }
197        }
198    }

200    private int[] coherentSynthesis(int[] sample, int SW, int SH, int[][] sets, int N, int OW, int OH, double t, boolean indexed)
201    {
202        int[] result = new int[OW * OH];
203        Integer[] origins = new Integer[OW * OH];
204        boolean[][] mask = new boolean[SW][SH], cleanMask = new boolean[SW][SH];
205
206        for (int i = 0; i < OW * OH; i++)
207        {
208            int x = i % OW, y = i / OW;
209            IntDoubleOrderedMap candidates = new IntDoubleOrderedMap();
210            ArrayTools.insert(cleanMask, mask, 0, 0);
211
212            for (int dy = -1; dy <= 1; dy++){
213                for (int dx = -1; dx <= 1; dx++)
214                {
215                    int sx = (x + dx + OW) % OW, sy = (y + dy + OH) % OH;
216                    Integer origin = origins[sy * OW + sx];
217                    if ((dx != 0 || dy != 0) && origin != null)
218                    {
219                        for (int pi = 0, p; pi < sets[origin].length; pi++)
220                        {
221                            p = sets[origin][pi];
222                            int ox = (p % SW - dx + SW) % SW, oy = (p / SW - dy + SH) % SH;
223                            double s = similarity(oy * SW + ox, sample, SW, SH, i, result, OW, OH, N, origins, indexed);
224
225                            if (!mask[ox][oy]) candidates.put(ox + oy * SW, Math.pow(100, s / t));
226                            mask[ox][oy] = true;
227                        }
228                    }
229                }
230            }
231
232            int shifted = candidates.isEmpty() ? random.nextInt(SW) + random.nextInt(SH) * SW : weightedRandom(candidates, random.nextDouble());
233            origins[i] = shifted;
234            result[i] = sample[shifted];
235        }
236
237        return result;
238    }

240    private int[] fullSynthesis(int[] sample, int SW, int SH, int N, int OW, int OH, double t, boolean indexed)
241    {
242        int[] result = new int[OW * OH];
243        Integer[] origins = new Integer[OW * OH];
244
245        if (!indexed) for (int y = 0; y < OH; y++) {
246            for (int x = 0; x < OW; x++){
247                if (y + N >= OH)
248                {
249                    result[x + y * OW] = sample[random.nextInt(SW * SH)];
250                    origins[x + y * OW] = -1;
251                }
252            }
253        }
254
255        for (int i = 0; i < OW * OH; i++)
256        {
257            double[] candidates = new double[SW * SH];
258            double max = -10000;
259            int argmax = -1;
260
261            for (int j = 0; j < SW * SH; j++)
262            {
263                double s = similarity(j, sample, SW, SH, i, result, OW, OH, N, origins, indexed);
264                if (s > max)
265                {
266                    max = s;
267                    argmax = j;
268                }
269
270                if (indexed) candidates[j] = Math.pow(100.0, s / t);
271            }
272
273            if (indexed) argmax = weightedRandom(candidates, random.nextDouble());
274            result[i] = sample[argmax];
275            origins[i] = -1;
276        }
277
278        return result;
279    }

282    private int[] reSynthesis(int[] sample, int SW, int SH, int N, int M, int polish, boolean indexed, int OW, int OH)
283    {
284        IntVLA colors = new IntVLA();
285        int[] indexedSample = new int[Math.min(SW * SH, sample.length)];
286        for (int j = 0; j < indexedSample.length; j++)
287        {
288            int color = sample[j];
289
290            int i = 0;
291            for (int cn = 0; cn < colors.size; cn++)
292            {
293                if(colors.get(cn) == color) break;
294                i++;
295            }
296
297            if (i == colors.size) colors.add(color);
298            indexedSample[j] = i;
299        }
300
301        int colorsNumber = colors.size;
302
303        double[][] colorMetric = null;
304        if (!indexed && colorsNumber <= 1024)
305        {
306            colorMetric = new double[colorsNumber][colorsNumber];
307            for (int x = 0; x < colorsNumber; x++)
308            {
309                for (int y = 0; y < colorsNumber; y++)
310                {
311                    int cx = colors.get(x), cy = colors.get(y);
312                    colorMetric[x][y] = difference.difference(cx, cy);
313                }
314            }
315        }
316
317        int[] origins = new int[OW * OH];
318        for (int i = 0; i < origins.length; i++) origins[i] = -1;
319
320        int[] shuffle = new int[OW * OH];
321        for (int i = 0; i < shuffle.length; i++)
322        {
323            int j = random.nextInt(i + 1);
324            if (j != i) shuffle[i] = shuffle[j];
325            shuffle[j] = i;
326        }
327
328        for (int round = 0; round <= polish; round++) for (int counter = 0; counter < shuffle.length; counter++)
329        {
330            int f = shuffle[counter];
331            int fx = f % OW, fy = f / OW;
332            int neighborsNumber = round > 0 ? 8 : Math.min(8, counter);
333            int neighborsFound = 0;
334
335            int[] candidates = new int[neighborsNumber + M];
336
337            if (neighborsNumber > 0)
338            {
339                int[] neighbors = new int[neighborsNumber];
340                int[] x = new int[4], y = new int[4];
341
342                for (int radius = 1; neighborsFound < neighborsNumber; radius++)
343                {
344                    x[0] = fx - radius;
345                    y[0] = fy - radius;
346                    x[1] = fx - radius;
347                    y[1] = fy + radius;
348                    x[2] = fx + radius;
349                    y[2] = fy + radius;
350                    x[3] = fx + radius;
351                    y[3] = fy - radius;
352
353                    for (int k = 0; k < 2 * radius; k++)
354                    {
355                        for (int d = 0; d < 4; d++)
356                        {
357                            x[d] = (x[d] + 10 * OW) % OW;
358                            y[d] = (y[d] + 10 * OH) % OH;
359
360                            if (neighborsFound >= neighborsNumber) continue;
361                            int point = x[d] + y[d] * OW;
362                            if (origins[point] != -1)
363                            {
364                                neighbors[neighborsFound] = point;
365                                neighborsFound++;
366                            }
367                        }
368
369                        y[0]++;
370                        x[1]++;
371                        y[2]--;
372                        x[3]--;
373                    }
374                }
375
376
377                for (int n = 0; n < neighborsNumber; n++)
378                {
379                    int cx = (origins[neighbors[n]] + (f - neighbors[n]) % OW + 100 * SW) % SW;
380                    int cy = (origins[neighbors[n]] / SW + f / OW - neighbors[n] / OW + 100 * SH) % SH;
381                    candidates[n] = cx + cy * SW;
382                }
383            }
384
385            for (int m = 0; m < M; m++) candidates[neighborsNumber + m] = random.nextInt(SW * SH);
386
387            double max = -1E+10;
388            int argmax = -1;
389
390            for (int c = 0; c < candidates.length; c++)
391            {
392                double sum = random.nextDouble(0.000001);
393                int ix = candidates[c] % SW, iy = candidates[c] / SW, jx = f % OW, jy = f / OW;
394                int SX, SY, FX, FY, S, F;
395                int origin;
396
397                for (int dy = -N; dy <= N; dy++)
398                {
399                    for (int dx = -N; dx <= N; dx++)
400                    {
401                        if (dx != 0 || dy != 0)
402                        {
403                            SX = ix + dx;
404                            if (SX < 0) SX += SW;
405                            else if (SX >= SW) SX -= SW;
406
407                            SY = iy + dy;
408                            if (SY < 0) SY += SH;
409                            else if (SY >= SH) SY -= SH;
410
411                            FX = jx + dx;
412                            if (FX < 0) FX += OW;
413                            else if (FX >= OW) FX -= OW;
414
415                            FY = jy + dy;
416                            if (FY < 0) FY += OH;
417                            else if (FY >= OH) FY -= OH;
418
419                            S = SX + SY * SW;
420                            F = FX + FY * OW;
421
422                            origin = origins[F];
423                            if (origin != -1)
424                            {
425                                if (indexed) sum += sample[origin] == sample[S] ? 1 : -1;
426                                else if (colorMetric != null) sum += colorMetric[indexedSample[origin]][indexedSample[S]];
427                                else sum += difference.difference(sample[origin], sample[S]);
428                            }
429                        }
430                    }
431                }
432
433                if (sum >= max)
434                {
435                    max = sum;
436                    argmax = candidates[c];
437                }
438            }
439
440            origins[f] = argmax;
441        }
442
443        int[] result = new int[OW * OH];
444        for (int i = 0; i < result.length; i++) result[i] = sample[origins[i]];
445        return result;
446    }

448    private double similarity(int i1, int[] b1, int w1, int h1, int i2, int[] b2, int w2, int h2, int N, Integer[] origins, boolean indexed)
449    {
450        double sum = 0;
451        int x1 = i1 % w1, y1 = i1 / w1, x2 = i2 % w2, y2 = i2 / w2;
452
453        for (int dy = -N; dy <= 0; dy++) for (int dx = -N; (dy < 0 && dx <= N) || (dy == 0 && dx < 0); dx++)
454        {
455            int sx1 = (x1 + dx + w1) % w1, sy1 = (y1 + dy + h1) % h1;
456            int sx2 = (x2 + dx + w2) % w2, sy2 = (y2 + dy + h2) % h2;
457
458            int c1 = b1[sx1 + sy1 * w1];
459            int c2 = b2[sx2 + sy2 * w2];
460
461            if (origins == null || origins[sy2 * w2 + sx2] != null)
462            {
463                if (indexed)
464                    sum += c1 == c2 ? 1 : -1;
465                else
466                    sum -= difference.difference(c1, c2);
467                    /*
468                    Color C1 = Color.FromArgb(c1), C2 = Color.FromArgb(c2);
469                    sum -= (double)((C1.R - C2.R) * (C1.R - C2.R) + (C1.G - C2.G) * (C1.G - C2.G) + (C1.B - C2.B) * (C1.B - C2.B)) / 65536.0;
470                    */
471            }
472        }
473
474        return sum;
475    }

477    static int weightedRandom(double[] array, double r)
478    {
479        double sum = 0;
480        for (int j = 0; j < array.length; j++)
481            sum += Math.max(1.0, array[j]);
482
483        for (int j = 0; j < array.length; j++)
484            array[j] /= sum;
485
486        int i = 0;
487        double x = 0;
488
489        while (i < array.length)
490        {
491            x += array[i];
492            if (r <= x) return i;
493            i++;
494        }
495
496        return 0;
497    }
498
499    static int weightedRandom(IntDoubleOrderedMap dic, double r) {
500        int[] ints = dic.keySet().toIntArray();
501        double[] doubles =  dic.values().toDoubleArray();
502        return ints[weightedRandom(doubles, r)];
503    }

505    /**
506     * Utility method to produce 1D int arrays this can process when discrete is true or difference is null.
507     * @param map a 2D char array
508     * @return an int array that can be used as a sample
509     */
510    public static int[] convertCharToInt(char[][] map)
511    {
512        int w = map.length, h = map[0].length;
513        int[] result = new int[w * h];
514        for (int x = 0; x < w; x++) {
515            for (int y = 0; y < h; y++) {
516                result[x * h + y] = map[x][y];
517            }
518        }
519        return result;
520    }

522    /**
523     * Utility method that takes a 1D int array that represents chars (such as a sample produced by
524     * {@link #convertCharToInt(char[][])} or, more likely, the result of processing such a sample with this class) and
525     * returns a 2D char array with the requested width and height (which should match the targetWidth and targetHeight
526     * given during processing).
527     * @param arr a 1D int array that represents char values
528     * @param w the width that arr can be interpreted as; should probably match the targetWidth given in processing
529     * @param h the height that arr can be interpreted as; should probably match the targetHeight given in processing
530     * @return a 2D char array with the given width and height, probably filled with the data from arr
531     */
532    public static char[][] convertIntToChar(int[] arr, int w, int h)
533    {
534        char[][] result = new char[w][h];
535        if(arr == null)
536            return result;
537        for (int x = 0; x < w; x++) {
538            for (int y = 0; y < h; y++) {
539                if(x * h + y >= arr.length)
540                    return result;
541                result[x][y] = (char) arr[x * h + y];
542            }
543        }
544        return result;
545    }
