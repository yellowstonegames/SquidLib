009/**
010 * An implementation of the Wave Function Collapse algorithm for placing tiles in a way that satisfies constraints; the
011 * name both sounds like Wave Function and loosely describes how the algorithm works. All tiles start off in an
012 * indeterminate state, and as tiles are placed the states of their neighbors become more certain; the "Collapse" in
013 * Wave Function Collapse refers to the quantum-mechanics term for a decline in unpredictability as a superposition is
014 * measured, but here we use the term "Waffle" for its meaning as switching between positions or viewpoints until forced
015 * into a concrete state. This is meant to act like {@link DetailedMimic} or maybe {@link MimicFill}, but only requires
016 * tiles to be given to it with specific constraints on placement instead of requiring an existing design to mimic.
017 * <br>
018 * Massive credit goes to ExUtumno/mxgmn for developing the amazing Wave Function Collapse algorithm, which this code is
019 * closely based off of, but is not at all a verbatim copy. You can find out more at mxgmn's primary repository for WFC
020 * code, https://github.com/mxgmn/WaveFunctionCollapse , and see the incredible things more creative minds than my own
021 * have come up with using WFC or modifications on it. This file only implements the Simple Tiled Model for now.
022 * Created by Tommy Ettinger on 12/22/2016.
023 */

038    protected Boolean observe()
039    {
040        double min = 1E+3, sum, mainSum, logSum, noise, entropy;
041        int argminx = -1, argminy = -1, amount;
042        boolean[] w;
043
044        for (int x = 0; x < FMX; x++) for (int y = 0; y < FMY; y++)
045        {
046            //if (OnBoundary(x, y)) continue;
047
048            //w = wave[x][y];
049            amount = 0;
050            sum = 0;
051
052            for (int t = 0; t < T; t++) if (wave[t].contains(x, y))
053            {
054                amount += 1;
055                sum += stationary[t];
056            }
057
058            if (sum == 0) return false;
059
060            noise = random.nextDouble(1E-6);
061
062            if (amount == 1) entropy = 0;
063            else if (amount == T) entropy = logT;
064            else
065            {
066                mainSum = 0;
067                logSum = Math.log(sum);
068                for (int t = 0; t < T; t++) if (wave[t].contains(x, y)) mainSum += stationary[t] * logProb[t];
069                entropy = logSum - mainSum / sum;
070            }
071
072            if (entropy > 0 && entropy + noise < min)
073            {
074                min = entropy + noise;
075                argminx = x;
076                argminy = y;
077            }
078        }
079
080        if (argminx == -1 && argminy == -1)
081        {
082            observed = new int[FMX][FMY];
083            for (int x = 0; x < FMX; x++)
084            {
085                for (int y = 0; y < FMY; y++) for (int t = 0; t < T; t++) if (wave[t].contains(x, y))
086                {
087                    observed[x][y] = t;
088                    break;
089                }
090            }
091
092            return true;
093        }
094
095        double[] distribution = new double[T];
096        for (int t = 0; t < T; t++) distribution[t] = wave[t].contains(argminx, argminy) ? stationary[t] : 0;
097        int r = DetailedMimic.weightedRandom(distribution, random.nextDouble());
098        for (int t = 0; t < T; t++) wave[t].set(t == r, argminx, argminy);
099        changes.insert(argminx, argminy);
100
101        return null;
102    }

104    public boolean Run(RNG rng, int limit)
105    {
106        logT = Math.log(T);
107        logProb = new double[T];
108        for (int t = 0; t < T; t++) logProb[t] = Math.log(stationary[t]);
109
110        Clear();
111
112        random = rng;
113
114        for (int l = 0; l < limit || limit == 0; l++)
115        {
116            Boolean result = observe();
117            if (result != null) return result;
118            while (propagate());
119        }
120
121        return true;
122    }

124    protected void Clear()
125    {
126        for (int t = 0; t < T; t++) wave[t].allOn();
127        changes.clear();
128    }

367    protected boolean propagate()
368    {
369        boolean change = false, b;
370        for (int x2 = 0; x2 < FMX; x2++) {
371            for (int y2 = 0; y2 < FMY; y2++) {
372                for (int d = 0; d < 4; d++)
373                {
374                    int x1 = x2, y1 = y2;
375                    if (d == 0)
376                    {
377                        if (x2 == 0)
378                        {
379                            if (!periodic) continue;
380                            else x1 = FMX - 1;
381                        }
382                        else x1 = x2 - 1;
383                    }
384                    else if (d == 1)
385                    {
386                        if (y2 == FMY - 1)
387                        {
388                            if (!periodic) continue;
389                            else y1 = 0;
390                        }
391                        else y1 = y2 + 1;
392                    }
393                    else if (d == 2)
394                    {
395                        if (x2 == FMX - 1)
396                        {
397                            if (!periodic) continue;
398                            else x1 = 0;
399                        }
400                        else x1 = x2 + 1;
401                    }
402                    else
403                    {
404                        if (y2 == 0)
405                        {
406                            if (!periodic) continue;
407                            else y1 = FMY - 1;
408                        }
409                        else y1 = y2 - 1;
410                    }
411
412                    if (!changes.contains(x1, y1)) continue;
413
414                    for (int t2 = 0; t2 < T; t2++)
415                    {
416                        if (!wave[t2].contains(x2, y2)) continue;
417
418                        b = false;
419                        int[] prop = propagator[d][t2];
420                        for (int i1 = 0; i1 < prop.length && !b; i1++)
421                        {
422                            b = wave[prop[i1]].contains(x1, y1);
423                        }
424
425                        if (!b)
426                        {
427                            changes.insert(x2, y2);
428                            change = true;
429                            wave[t2].remove(x2, y2);
430                        }
431                    }
432                }
433            }
434        }
435
436        return change;
437    }

461        public WaffleTile(char[] data, char routerID, int w, int h, int weight, char symmetry, String name)
462        {
463            this.data = new char[data.length];
464            System.arraycopy(data, 0, this.data, 0, data.length);
465            this.w = w;
466            this.h = h;
467            this.weight = weight;
468            this.symmetry = symmetry;
469            this.name = name;
470            this.routerID = routerID;
471
472            if((routerID & 3) != 0 && (routerID & 12) != 0)
473                indexer = Router.Generator.chain(w, h,
474                        Router.Generator.rotate(w, h, routerID & 3),
475                        Router.Generator.flip(w, h, (routerID & 4) != 0, (routerID & 8) != 0));
476            else if((routerID & 3) != 0)
477                indexer = Router.Generator.rotate(w, h, routerID & 3);
478            else if((routerID & 12) != 0)
479                indexer = Router.Generator.flip(w, h, (routerID & 4) != 0, (routerID & 8) != 0);
480            else
481                indexer = Router.Generator.simple(w, h);

483        }

492            symmetry = encoded.charAt(data.length + 25);

505        public WaffleTile(char[][] map, char symmetry)
506        {
507            w = map.length;
508            h = map[0].length;
509            this.symmetry = symmetry;
510            data = new char[w * h];

553            this.symmetry = symmetry;
554            data = new char[w * h];
555            for (int x = 0; x < w; x++) {
556                for (int y = 0; y < h; y++) {
557                    data[x + y * w] = map[x][y];
558                }

562                indexer = Router.Generator.chain(w, h,

564                        Router.Generator.flip(w, h, flipX, flipY));

581            if(flipY)

602        public WaffleTile rotateCopy(int rotation)
