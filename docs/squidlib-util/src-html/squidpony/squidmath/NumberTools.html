003/**
004 * Various numeric functions that are important to performance but need alternate implementations on GWT to obtain it.
005 * Super-sourced on GWT, but most things here are direct calls to JDK methods when on desktop or Android.
006 * Some of this code makes use of "creative" bit manipulation of floats and doubles, which can sometimes allow uncommon
007 * input-to-output patterns (as in {@link #bounce(float)}), or even can yield a performance boost (compare
008 * {@link #zigzag(float)} to using modulus to accomplish the same results). The bit manipulation has good performance on
009 * GWT thanks to JS typed arrays, which are well-supported now across all recent browsers and have fallbacks in GWT in
010 * the unlikely event of a browser not supporting them.
011 */
012public class NumberTools {
013    /**
014     * Identical to {@link Double#doubleToLongBits(double)} on desktop; optimized on GWT. When compiling to JS via GWT,
015     * there is no way to distinguish NaN values with different bits but that are still NaN, so this doesn't try to
016     * somehow permit that. Uses JS typed arrays on GWT, which are well-supported now across all recent browsers and
017     * have fallbacks in GWT in the unlikely event of a browser not supporting them. JS typed arrays support double, but
018     * not long, so this needs to compose a long from two ints, which means the double-to/from-long conversions aren't
019     * as fast as float-to/from-int conversions.
020     * @param value a {@code double} floating-point number.
021     * @return the bits that represent the floating-point number.
022     */
023    public static long doubleToLongBits(final double value)
024    {
025        return Double.doubleToLongBits(value);
026    }

056     * Converts {@code value} to a long and gets the lower 32 bits of that long, as an int.

075    /**
076     * Converts {@code value} to a long and gets the XOR of its upper and lower 32-bit sections. Useful for numerical
077     * code where a 64-bit double needs to be reduced to a 32-bit value with some hope of keeping different doubles
078     * giving different ints.
079     * @param value a {@code double} precision floating-point number.
080     * @return the XOR of the lower and upper halves of the bits that represent the floating-point number.
081     */
082    public static int doubleToMixedIntBits(final double value)
083    {
084        final long l = Double.doubleToLongBits(value);
085        return (int)(l & 0xFFFFFFFFL) ^ (int)(l >>> 32);
086    }

088    /**
089     * Makes a modified version of value that uses the specified bits (up to 12) for its exponent and sign.
090     * Meant for some specific cases, like adjusting the exponent on an unknown double to the 1.0 to 2.0 range (which
091     * would pass 0x3ff for exponentBits). If you have a double from 1.0 to 2.0, you can subtract 1.0 from it to get the
092     * often-desirable 0.0-1.0 range. Other common cases are 0x400, which adjusts to between 2.0 and 4.0 (subtracting
093     * 3.0 from this gives the -1.0 to 1.0 range, useful for noise), and 0xBFF, which adjusts to between -2.0 and -1.0.
094     * For the last case, you might think that -0x3ff would work, but sadly it doesn't. You can use
095     * {@code exponentBits |= 0x800} to set the sign bit to negative, or {@code exponentBits &= 0x7ff} for positive.
096     * @param value a double that will have its sign and exponent set to the specified bits
097     * @param exponentBits the bits to use for the sign and exponent section of the returned modification of value
098     * @return the double produced by keeping the significand of value but changing its exponent and sign as given
099     */
100    public static double setExponent(final double value, final int exponentBits)
101    {
102        return Double.longBitsToDouble((Double.doubleToLongBits(value) & 0xfffffffffffffL) | ((long) exponentBits << 52));
103    }

106     * Gets an 8-bit section of the given double {@code value}, using {@code whichByte} to select whether this should
107     * return byte 0 (least significant), 1, 2, and so on up to 7 (most significant).
108     * @param value a float
109     * @param whichByte an int that will be used to select the byte to take from value (any int is allowed, only the bottom 3 bits are used to select)
110     * @return the selected byte from the given float
111     */
112    public static byte getSelectedByte(final double value, final int whichByte)
113    {
114        return (byte)(Double.doubleToLongBits(value) >>> ((whichByte & 7) << 3));
115    }

117    /**
118     * Like {@link #getSelectedByte(double, int)}, this sets the byte at a selected position in the int representation of
119     * a double, then returns the double produced by the bit change. Uses {@code whichByte} to select whether this should
120     * set byte 0 (least significant), 1, 2, and so on up to 7 (most significant). {@code newValue} is a byte.
121     * @param value a double
122     * @param whichByte an int that will be used to select the byte to take from value (any int is allowed, only the bottom 3 bits are used to select)
123     * @param newValue a byte that will be placed into the returned double's bits at the selected position
124     * @return a double that results from changing the bits at the selected position to match newValue
125     */
126    public static double setSelectedByte(final double value, final int whichByte, final byte newValue)
127    {
128        return Double.longBitsToDouble((Double.doubleToLongBits(value) & ~(255 << ((whichByte & 7) << 3)))
129                | ((newValue & 255) << ((whichByte & 7) << 3)));
130    }

132    /**
133     * Very limited-use; takes any double and produces a double in the -1.0 to 1.0 range, with similar inputs producing
134     * close to a consistent rate of up and down through the range. This is meant for noise, where it may be useful to
135     * limit the amount of change between nearby points' noise values and prevent sudden "jumps" in noise value.
136     * @param value any double
137     * @return a double from -1.0 (inclusive) to 1.0 (exclusive)
138     */
139    public static double bounce(final double value)
140    {
141        final long s = Double.doubleToLongBits(value);
142        return Double.longBitsToDouble(((s ^ -((s & 0x8000000000000L)>>51)) & 0xfffffffffffffL)
143                | 0x4010000000000000L) - 5.0;
144    }

172    /**
173     * Very limited-use; takes the significand bits of a double, represented as a pair of ints {@code valueLow} and
174     * {@code valueHigh}, using all bits in valueLow and the least-significant 20 bits of valueHigh, and
175     * produces a double in the -1.0 to 1.0 range, with similar inputs producing close to a consistent rate of up and
176     * down through the range. This is meant for noise, where it may be useful to limit the amount of change between
177     * nearby points' noise values and prevent sudden "jumps" in noise value.
178     * @param valueLow any int; all bits will be used as the less-significant bits of the significand
179     * @param valueHigh any int; only the bottom 20 bits will be used as the more-significant bits of the significand
180     * @return a double from -1.0 (inclusive) to 1.0 (exclusive)
181     */
182
183    public static double bounce(final int valueLow, final int valueHigh)
184    {
185        final long s = (((long) valueHigh) << 32 | valueLow);
186        return Double.longBitsToDouble(((s ^ -((s & 0x8000000000000L)>>51)) & 0xfffffffffffffL)
187                | 0x4010000000000000L) - 5.0;
188    }

190    /**
191     * Limited-use; takes any double and produces a double in the -1.0 to 1.0 range, with similar inputs producing
192     * close to a consistent rate of up and down through the range. This is meant for noise, where it may be useful to
193     * limit the amount of change between nearby points' noise values and prevent sudden "jumps" in noise value. It is
194     * very similar to {@link #bounce(double)}, but unlike bounce() this will maintain a continuous rate regardless of
195     * the magnitude of its input. An input of any even number should produce something very close to -1.0, any odd
196     * number should produce something very close to 1.0, and any number halfway between two incremental integers (like
197     * 8.5 or -10.5) should produce 0.0 or a very small fraction.
198     * @param value any double
199     * @return a double from -1.0 (inclusive) to 1.0 (inclusive)
200     */
201    public static double zigzag(final double value)
202    {
203        final long s = Double.doubleToLongBits(value + (value < 0f ? -2.0 : 2.0)), m = (s >>> 52 & 0x7FFL) - 0x400, sm = s << m;
204        return (Double.longBitsToDouble(((sm ^ -((sm & 0x8000000000000L)>>51)) & 0xfffffffffffffL)
205                | 0x4010000000000000L) - 5.0);
206    }

209     * Limited-use; takes any float and produces a float in the -1f to 1f range, with similar inputs producing
210     * close to a consistent rate of up and down through the range. This is meant for noise, where it may be useful to
211     * limit the amount of change between nearby points' noise values and prevent sudden "jumps" in noise value. It is
212     * very similar to {@link #bounce(float)}, but unlike bounce() this will maintain a continuous rate regardless of
213     * the magnitude of its input. An input of any even number should produce something very close to -1f, any odd
214     * number should produce something very close to 1f, and any number halfway between two incremental integers (like

217     * @return a float from -1f (inclusive) to 1f (inclusive)
218     */

220    {

224    }

226    /**
227     * Limited-use; takes any double and produces a double in the -1 to 1 range, with a graph of input to output that
228     * looks much like a sine wave, curving to have a flat slope when given an integer input and a steep slope when the
229     * input is halfway between two integers, smoothly curving at any points between those extremes. This is meant for
230     * noise, where it may be useful to limit the amount of change between nearby points' noise values and prevent both
231     * sudden "jumps" in noise value and "cracks" where a line takes a sudden jagged movement at an angle. It is very
232     * similar to {@link #bounce(double)} and {@link #zigzag(double)}, but unlike bounce() this will maintain its
233     * frequency of returning max or min values, regardless of the magnitude of its input, and unlike zigzag() this
234     * will smooth its path. An input of any even number should produce something very close to -1.0, any odd
235     * number should produce something very close to 1.0, and any number halfway between two incremental integers (like
236     * 8.5 or -10.5) should produce 0.0 or a very small fraction.
237     * @param value any double other than NaN or infinite values
238     * @return a double from -1.0 (inclusive) to 1.0 (inclusive)
239     */
240    public static double sway(final double value)
241    {
242        final long s = Double.doubleToLongBits(value + (value < 0.0 ? -2.0 : 2.0)), m = (s >>> 52 & 0x7FFL) - 0x400, sm = s << m;
243        final double a = (Double.longBitsToDouble(((sm ^ -((sm & 0x8000000000000L)>>51)) & 0xfffffffffffffL)
244                | 0x4000000000000000L) - 2.0);
245        return a * a * a * (a * (a * 6.0 - 15.0) + 10.0) * 2.0 - 1.0;
246    }

377    /**
378     * Get a pseudo-random long from this with {@code splitMix64(z += 0x9E3779B97F4A7C15L)}, where z is a long to use
379     * as state. 0x9E3779B97F4A7C15L can be changed for any odd long if the same number is used across calls.
380     * @param state long, must be changed with each call; {@code splitMix64(z += 0x9E3779B97F4A7C15L)} is recommended
381     * @return a pseudo-random long
382     */
383    public static long splitMix64(long state) {
384        state = ((state >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
385        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
386        return state ^ (state >>> 31);
387    }

389    /**
390     * Generates a pseudo-random double between 0.0 (inclusive) and 1.0 (exclusive) using the given int seed, passing it
391     * once through the (very high-quality and rather fast) {@link LightRNG} algorithm, also called SplitMix64. This
392     * produces a pair of random ints, which this produces a double from using the equivalent of
393     * {@link #longBitsToDouble(long)} or something functionally equivalent on GWT.
394     * <br>
395     * Consider calling this with {@code NumberTools.randomDouble(++seed)} for an optimal period of 2 to the 32 when
396     * repeatedly called, but {@code NumberTools.randomDouble(seed += ODD_INT)} will also work just fine if ODD_INT is
397     * any odd-number integer, positive or negative.
398     * @param seed any int to be used as a seed
399     * @return a pseudo-random double from 0.0 (inclusive) to 1.0 (exclusive)
400     */
401    public static double randomDouble(int seed)
402    {
403        long state = seed * 0x9E3779B97F4A7C15L;
404        state = ((state >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
405        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
406        return Double.longBitsToDouble(((state ^ (state >>> 31)) >>> 12) | 0x3ff0000000000000L) - 1.0;
407    }
408    /**
409     * Generates a pseudo-random float between 0.0f (inclusive) and 1.0f (exclusive) using the given int seed, passing
410     * it once through the (very high-quality and rather fast) {@link LightRNG} algorithm, also called SplitMix64. This
411     * produces a random int, which this produces a float from using {@link #intBitsToFloat(int)} (long)} or something
412     * functionally equivalent on GWT.
413     * <br>
414     * Consider calling this with {@code NumberTools.randomFloat(++seed)} for an optimal period of 2 to the 32 when
415     * repeatedly called, but {@code NumberTools.randomFloat(seed += ODD_INT)} will also work just fine if ODD_INT is
416     * any odd-number integer, positive or negative.
417     * @param seed any int to be used as a seed
418     * @return a pseudo-random float from -1.0f (exclusive) to 1.0f (exclusive)
419     */
420    public static float randomFloat(int seed)
421    {
422        long state = seed * 0x9E3779B97F4A7C15L;
423        state = ((state >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
424        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
425        return Float.intBitsToFloat((int)(state >>> 41) | 0x3f800000) - 1f;
426    }
427    /**
428     * Generates a pseudo-random float between -1.0f (exclusive) and 1.0f (exclusive) using the given int seed, passing
429     * it once through the (very high-quality and rather fast) {@link LightRNG} algorithm, also called SplitMix64. This
430     * produces a random int, which this produces a float from using {@link #intBitsToFloat(int)} (long)} or something
431     * functionally equivalent on GWT. The sign bit of the result is determined by data that is not used by the float
432     * otherwise, and keeps the results almost linear in distribution between -1.0 and 1.0, exclusive for both (0 shows
433     * up twice as often as any single other result, but this shouldn't affect the odds very strongly; it's about a 1 in
434     * 8 million chance of exactly 0 occurring vs. a 1 in 16 million of any other specific float this can produce).
435     * <br>
436     * Consider calling this with {@code NumberTools.randomSignedFloat(++seed)} for an optimal period of 2 to the 32
437     * when repeatedly called, but {@code NumberTools.randomSignedFloat(seed += ODD_INT)} will also work just fine if
438     * ODD_INT is any odd-number integer, positive or negative.
439     * @param seed any int to be used as a seed
440     * @return a pseudo-random float from -1.0f (exclusive) to 1.0f (exclusive)
441     */
442    public static float randomSignedFloat(int seed)
443    {
444        long state = seed * 0x9E3779B97F4A7C15L;
445        state = ((state >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
446        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
447        return (Float.intBitsToFloat((int)(state >>> 40) | 0x3f800000) - 1f) * (state >> 63 | 1L);
448    }

450    /**
451     * Generates a pseudo-random double between -1.0 (exclusive) and 1.0 (exclusive) with a distribution that has a
452     * strong central bias (around 0.0). Uses the given int seed, passing it once through the (very high-quality and
453     * rather fast) {@link LightRNG} algorithm, also called SplitMix64. This produces a pair of random ints, which this
454     * uses to generate a pair of floats between 0.0 (inclusive)and 1.0 (exclusive) using the equivalent of
455     * {@link #intBitsToFloat(int)} or something functionally equivalent on GWT, multiplies the floats, and sets the
456     * sign pseudo-randomly based on an unused bit from earlier.
457     * <br>
458     * Consider calling this with {@code NumberTools.randomFloatCurved(++seed)} for an optimal period of 2 to the 32
459     * when repeatedly called, but {@code NumberTools.randomFloatCurved(seed += ODD_INT)} will also work just fine if
460     * ODD_INT is any odd-number integer, positive or negative.
461     * @param seed any int to be used as a seed
462     * @return a pseudo-random double from -1.0 (exclusive) to 1.0 (exclusive), distributed on a curve centered on 0.0
463     */
464    public static float randomFloatCurved(int seed)
465    {
466        long state = seed * 0x9E3779B97F4A7C15L;
467        state = ((state >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
468        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
469        state ^= (state >>> 31);
470        return (Float.intBitsToFloat((int)(state & 0x7FFFFF) | 0x3f800000) - 1f)
471                * (Float.intBitsToFloat((int)(state >>> 40) | 0x3f800000) - 1f)
472                * (state >> 63 | 1L);
473    }

496    /**
497     * A different kind of determine-like method that expects to be given a random long and produces a random float with
498     * a curved distribution that centers on 0 (where it has a bias) and can (rarely) approach -1f and 1f.
499     * The distribution for the values is similar to Irwin-Hall, and is frequently near 0 but not too-rarely near -1f or
500     * 1f. It cannot produce values greater than or equal to 1f, or less than -1f, but it can produce -1f.
501     * @param start a long, usually random, such as one produced by any RandomnessSource
502     * @return a deterministic float between -1f (inclusive) and 1f (exclusive), that is very likely to be close to 0f
503     */
504    public static float formCurvedFloat(final long start) {
505        return   (intBitsToFloat((int)start >>> 9 | 0x3F000000)
506                + intBitsToFloat(((int)~start & 0x007FFFFF) | 0x3F000000)
507                + intBitsToFloat((int) (start >>> 41) | 0x3F000000)
508                + intBitsToFloat(((int) (~start >>> 32) & 0x007FFFFF) | 0x3F000000)
509                - 3f);
510    }

531     * The distribution for the values is similar to Irwin-Hall, and is frequently near 0 but not too-rarely near -1f or

544    /**
545     * Returns an int value with at most a single one-bit, in the position of the lowest-order ("rightmost") one-bit in
546     * the specified int value. Returns zero if the specified value has no one-bits in its two's complement binary
547     * representation, that is, if it is equal to zero.
548     * <br>
549     * Identical to {@link Integer#lowestOneBit(int)}, but super-sourced to act correctly on GWT. If you have GWT as a
550     * target and do bit manipulation work, double-check everything! An int can be higher than {@link Integer#MAX_VALUE}
551     * or lower than {@link Integer#MIN_VALUE} on GWT, without actually being a long (internally it's a double). This
552     * is especially relevant for the overload of this method that takes and returns a long;
553     * {@link Long#lowestOneBit(long)} does not provide correct results for certain inputs on GWT, such as
554     * -17592186044416L, which it mysteriously returns 0L on, so you should use {@link #lowestOneBit(long)}.
555     * @param num the value whose lowest one bit is to be computed
556     * @return an int value with a single one-bit, in the position of the lowest-order one-bit in the specified value,
557     *         or zero if the specified value is itself equal to zero.
558     */
559    public static int lowestOneBit(int num)
560    {
561        return num & -num;
