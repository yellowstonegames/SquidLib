003import java.util.Arrays;

005/**
006 * This is a port of the public domain Isaac (cryptographic) random number generator to Java, by Bob Jenkins.
007 * It is a RandomnessSource here, so it should generally be used to make an RNG, which has more features.
008 * Isaac32RNG is slower than the non-cryptographic RNGs in SquidLib, but much faster than cryptographic RNGs
009 * that need SecureRandom, and it's compatible with GWT and Android to boot! Isaac32RNG should perform better
010 * than IsaacRNG on GWT, or when you specifically need a large amount of int values to be set using
011 * {@link #fillBlock(int[])}. If you don't need GWT support, then {@link IsaacRNG} will have better properties.
012 * Created by Tommy Ettinger on 8/1/2016.
013 */
014public class Isaac32RNG implements RandomnessSource {
015    private int count;                           /* count through the results in results[] */
016    private int results[];                                /* the results given to the user */
017    private int mem[];                                   /* the internal state */
018    private int a;                                              /* accumulator */
019    private int b;                                          /* the last result */
020    private int c;              /* counter, guarantees cycle is at least 2^^40 */
021
022
023    /* no seed, equivalent to randinit(ctx,FALSE) in C */
024    public Isaac32RNG() {
025        mem = new int[256];
026        results = new int[256];
027        init(false);
028    }
029
030    /* equivalent to randinit(ctx, TRUE) after putting seed in randctx in C */
031    public Isaac32RNG(final int seed[]) {
032        mem = new int[256];
033        results = new int[256];
034        if(seed == null)
035            init(false);
036        else {
037            System.arraycopy(seed, 0, results, 0, Math.min(256, seed.length));
038            init(true);
039        }
040    }
041    /**
042     * Constructs an IsaacRNG with its state filled by the value of seed, run through the LightRNG algorithm.
043     * @param seed any long; will have equal influence on all bits of state
044     */
045    public Isaac32RNG(long seed) {
046        mem = new int[256];
047        results = new int[256];
048        long z;
049        for (int i = 0; i < 256; i++) {
050            z = seed += 0x9E3779B97F4A7C15L;
051            z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
052            z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
053            results[i++] = (int) ((z ^ (z >>> 31)) & 0xffffffffL);
054            results[i] = (int) ((z ^ (z >>> 31)) >>> 32);
055        }
056        init(true);
057    }
058
059    /**
060     * Constructs an IsaacRNG with its state filled by repeated hashing of seed.
061     * @param seed a String that should be exceptionally long to get the best results.
062     */
063    public Isaac32RNG(String seed) {
064        mem = new int[256];
065        results = new int[256];
066        if(seed == null)
067            init(false);
068        else {
069            char[] chars = seed.toCharArray();
070            int slen = chars.length, i = 0;
071            for (; i < 256 && i < slen; i++) {
072                results[i] = CrossHash.Wisp.hash(chars, i, slen);
073            }
074            for (; i < 256; i++) {
075                results[i] = CrossHash.Wisp.hash(results);
076            }
077            init(true);
078        }
079    }
080
081    private Isaac32RNG(Isaac32RNG other)
082    {
083        this(other.results);
084    }

087    /**
088     * Generates 256 results to be used by later calls to next() or nextLong().
089     * This is a fast (not small) implementation.
090     */
091    public final void regen() {
092        int i, j, x, y;
093
094        b += ++c;
095        for (i = 0, j = 128; i < 128; ) {
096            x = mem[i];
097            a = (a ^ a << 13) + mem[j++];
098            mem[i] = y = mem[x >> 2 & 255] + a + b;
099            results[i++] = b = mem[y >> 10 & 255] + x;
100
101            x = mem[i];
102            a = (a ^ a >>> 6) + mem[j++];
103            mem[i] = y = mem[x >> 2 & 255] + a + b;
104            results[i++] = b = mem[y >> 10 & 255] + x;
105
106            x = mem[i];
107            a = (a ^ a << 2) + mem[j++];
108            mem[i] = y = mem[x >> 2 & 255] + a + b;
109            results[i++] = b = mem[y >> 10 & 255] + x;
110
111            x = mem[i];
112            a = (a ^ a >>> 16) + mem[j++];
113            mem[i] = y = mem[x >> 2 & 255] + a + b;
114            results[i++] = b = mem[y >> 10 & 255] + x;
115        }
116
117        for (j = 0; j < 128; ) {
118            x = mem[i];
119            a = (a ^ a << 13) + mem[j++];
120            mem[i] = y = mem[x >> 2 & 255] + a + b;
121            results[i++] = b = mem[y >> 10 & 255] + x;
122
123            x = mem[i];
124            a = (a ^ a >>> 6) + mem[j++];
125            mem[i] = y = mem[x >> 2 & 255] + a + b;
126            results[i++] = b = mem[y >> 10 & 255] + x;
127
128            x = mem[i];
129            a = (a ^ a << 2) + mem[j++];
130            mem[i] = y = mem[x >> 2 & 255] + a + b;
131            results[i++] = b = mem[y >> 10 & 255] + x;
132
133            x = mem[i];
134            a = (a ^ a >>> 16) + mem[j++];
135            mem[i] = y = mem[x >> 2 & 255] + a + b;
136            results[i++] = b = mem[y >> 10 & 255] + x;
137        }
138    }

141    /**
142     * Initializes this IsaacRNG; typically used from the constructor but can be called externally.
143     *
144     * @param flag if true, use data from seed; if false, initializes this to unseeded random state
145     */
146    public final void init(boolean flag) {
147        int i;
148        int a, b, c, d, e, f, g, h;
149        a = b = c = d = e = f = g = h = 0x9e3779b9;                        /* the golden ratio */
150
151        for (i = 0; i < 4; ++i) {
152            a ^= b << 11;
153            d += a;
154            b += c;
155            b ^= c >>> 2;
156            e += b;
157            c += d;
158            c ^= d << 8;
159            f += c;
160            d += e;
161            d ^= e >>> 16;
162            g += d;
163            e += f;
164            e ^= f << 10;
165            h += e;
166            f += g;
167            f ^= g >>> 4;
168            a += f;
169            g += h;
170            g ^= h << 8;
171            b += g;
172            h += a;
173            h ^= a >>> 9;
174            c += h;
175            a += b;
176        }
177
178        for (i = 0; i < 256; i += 8) {              /* fill in mem[] with messy stuff */
179            if (flag) {
180                a += results[i];
181                b += results[i + 1];
182                c += results[i + 2];
183                d += results[i + 3];
184                e += results[i + 4];
185                f += results[i + 5];
186                g += results[i + 6];
187                h += results[i + 7];
188            }
189            a ^= b << 11;
190            d += a;
191            b += c;
192            b ^= c >>> 2;
193            e += b;
194            c += d;
195            c ^= d << 8;
196            f += c;
197            d += e;
198            d ^= e >>> 16;
199            g += d;
200            e += f;
201            e ^= f << 10;
202            h += e;
203            f += g;
204            f ^= g >>> 4;
205            a += f;
206            g += h;
207            g ^= h << 8;
208            b += g;
209            h += a;
210            h ^= a >>> 9;
211            c += h;
212            a += b;
213            mem[i] = a;
214            mem[i + 1] = b;
215            mem[i + 2] = c;
216            mem[i + 3] = d;
217            mem[i + 4] = e;
218            mem[i + 5] = f;
219            mem[i + 6] = g;
220            mem[i + 7] = h;
221        }
222
223        if (flag) {           /* second pass makes all of seed affect all of mem */
224            for (i = 0; i < 256; i += 8) {
225                a += mem[i];
226                b += mem[i + 1];
227                c += mem[i + 2];
228                d += mem[i + 3];
229                e += mem[i + 4];
230                f += mem[i + 5];
231                g += mem[i + 6];
232                h += mem[i + 7];
233                a ^= b << 11;
234                d += a;
235                b += c;
236                b ^= c >>> 2;
237                e += b;
238                c += d;
239                c ^= d << 8;
240                f += c;
241                d += e;
242                d ^= e >>> 16;
243                g += d;
244                e += f;
245                e ^= f << 10;
246                h += e;
247                f += g;
248                f ^= g >>> 4;
249                a += f;
250                g += h;
251                g ^= h << 8;
252                b += g;
253                h += a;
254                h ^= a >>> 9;
255                c += h;
256                a += b;
257                mem[i] = a;
258                mem[i + 1] = b;
259                mem[i + 2] = c;
260                mem[i + 3] = d;
261                mem[i + 4] = e;
262                mem[i + 5] = f;
263                mem[i + 6] = g;
264                mem[i + 7] = h;
265            }
266        }
267
268        regen();
269        count = 256;
270    }
271
272    public final int nextInt() {
273        if (0 == count--) {
274            regen();
275            count = 255;
276        }
277        return results[count];
278    }
279    /**
280     * Generates and returns a block of 256 pseudo-random int values.
281     * @return a freshly-allocated array of 256 pseudo-random ints, with all bits possible
282     */
283    public final int[] nextBlock()
284    {
285        regen();
286        final int[] block = new int[256];
287        System.arraycopy(results, 0, block, 0, 256);
288        count = 0;
289        return block;
290    }
291
292    /**
293     * Generates enough pseudo-random int values to fill {@code data} and assigns them to it.
294     */
295    public final void fillBlock(final int[] data)
296    {
297        int len, i;
298        if(data == null || (len = data.length) == 0) return;
299        for (i = 0; len > 256; i += 256, len -= 256) {
300            regen();
301            System.arraycopy(results, 0, data, i, 256);
302        }
303        regen();
304        System.arraycopy(results, 0, data, i, len);
305        count = len & 255;
306    }

308    /**
309     * Generates enough pseudo-random float values between 0f and 1f to fill {@code data} and assigns them to it.
310     * Inclusive on 0f, exclusive on 1f. Intended for cases where you need some large source of randomness to be checked
311     * later repeatedly, such as how permutation tables are used in Simplex noise.
312     */
313    public final void fillFloats(final float[] data)
314    {
315        int len, n;
316        if(data == null || (len = data.length) == 0) return;
317        for (int i = 0; i < len; i++) {
318            data[i] = NumberTools.intBitsToFloat(((n = nextInt()) >>> 9 ^ (n & 0x7fffff)) | 0x3f800000) - 1f;
319        }
320    }
321
322    /**
323     * Generates enough pseudo-random float values between -1f and 1f to fill {@code data} and assigns them to it.
324     * Inclusive on -1f, exclusive on 1f. Intended for cases where you need some large source of randomness to be
325     * checked later repeatedly, such as how permutation tables are used in Simplex noise.
326     */
327    public final void fillSignedFloats(final float[] data)
328    {
329        int len, n;
330        if(data == null || (len = data.length) == 0) return;
331        for (int i = 0; i < len; i++) {
332            data[i] = NumberTools.intBitsToFloat(((n = nextInt()) >>> 9 ^ (n & 0x7fffff)) | 0x40000000) - 3f;
333        }
334    }

336    @Override
337    public final int next( int bits ) {
338        return nextInt() >>> 32 - bits;
339    }
340
341    /**
342     * Using this method, any algorithm that needs to efficiently generate more
343     * than 32 bits of random data can interface with this randomness source.
344     * <p>
345     * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
346     *
347     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
348     */
349    @Override
350    public long nextLong() {
351        return (nextInt() & 0xffffffffL) | (nextInt() & 0xffffffffL) << 32;
352    }

355    /**
356     * Produces another RandomnessSource, but the new one will not produce the same data as this one.
357     * This is meant to be a "more-secure" generator, so this helps reduce the ability to guess future
358     * results from a given sequence of output.
359     * @return another RandomnessSource with the same implementation but no guarantees as to generation
360     */
361    @Override
362    public RandomnessSource copy() {
363        return new Isaac32RNG(results);
364    }
365
366    @Override
367    public boolean equals(Object o) {
368        if (this == o) return true;
369        if (o == null || getClass() != o.getClass()) return false;
370
371        Isaac32RNG isaac32RNG = (Isaac32RNG) o;
372
373        if (count != isaac32RNG.count) return false;
374        if (a != isaac32RNG.a) return false;
375        if (b != isaac32RNG.b) return false;
376        if (c != isaac32RNG.c) return false;
377        if (!Arrays.equals(results, isaac32RNG.results)) return false;
378        return Arrays.equals(mem, isaac32RNG.mem);
379    }
380
381    @Override
382    public int hashCode() {
383        int result = count;
384        result = 31 * result + CrossHash.Wisp.hash(results);
385        result = 31 * result + CrossHash.Wisp.hash(mem);
386        result = 31 * result + a;
387        result = 31 * result + b;
388        result = 31 * result + c;
389        return result;
390    }
391
392    @Override
393    public String toString()
394    {
395        return "Isaac32RNG with a hidden state (id is " + System.identityHashCode(this) + ')';
396    }
