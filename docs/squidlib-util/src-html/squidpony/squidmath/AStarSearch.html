010/**
011 * Performs A* search.
012 *
013 * A* is a best-first search algorithm for pathfinding. It uses a heuristic
014 * value to reduce the total search space. If the heuristic is too large then
015 * the optimal path is not guaranteed to be returned.
016 * <br>
017 * This implementation outperforms DijkstraMap (it can be about 8x faster) on relatively short
018 * paths (8 or less cells), but when an especially long path is requested (20 or more cells),
019 * this can be slower by a significant degree (it can be 10x slower for paths of 40-50 cells).
020 * This replaces an earlier version of AStarSearch that was almost always slower than
021 * DijkstraMap and had serious issues. One issue was recursion-related, and previously led to
022 * very long paths taking 200x as much time as DijkstraMap instead of this version's comparably
023 * modest 10x slowdown on the same paths. That recursion-related issue could have caused
024 * StackOverflowExceptions to be thrown when finding very long paths, though that may not have
025 * occurred "in the wild." The only caveat to the optimizations here is that an AStarSearch
026 * object has slightly more state that it stores now, though it also reuses more state and
027 * produces less garbage data to collect.
028 * <br>
029 * If you want pathfinding over an arbitrary graph or need really fast searches, you may want
030 * to use gdx-ai's pathfinding code in its {@code com.badlogic.gdx.ai.pfa} package. Their
031 * {@code IndexedAStarPathFinder} class outperforms all of the pathfinders in squidlib in all
032 * cases tested, though it has less features than DijkstraMap. You would need a dependency on
033 * gdx-ai and libGDX, which the squidlib-util module does not have, but if you use the squidlib
034 * display module, then you already depend on libGDX.
035 * @see squidpony.squidai.DijkstraMap a sometimes-faster pathfinding algorithm that can pathfind to multiple goals
036 * @see squidpony.squidai.CustomDijkstraMap an alternative to DijkstraMap; faster and supports complex adjacency rules
037 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
038 * @author Tommy Ettinger - optimized code
039 */
040public class AStarSearch implements Serializable {
041    private static final long serialVersionUID = 1248976538417655312L;
042    /**
043     * The type of heuristic to use.
044     */
045    public enum SearchType {
046
047        /**
048         * The distance it takes when only the four primary directions can be
049         * moved in.
050         */
051        MANHATTAN,
052        /**
053         * The distance it takes when diagonal movement costs the same as
054         * cardinal movement.
055         */
056        CHEBYSHEV,
057        /**
058         * The distance it takes as the crow flies.
059         */
060        EUCLIDEAN,
061        /**
062         * Full space search. Least efficient but guaranteed to return a path if
063         * one exists. See also DijkstraMap class.
064         */
065        DIJKSTRA
066    }

077    private int dirCount;
078    private transient Direction inner = Direction.DOWN;
079    private boolean[][] finished;
080    protected AStarSearch()
081    {
082        width = 0;
083        height = 0;
084        type = SearchType.MANHATTAN;
085        map = new double[width][height];
086        parent = new byte[width][height];
087        gCache = new double[width][height];
088        finished = new boolean[width][height];
089        dirs = Direction.CARDINALS;
090        dirCount = 4;
091    }
092    /**
093     * Builds a pathing object to run searches on.
094     *
095     * Values in the map are treated as positive values (and 0) being legal
096     * weights, with higher values being harder to pass through. Any negative
097     * value is treated as being an impassible space.
098     *
099     * If the type is Manhattan, only the cardinal directions will be used. All
100     * other search types will return results based on intercardinal and
101     * cardinal pathing.
102     *
103     * @param map
104     *            the search map. It is not modified by this class, hence you can
105     *            share this map among multiple instances.
106     * @param type the manner of search
107     */
108    public AStarSearch(double[][] map, SearchType type) {
109        if (map == null)
110            throw new NullPointerException("map should not be null when building an AStarSearch");
111        this.map = map;
112        width = map.length;
113        height = width == 0 ? 0 : map[0].length;
114        parent = new byte[width][height];
115        gCache = new double[width][height];
116        finished = new boolean[width][height];
117        this.type = type == null ? SearchType.DIJKSTRA : type;
118        switch (this.type) {
119            case MANHATTAN:
120                dirs = Direction.CARDINALS;
121                dirCount = 4;
122                break;
123            case CHEBYSHEV:
124            case EUCLIDEAN:
125            case DIJKSTRA:
126            default:
127                dirs = Direction.OUTWARDS;
128                dirCount = 8;
129                break;
130        }
131    }
132
133    /**
134     * Finds an A* path to the target from the start. If no path is possible,
135     * returns null.
136     *
137     * @param startx the x coordinate of the start location
138     * @param starty the y coordinate of the start location
139     * @param targetx the x coordinate of the target location
140     * @param targety the y coordinate of the target location
141     * @return the shortest path, or null
142     */
143    public Queue<Coord> path(int startx, int starty, int targetx, int targety) {
144        return path(Coord.get(startx, starty), Coord.get(targetx, targety));
145    }
146    /**
147     * Finds an A* path to the target from the start. If no path is possible,
148     * returns null.
149     *
150     * @param start the start location
151     * @param target the target location
152     * @return the shortest path, or null
153     */
154    public Queue<Coord> path(Coord start, Coord target) {
155        this.start = start;
156        this.target = target;
157        open.clear();
158        ArrayTools.fill(finished, false);
159        ArrayTools.fill(parent, (byte)-1);
160        ArrayTools.fill(gCache, -1.0);
161        gCache[start.x][start.y] = 0;
162
163
164        /* Not using Deque nor ArrayDeque, they aren't Gwt compatible */
165        final LinkedList<Coord> deq = new LinkedList<>();
166
167        Coord p = start;
168        open.add(p);
169        Direction dir;
170        byte turn;
171        while (!p.equals(target)) {
172            finished[p.x][p.y] = true;
173            open.remove(p);
174            for (byte d = 0; d < dirCount; d++) {
175                dir = dirs[d];
176                int x = p.x + dir.deltaX;
177                if (x < 0 || x >= width) {
178                    continue;//out of bounds so skip ahead
179                }
180
181                int y = p.y + dir.deltaY;
182                if (y < 0 || y >= height) {
183                    continue;//out of bounds so skip ahead
184                }
185
186                if (!finished[x][y]) {
187                    Coord test = Coord.get(x, y);
188                    if (open.contains(test)) {
189                        turn = parent[x][y];
190                        if(turn < 0)
191                            continue;
192                        // look back and find what we had in gCache
193                        inner = dirs[turn];
194                        double parentG = g(x - inner.deltaX, y - inner.deltaY);
195                        //double parentG = g(parent[x][y].x, parent[x][y].y);
196                        if (parentG < 0) {
197                            continue;//not a valid point so skip ahead
198                        }
199                        double g = g(p.x, p.y);
200                        if (g < 0) {
201                            continue;//not a valid point so skip ahead
202                        }
203                        if (parentG > g) {
204                            parent[x][y] = d;
205                        }
206                    } else {
207                        open.add(test);
208                        parent[x][y] = d;
209                    }
210                }
211            }
212            p = smallestF();
213            if (p == null) {
214                return deq;//no path possible
215            }
216        }
217
218        while (!p.equals(start)) {
219            deq.addFirst(p);
220            inner = dirs[parent[p.x][p.y]];
221            p = p.translate(-inner.deltaX, -inner.deltaY);
222        }
223        return deq;
224    }

238    /**
239     * Finds the g value (start to current) for the given location.
240     *
241     * If the given location is not valid or not attached to the pathfinding
242     * then -1 is returned.
243     *
244     * @param x coordinate
245     * @param y coordinate
246     */
247    protected double g(int x, int y) {
248        if (x == start.x && y == start.y) {
249            gCache[x][y] = 0;
250            return 0;
251        }
252        if (x < 0 || y < 0 || x >= width || y >= height || map[x][y] < 0 || parent[x][y] < 0) {
253            gCache[x][y] = -1;
254            return -1;//not a valid location
255        }
256        inner = dirs[parent[x][y]];
257        double parentG = gCache[x - inner.deltaX][y - inner.deltaY];
258        //double parentG = g(parent[x][y].x, parent[x][y].y);
259        if (parentG < 0) {
260            gCache[x][y] = -1;
261            return -1;//if any part of the path is not valid, this part is not valid
262        }
263
264        return (gCache[x][y] = map[x][y] + parentG + 1);//follow path back to start
265    }

267    /**
268     * Returns the heuristic distance from the current cell to the goal location\
269     * using the current calculation type.
270     *
271     * @param x coordinate
272     * @param y coordinate
273     * @return distance
274     */
275    protected double h(int x, int y) {
276        switch (type) {
277            case MANHATTAN:
278                return Math.abs(x - target.x) + Math.abs(y - target.y);
279            case CHEBYSHEV:
280                return Math.max(Math.abs(x - target.x), Math.abs(y - target.y));
281            case EUCLIDEAN:
282                int xDist = Math.abs(x - target.x);
283                xDist *= xDist;
284                int yDist = Math.abs(y - target.y);
285                yDist *= yDist;
286                return Math.sqrt(xDist + yDist);
287            case DIJKSTRA:
288            default:
289                return 0;
290
291        }
292    }
293
294    /**
295     * Combines g and h to get the estimated distance from start to goal going on the current route.
296     * @param x coordinate
297     * @param y coordinate
298     * @return The current known shortest distance to the start position from
299     *         the given position. If the current position cannot reach the
300     *         start position or is invalid, -1 is returned.
301     */
302    protected double f(int x, int y) {
303        double foundG = g(x, y);
304        if (foundG < 0) {
305            return -1;
306        }
307        return h(x, y) + foundG;
308    }
309
310    /**
311     * @return the current open point with the smallest F
312     */
313    protected Coord smallestF() {
314        Coord smallest = null;
315        double smallF = Double.POSITIVE_INFINITY;
316        double f;
317        int sz = open.size();
318        Coord p;
319        for (int o = 0; o < sz; o++) {
320            p = open.getAt(o);
321            if(p == null)
322                continue;
323            f = f(p.x, p.y);
324            if (f < 0) {
325                continue;//current tested point is not valid so skip it
326            }
327            if (smallest == null || f < smallF) {
328                smallest = p;
329                smallF = f;
330            }
331        }
332
333        return smallest;
334    }

336        @Override
337        public String toString() {
338        final StringBuilder result = new StringBuilder(width * height);
339        int maxLen = 0;
340                /*
341                 * First we compute the longest (String-wise) entry, so that we can
342                 * "indent" shorter cells, so that the output looks good (and is hereby
343                 * readable).
344                 */
345        for (int y = 0; y < height; y++) {
346            for (int x = 0; x < width; x++) {
347                final String output = String.valueOf(Math.round(map[x][y]));
348                final int locLen = output.length();
349                if (maxLen < locLen)
350                    maxLen = locLen;
351            }
352        }
353        for (int y = 0; y < height; y++) {
354            for (int x = 0; x < width; x++) {
355                final long v = Math.round(map[x][y]);
356                final String s = String.valueOf(v);
357                final int slen = s.length();
358                assert slen <= maxLen;
359                int diff = maxLen - slen;
360                while (0 < diff) {
361                    result.append(" ");
362                    diff--;
363                }
364                result.append(s);
365            }
366            if (y < height - 1)
367                result.append('\n');
368        }
369        return result.toString();
370    }
