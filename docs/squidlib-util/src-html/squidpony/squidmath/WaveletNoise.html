018/**
019 * Wavelet Noise, a kind of multi-dimensional noise that is technically unrelated to classic Perlin or Simplex Noise,
020 * developed by Pixar to solve some difficult issues of 2D textures being displayed in 3D scenes. A good source for the
021 * benefits this has is <a href="http://graphics.pixar.com/library/WaveletNoise/paper.pdf">The original Pixar paper</a>.
022 * This is in SquidLib to experiment with using it for higher-dimensional noise. It is unusual in that it generates a
023 * large number of pseudo-random floats when a WaveletNoise object is constructed, and uses the same block of random
024 * numbers (the default is 48 * 48 * 48 numbers in a cube) in different ways as different areas are sampled. It has
025 * somewhat-noticeable axis-aligned bias, and isn't as fast as something like SeededNoise or WhirlingNoise. These flaws
026 * may be corrected at some point. It currently doesn't implement Noise2D or Noise3D, but it should be able to soon.
027 * <br>
028 * This code is originally from the Apache-licensed OcularEngine project, fixing small issues (float distribution from
029 * the RNG): https://github.com/ssell/OcularEngine/blob/master/OcularCore/src/Math/Noise/WaveletNoise.cpp .
030 * <br>
031 * Created by Tommy Ettinger on 9/8/2017.
032 */

034    private static int mod(final int x, final int n) {
035        final int m = x % n;
036        return (m < 0) ? m + n : m;

128    /**
129     * Like {@link Math#floor}, but returns an int. Doesn't consider weird floats like INFINITY and NaN.
130     * @param t the float to find the floor for
131     * @return the floor of t, as an int
132     */
133    public static int fastFloor(float t) {
134        return t >= 0 ? (int) t : (int) t - 1;
135    }

137    /**
138     * Like {@link Math#ceil(double)}, but returns an int. Doesn't consider weird floats like INFINITY and NaN.
139     * @param t the float to find the ceiling for
140     * @return the ceiling of t, as an int
141     */
142    private static int fastCeil(float t) {
143        return t >= 0 ? -(int) -t + 1: -(int)-t;    }

145    /**
146     * The basis for all getNoise methods in this class; takes x, y, and z coordinates as floats, plus a seed that will
147     * alter the noise effectively by just moving the section this samples in an unrelated way to changing x, y, and z
148     * normally. Returns a float between -1.0f (inclusive, in theory) and 1.0f exclusive.
149     * @param x x position
150     * @param y y position
151     * @param z z position
152     * @param seed seed as an int to modify the noise produced
153     * @return a float between -1.0f (inclusive) and 1.0f (exclusive)
154     */
155    public float getRawNoise(float x, float y, float z, int seed) {
156        int n = m_Dimensions;
157        int[] f = this.f;
158        int[] c = this.c;
159        int[] mid = this.mid;
160        float[][] w = this.w;
161        float t;
162        float result = 0.0f;
163        final float rnd = (ThrustRNG.determine(seed) >> 41) * 0x1p-15f;
164        x += rnd;
165        y += rnd;
166        z += rnd;
167        //---------------------------------------------------
168        // Evaluate quadratic B-spline basis functions
169        mid[0] = fastCeil(x - 0.5f); //Math.round(x);
170        t = mid[0] - (x - 0.5f);
171        w[0][0] = t * t * 0.5f;
172        w[0][2] = (1.0f - t) * (1.0f - t) * 0.5f;
173        w[0][1] = 1.0f - w[0][0] - w[0][2];
174        mid[1] = fastCeil(y - 0.5f); //Math.round(y);
175        t = mid[1] - (y - 0.5f);
176        w[1][0] = t * t * 0.5f;
177        w[1][2] = (1.0f - t) * (1.0f - t) * 0.5f;
178        w[1][1] = 1.0f - w[1][0] - w[1][2];
179        mid[2] = fastCeil(z - 0.5f); //Math.round(z);
180        t = mid[2] - (z - 0.5f);
181        w[2][0] = t * t * 0.5f;
182        w[2][2] = (1.0f - t) * (1.0f - t) * 0.5f;
183        w[2][1] = 1.0f - w[2][0] - w[2][2];
184
185        float weight;
186
187        for (f[2] = -1; f[2] <= 1; f[2]++) {
188            for (f[1] = -1; f[1] <= 1; f[1]++) {
189                for (f[0] = -1; f[0] <= 1; f[0]++) {
190                    weight = 1.0f;
191
192                    for (int i = 0; i < 3; i++) {
193                        c[i] = mod((mid[i] + f[i]), n);
194                        weight *= w[i][f[i] + 1];
195                    }
196
197                    result += weight * m_Noise[(c[2] * n * n) + (c[1] * n) + c[0]];
198                }
199            }
200        }
201//        if(result < -1f || result >= 1f)
202//            System.out.println("BAD: result "+ result + ", x " + x + ", y " + y + ", z" + z + ", mid[0] " + mid[0] + ", mid[1] " + mid[1] + ", mid[2] " + mid[2]);
203        return result;
204    }

206    /**
207     * Makes a float array that can be passed as bands to {@link #getBandedNoise(float, float, float, int, float[])}.
208     * This takes an array or varargs of floats, which are used in order as weights for successively doubled frequencies
209     * for the noise. It may be good to experiment with this; {@code 0.5f, 1.1f, 1.9f, 1.2f, 0.6f} will be different
210     * from {@code 2f, 0.8f, 0f, 0.4f, 1.3f}. Lengths for the weights array of 10 or more can yield nice results, but
211     * take longer to compute. Weights of 0 do not require calculating any noise when they are used.
212     * @param weights an array or varargs of float, where each float, in order, corresponds to a weight for a higher frequency of noise
213     * @return a float array that can be passed as bands to {@link #getBandedNoise(float, float, float, int, float[])}
214     */
215    public static float[] prepareBands(float... weights)
216    {
217        final int len = weights.length;
218        final float[] bands = new float[len+1];
219        float variance = 0f, t;
220        for (int i = 0; i < len; i++) {
221            t = (bands[i] = Math.max(0f, weights[i]));
222            variance += t * t;
223        }
224        bands[len] = (variance != 0f)
225                ? (float)(1.0 / Math.sqrt(variance * len * 0.11))
226                : 1f;
227        return bands;
228    }

230    /**
231     * A 3D noise function that allows different frequencies to be mixed in unusual ways, similarly to octaves in other
232     * noise functions but allowing arbitrary weights for frequencies. This takes x, y, and z coordinates as floats, a
233     * seed that will alter the noise calls like it does in {@link #getRawNoise(float, float, float, int)}, and a float
234     * array of bands. The bands must be readied by {@link #prepareBands(float...)}, which creates a specially formatted
235     * float array that can (and probably should) be reused; the items given to prepareBands are each weights for
236     * successively doubled frequencies. Returns a float between -1.0f (inclusive, in theory) and 1.0f exclusive.
237     * @param x x position
238     * @param y y position
239     * @param z z position
240     * @param seed seed as an int to modify the noise produced
241     * @param bands a float array that was created by {@link #prepareBands(float...)}; holds frequency band data
242     * @return a float between -1.0f (inclusive) and 1.0f (exclusive)
243     */
244    public float getBandedNoise(float x, float y, float z, int seed, float[] bands) {
245        float result=0, ax, ay, az, t;
246        final int len = bands.length - 1;
247        for (int b = 0; b < len; b++) {
248            if((t = bands[b]) == 0)
249                continue;;
250            ax = x * (2 << b);
251            ay = y * (2 << b);
252            az = z * (2 << b);
253            result += t * getRawNoise(ax, ay, az, seed);
254        }
255        return result * bands[len];
256//        if(result < -1f || result >= 1f)
257//            System.out.println("BAD: result "+ result + ", x " + x + ", y " + y + ", z" + z);
258//        return result;
259    }

261    /**
262     * Only needs to be called if you change the {@link #seed} field and want the cube of random values re-created.
263     */
264    public void generate() {
265        int x;
266        int y;
267        int z;
268        int i;
269
270        float[] temp1 = this.temp1;
271        float[] temp2 = this.temp2;
272        float[] noise = this.noise;
273
274        //---------------------------------------------------
275        // Step 1: Fill the tile with random numbers on range [-1.0, 1.0)
276
277
278        for (i = 0; i < m_NoiseSize; i++) {
279            noise[i] = //(ThrustRNG.determine(seed + i * 421L) >> 41) * 0x1p-22f;
280                    NumberTools.formCurvedFloat(ThrustRNG.determine(seed + i * 181L));
281        }
282
283        //---------------------------------------------------
284        // Step 2 & 3: Downsample and then Upsample
285
286        for (y = 0; y < m_Dimensions; y++) {
287            for (z = 0; z < m_Dimensions; z++) {
288                i = (y * m_Dimensions) + (z * m_Dimensions * m_Dimensions);
289
290                downsample(noise, temp1, i, m_Dimensions, 1);
291                upsample(temp1, temp2, i, m_Dimensions, 1);
292            }
293        }
294
295        for (x = 0; x < m_Dimensions; x++) {
296            for (z = 0; z < m_Dimensions; z++) {
297                i = x + (z * m_Dimensions * m_Dimensions);
298
299                downsample(temp2, temp1, i, m_Dimensions, m_Dimensions);
300                upsample(temp1, temp2, i, m_Dimensions, m_Dimensions);
301            }
302        }
303
304        for (x = 0; x < m_Dimensions; x++) {
305            for (y = 0; y < m_Dimensions; y++) {
306                i = x + (y * m_Dimensions);
307
308                downsample(temp2, temp1, i, m_Dimensions, (m_Dimensions * m_Dimensions));
309                upsample(temp1, temp2, i, m_Dimensions, (m_Dimensions * m_Dimensions));
310            }
311        }
312
313        //---------------------------------------------------
314        // Step 4: Substract out the coarse-scale constribution (original - (downsample upsample))
315
316        for (i = 0; i < m_NoiseSize; i++) {
317            noise[i] -= temp2[i];
318        }
319
320        //---------------------------------------------------
321        // Step 5: Avoid even/odd variance difference by adding odd-offset version of noise to itself
322
323        int offset = m_Dimensions >> 1 | 1;
324
325        for (i = 0, x = 0; x < m_Dimensions; x++) {
326            for (y = 0; y < m_Dimensions; y++) {
327                for (z = 0; z < m_Dimensions; z++) {
328                    temp1[i++] = noise[mod((x + offset), m_Dimensions) + (mod((y + offset), m_Dimensions) * m_Dimensions) + (mod((z + offset), m_Dimensions) * m_Dimensions * m_Dimensions)];
329                }
330            }
331        }
332
333        for (i = 0; i < m_NoiseSize; i++) {
334            m_Noise[i] = (noise[i] + temp1[i]);// + 1.3125f);
335        }
336    }

338    protected void downsample(float[] from, float[] to, int idx, int n, int stride) {
339
340        int tindex;
341        int findex;
342        int cindex;
343
344        for (int i = 0; i < (n >> 1); i++) {
345            tindex = i * stride + idx;
346            to[tindex] = 0.0f;
347
348            for (int j = ((2 * i) - 16); j < ((2 * i) + 16); j++) {
349                cindex = 16 + (j - 2 * i);
350                findex = mod(j, n) * stride + idx;
351
352                to[tindex] += downCoefficients[cindex] * from[findex];
353            }
354        }
355    }
