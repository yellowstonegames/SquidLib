022    public MummyNoise(final int seed) {
023        seedX = 0x9E3779B97F4A7C15L * LinnormRNG.determine(seed + 0xC6BC279692B5CC83L);
024        seedY = 0x9E3779B97F4A7C15L * LinnormRNG.determine(seedX ^ 0xC7BC279692B5CB83L);
025        seedZ = 0x9E3779B97F4A7C15L * LinnormRNG.determine(seedY ^ 0xC8BC279692B5CA83L);
026        seedW = 0x9E3779B97F4A7C15L * LinnormRNG.determine(seedZ ^ 0xC9BC279692B5C983L);
027        seedU = 0x9E3779B97F4A7C15L * LinnormRNG.determine(seedW ^ 0xCABC279692B5C883L);
028        seedV = 0x9E3779B97F4A7C15L * LinnormRNG.determine(seedU ^ 0xCBBC279692B5C783L);
029    }

031    /**
032     * The same as {@link LinnormRNG#determine(long)}, except this assumes state has already been multiplied by
033     * 0x632BE59BD9B4E019L.
034     * @param state a long that should change in increments of 0x632BE59BD9B4E019L
035     * @return a pseudo-random permutation of state
036     */
037    public static long determine(long state)
038    {
039        return (state = ((state = ((state ^ 0x9E3779B97F4A7C15L) * 0xC6BC279692B5CC83L)) ^ state >>> 27) * 0xAEF17502108EF2D9L) ^ state >>> 25;
040    }

068    public static double getNoiseWithSeeds(final double x, final double y,
069                                    final long seedX, final long seedY) {
070        final long
071                xf = longFloor(x),
072                yf = longFloor(y),
073                bx0 = xf * seedX,
074                by0 = yf * seedY,
075                bx1 = bx0+seedX,
076                by1 = by0+seedY;
077        return
078                NumberTools.sway(
079                        cerp(cerp(determine(bx0 + by0) * 0x1.25p-62, determine(bx1 + by0) * 0x1.25p-62,
080                                x - xf),
081                                cerp(determine(bx0 + by1) * 0x1.25p-62, determine(bx1 + by1) * 0x1.25p-62, x - xf),
082                                y - yf));
083    }

320    private transient long[] scratch3;
321    private transient double[] scratch;
322    private transient int lastLen = -1;
323    private transient double lastEffect = 0x1.12bp-61;
324    public final double arbitraryNoise(long seed, double... coordinates) {
325        final int len = coordinates.length, upper = 1 << len;
326        final double effect;
327        if(len != lastLen)
328        {
329            lastLen = len;
330            lastEffect = effect = 0x1.81p-62 * Math.pow(1.1875, len);
331        }
332        else
333            effect = lastEffect;
334        if(scratch3 == null || scratch3.length < len * 3)
335            scratch3 = new long[len * 3];
336        if(scratch == null || scratch.length < upper)
337            scratch = new double[upper];
338        for (int i = 0; i < len; i++) {
339            seed = LinnormRNG.determine(seed + 0xC6BC279692B5CC83L ^ ~seed << 32);
340            scratch3[i * 3 + 1] = (scratch3[i * 3] = (scratch3[i * 3 + 2] = longFloor(coordinates[i])) * seed) + seed;
341        }
342        long working;
343        for (int i = 0; i < upper; i++) {
344            working = 0L;
345            for (int j = 0; j < len; j++) {
346                working += scratch3[j * 3 + (i >> j & 1)];
347            }
348            scratch[i] = determine(working) * effect;
349        }
350        for (int i = 0; i < len; ++i) {
351            for (int j = 0, t = upper >> i; j < t; j += 2) {
352                scratch[j >> 1] = cerp(scratch[j], scratch[j + 1], coordinates[i] - scratch3[i * 3 + 2]);
353            }
354        }
355        return NumberTools.sway(scratch[0]);
356    }
