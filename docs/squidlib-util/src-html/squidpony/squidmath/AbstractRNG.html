011/**
012 * A helper class for implementing {@link IRNG} without so much busy-work.
013 * You need to provide implementations for the abstract methods {@link #nextInt()}, {@link #next(int)},
014 * {@link #nextLong()}, {@link #nextBoolean()}, {@link #nextDouble()}, {@link #nextFloat()}, {@link #copy()}, and
015 * {@link #toSerializable()}, many of which may be built off of each other and some of which have sample code in their
016 * documentation strings in this class.
017 * <br>
018 * Big thanks to smelc for the concept in his hgameshrogue library.
019 * @author Tommy Ettinger
020 * @author smelC
021 */
022public abstract class AbstractRNG implements IRNG {
023    /**
024     * Get up to 32 bits (inclusive) of random output; the int this produces
025     * will not require more than {@code bits} bits to represent.
026     *
027     * @param bits an int between 1 and 32, both inclusive
028     * @return a random number that fits in the specified number of bits
029     */
030    @Override
031    public abstract int next(int bits);
032    /**
033     * Get a random integer between Integer.MIN_VALUE to Integer.MAX_VALUE (both inclusive).
034     *
035     * @return a 32-bit random int.
036     */
037    @Override
038    public abstract int nextInt();
039
040    /**
041     * Returns a random non-negative integer below the given bound, or 0 if the bound is 0 or
042     * negative.
043     *
044     * @param bound the upper bound (exclusive)
045     * @return the found number
046     */
047    @Override
048    public int nextInt(int bound) {
049        return (int) ((bound * ((long)next(31))) >>> 31) & ~(bound >> 31);
050    }
051
052    /**
053     * Get a random long between Long.MIN_VALUE to Long.MAX_VALUE (both inclusive).
054     *
055     * @return a 64-bit random long.
056     */
057    @Override
058    public abstract long nextLong();
059
060    /**
061     * Returns a random long below the given bound, or 0 if the bound is 0 or
062     * negative.
063     *
064     * @param bound the upper bound (exclusive)
065     * @return the found number
066     */
067    @Override
068    /**
069     * Exclusive on bound (which must be positive), with an inner bound of 0.
070     * If bound is negative or 0 this always returns 0.
071     * <br>
072     * Credit for this method goes to <a href="https://oroboro.com/large-random-in-range/">Rafael Baptista's blog</a>,
073     * with some adaptation for signed long values and a 64-bit generator. This method is drastically faster than the
074     * previous implementation when the bound varies often (roughly 4x faster, possibly more). It also always gets
075     * exactly one random number, so it advances the state as much as {@link #nextInt(int)}.
076     * @param bound the outer exclusive bound; should be positive, otherwise this always returns 0L
077     * @return a random long between 0 (inclusive) and bound (exclusive)
078     */
079    public long nextLong(long bound) {
080        long rand = nextLong();
081        if (bound <= 0) return 0;
082        final long randLow = rand & 0xFFFFFFFFL;
083        final long boundLow = bound & 0xFFFFFFFFL;
084        rand >>>= 32;
085        bound >>= 32;
086        final long z = (randLow * boundLow >> 32);
087        final long t = rand * boundLow + z;
088        return rand * bound + (t >> 32) + ((t & 0xFFFFFFFFL) + randLow * bound >> 32) - (z >> 63);
089    }

091    /**
092     * Get a random bit of state, interpreted as true or false with approximately equal likelihood.
093     * <br>
094     * Note: This is abstract because some implementations may be best served by using {@link #next(int)} to get 1 bit,
095     * returning {@code next(1) == 1}, but others will get much better results with a sign check by calling their choice
096     * of {@link #nextInt()} or {@link #nextLong()} and returning {@code nextInt() < 0} or {@code nextLong < 0L}. For
097     * example, an implementation that uses a linear congruential generator without truncating some lower bits will have
098     * very-low-period results for the bottom bit (alternating true and false), but perfectly fine results from a sign
099     * check. There are tested issues on the bottom (at least 2) bits of {@link XoRoRNG}, but again not on a sign check.
100     * @return a random boolean.
101     */
102    @Override
103    public abstract boolean nextBoolean();
104    /**
105     * Gets a random double between 0.0 inclusive and 1.0 exclusive.
106     * This returns a maximum of 0.9999999999999999 because that is the largest double value that is less than 1.0 .
107     * <br>
108     * This is abstract because some generators may natively work with double or float values, but others may need to
109     * convert a long to a double as with {@code (nextLong() & 0x1fffffffffffffL) * 0x1p-53}, which is recommended if
110     * longs are fast to produce.
111     * @return a double between 0.0 (inclusive) and 0.9999999999999999 (inclusive)
112     */
113    @Override
114    public abstract double nextDouble();
115    /**
116     * This returns a random double between 0.0 (inclusive) and outer (exclusive). The value for outer can be positive
117     * or negative. Because of how math on doubles works, there are at most 2 to the 53 values this can return for any
118     * given outer bound, and very large values for outer will not necessarily produce all numbers you might expect.
119     *
120     * @param outer the outer exclusive bound as a double; can be negative or positive
121     * @return a double between 0.0 (inclusive) and outer (exclusive)
122     */
123    @Override
124    public double nextDouble(double outer) {
125        return nextDouble() * outer;
126    }
127
128    /**
129     * Gets a random float between 0.0f inclusive and 1.0f exclusive.
130     * This returns a maximum of 0.99999994 because that is the largest float value that is less than 1.0f .
131     * <br>
132     * This is abstract because some generators may natively work with double or float values, but others may need to
133     * convert an int or long to a float as with {@code (nextInt() & 0xffffff) * 0x1p-24f},
134     * {@code (nextLong() & 0xffffffL) * 0x1p-24f}, or {@code next(24) * 0x1p-24f}, any of which can work when the
135     * method they call is high-quality and fast. You probably would want to use nextInt() or next() if your
136     * implementation is natively 32-bit and is slower at producing longs, for example.
137     * @return a float between 0f (inclusive) and 0.99999994f (inclusive)
138     */
139    @Override
140    public abstract float nextFloat();
141    /**
142     * This returns a random float between 0.0f (inclusive) and outer (exclusive). The value for outer can be positive
143     * or negative. Because of how math on floats works, there are at most 2 to the 24 values this can return for any
144     * given outer bound, and very large values for outer will not necessarily produce all numbers you might expect.
145     *
146     * @param outer the outer exclusive bound as a float; can be negative or positive
147     * @return a float between 0f (inclusive) and outer (exclusive)
148     */
149    @Override
150    public float nextFloat(float outer) {
151        return nextFloat() * outer;
152    }
153    /**
154     * Exclusive on bound (which may be positive or negative), with an inner bound of 0.
155     * If bound is negative this returns a negative long; if bound is positive this returns a positive long. The bound
156     * can even be 0, which will cause this to return 0L every time. This uses a biased technique to get numbers from
157     * large ranges, but the amount of bias is incredibly small (expected to be under 1/1000 if enough random ranged
158     * numbers are requested, which is about the same as an unbiased method that was also considered). It may have
159     * noticeable bias if the generator's period is exhausted by only calls to this method. Unlike all unbiased methods,
160     * this advances the state by an equivalent to exactly one call to {@link #nextLong()}, where rejection sampling
161     * would sometimes advance by one call, but other times by arbitrarily many more.
162     * <br>
163     * Credit for this method goes to <a href="https://oroboro.com/large-random-in-range/">Rafael Baptista's blog</a>,
164     * with some adaptation for signed long values and a 64-bit generator. This method is drastically faster than the
165     * previous implementation when the bound varies often (roughly 4x faster, possibly more). It also always gets at
166     * most one random number, so it advances the state as much as {@link #nextInt(int)} or {@link #nextLong()}.
167     * @param bound the outer exclusive bound; can be positive or negative
168     * @return a random long between 0 (inclusive) and bound (exclusive)
169     */
170    @Override
171    public long nextSignedLong(long bound) {
172        long rand = nextLong();
173        final long randLow = rand & 0xFFFFFFFFL;
174        final long boundLow = bound & 0xFFFFFFFFL;
175        rand >>>= 32;
176        bound >>= 32;
177        final long z = (randLow * boundLow >> 32);
178        final long t = rand * boundLow + z;
179        return rand * bound + (t >> 32) + ((t & 0xFFFFFFFFL) + randLow * bound >> 32) - (z >> 63);
180    }

182    /**
183     * Returns a random non-negative integer between 0 (inclusive) and the given bound (exclusive),
184     * or 0 if the bound is 0. The bound can be negative, which will produce 0 or a negative result.
185     * <br>
186     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
187     *
188     * @param bound the outer bound (exclusive), can be negative or positive
189     * @return the found number
190     */
191    @Override
192    public int nextSignedInt(final int bound) {
193        return (int) ((bound * (nextInt() & 0xFFFFFFFFL)) >> 32);
194    }

196    /**
197     * Returns a value between min (inclusive) and max (exclusive) as ints.
198     * <br>
199     * The inclusive and exclusive behavior is to match the behavior of the similar
200     * method that deals with floating point values.
201     * <br>
202     * If {@code min} and {@code max} happen to be the same, {@code min} is returned
203     * (breaking the exclusive behavior, but it's convenient to do so).
204     *
205     * @param min the minimum bound on the return value (inclusive)
206     * @param max the maximum bound on the return value (exclusive)
207     * @return the found value
208     */
209    @Override
210    public int between(int min, int max) {
211        return nextInt(max - min) + min;
212    }
213
214    /**
215     * Returns a value between min (inclusive) and max (exclusive) as longs.
216     * <br>
217     * The inclusive and exclusive behavior is to match the behavior of the similar
218     * method that deals with floating point values.
219     * <br>
220     * If {@code min} and {@code max} happen to be the same, {@code min} is returned
221     * (breaking the exclusive behavior, but it's convenient to do so).
222     *
223     * @param min the minimum bound on the return value (inclusive)
224     * @param max the maximum bound on the return value (exclusive)
225     * @return the found value
226     */
227    @Override
228    public long between(long min, long max) {
229        return nextLong(max - min) + min;
230    }

232    /**
233     * Returns a value from a uniform distribution from min (inclusive) to max
234     * (exclusive).
235     *
236     * @param min the minimum bound on the return value (inclusive)
237     * @param max the maximum bound on the return value (exclusive)
238     * @return the found value
239     */
240    @Override
241    public double between(double min, double max) {
242        return nextDouble(max - min) + min;
243    }

245    /**
246     * Returns a random element from the provided array and maintains object
247     * type.
248     *
249     * @param array the array to get an element from
250     * @return the randomly selected element
251     */
252    @Override
253    public <T> T getRandomElement(T[] array) {
254        if (array.length < 1) {
255            return null;
256        }
257        return array[nextInt(array.length)];
258    }
259
260    /**
261     * Returns a random element from the provided list. If the list is empty
262     * then null is returned. This will perform well on Lists that allow random access,
263     * but will not perform as well on {@link java.util.LinkedList} or similar classes
264     * that need to iterate one-by-one in their {@link List#get(int)} method.
265     *
266     * @param list the list to get an element from
267     * @return the randomly selected element
268     */
269    @Override
270    public <T> T getRandomElement(List<T> list) {
271        if (list.isEmpty()) {
272            return null;
273        }
274        return list.get(nextInt(list.size()));
275    }
276
277    /**
278     * Returns a random element from the provided Collection, which should have predictable iteration order if you want
279     * predictable behavior for identical RNG seeds, though it will get a random element just fine for any Collection
280     * (just not predictably in all cases). If you give this a Set, it should be a LinkedHashSet or some form of sorted
281     * Set like TreeSet if you want predictable results. Any List or Queue should be fine. Map does not implement
282     * Collection, thank you very much Java library designers, so you can't actually pass a Map to this, though you can
283     * pass the keys or values. If coll is empty, returns null.
284     * <br>
285     * Requires iterating through a random amount of coll's elements, so performance depends on the size of coll but is
286     * likely to be decent, as long as iteration isn't unusually slow. This replaces {@code getRandomElement(Queue)},
287     * since Queue implements Collection and the older Queue-using implementation was probably less efficient.
288     * <br>
289     * You should generally prefer {@link #getRandomElement(List)} whenever possible, or in some cases you can use
290     * methods that get a random value on the Collection (or Map, in the case of OrderedMap) itself.
291     *
292     * @param coll the Collection to get an element from; remember, Map does not implement Collection
293     * @return the randomly selected element
294     */
295    @Override
296    public <T> T getRandomElement(Collection<T> coll) {
297        int n;
298        if ((n = coll.size()) <= 0) {
299            return null;
300        }
301        n = nextInt(n);
302        T t = null;
303        Iterator<T> it = coll.iterator();
304        while (n-- >= 0 && it.hasNext())
305            t = it.next();
306        return t;
307    }

309    /**
310     * Mutates the array arr by switching the contents at pos1 and pos2.
311     * @param arr an array of T; must not be null
312     * @param pos1 an index into arr; must be at least 0 and no greater than arr.length
313     * @param pos2 an index into arr; must be at least 0 and no greater than arr.length
314     */
315    protected static <T> void swap(T[] arr, int pos1, int pos2) {
316        final T tmp = arr[pos1];
317        arr[pos1] = arr[pos2];
318        arr[pos2] = tmp;
319    }

321    /**
322     * Shuffle an array using the Fisher-Yates algorithm and returns a shuffled copy, freshly-allocated, without
323     * modifying elements.
324     * <br>
325     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
326     *
327     * @param elements an array of T; will not be modified
328     * @return a shuffled copy of elements
329     */
330    @Override
331    public <T> T[] shuffle(T[] elements) {
332        final int size = elements.length;
333        final T[] array = Arrays.copyOf(elements, size);
334        for (int i = size; i > 1; i--) {
335            swap(array, i - 1, nextInt(i));
336        }
337        return array;
338    }

340    /**
341     * Shuffles an array in-place using the Fisher-Yates algorithm.
342     * If you don't want the array modified, use {@link #shuffle(Object[], Object[])}.
343     * <br>
344     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
345     *
346     * @param elements an array of T; <b>will</b> be modified
347     * @return elements after shuffling it in-place
348     */
349    @Override
350    public <T> T[] shuffleInPlace(T[] elements) {
351        final int size = elements.length;
352        for (int i = size; i > 1; i--) {
353            swap(elements, i - 1, nextInt(i));
354        }
355        return elements;
356    }

358    /**
359     * Shuffle an array using the Fisher-Yates algorithm. If possible, create a new array or reuse an existing array
360     * with the same length as elements and pass it in as dest; the dest array will contain the shuffled contents of
361     * elements and will also be returned as-is. If dest does not have the same length as elements, this will throw an
362     * IllegalArgumentException
363     * <br>
364     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
365     *
366     * @param elements an array of T; will not be modified
367     * @param <T>      can be any non-primitive type.
368     * @param dest     Where to put the shuffle. If it does not have the same length as {@code elements}, this will
369     *                 throw an IllegalArgumentException.
370     * @return {@code dest} after modifications
371     */
372    @Override
373    public <T> T[] shuffle(T[] elements, T[] dest) {
374        if (dest.length != elements.length)
375            throw new IllegalArgumentException("Not allowed! In AbstractRNG.shuffle(), elements and dest do not have the same length.");
376        System.arraycopy(elements, 0, dest, 0, elements.length);
377        shuffleInPlace(dest);
378        return dest;
379    }

381    /**
382     * Shuffles a {@link Collection} of T using the Fisher-Yates algorithm and returns an ArrayList of T.
383     * <br>
384     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
385     *
386     * @param elements a Collection of T; will not be modified
387     * @return a shuffled ArrayList containing the whole of elements in pseudo-random order.
388     */
389    @Override
390    public <T> ArrayList<T> shuffle(Collection<T> elements) {
391        return shuffle(elements, null);
392    }
393
394    /**
395     * Shuffles a {@link Collection} of T using the Fisher-Yates algorithm and puts it in a buffer.
396     * The result is allocated if {@code buf} is null or if {@code buf} isn't empty,
397     * otherwise {@code elements} is poured into {@code buf}.
398     * <br>
399     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
400     *
401     * @param elements a Collection of T; will not be modified
402     * @param buf      a buffer as an ArrayList that will be filled with the shuffled contents of elements;
403     *                 if null or non-empty, a new ArrayList will be allocated and returned
404     * @return a shuffled ArrayList containing the whole of elements in pseudo-random order, which may be {@code buf}
405     */
406    @Override
407    public <T> ArrayList<T> shuffle(Collection<T> elements, ArrayList<T> buf) {
408        final ArrayList<T> al;
409        if (buf == null || !buf.isEmpty())
410            al = new ArrayList<>(elements);
411        else {
412            al = buf;
413            al.addAll(elements);
414        }
415        final int n = al.size();
416        for (int i = n; i > 1; i--) {
417            Collections.swap(al, nextInt(i), i - 1);
418        }
419        return al;
420    }

423     * Shuffles a Collection of T items in-place using the Fisher-Yates algorithm.
424     * This only shuffles List data structures.
425     * If you don't want the array modified, use {@link #shuffle(Collection)}, which returns a List as well.
426     * <br>
427     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
428     *
429     * @param elements a Collection of T; <b>will</b> be modified
430     * @return elements after shuffling it in-place
431     */
432    @Override
433    public <T> List<T> shuffleInPlace(List<T> elements) {
434        final int n = elements.size();
435        for (int i = n; i > 1; i--) {
436            Collections.swap(elements, nextInt(i), i - 1);
437        }
438        return elements;
439    }

441    /**
442     * Generates a random permutation of the range from 0 (inclusive) to length (exclusive).
443     * Useful for passing to OrderedMap or OrderedSet's reorder() methods.
444     *
445     * @param length the size of the ordering to produce
446     * @return a random ordering containing all ints from 0 to length (exclusive)
447     */
448    @Override
449    public int[] randomOrdering(int length) {
450        if (length <= 0)
451            return new int[0];
452        return randomOrdering(length, new int[length]);
453    }
454
455    /**
456     * Generates a random permutation of the range from 0 (inclusive) to length (exclusive) and stores it in
457     * the dest parameter, avoiding allocations.
458     * Useful for passing to OrderedMap or OrderedSet's reorder() methods.
459     *
460     * @param length the size of the ordering to produce
461     * @param dest   the destination array; will be modified
462     * @return dest, filled with a random ordering containing all ints from 0 to length (exclusive)
463     */
464    @Override
465    public int[] randomOrdering(int length, int[] dest) {
466        if (dest == null) return null;
467
468        final int n = Math.min(length, dest.length);
469        for (int i = 0; i < n; i++) {
470            dest[i] = i;
471        }
472        for (int i = n - 1; i > 0; i--) {
473            final int r = nextInt(i+1),
474                    t = dest[r];
475            dest[r] = dest[i];
476            dest[i] = t;
477        }
478        return dest;
479    }

483     * IRNG at the point copy() is called. The copy will not share references with this IRNG. If this IRNG does not

487     * This is abstract because every implementation is likely to have different specifics for this.
488     * @return a copy of this IRNG
489     */
490    @Override
491    public abstract IRNG copy();

494     * Gets a view of this IRNG in a way that implements {@link Serializable}, which may simply be this IRNG if it

496     * <br>
497     * For implementors: It is suggested to return an {@link RNG} initialized by calling
