008package squidpony.squidmath;
009
010import squidpony.StringKit;
011
012import java.io.Serializable;
013
014/**
015 * A modification of Blackman and Vigna's xoroshiro128+ generator using two 32-bit ints of state instead of two 64-bit
016 * longs, as well as modifying the output with two additional operations on the existing state; this is both the fastest
017 * generator on GWT I have found without statistical failures, and a StatefulRandomness. This algorithm is sometimes
018 * called xoroshiro64++ and is mentioned in <a href="http://vigna.di.unimi.it/ftp/papers/ScrambledLinear.pdf">this paper
019 * by Blackman and Vigna</a> (in section 10.7; the 'r' constant is 10, which seems to usually do well on those same
020 * authors' HWD test). Lathe32RNG passes the full 32TB battery of PractRand's statistical tests, and does so with 3
021 * "unusual" anomalies, no more-serious anomalies, and no failures. It isn't especially likely that this can pass much
022 * more than 32TB of testing (judging by related attempts, 128TB would be a likely failure point), but because
023 * multi-threaded code is either impossible or impractical on GWT, actually using that many numbers would take a very
024 * long time (generating them would take about 3 nanoseconds per int, but it would take more than 2 to the 43 ints to
025 * start to approach detectable failures, and detecting the failures in anything but the worst case would take more than
026 * a day). In statistical testing, xoroshiro with the '+' scrambler always fails some binary matrix rank tests, but
027 * smaller-state versions fail other tests as well. The changes Lathe makes apply only to the output of xoroshiro64+ (in
028 * Vigna's and Blackman's terms, they are a scrambler), not its well-tested state transition, and these changes
029 * eliminate all statistical failures on 32TB of tested data, avoiding the failures the small-state variant of xoroshiro
030 * suffers on BinaryRank, BCFN, DC6, and FPF. It avoids multiplication (except in {@link #setSeed(int)}, which needs to
031 * use a different algorithm to spread a seed out across twice as much state), like xoroshiro and much of the xorshift
032 * family of generators, and any arithmetic it performs is safe for GWT. Lathe makes an extremely small set of changes
033 * to xoroshiro64+, running xoroshiro64+ as normal (holding on to the result as well as the initial stateA, called s[0]
034 * in the original xoroshiro code) and then bitwise-rotating the result and adding the (now previous) stateA. Although
035 * no bits of xoroshiro are truly free of artifacts, some are harder to find issues with
036 * (see <a href="http://www.pcg-random.org/posts/xoroshiro-fails-truncated.html">this article by PCG-Random's author</a>
037 * for more detail). It is unclear if the changes made here would improve the larger-state version, but they probably
038 * would help to some extent with at least the binary rank failures. The period is identical to xoroshiro with two
039 * 32-bit states, at 0xFFFFFFFFFFFFFFFF or 2 to the 64 minus 1. This generator is slightly slower than xoroshiro without
040 * the small extra steps applied to the output, but about as fast as {@link Oriole32RNG} (this has a smaller period and
041 * smaller state but implements StatefulRandomness). Some simple tests on bytes instead of ints showed that the
042 * technique used here produces all possible bytes with equal frequency when run on bytes as state, with the exception
043 * of producing 0 one less time (because both states cannot be 0 at the same time). This gives some confidence for the
044 * algorithm used here, but doesn't say anything about how equidistributed this is across more than one dimension (it
045 * could be better or worse than xoroshiro128+).
046 * <br>
047 * The name comes from a tool that rotates very quickly to remove undesirable parts of an object, akin to how this
048 * generator adds an extra bitwise rotation to xoroshiro64+ to remove several types of
049 * undesirable statistical failures from its test results.
050 * <br>
051 * <a href="http://xoroshiro.di.unimi.it/xoroshiro128plus.c">Original version here for xorshiro128+</a>; this version
052 * uses <a href="https://groups.google.com/d/msg/prng/Ll-KDIbpO8k/bfHK4FlUCwAJ">different constants</a> by the same
053 * author, Sebastiano Vigna. It does not use <a href="http://xoshiro.di.unimi.it/xoroshiro64star.c">the constants used
054 * in other xoroshiro64 scrambled generators</a>, instead using similar-quality ones from the earlier constants link.
055 * <br>
056 * Written in 2016 by David Blackman and Sebastiano Vigna (vigna@acm.org)
057 * Ported and modified in 2018 by Tommy Ettinger
058 * @author Sebastiano Vigna
059 * @author David Blackman
060 * @author Tommy Ettinger (if there's a flaw, use SquidLib's or Sarong's issues and don't bother Vigna or Blackman, it's probably a mistake in SquidLib's implementation)
061 */
062public final class Lathe32RNG implements StatefulRandomness, Serializable {
063
064    private static final long serialVersionUID = 1L;
065
066    private int stateA, stateB;
067
068    /**
069     * Creates a new generator seeded using two calls to Math.random().
070     */
071    public Lathe32RNG() {
072        setState((int)((Math.random() * 2.0 - 1.0) * 0x80000000), (int)((Math.random() * 2.0 - 1.0) * 0x80000000));
073    }
074    /**
075     * Constructs this Lathe32RNG by dispersing the bits of seed using {@link #setSeed(int)} across the two parts of state
076     * this has.
077     * @param seed an int that won't be used exactly, but will affect both components of state
078     */
079    public Lathe32RNG(final int seed) {
080        setSeed(seed);
081    }
082    /**
083     * Constructs this Lathe32RNG by splitting the given seed across the two parts of state this has with
084     * {@link #setState(long)}.
085     * @param seed a long that will be split across both components of state
086     */
087    public Lathe32RNG(final long seed) {
088        setState(seed);
089    }
090    /**
091     * Constructs this Lathe32RNG by calling {@link #setState(int, int)} on stateA and stateB as given; see that method
092     * for the specific details (stateA and stateB are kept as-is unless they are both 0).
093     * @param stateA the number to use as the first part of the state; this will be 1 instead if both seeds are 0
094     * @param stateB the number to use as the second part of the state
095     */
096    public Lathe32RNG(final int stateA, final int stateB) {
097        setState(stateA, stateB);
098    }
099    
100    @Override
101    public final int next(int bits) {
102        final int s0 = stateA;
103        int s1 = stateB;
104        final int result = s0 + s1;
105        s1 ^= s0;
106        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
107        stateB = (s1 << 28 | s1 >>> 4); // c
108        return (result << 10 | result >>> 22) + s0 >>> (32 - bits);
109    }
110
111    /**
112     * Can return any int, positive or negative, of any size permissible in a 32-bit signed integer.
113     * @return any int, all 32 bits are random
114     */
115    public final int nextInt() {
116        final int s0 = stateA;
117        int s1 = stateB;
118        final int result = s0 + s1;
119        s1 ^= s0;
120        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
121        stateB = (s1 << 28 | s1 >>> 4); // c
122        return (result << 10 | result >>> 22) + s0;
123    }
124
125    @Override
126    public final long nextLong() {
127        final int s0 = stateA;
128        int s1 = stateB;
129        final int high = s0 + s1;
130        s1 ^= s0;
131        final int s00 = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
132        s1 = (s1 << 28 | s1 >>> 4); // c
133        final int low = s00 + s1;
134        s1 ^= s00;
135        stateA = (s00 << 13 | s00 >>> 19) ^ s1 ^ (s1 << 5); // a, b
136        stateB = (s1 << 28 | s1 >>> 4); // c
137        return (long)((high << 10 | high >>> 22) + s0) << 32 ^ ((low << 10 | low >>> 22) + s00);
138    }
139
140    /**
141     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
142     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
143     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
144     *
145     * @return a copy of this RandomnessSource
146     */
147    @Override
148    public Lathe32RNG copy() {
149        return new Lathe32RNG(stateA, stateB);
150    }
151
152    /**
153     * Sets the state of this generator using one int, running it through Zog32RNG's algorithm two times to get 
154     * two ints. If the states would both be 0, state A is assigned 1 instead.
155     * @param seed the int to use to produce this generator's state
156     */
157    public void setSeed(final int seed) {
158        int z = seed + 0xC74EAD55, a = seed ^ z;
159        a ^= a >>> 14;
160        z = (z ^ z >>> 10) * 0xA5CB3;
161        a ^= a >>> 15;
162        stateA = (z ^ z >>> 20) + (a ^= a << 13);
163        z = seed + 0x8E9D5AAA;
164        a ^= a >>> 14;
165        z = (z ^ z >>> 10) * 0xA5CB3;
166        a ^= a >>> 15;
167        stateB = (z ^ z >>> 20) + (a ^ a << 13);
168        if((stateA | stateB) == 0)
169            stateA = 1;
170    }
171
172    public int getStateA()
173    {
174        return stateA;
175    }
176    /**
177     * Sets the first part of the state to the given int. As a special case, if the parameter is 0 and stateB is
178     * already 0, this will set stateA to 1 instead, since both states cannot be 0 at the same time. Usually, you
179     * should use {@link #setState(int, int)} to set both states at once, but the result will be the same if you call
180     * setStateA() and then setStateB() or if you call setStateB() and then setStateA().
181     * @param stateA any int
182     */
183
184    public void setStateA(int stateA)
185    {
186        this.stateA = (stateA | stateB) == 0 ? 1 : stateA;
187    }
188    public int getStateB()
189    {
190        return stateB;
191    }
192
193    /**
194     * Sets the second part of the state to the given int. As a special case, if the parameter is 0 and stateA is
195     * already 0, this will set stateA to 1 and stateB to 0, since both cannot be 0 at the same time. Usually, you
196     * should use {@link #setState(int, int)} to set both states at once, but the result will be the same if you call
197     * setStateA() and then setStateB() or if you call setStateB() and then setStateA().
198     * @param stateB any int
199     */
200    public void setStateB(int stateB)
201    {
202        this.stateB = stateB;
203        if((stateB | stateA) == 0) stateA = 1;
204    }
205
206    /**
207     * Sets the current internal state of this Lathe32RNG with three ints, where stateA and stateB can each be any int
208     * unless they are both 0 (which will be treated as if stateA is 1 and stateB is 0).
209     * @param stateA any int (if stateA and stateB are both 0, this will be treated as 1)
210     * @param stateB any int
211     */
212    public void setState(int stateA, int stateB)
213    {
214        this.stateA = (stateA | stateB) == 0 ? 1 : stateA;
215        this.stateB = stateB;
216    }
217
218    /**
219     * Get the current internal state of the StatefulRandomness as a long.
220     *
221     * @return the current internal state of this object.
222     */
223    @Override
224    public long getState() {
225        return stateA & 0xFFFFFFFFL | ((long)stateB) << 32;
226    }
227
228    /**
229     * Set the current internal state of this StatefulRandomness with a long.
230     *
231     * @param state a 64-bit long. You should avoid passing 0; this implementation will treat it as 1.
232     */
233    @Override
234    public void setState(long state) {
235        stateA = state == 0 ? 1 : (int)(state & 0xFFFFFFFFL);
236        stateB = (int)(state >>> 32);
237    }
238
239    @Override
240    public String toString() {
241        return "Lathe32RNG with stateA 0x" + StringKit.hex(stateA) + " and stateB 0x" + StringKit.hex(stateB);
242    }
243
244    @Override
245    public boolean equals(Object o) {
246        if (this == o) return true;
247        if (o == null || getClass() != o.getClass()) return false;
248
249        Lathe32RNG lathe32RNG = (Lathe32RNG) o;
250
251        return stateA == lathe32RNG.stateA && stateB == lathe32RNG.stateB;
252    }
253
254    @Override
255    public int hashCode() {
256        return 31 * stateA + stateB;
257    }
258}
