007/**
008 * An RNG that has a drastically longer period than the other generators in SquidLib, other than MersenneTwister,
009 * without sacrificing speed or HTML target compatibility. If you don't already know what the period of an RNG is, this
010 * probably isn't needed for your purposes, or many purposes in games at all. It is primarily meant for applications
011 * that need to generate massive amounts of random numbers, more than pow(2, 64) (18,446,744,073,709,551,616), without
012 * repeating sequences of generated numbers. An RNG's period refers to the number of numbers it generates given a single
013 * seed before the sequence repeats from the beginning. The period of this class is pow(2, 1024) minus 1
014 * (179,769,313,486,231,590,772,930,519,078,902,473,361,797,697,894,230,657,273,430,081,157,732,675,805,500,963,132,708,
015 * 477,322,407,536,021,120,113,879,871,393,357,658,789,768,814,416,622,492,847,430,639,474,124,377,767,893,424,865,485,
016 * 276,302,219,601,246,094,119,453,082,952,085,005,768,838,150,682,342,462,881,473,913,110,540,827,237,163,350,510,684,
017 * 586,298,239,947,245,938,479,716,304,835,356,329,624,224,137,215).
018 * This class may be particularly useful in conjunction with the shuffle method of RNG; the period of an RNG determines
019 * how many possible "shuffles", a.k.a orderings or permutations, can be produced over all calls to a permuting method
020 * like shuffle. A LightRNG-backed RNG with a period of pow(2, 64) will only be able to produce all possible "shuffles"
021 * for lists or arrays of 20 items or less. If a LongPeriodRNG is given to the constructor of an RNG and a large enough
022 * state has been given to the LongPeriodRNG (the String or long[] constructors can allow this), then lists or arrays of up
023 * to 170 elements can have all possible orderings produced by shuffle(), though it will take near-impossibly-many calls.
024 * This class has 128 bytes of state plus more in overhead (compare to the 16-byte-with-overhead LightRNG), but due to
025 * its massive period and createMany() static method, you can create a large number of subsequences with rather long
026 * periods themselves from a single seed. This uses the xorshift-1024* algorithm, and has competitive speed.
027 * Created by Tommy Ettinger on 3/21/2016.
028 * Ported from CC0-licensed C code by Sebastiano Vigna, at http://xorshift.di.unimi.it/xorshift1024star.c
029 */
030public class LongPeriodRNG implements RandomnessSource {

052    /**
053     * Builds a LongPeriodRNG and initializes this class' 1024 bits of state with many calls to a SplitMix64-based RNG
054     * with a random seed influenced by Math.random() and also the time (in milliseconds to keep GWT compatibility),
055     * mixing Math.random() calls in to alter the SplitMix64 state at uneven intervals.
056     * @param seed a 64-bit seed; can be any value.
057     */
058    public LongPeriodRNG(long seed) {
059        reseed(seed);
060    }

062    public void reseed()
063    {
064        LightRNG lr = new LightRNG(System.currentTimeMillis() ^
065                (Double.doubleToLongBits(Math.random()) << 32) |
066                (Double.doubleToLongBits(Math.random()) & 0xffffffffL));
067        long ts = lr.nextLong() ^ lr.nextLong() ^ lr.nextLong();
068        if (ts == 0)
069            ts++;
070        choice = (int) (ts & 15);
071        state[0] = ts;
072        for (int i = 1; i < 16; i++) {
073            //Chosen by trial and error to unevenly reseed 4 times, where i is 2, 5, 10, or 13
074            if((6 & (i * 1281783497376652987L)) == 6)
075                lr.state ^= (Double.doubleToLongBits(Math.random()) << 32) |
076                    (Double.doubleToLongBits(Math.random()) & 0xffffffffL);
077            state[i] = lr.nextLong() ^ lr.nextLong() ^ lr.nextLong();
078            state[i - 1] ^= state[i];
079            if(state[i - 1] == 0) state[i - 1]++;
080        }
081    }
082    /**
083     * Reinitializes this class' 1024 bits of state with the given seed passed into SplitMix64, the algorithm also used by
084     * LightRNG. A different algorithm is used in actual number generating code, which is a recommended technique to
085     * generate seeds.
086     *
087     * @param seed a 64-bit seed; can be any value.
088     */
089    public void reseed(long seed) {
090        state = init(seed);
091        choice = (int) (seed & 15);
092    }
093    /**
094     * Builds a LongPeriodRNG and initializes this class' 1024 bits of state with the given seed, using a different
095     * strategy depending on the seed. If seed is null, this uses the same state as any other null seed. If seed is a
096     * String with length 15 or less, this generates a 64-bit hash of the seed and uses it in the same way the constructor
097     * that takes a long creates 1024 bits of state from a 64-bit seed. If seed is a String with length 16 or more, this
098     * splits the string up and generates 16 hashes from progressively smaller substrings of seed. The highest quality
099     * states will result from passing this a very long String.
100     *
101     * @param seed a String seed; can be any value, but produces the best results if it at least 16 characters long
102     */
103    public LongPeriodRNG(String seed) {
104        reseed(seed);
105    }

107    /**
108     * Reinitializes this class' 1024 bits of state with the given seed, using a different strategy depending on the seed.
109     * If seed is null, this uses the same state as any other null seed. If seed is a String with length 15 or less, this
110     * generates a 64-bit hash of the seed and uses it in the same way the constructor that takes a long creates 1024 bits
111     * of state from a 64-bit seed. If seed is a String with length 16 or more, this splits the string up and generates 16
112     * hashes from progressively smaller substrings of seed. The highest quality states will result from passing this a
113     * very long String.
114     *
115     * @param seed a String seed; can be any value, but produces the best results if it at least 16 characters long
116     */
117    public void reseed(String seed)
118    {
119        if (seed == null) {
120            state = new long[]{0x0101, 0x1212, 0x2323, 0x3434, 0x4545, 0x5656, 0x6767, 0x7878,
121                    0x8989, 0x9A9A, 0xABAB, 0xBCBC, 0xCDCD, 0xDEDE, 0xEFEF, 0xF0F0};
122            choice = 0;
123        } else {
124            int len = seed.length();
125            if (len < 16) {
126                long h = CrossHash.hash64(seed);
127                state = init(h);
128                choice = (int) (h & 15);
129            } else {
130                char[] chars = seed.toCharArray();
131                state = new long[]
132                        {
133                                validate(CrossHash.hash64(chars)),
134                                validate(CrossHash.hash64(chars, len / 16, len)),
135                                validate(CrossHash.hash64(chars, 2 * len / 16, len)),
136                                validate(CrossHash.hash64(chars, 3 * len / 16, len)),
137                                validate(CrossHash.hash64(chars, 4 * len / 16, len)),
138                                validate(CrossHash.hash64(chars, 5 * len / 16, len)),
139                                validate(CrossHash.hash64(chars, 6 * len / 16, len)),
140                                validate(CrossHash.hash64(chars, 7 * len / 16, len)),
141                                validate(CrossHash.hash64(chars, 8 * len / 16, len)),
142                                validate(CrossHash.hash64(chars, 9 * len / 16, len)),
143                                validate(CrossHash.hash64(chars, 10 * len / 16, len)),
144                                validate(CrossHash.hash64(chars, 11 * len / 16, len)),
145                                validate(CrossHash.hash64(chars, 12 * len / 16, len)),
146                                validate(CrossHash.hash64(chars, 13 * len / 16, len)),
147                                validate(CrossHash.hash64(chars, 14 * len / 16, len)),
148                                validate(CrossHash.hash64(chars, 15 * len / 16, len)),
149                        };
150                choice = (int) (state[0] & 15);
151            }
152        }
153    }
154    /**
155     * Builds a LongPeriodRNG and initializes this class' 1024 bits of state with the given seed as a long array, which
156     * may or may not have 16 elements (though it is less wasteful to run this with 16 longs since that is exactly 1024
157     * bits). If seed is null, this produces the same state as the String constructor does when given a null seed. If seed
158     * has fewer than 16 elements, this repeats earlier elements once it runs out of unused longs. If seed has 16 or more
159     * elements, this exclusive-ors elements after the sixteenth with longs it has already placed into the state, causing
160     * all elements of the seed to have an effect on the state, and making the 16-element case copy all longs exactly.
161     * @param seed a long array seed; can have any number of elements, though 16 is ideal
162     */
163    public LongPeriodRNG(long[] seed) {
164        reseed(seed);
165    }

167    /**
168     * Reinitializes this class' 1024 bits of state with the given seed as a long array, which may or may not have 16
169     * elements (though it is less wasteful to run this with 16 longs since that is exactly 1024 bits). If seed is null,
170     * this produces the same state as the String constructor does when given a null seed. If seed has fewer than 16
171     * elements, this repeats earlier elements once it runs out of unused longs. If seed has 16 or more elements, this
172     * exclusive-ors elements after the sixteenth with longs it has already placed into the state, causing all elements of
173     * the seed to have an effect on the state, and making the 16-element case copy all longs exactly.
174     * @param seed a long array seed; can have any number of elements, though 16 is ideal
175     */
176    public void reseed(long[] seed)
177    {
178        if (seed == null || seed.length == 0) {
179            state = new long[]{0x0101, 0x1212, 0x2323, 0x3434, 0x4545, 0x5656, 0x6767, 0x7878,
180                    0x8989, 0x9A9A, 0xABAB, 0xBCBC, 0xCDCD, 0xDEDE, 0xEFEF, 0xF0F0};
181            choice = 0;
182        } else if(seed.length < 16)
183        {
184            for (int i = 0, s = 0; i < 16; i++, s = (s+1) % seed.length) {
185                state[i] = seed[s];
186                if(state[i] == 0) state[i]++;
187            }
188            choice = (int) (state[0] & 15);
189        }
190        else
191        {
192            for (int i = 0, s = 0; s < seed.length; s++, i = (i+1) % 16) {
193                state[i] ^= seed[s];
194                if(state[i] == 0) state[i]++;
195            }
196            choice = (int) (state[0] & 15);
197        }
198    }

200    private static long validate(long seed)
201    {
202        if(seed == 0) return 1;
203        else return seed;
204    }
205    private static long[] init(long seed)
206    {
207        long[] state = new long[16];
208        for (int i = 0; i < 16; i++) {
209            long z = ( seed += 0x9E3779B97F4A7C15L );
210            z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
211            z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
212            state[i] = z ^ (z >>> 31);
213            if(state[i] == 0) state[i]++;
214        }
215        return state;
216    }
217    public LongPeriodRNG(LongPeriodRNG other)
218    {
219        choice = other.choice;
220        System.arraycopy(other.state, 0, state, 0, 16);
221    }

223    @Override
224    public int next( int bits ) {
225        return (int)( nextLong() & ( 1L << bits ) - 1 );
226    }

228    /**
229     * Can return any long, positive or negative, of any size permissible in a 64-bit signed integer.
230     * <br>
231     * Written by Sebastiano Vigna, from http://xorshift.di.unimi.it/xorshift1024star.c
232     * @return any long, all 64 bits are random
233     */
234    @Override
235    public long nextLong() {
236        final long s0 = state[choice];
237        long s1 = state[choice = (choice + 1) & 15];
238        s1 ^= s1 << 31; // a
239        state[choice] = s1 ^ s0 ^ (s1 >>> 11) ^ (s0 >>> 30); // b,c
240        return state[choice] * 1181783497276652981L;
241    }

243    /**
244     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
245     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
246     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
247     *
248     * @return a copy of this RandomnessSource
249     */
250    @Override
251    public RandomnessSource copy() {
252        LongPeriodRNG next = new LongPeriodRNG();
253        System.arraycopy(state, 0, next.state, 0, 16);
254        next.choice = choice;
255        return next;
256
257    }

259    /**
260     * This is the jump function for the generator. It is equivalent to 2^512 calls to nextLong(); it can be used to
261     * generate 2^512 non-overlapping subsequences for parallel computations. Alters the state of this object.
262     * <br>
263     * Written by Sebastiano Vigna, from http://xorshift.di.unimi.it/xorshift1024star.c , don't ask how it works.
264     * */
265    public void jump() {
266
267        long[] t = new long[16];
268        for(int i = 0; i < 16; i++)
269        for(int b = 0; b < 64; b++) {
270            if ((jumpTable[i] & 1L << b) != 0) {
271                for (int j = 0; j < 16; j++)
272                    t[j] ^= state[(j + choice) & 15];
273            }
274            nextLong();
275        }
276
277        for(int j = 0; j < 16; j++)
278            state[(j + choice) & 15] = t[j];
279    }

281    /**
282     * Creates many LongPeriodRNG objects in an array, where each will generate a sequence of 2 to the 512 numbers that
283     * will not overlap with other sequences in the array. The number of items in the array is specified by count.
284     * @param count the number of LongPeriodRNG objects to generate in the array.
285     * @return an array of LongPeriodRNG where none of the RNGs will generate overlapping sequences.
286     */
287    public static LongPeriodRNG[] createMany(int count)
288    {
289        if(count < 1) count = 1;
290        LongPeriodRNG origin = new LongPeriodRNG();
291        LongPeriodRNG[] values = new LongPeriodRNG[count];
292        for (int i = 0; i < count; i++) {
293            values[i] = new LongPeriodRNG(origin);
294            origin.jump();
295        }
296        return values;
297    }

375    public boolean equals(Object o) {
376        if (this == o) return true;
377        if (o == null || getClass() != o.getClass()) return false;
