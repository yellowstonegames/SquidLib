003import java.io.Serializable;

005/**
006 * Gets a sequence of distinct pseudo-random ints (typically used as indices) from 0 to some bound, without storing all
007 * of the sequence in memory. Uses a Feistel network, as described in
008 * <a href="https://blog.demofox.org/2013/07/06/fast-lightweight-random-shuffle-functionality-fixed/">this post by Alan Wolfe</a>.
009 * The API is very simple; you construct a LowStorageShuffler by specifying how many items it can shuffle, and you can
010 * optionally use a seed (it will be random if you don't specify a seed). Call {@link #next()} on a LowStorageShuffler
011 * to get the next distinct int in the shuffled ordering; next() will return -1 if there are no more distinct ints (if
012 * {@link #bound} items have already been returned). You can go back to the previous item with {@link #previous()},
013 * which similarly returns -1 if it can't go earlier in the sequence. You can restart the sequence with
014 * {@link #restart()} to use the same sequence over again, or {@link #restart(int)} to use a different seed (the bound
015 * is fixed).
016 * <br>
017 * This differs from the version in Alan Wolfe's example code and blog post; it uses a very different round function,
018 * and it only uses 2 rounds of it (instead of 4). Wolfe's round function is MurmurHash2, but as far as I can tell the
019 * version he uses doesn't have anything like MurmurHash3's fmix32() to adequately avalanche bits, and since all keys
020 * are small keys with the usage of MurmurHash2 in his code, avalanche is the most important thing. It's also perfectly
021 * fine to use irreversible operations in a Feistel network round function, and I do that since it seems to improve
022 * randomness slightly. The {@link #round(int, int)} method used here acts like {@link Coord#hashCode()}, but with two
023 * small multiplications included to increase randomness significantly. Using 4 rounds turns out to be overkill in this
024 * case. This also uses a different seed for each round.
025 * <br>
026 * This class is extremely similar to {@link SwapOrNotShuffler}, but LowStorageShuffler is optimized for usage on GWT
027 * while SwapOrNotShuffler is meant to have higher quality in general. There's also {@link ShuffledIntSequence}, which
028 * extends this class and uses different behavior so it "re-shuffles" the results when all results have been produced,
029 * and {@link SNShuffledIntSequence}, which extends SwapOrNotShuffler but is otherwise like ShuffledIntSequence.
030 * <br>
031 * Created by Tommy Ettinger on 9/22/2018.
032 * @author Alan Wolfe
033 * @author Tommy Ettinger
034 */
035public class LowStorageShuffler implements Serializable {
036    private static final long serialVersionUID = 1L;
037    public final int bound;
038    protected int index, pow4, halfBits, leftMask, rightMask;
039    protected int key0, key1;//, key2, key3;
040
041    /**
042     * Constructs a LowStorageShuffler with a random seed and a bound of 10.
043     */
044    public LowStorageShuffler(){
045        this(10);
046    }
047    /**
048     * Constructs a LowStorageShuffler with the given exclusive upper bound and a random seed.
049     * @param bound how many distinct ints this can return
050     */
051    public LowStorageShuffler(int bound)
052    {
053        this(bound, (int)((Math.random() * 2.0 - 1.0) * 0x80000000));
054    }
055
056    /**
057     * Constructs a LowStorageShuffler with the given exclusive upper bound and int seed.
058     * @param bound how many distinct ints this can return
059     * @param seed any int; will be used to get several seeds used internally
060     */
061    public LowStorageShuffler(int bound, int seed)
062    {
063        // initialize our state
064        this.bound = bound;
065        restart(seed);
066        // calculate next power of 4.  Needed since the balanced Feistel network needs
067        // an even number of bits to work with
068        pow4 = HashCommon.nextPowerOfTwo(bound);
069        pow4 = ((pow4 | pow4 << 1) & 0x55555554) - 1;
070        // calculate our left and right masks to split our indices for the Feistel network
071        halfBits = Integer.bitCount(pow4) >>> 1;
072        rightMask = pow4 >>> halfBits;
073        leftMask = pow4 ^ rightMask;
074    }
075
076    /**
077     * Gets the next distinct int in the sequence, or -1 if all distinct ints have been returned that are non-negative
078     * and less than {@link #bound}.
079     * @return the next item in the sequence, or -1 if the sequence has been exhausted
080     */
081    public int next()
082    {
083        int shuffleIndex;
084        // index is the index to start searching for the next number at
085        while (index <= pow4)
086        {
087            // get the next number
088            shuffleIndex = encode(index++);
089
090            // if we found a valid index, return it!
091            if (shuffleIndex < bound)
092                return shuffleIndex;
093        }
094
095        // end of shuffled list if we got here.
096        return -1;
097    }
098    /**
099     * Gets the previous returned int from the sequence (as yielded by {@link #next()}), or -1 if next() has never been
100     * called (or the LowStorageShuffler has reached the beginning from repeated calls to this method).
101     * @return the previously-given item in the sequence, or -1 if this can't go earlier
102     */
103    public int previous()
104    {
105        int shuffleIndex;
106        // index is the index to start searching for the next number at
107        while (index > 0)
108        {
109            // get the next number
110            shuffleIndex = encode(--index);
111
112            // if we found a valid index, return success!
113            if (shuffleIndex < bound)
114                return shuffleIndex;
115        }
116
117        // end of shuffled list if we got here.
118        return -1;
119    }
120
121    /**
122     * Starts the same sequence over from the beginning.
123     */
124    public void restart()
125    {
126        index = 0;
127    }
128
129    public static int determine(int z)
130    {
131        return (z = ((z = ((z = ((z *= 0xBDEAD) ^ z >>> 13) * 0x7FFFF) ^ z >>> 12) * 0x1FFFF) ^ z >>> 14) * 0x1FFF) ^ z >>> 15;
132
133    }    
134    /**
135     * Starts the sequence over, but can change the seed (completely changing the sequence). If {@code seed} is the same
136     * as the seed given in the constructor, this will use the same sequence, acting like {@link #restart()}.
137     * @param seed any int; will be used to get several seeds used internally
138     */
139    public void restart(int seed)
140    {
141        index = 0;
142        key0 = determine(seed ^ 0xDE4D * ~bound);
143        key1 = determine(key0 ^ 0xBA55 * bound);
144        key0 ^= determine(~key1 ^ 0xBEEF * bound);
145        key1 ^= determine(~key0 ^ 0x1337 * bound);
146    }
147
148    /**
149     * An irreversible mixing function that seems to give good results; GWT-compatible.
150     * This is mostly the same as {@link Coord#hashCode()}, with data acting like x and seed acting like y, but also
151     * includes two small multiplications (each by 16-bit numbers, which doesn't risk GWT safety). The algorithm is a
152     * little complicated, and is much like the 32-bit xoroshiro variant used by {@link Lathe32RNG}, but changes any
153     * additive operations to use XOR. Less involved schemes did not do well at all. There's four bitwise rotations and
154     * two multiplications used here, plus several XOR and bitwise shift operations.
155     * @param data the data being ciphered
156     * @param seed the current seed
157     * @return the ciphered data
158     */
159    public int round(int data, int seed)
160    {
161        
162        seed ^= data * 0xBCFD;
163        seed ^= (data << 13 | data >>> 19) ^ (seed << 5) ^ (seed << 28 | seed >>> 4);
164        data ^= (seed << 11 | seed >>> 21) * 0xC6D5;
165        return data ^ (data << 25 | data >>> 7);
166
167//        seed ^= data * 0xC6D5 + 0xB531A935;
168//        data ^= seed * 0xBCFD + 0x41C64E6D;
169//        seed ^= data * 0xACED;
170//        data ^= seed * 0xBA55;
171//        data += data >>> 21;
172//        seed += seed >>> 22;
173//        data += data << 8;
174//        seed += seed << 5;
175//        return data ^ seed;
176
177//        data += data >>> 21;
178//        seed += seed >>> 22;
179//        data += data << 8;
180//        seed += seed << 5;
181//        data += data >>> 16;
182//        seed += seed >>> 13;
183//        data += data << 9;
184//        seed += seed << 11;
185//        return data ^ seed;
186    }

188    /**
189     * Encodes an index with a 2-round Feistel network. It is possible that someone would want to override this method
190     * to use more or less rounds, but there must always be an even number.
191     * @param index the index to cipher; must be between 0 and {@link #pow4}, inclusive
192     * @return the ciphered index, which might not be less than bound but will be less than or equal to {@link #pow4}
193     */
194    public int encode(int index)
195    {
196        // break our index into the left and right half
197        int left = (index & leftMask) >>> halfBits;
198        int right = (index & rightMask);
199        // do 2 Feistel rounds
200        int newRight = left ^ (round(right, key0) & rightMask);
201        left = right;
202        right = newRight;
203        newRight = left ^ (round(right, key1) & rightMask);
204//        left = right;
205//        right = newRight;
206//        newRight = left ^ (round(right, key2) & rightMask);
207//        left = right;
208//        right = newRight;
209//        newRight = left ^ (round(right, key3) & rightMask);
210
211//        left = right;
212//        right = newRight;
213
214        // put the left and right back together to form the encrypted index
215//        return left << halfBits | right;
216        return right << halfBits | newRight;
217    }
