003import java.io.Serializable;

005/**
006 * Gets a sequence of distinct pseudo-random ints (typically used as indices) from 0 to some bound, without storing all
007 * of the sequence in memory. Uses a Feistel network, as described in
008 * <a href="https://blog.demofox.org/2013/07/06/fast-lightweight-random-shuffle-functionality-fixed/">this post by Alan Wolfe</a>.
009 * The API is very simple; you construct a LowStorageShuffler by specifying how many items it can shuffle, and you can
010 * optionally use a seed (it will be random if you don't specify a seed). Call {@link #next()} on a LowStorageShuffler
011 * to get the next distinct int in the shuffled ordering; next() will return -1 if there are no more distinct ints (if
012 * {@link #bound} items have already been returned). You can go back to the previous item with {@link #previous()},
013 * which similarly returns -1 if it can't go earlier in the sequence. You can restart the sequence with
014 * {@link #restart()} to use the same sequence over again, or {@link #restart(int)} to use a different seed (the bound
015 * is fixed).
016 * <br>
017 * This differs from the version in Alan Wolfe's example code and blog post; it uses a very different round function,
018 * and it only uses 2 rounds of it (instead of 4). Wolfe's round function is MurmurHash2, but as far as I can tell the
019 * version he uses doesn't have anything like MurmurHash3's fmix32() to adequately avalanche bits, and since all keys
020 * are small keys with the usage of MurmurHash2 in his code, avalanche is the most important thing. It's also perfectly
021 * fine to use irreversible operations in a Feistel network round function, and I do that since it seems to improve
022 * randomness slightly. The round function used here acts like a rather-robust 2-item hash. First, it runs a step where
023 * it takes one of the two parameters, multiplies it by a 16-bit value, "sometimes" adds a 32-bit value, and XORs that
024 * with the other parameter. It It runs this step twice each on both combinations of the two parameters, once doing the
025 * "sometimes" adding and once not, then runs a different Overton iadla generator on each parameter (one such generator
026 * is {@code x += x >>> 21; return (x += x << 8);}, for reference). It then returns data XORed with seed. Using 4 rounds
027 * turns out to be overkill in this case. This also uses a different seed for each round.
028 * <br>
029 * Created by Tommy Ettinger on 9/22/2018.
030 * @author Alan Wolfe
031 * @author Tommy Ettinger
032 */
033public class LowStorageShuffler implements Serializable {
034    private static final long serialVersionUID = 1L;
035    public final int bound;
036    protected int index, pow4, halfBits, leftMask, rightMask;
037    protected int key0, key1;//, key2, key3;
038
039    /**
040     * Constructs a LowStorageShuffler with the given exclusive upper bound and a random seed.
041     * @param bound how many distinct ints this can return
042     */
043    public LowStorageShuffler(int bound)
044    {
045        this(bound, (int)((Math.random() * 2.0 - 1.0) * 0x80000000));
046    }
047
048    /**
049     * Constructs a LowStorageShuffler with the given exclusive upper bound and int seed.
050     * @param bound how many distinct ints this can return
051     * @param seed any int; will be used to get several seeds used internally
052     */
053    public LowStorageShuffler(int bound, int seed)
054    {
055        // initialize our state
056        this.bound = bound;
057        restart(seed);
058        // calculate next power of 4.  Needed since the balanced Feistel network needs
059        // an even number of bits to work with
060        pow4 = HashCommon.nextPowerOfTwo(bound);
061        pow4 = ((pow4 | pow4 << 1) & 0x55555554) - 1;
062        // calculate our left and right masks to split our indices for the Feistel network
063        halfBits = Integer.bitCount(pow4) >>> 1;
064        rightMask = pow4 >>> halfBits;
065        leftMask = pow4 ^ rightMask;
066    }
067
068    /**
069     * Gets the next distinct int in the sequence, or -1 if all distinct ints have been returned that are non-negative
070     * and less than {@link #bound}.
071     * @return the next item in the sequence, or -1 if the sequence has been exhausted
072     */
073    public int next()
074    {
075        int shuffleIndex;
076        // index is the index to start searching for the next number at
077        while (index <= pow4)
078        {
079            // get the next number
080            shuffleIndex = encode(index++);
081
082            // if we found a valid index, return it!
083            if (shuffleIndex < bound)
084                return shuffleIndex;
085        }
086
087        // end of shuffled list if we got here.
088        return -1;
089    }
090    /**
091     * Gets the previous returned int from the sequence (as yielded by {@link #next()}), or -1 if next() has never been
092     * called (or the LowStorageShuffler has reached the beginning from repeated calls to this method).
093     * @return the previously-given item in the sequence, or -1 if this can't go earlier
094     */
095    public int previous()
096    {
097        int shuffleIndex;
098        // index is the index to start searching for the next number at
099        while (index > 0)
100        {
101            // get the next number
102            shuffleIndex = encode(index--);
103
104            // if we found a valid index, return success!
105            if (shuffleIndex < bound)
106                return shuffleIndex;
107        }
108
109        // end of shuffled list if we got here.
110        return -1;
111    }
112
113    /**
114     * Starts the same sequence over from the beginning.
115     */
116    public void restart()
117    {
118        index = 0;
119    }
120
121    /**
122     * Starts the sequence over, but can change the seed (completely changing the sequence). If {@code seed} is the same
123     * as the seed given in the constructor, this will use the same sequence, acting like {@link #restart()}.
124     * @param seed any int; will be used to get several seeds used internally
125     */
126    public void restart(int seed)
127    {
128        index = 0;
129        key0 = Light32RNG.determine(seed ^ 0xDE4D * ~bound);
130        key1 = Light32RNG.determine(key0 ^ 0xBA55 * bound);
131        key0 ^= Light32RNG.determine(~key1 ^ 0xBEEF * bound);
132        key1 ^= Light32RNG.determine(~key0 ^ 0x1337 * bound);
133    }
134
135    /**
136     * An irreversible mixing function that seems to give good results; GWT-compatible.
137     * First, it runs a step where it takes one of the two parameters, multiplies it by a 16-bit value, "sometimes" adds
138     * a 32-bit value, and XORs that with the other parameter. It It runs this step twice each on both combinations of
139     * the two parameters, once doing the "sometimes" adding and once not, then runs a different Overton iadla generator
140     * on each parameter (one such generator is {@code x += x >>> 21; return (x += x << 8);}, for reference). It then
141     * returns data XORed with seed. This is complicated, but less involved schemes did not do well at all. There's four
142     * multiplications used here, all by small enough values to not risk GWT safety.
143     * @param data the data being ciphered
144     * @param seed the current seed
145     * @return the ciphered data
146     */
147    protected int round(int data, int seed)
148    {
149        seed ^= data * 0xC6D5 + 0xB531A935;
150        data ^= seed * 0xBCFD + 0x41C64E6D;
151        seed ^= data * 0xACED;
152        data ^= seed * 0xBA55;
153        data += data >>> 21;
154        seed += seed >>> 22;
155        data += data << 8;
156        seed += seed << 5;
157//        data += data >>> 21;
158//        seed += seed >>> 22;
159//        data += data << 8;
160//        seed += seed << 5;
161//        data += data >>> 16;
162//        seed += seed >>> 13;
163//        data += data << 9;
164//        seed += seed << 11;
165        return data ^ seed;
166    }
167
168    /**
169     * Encodes an index with a 2-round Feistel network. It is possible that someone would want to override this method
170     * to use more or less rounds, but there must always be an even number.
171     * @param index the index to cipher; must be betweeen 0 and {@link #pow4}, inclusive
172     * @return the ciphered index, which might not be less than bound but will be less than or equal to {@link #pow4}
173     */
174    protected int encode(int index)
175    {
176        // break our index into the left and right half
177        int left = (index & leftMask) >>> halfBits;
178        int right = (index & rightMask);
179        // do 2 Feistel rounds
180        int newRight = left ^ (round(right, key0) & rightMask);
181        left = right;
182        right = newRight;
183        newRight = left ^ (round(right, key1) & rightMask);
184//        left = right;
185//        right = newRight;
186//        newRight = left ^ (round(right, key2) & rightMask);
187//        left = right;
188//        right = newRight;
189//        newRight = left ^ (round(right, key3) & rightMask);
190
191//        left = right;
192//        right = newRight;
193
194        // put the left and right back together to form the encrypted index
195//        return left << halfBits | right;
196        return right << halfBits | newRight;
197    }
198}
