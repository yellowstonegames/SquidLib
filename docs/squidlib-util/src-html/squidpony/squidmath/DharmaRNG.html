007/**
008 * An alteration to a RandomnessSource that attempts to produce values that are perceived as fair to an imperfect user.
009 * <p>
010 * This takes a RandomnessSource, defaulting to a LinnormRNG, and uses it to generate random values, but tracks the total
011 * and compares it to the potential total of a generator of only numbers with a desired value (default 0.54,
012 * so it compares against a sequence of all 0.54). If the current generated total is too high or low compared to the
013 * desired total, the currently used seed is possibly changed, the generated number is moved in the direction of the
014 * desired fairness, and it returns that instead of the number that would have pushed the current generated total
015 * beyond the desired threshold. The new number, if one is changed, will always be closer to the desired fairness.
016 * This is absolutely insecure for cryptographic purposes, but should seem more "fair" to a player than a
017 * random number generator that seeks to be truly random.
018 * You can create multiple DharmaRNG objects with different fairness values and seeds, and use favorable generators
019 * (with fairness greater than 0.54) for characters that need an easier time, or unfavorable generators if you want
020 * the characters that use that RNG to be impeded somewhat.
021 * The name comes from the Wheel of Dharma.
022 * This class currently will have a slight bias toward lower numbers with many RNGs unless fairness is tweaked; 0.54
023 * can be used as a stand-in because 0.5 leans too low.
024 *
025 * <p>
026 * You can get values from this generator with: {@link #nextDouble()}, {@link #nextInt()},
027 *   {@link #nextLong()}, and the bounded variants on each of those.
028 * <p>
029 * You can alter the tracking information or requested fairness with {@link #resetFortune()},
030 *   {@link #setFairness(double)}, and {@link #getFairness()}.
031 *
032 * Created by Tommy Ettinger on 5/2/2015.
033 */
034public class DharmaRNG extends RNG implements Serializable{
035
036        /** Used to tweak the generator toward high or low values. */
037    private double fairness = 0.54;
038
039    /** Running total for what this has actually produced. */
040    private double produced = 0.0;
041
042    /** Running total for what this would produce if it always produced a value equal to fairness. */
043    private double baseline = 0.0;
044
045        private static final long serialVersionUID = -8919455766853811999L;

047    /**
048     * Constructs a DharmaRNG with a pseudo-random seed from Math.random().
049     */
050    public DharmaRNG()
051    {
052        this((long)(Math.random() * ((1L << 50) - 1)));
053    }

055     * Construct a new DharmaRNG with the given seed.

063    /**
064     * Construct a new DharmaRNG with the given seed.
065     *
066     * @param seed used to seed the default RandomnessSource.
067     * @param fairness the desired fairness metric, which must be between 0.0 and 1.0
068     */
069    public DharmaRNG(final long seed, final double fairness) {
070        super(seed);
071        if(fairness < 0.0 || fairness >= 1.0)
072            this.fairness = 0.54;
073        else
074            this.fairness = fairness;
075    }

127    /**
128     * Generate a random double, altering the result if recently generated results have been leaning
129     * away from this class' fairness value.
130     * @return a double between 0.0 (inclusive) and 1.0 (exclusive)
131     */
132    @Override
133    public double nextDouble() {
134        double gen = (random.nextLong() & 0x1fffffffffffffL) * 0x1p-53;
135        /*if(Math.abs((produced + gen) - (baseline + fairness)) > 1.5) {
136            //do some reseeding here if possible
137        }*/
138        if(Math.abs((produced + gen) - (baseline + fairness)) > 0.5)
139        {
140            gen = (gen + fairness) / 2.0;
141            produced *= 0.5;
142            baseline *= 0.5;
143            produced += gen;
144            baseline += fairness;
145            return gen;
146        }
147        else
148        {
149            produced += gen;
150            baseline += fairness;
151            return gen;
152        }
153    }

155    /**
156     * Returns a random integer below the given bound, or 0 if the bound is 0 or
157     * negative. Affects the current fortune.
158     *
159     * @param bound the upper bound (exclusive)
160     * @return the found number
161     */
162    @Override
163    public int nextInt(int bound) {
164        if (bound <= 0) {
165            return 0;
166        }
167
168        return (int)(nextDouble() * bound);
169    }
170
171    /**
172     * Returns a random integer, which may be any positive or negative value. Affects the current fortune.
173     * @return A random int (can be any int, without restriction)
174     */
175    @Override
176    public int nextInt() {
177        return (int)((nextDouble() - 0.5) * 2.0 * 0x7FFFFFFF);
178    }
179
180    /**
181     * Returns a random long, which may be any positive or negative value. Affects the current fortune.
182     * @return A random long (can be any long, without restriction)
183     */
184    @Override
185    public long nextLong() {
186        return (long)((nextDouble() - 0.5) * 2.0 * 0x7FFFFFFFFFFFFFFFL);
187    }
188
189    /**
190     * Returns a random long below the given bound, or 0 if the bound is 0 or
191     * negative.
192     *
193     * @param bound the upper bound (exclusive)
194     * @return the found number
195     */
196    @Override
197        public long nextLong(long bound) {
198        if (bound <= 0) {
199            return 0;
200        }
201        return (long)(nextDouble() * bound);
202    }
203    /**
204     * Gets the measure that this class uses for RNG fairness, defaulting to 0.54 (always between 0.0 and 1.0).
205     * @return the current fairness metric.
206     */
207    public double getFairness() {
208        return fairness;
209    }
210
211    /**
212     * Sets the measure that this class uses for RNG fairness, which must always be between 0.0 and 1.0, and will be
213     * set to 0.54 if an invalid value is passed.
214     * @param fairness the desired fairness metric, which must be 0.0 &lt;= fairness &lt; 1.0
215     */
216    public void setFairness(double fairness) {
217        if(fairness < 0.0 || fairness >= 1.0)
218            this.fairness = 0.54;
219        else
220            this.fairness = fairness;
221    }

223    /**
224     * Gets the status of the fortune used when calculating fairness adjustments.
225     * @return the current value used to determine whether the results should be adjusted toward fairness.
226     */
227    public double getFortune()
228    {
229        return Math.abs(produced - baseline);
230    }
231
232    /**
233     * Resets the stored history this RNG uses to try to ensure fairness.
234     */
235    public void resetFortune()
236    {
237        produced = 0.0;
238        baseline = 0.0;
239    }
240    /**
241     *
242     * @param bits the number of bits to be returned
243     * @return a random int of the number of bits specified.
244     */
245    @Override
246    public int next(int bits) {
247        if(bits <= 0)
248            return 0;
249        if(bits > 32)
250            bits = 32;
251        return (int)(nextDouble() * (1l << bits));
252
253    }
254
255    /**
256     * Returns a value between min (inclusive) and max (exclusive).
257     * <p>
258     * The inclusive and exclusive behavior is to match the behavior of the
259     * similar method that deals with floating point values.
260     *
261     * @param min the minimum bound on the return value (inclusive)
262     * @param max the maximum bound on the return value (exclusive)
263     * @return the found value
264     */
265    @Override
266    public long between(long min, long max) {
267        return min + nextLong(max - min);
268    }
269
270    /**
271     * Returns a random non-negative integer below the given bound, or 0 if the bound is 0.
272     * Uses a slightly optimized technique. This method is considered "hasty" since
273     * it should be faster than nextInt() doesn't check for "less-valid" bounds values. It also
274     * has undefined behavior if bound is negative, though it will probably produce a negative
275     * number (just how negative is an open question).
276     *
277     * @param bound the upper bound (exclusive); behavior is undefined if bound is negative
278     * @return the found number
279     */
280    @Override
281    public int nextIntHasty(int bound) {
282        return (int)(nextDouble() * bound);
283    }

286    public float nextFloat() {
287        return (float)nextDouble();
288    }

290    @Override
291    public boolean nextBoolean() {
292        return nextDouble() >= 0.5;
293    }

305    /**
306     * Creates a copy of this DharmaRNG; it will generate the same random numbers, given the same calls in order, as
307     * this DharmaRNG at the point copy() is called. The copy will not share references with this DharmaRNG.
308     *
309     * @return a copy of this DharmaRNG
310     */
311    @Override
312    public DharmaRNG copy() {
313        DharmaRNG next = new DharmaRNG(random.copy(), fairness);
314        next.produced = produced;
315        next.baseline = baseline;
316        return next;
317    }

319    /**
320     * Gets a random portion of data (an array), assigns that portion to output (an array) so that it fills as much as
321     * it can, and then returns output. Will only use a given position in the given data at most once; does this by
322     * shuffling a copy of data and getting a section of it that matches the length of output.
323     *
324     * Based on http://stackoverflow.com/a/21460179 , credit to Vincent van der Weele; modifications were made to avoid
325     * copying or creating a new generic array (a problem on GWT).
326     * @param data an array of T; will not be modified.
327     * @param output an array of T that will be overwritten; should always be instantiated with the portion length
328     * @param <T> can be any non-primitive type.
329     * @return an array of T that has length equal to output's length and may contain null elements if output is shorter
330     * than data
331     */
332    @Override
333    public <T> T[] randomPortion(T[] data, T[] output) {
334        return super.randomPortion(data, output);
335    }

342     * @param count the non-negative number of elements to randomly take from data

374    @Override
375    public boolean equals(Object o) {
376        if (this == o) return true;
377        if (o == null || getClass() != o.getClass()) return false;
378
379        DharmaRNG dharmaRNG = (DharmaRNG) o;
380
381        if (Double.compare(dharmaRNG.fairness, fairness) != 0) return false;
382        return Double.compare(dharmaRNG.produced, produced) == 0 && Double.compare(dharmaRNG.baseline, baseline) == 0;
383
384    }

386    @Override
387    public int hashCode() {
388        int result;
389        long temp;
390        temp = NumberTools.doubleToLongBits(fairness);
391        result = (int) (temp ^ (temp >>> 32));
392        temp = NumberTools.doubleToLongBits(produced);
393        result = 31 * result + (int) (temp ^ (temp >>> 32));
394        temp = NumberTools.doubleToLongBits(baseline);
395        result = 31 * result + (int) (temp ^ (temp >>> 32));
396        return result;
397    }
