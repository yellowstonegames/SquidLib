065    public static int rawNoise2D(final int x, final int y) {
066        return ((x * 0x9E3779B9 ^ 0xC6BC2796) * (y * 0x92B5CC83 + 0xD9B4E019) - (x ^ ~y) * 0x632BE59B * (~x + y) * 0x7F4A7C15) >>> 8 & 0xff;
067    }

070        return (((z * 0xD0E89D2D + 0x311E289F) ^ x * 0x9E3779B9 ^ 0xC6BC2796 ^ (y * 0x92B5CC83 + 0xD9B4E019)) +
071                //(z * 0xD0E89D2D ^ x * 0xC6BC2796 ^ y * 0x92B5CC83 ^
072                (x ^ ~y + z) * 0x632BE59B + (z ^ ~x + y) * 0x7F4A7C15 + (y ^ ~z + x) * 0x9E3779B9) >>> 8 & 0xff; //0x311E289F
073    }
074    public static int rawNoise2D_alt(final int x, final int y) {
075
076        int s,
077                result = (s = 0x632BE5AB ^ x * 0x85157AF5) + 0xC5C5581A // equal to 0x62E2AC0D + 0x62E2AC0D
078                        + (s ^= y * 0x85157AF5);
079        return (result ^ ((s ^ result) >>> 8) * 0x9E3779B9) >>> 24;
080    }

086                        + (s ^= z * 0x85157AF5);

097        return (rawNoise2D(x - 1, y - 1) & 43) + (rawNoise2D(x - 1, y + 1) & 43) +
098                (rawNoise2D(x + 1, y - 1) & 43) + (rawNoise2D(x + 1, y + 1) & 43) +

100                (rawNoise2D(x, y - 1) & 53) + (rawNoise2D(x, y - 1) & 53) +
101                (rawNoise2D(x, y) & 127) >>> 1;
102    }/**

231        if (zoom < 1)
232            return 0;
233        int v = 0, a = x << 1, b = y << 1;
234        for (int s = zoom; s > 0; s--) {
235            v += noise2D((a += 379) / s, (b += 379) / s) + noise2D(a / s, (b + 1) / s) + noise2D((a + 1) / s, b / s);
236        }
237        //return v / zoom;

239        double adj = Math.sin((v / (zoom * 1530.0)) * Math.PI);
240        return (int)(adj * adj * 255.5);
241        */
242        double adj = Math.sin((v / (zoom * 1530.0)) * Math.PI);
243        adj *= adj;
244        return (int) (Math.pow(adj, 2.0 - 2.0 * adj) * 255.5);
245    }
246    public static int noise2D_alt(final int x, final int y, final int zoom) {
247        if (zoom < 1)
248            return 0;
249        int v = 0, a = x, b = y, t = 0, m;
250        for (int s = zoom, u = 1; s > 0; s--, u++) {
251            v += ((m = noise2D_alt((a += 76379) >>> s, (b += 76379) >>> s))
252                    + noise2D_alt((a+(m&2)-1) >>> s, (b+((m&1)<<1)-1) >>> s)
253            ) << u;
254            t += 1 << u;
255        }
256        return lookup[(v / (t << 1))];
257    }
