001package squidpony.squidmath;

003import squidpony.StringKit;

005import java.io.Serializable;

007/**
008 * A RandomnessSource based on PCG-Random that has a single int of state. Its period is extremely short at 2 to the 32,
009 * but its quality over that period is high even though its speed is not especially noteworthy. This generator is not
010 * suitable for GWT; for that you should use {@link Zag32RNG} if you need a StatefulRandomness or {@link Oriole32RNG} if
011 * you need higher period and speed but don't need StatefulRandomness' state adjustment methods.
012 * <br>
013 * Quality should be excellent in this version (at least for a generator with so little state) since it's based directly
014 * on PCG-Random's choices of numerical constants. Visual tests, at least, appear indistinguishable from other PRNGs.
015 * Period is very low, at 2 to the 32, but all seeds should be valid, including 0. Generating 64 bits of random data
016 * takes a little less than twice as much time as generating 32 bits, since this can avoid some overhead via inlining.
017 * <br>
018 * The name can be construed as Pint-Size, since this has a small period and uses a smaller amount of space, or as
019 * Permuted Int, since this is based on PermutedRNG, changed to use 32-bit operations on ints.
020 * <br>
021 * Based on work by Melissa E. O'Neill for PCG-Random; some code has been ported more-directly than other sections, but
022 * the foundation for this class would not be possible without O'Neill's work.
023 * <br>
024 * Created by Tommy Ettinger on 11/15/2016.
025 */
026public final class PintRNG implements RandomnessSource, StatefulRandomness, Serializable {
027
028    /** 2 raised to the 53, - 1. */
029    private static final long DOUBLE_MASK = ( 1L << 53 ) - 1;
030    /** 2 raised to the -53. */
031    private static final double NORM_53 = 1. / ( 1L << 53 );
032    /** 2 raised to the 24, -1. */
033    private static final long FLOAT_MASK = ( 1L << 24 ) - 1;
034    /** 2 raised to the -24. */
035    private static final double NORM_24 = 1. / ( 1L << 24 );
036
037    private static final long serialVersionUID = -374415589203474497L;
038
039    public int state; /* The state can be seeded with any value. */
040
041    /** Creates a new generator seeded using Math.random. */
042    public PintRNG() {
043        this((int)((Math.random() - 0.5) * 4.294967296E9));
044    }
045
046    public PintRNG( final long seed ) {
047        setState(seed);
048    }
049
050    public PintRNG(final int a)
051    {
052        state = a;
053    }
054
055    @Override
056    public int next( int bits )
057    {
058        int p = (state = state * 0x2C9277B5 + 0xAC564B05);
059        p ^= p >>> (4 + (p >>> 28));
060        return (((p *= 0x108EF2D9) >>> 22) ^ p) >>> (32 - bits);
061    }
062
063    /**
064     * Can return any int, positive or negative, of any size permissible in a 32-bit signed integer.
065     * @return any int, all 32 bits are random
066     */
067    public int nextInt() {
068        // increment  = 2891336453;
069        // multiplier = 747796405;
070
071//        int p = state;
072//        p ^= p >>> (4 + (p >>> 28));
073//        state = state * 0x2C9277B5 + 0xAC564B05;
074        int p = (state = state * 0x2C9277B5 + 0xAC564B05);
075        p ^= p >>> (4 + (p >>> 28));
076        return ((p *= 0x108EF2D9) >>> 22) ^ p;
077    }
078
079    /**
080     * Can return any long, positive or negative, of any size permissible in a 64-bit signed integer.
081     * Internally, generates two random 32-bit values and combines them into one random long.
082     * @return any long, all 64 bits are random
083     */
084    @Override
085    public long nextLong() {
086        int p = (state = state * 0x2C9277B5 + 0xAC564B05);
087        p ^= p >>> (4 + (p >>> 28));
088        int q = (state = state * 0x2C9277B5 + 0xAC564B05);
089        q ^= q >>> (4 + (q >>> 28));
090        return (((p *= 0x108EF2D9) >>> 22) ^ p) | ((((q *= 0x108EF2D9) >>> 22) ^ q) & 0xffffffffL) << 32;
091    }
092
093    /**
094     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
095     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
096     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
097     *
098     * @return a copy of this RandomnessSource
099     */
100    @Override
101    public PintRNG copy() {
102        return new PintRNG(state);
103    }
104
105    /**
106     * Exclusive on the upper bound.  The lower bound is 0.
107     * <br>
108     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
109     * @param bound the upper bound; should be positive
110     * @return a random int less than n and at least equal to 0
111     */
112    public int nextInt( final int bound ) {
113        return (bound <= 0) ? 0 : (int)((bound * (nextInt() & 0x7FFFFFFFL)) >> 31);
114    }
115    /**
116     * Inclusive lower, exclusive upper.
117     * @param lower the lower bound, inclusive, can be positive or negative
118     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
119     * @return a random int at least equal to lower and less than upper
120     */
121    public int nextInt( final int lower, final int upper ) {
122        if ( upper - lower <= 0 ) throw new IllegalArgumentException("Upper bound must be greater than lower bound");
123        return lower + nextInt(upper - lower);
124    }
125
126    /**
127     * Gets a uniform random double in the range [0.0,1.0)
128     * @return a random double at least equal to 0.0 and less than 1.0
129     */
130    public double nextDouble() {
131        return ( nextLong() & DOUBLE_MASK ) * NORM_53;
132    }
133
134    /**
135     * Gets a uniform random double in the range [0.0,outer) given a positive parameter outer. If outer
136     * is negative, it will be the (exclusive) lower bound and 0.0 will be the (inclusive) upper bound.
137     * @param outer the exclusive outer bound, can be negative
138     * @return a random double between 0.0 (inclusive) and outer (exclusive)
139     */
140    public double nextDouble(final double outer) {
141        return nextDouble() * outer;
142    }
143
144    /**
145     * Gets a uniform random float in the range [0.0,1.0)
146     * @return a random float at least equal to 0.0 and less than 1.0
147     */
148    public float nextFloat() {
149        return (float)( ( nextInt() & FLOAT_MASK ) * NORM_24 );
150    }
151
152    /**
153     * Gets a random value, true or false.
154     * Calls nextInt() once.
155     * @return a random true or false value.
156     */
157    public boolean nextBoolean() {
158        return nextInt() > 0;
159    }
160
161    /**
162     * Given a byte array as a parameter, this will fill the array with random bytes (modifying it
163     * in-place). Calls nextInt() {@code Math.ceil(bytes.length / 4.0)} times.
164     * @param bytes a byte array that will have its contents overwritten with random bytes.
165     */
166    public void nextBytes( final byte[] bytes ) {
167        int i = bytes.length, n = 0;
168        while( i != 0 ) {
169            n = Math.min( i, 4 );
170            for ( int bits = nextInt(); n-- != 0; bits >>>= 8 ) bytes[ --i ] = (byte)bits;
171        }
172    }

176    /**
177     * Sets the current state of this generator (an int) using only the least-significant 32 bits of seed (by casting
178     * a mask of those bits in seed to int, which helps ensure that a full 32 bits of state are possible). Giving
179     * int seeds should set the seed to an identical int; long seeds will lose any information in higher bits (including
180     * the sign, so 0xFFFFFFFF00000000L, which is a negative long, would be treated as 0 since only the 0x00000000 part
181     * at the end is actually used).
182     * @param seed the seed to use for this PintRNG, as if it was constructed with this seed.
183     */
184    @Override
185    public void setState( final long seed ) {
186        state = (int)(seed & 0xFFFFFFFFL);
187    }
188    /**
189     * Gets the current state of this generator.
190     * @return the current seed of this PintRNG, changed once per call to nextInt()
191     */
192    @Override
193    public long getState() {
194        return state;
195    }
196
197    @Override
198    public String toString() {
199        return "PintRNG with state 0x" + StringKit.hex(state);
200    }

202    @Override
203    public boolean equals(Object o) {
204        if (this == o) return true;
205        if (o == null || getClass() != o.getClass()) return false;
206
207        PintRNG pintRNG = (PintRNG) o;
208
209        return state == pintRNG.state;
210
211    }

213    @Override
214    public int hashCode() {
215        return state;
216    }
217    /**
218     * Advances or rolls back the PintRNG's state without actually generating each number. Skip forward
219     * or backward a number of steps specified by advance, where a step is equal to one call to nextInt(),
220     * and returns the random number produced at that step (you can get the state with {@link #getState()}).
221     * <br>
222     * The method used here is based on Brown, "Random Number Generation with Arbitrary Stride,", Transactions of the
223     * American Nuclear Society (Nov. 1994). The code is mostly the same as in PCG-Random's C port by M.E. O'Neill,
224     * specifically <a href="https://github.com/imneme/pcg-c/blob/master/src/pcg-advance-32.c">this file</a>. Skipping
225     * ahead or behind takes more than constant time, unlike with {@link LightRNG}, but less time than calling nextInt()
226     * {@code advance} times. Skipping backwards by one step is the worst case for this.
227     * @param advance Number of future generations to skip past. Can be negative to backtrack.
228     * @return the int that would be generated after generating advance random numbers.
229     */
230    public int skip(int advance)
231    {
232        int acc_mult = 1;
233        int acc_plus = 0;
234        int cur_mult = 0x2C9277B5;
235        int cur_plus = 0xAC564B05;
236        while (advance != 0) {
237            if ((advance & 1) == 1) {
238                acc_mult *= cur_mult;
239                acc_plus = acc_plus * cur_mult + cur_plus;
240            }
241            cur_plus *= (cur_mult + 1);
242            cur_mult *= cur_mult;
243            advance >>>= 1;
244        }
245        int p = (state = acc_mult * state + acc_plus);
246        p ^= p >>> (4 + (p >>> 28));
247        return ((p *= 0x108EF2D9) >>> 22) ^ p;
248    }

250    /**
251     * Gets a pseudo-random int that is a permutation of {@code state}, which is an int.
252     * This should normally be called with a technique like
253     * {@code PintRNG.determine(state = state * 0x2C9277B5 + 0xAC564B05)}, where 0xAC564B05 can be changed to any odd
254     * constant as long as it is the same across calls to this. You can effectively produce multiple uncorrelated
255     * streams by adding different constants in place of 0xAC564B05, applied to different states.
256     * @param state any int, but should be updated like {@code state = state * 0x2C9277B5 + 0xAC564B05}, where 0xAC564B05 can be any odd-number constant
257     * @return any int, pseudo-randomly obtained from state
258     */
259    public static int determine(int state)
260    {
261        state ^= state >>> (4 + (state >>> 28));
262        return ((state *= 0x108EF2D9) >>> 22) ^ state;
263    }
264    /**
265     * Like {@link #determine(int)}, gets a pseudo-random int that is a permutation of {@code state}, which is an int.
266     * Unlike determine(), this static method performs an extra step to avoid correlation between similar inputs, such
267     * as 4, 5, and 6, and their outputs. If you already give very distant numbers as subsequent inputs to determine(),
268     * then you should continue to use that method unless you discover issues with correlation; otherwise it's not a bad
269     * idea to default to this method, though it is somewhat slower than determine(). This method is safe to use with
270     * sequential ints, so you can call it with the technique {@code PintRNG.disperse(++state)}, or just use it on int
271     * data as you obtain it to randomize its values.
272     * @param state any int
273     * @return any int, pseudo-randomly obtained from state
274     */
275    public static int disperse(int state)
276    {
277        state = ((state >>> 19 | state << 13) ^ 0x13A5BA1D);
278        state ^= state >>> (4 + (state >>> 28));
279        return ((state *= 0x108EF2D9) >>> 22) ^ state;
280    }
281    public static int determine(final int a, final int b)
282    {
283        int state = a * 0x9E3779B9 + b * 0x85157AF5;
284        state ^= state >>> (4 + (state >>> 28));
285        return ((state *= 0x108EF2D9) >>> 22) ^ state;
286    }

288    public static int determineBounded(int state, final int bound)
289    {
290        state ^= state >>> (4 + (state >>> 28));
291        return (int)((bound * ((((state *= 0x108EF2D9) >>> 22) ^ state) & 0x7FFFFFFFL)) >>> 31);
292    }
293    public static int disperseBounded(int state, final int bound)
294    {
295        state = ((state >>> 19 | state << 13) ^ 0x13A5BA1D);
296        state ^= state >>> (4 + (state >>> 28));
297        return (int)((bound * ((((state *= 0x108EF2D9) >>> 22) ^ state) & 0x7FFFFFFFL)) >>> 31);
298    }
