098    public static float[] randomUnitVectors(final int size, final int len) {
099        long seed = 0xC6BC279692B5C483L * len;
100        float[] vectors = new float[size * len];
101        float mag = 0.0f, v1, v2, s;
102        for (int u = 0; u < size; u++) {
103            for (int i = 0; i < len; ) {
104
105                do {
106                    v1 = ZapRNG.randomSignedFloat(seed, seed += 0xAE3779BE3779B9L); // between -1 and 1
107                    v2 = ZapRNG.randomSignedFloat(seed >> 1, seed + 0xDE3779BE3779B9L); // between -1 and 1
108                    s = v1 * v1 + v2 * v2;
109                } while (s >= 1 || s == 0);
110                double multiplier = Math.sqrt(-2 * Math.log(s) / s);
111                v1 *= multiplier;
112                vectors[u * len + i++] = v1;
113                mag += v1 * v1;
114                if (i < len) {
115                    vectors[u * len + i++] = (v2 *= multiplier);
116                    mag += v2 * v2;
117                }
118            }
119            if (mag == 0.0) {
120                vectors[u * len] = 1.0f;
121            } else {
122                mag = (float)Math.sqrt(mag);
123                for (int i = 0; i < len; i++) {
124                    vectors[u * len + i] /= mag;
125                }
126            }
127        }
128        return vectors;
129    }
