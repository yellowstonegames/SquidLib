029    /*
030     * Quintic-interpolates between start and end (valid floats), with a between 0 (yields start) and 1 (yields end).
031     * Will smoothly transition toward start or end as a approaches 0 or 1, respectively.
032     * @param start a valid float
033     * @param end a valid float
034     * @param a a float between 0 and 1 inclusive
035     * @return a float between x and y inclusive
036     */
037    public static double querp(final double start, final double end, double a){
038        return (1.0 - (a *= a * a * (a * (a * 6.0 - 15.0) + 10.0))) * start + a * end;
039    }
040    /**
041     * Like {@link Math#floor}, but returns an int. Doesn't consider weird doubles like INFINITY and NaN.
042     * @param t the double to find the floor for
043     * @return the floor of t, as an int
044     */
045    public static long fastFloor(double t) {
046        return t >= 0 ? (long) t : (long) t - 1;
047    }

056public static double gauss(long state) {

067    public double getNoiseWithSeed(final double x, final double y, final int seed) {
068        final long
069                rs  = ThrustRNG.determine(seed ^ (long)~seed << 32),
070                rx = (rs >>> 23 ^ rs << 23) * (rs | 1L),
071                ry = (rx >>> 23 ^ rx << 23) * (rx | 1L);
072        return getNoiseWithSeeds(x, y, rx, ry, gauss(rx) + 0.625, gauss(ry) + 0.625);

075    public double getNoiseWithSeeds(final double x, final double y,
076                                    final long seedX, final long seedY,
077                                    final double randX, final double randY) {
078        final double
079                grx = randX * 0.625,
080                gry = randY * 0.625,
081                cx = NumberTools.sway(x) * (gry + 1.125) * 0x0.93p-1,
082                cy = NumberTools.sway(y) * (grx + 1.125) * 0x0.93p-1,
083                ax = x + (cy * (0.35 + gry)),
084                ay = y + (cx * (0.35 + grx)),
085                mx = ((((seedX & 0x1F0L) | 0x2FL) - 0x1FFp-1) * 0x1.4p-8) * ax + ((((seedX & 0x1F000L) | 0x2F00L) - 0x1FF00p-1) * 0x1.1p-16) * ay,
086                my = ((((seedY & 0x1F0L) | 0x2FL) - 0x1FFp-1) * 0x1.4p-8) * ay + ((((seedY & 0x1F000L) | 0x2F00L) - 0x1FF00p-1) * 0x1.1p-16) * ax;
087        final long
088                xf = fastFloor(mx),
089                yf = fastFloor(my);
090        return
091                //NumberTools.bounce(5.875 + 2.0625 * (
092                NumberTools.sway( 2.1625 * (
093                        querp(gauss(xf * 0xAE3779B97F4A7E35L),
094                                gauss((xf+1) * 0xAE3779B97F4A7E35L),
095                                mx - xf)
096                                + querp(gauss(yf * 0xBE3779B97F4A7C55L),
097                                gauss((yf+1) * 0xBE3779B97F4A7C55L),
098                                my - yf)));
099
100    }
