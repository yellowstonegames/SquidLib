025/** A resizable, ordered or unordered variable-length int array. Avoids boxing that occurs with ArrayList of Integer.
026 * If unordered, this class avoids a memory copy when removing elements (the last element is moved to the removed
027 * element's position).
028 * <br>
029 * Was called IntArray in libGDX; to avoid confusion with the fixed-length primitive array type, VLA (variable-length
030 * array) was chosen as a different name.
031 * Copied from LibGDX by Tommy Ettinger on 10/1/2015.
032 * @author Nathan Sweet */
033public class IntVLA implements Serializable, Cloneable {
034    private static final long serialVersionUID = -2948161891082748626L;
035
036    public int[] items;
037    public int size;
038
039    /** Creates an ordered array with a capacity of 16. */
040    public IntVLA() {
041        this(16);
042    }
043
044    /** Creates an ordered array with the specified capacity. */
045    public IntVLA(int capacity) {
046        items = new int[capacity];
047    }
048
049    /** Creates a new array containing the elements in the specific array. The new array will be ordered if the specific array is
050     * ordered. The capacity is set to the number of elements, so any subsequent elements added will cause the backing array to be
051     * grown. */
052    public IntVLA(IntVLA array) {
053        size = array.size;
054        items = new int[size];
055        System.arraycopy(array.items, 0, items, 0, size);
056    }
057
058    /** Creates a new ordered array containing the elements in the specified array. The capacity is set to the number of elements,
059     * so any subsequent elements added will cause the backing array to be grown. */
060    public IntVLA(int[] array) {
061        this(array, 0, array.length);
062    }
063
064    /** Creates a new array containing the elements in the specified array. The capacity is set to the number of elements, so any
065     * subsequent elements added will cause the backing array to be grown.
066     * @param array the int array to copy from
067     * @param startIndex the first index in array to copy from
068     * @param count the number of ints to copy from array into this IntVLA
069     */
070    public IntVLA(int[] array, int startIndex, int count) {
071        this(count);
072        size = count;
073        System.arraycopy(array, startIndex, items, 0, count);
074    }

076    public void add (int value) {
077        int[] items = this.items;
078        if (size == items.length) items = resize(Math.max(8, (int)(size * 1.75f)));
079        items[size++] = value;
080    }

082    public void addAll (IntVLA array) {
083        addAll(array, 0, array.size);
084    }
085
086    public void addAll (IntVLA array, int offset, int length) {
087        if (offset + length > array.size)
088            throw new IllegalArgumentException("offset + length must be <= size: " + offset + " + " + length + " <= " + array.size);
089        addAll(array.items, offset, length);
090    }
091
092    public void addAll (int... array) {
093        addAll(array, 0, array.length);
094    }
095
096    public void addAll (int[] array, int offset, int length) {
097        int[] items = this.items;
098        int sizeNeeded = size + length;
099        if (sizeNeeded > items.length) items = resize(Math.max(8, (int)(sizeNeeded * 1.75f)));
100        System.arraycopy(array, offset, items, size, length);
101        size += length;
102    }
103
104    public void addRange (int start, int end) {
105        int[] items = this.items;
106        int sizeNeeded = size + end - start;
107        if (sizeNeeded > items.length) items = resize(Math.max(8, (int)(sizeNeeded * 1.75f)));
108        for(int r = start, i = size; r < end; r++, i++)
109        {
110            items[i] = (int)r;
111        }
112        size += end - start;
113    }
114
115    public void addFractionRange (int start, int end, int fraction) {
116        int[] items = this.items;
117        int sizeNeeded = size + (end - start) / fraction + 2;
118        if (sizeNeeded > items.length) items = resize(Math.max(8, (int)(sizeNeeded * 1.75f)));
119        for(int r = start, i = size; r < end; r = fraction * ((r / fraction) + 1), i++, size++)
120        {
121            items[i] = r;
122        }
123    }

125    public int get (int index) {
126        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
127        return items[index];
128    }
129
130    public void set (int index, int value) {
131        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
132        items[index] = value;
133    }

135    /**
136     * Adds value to the item in the IntVLA at index. Calling it "add" would overlap with the collection method.
137     * @param index
138     * @param value
139     */
140    public void incr (int index, int value) {
141        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
142        items[index] += value;
143    }
144
145    public void mul (int index, int value) {
146        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
147        items[index] *= value;
148    }
149
150    public void insert (int index, int value) {
151        if (index > size) throw new IndexOutOfBoundsException("index can't be > size: " + index + " > " + size);
152        int[] items = this.items;
153        if (size == items.length) items = resize(Math.max(8, (int)(size * 1.75f)));
154        System.arraycopy(items, index, items, index + 1, size - index);
155        size++;
156        items[index] = value;
157    }
158
159    public void swap (int first, int second) {
160        if (first >= size) throw new IndexOutOfBoundsException("first can't be >= size: " + first + " >= " + size);
161        if (second >= size) throw new IndexOutOfBoundsException("second can't be >= size: " + second + " >= " + size);
162        int[] items = this.items;
163        int firstValue = items[first];
164        items[first] = items[second];
165        items[second] = firstValue;
166    }

168    /**
169     * Given an array or varargs of replacement indices for the values of this IntVLA, reorders this so the first item
170     * in the returned version is the same as {@code get(ordering[0])} (with some care taken for negative or too-large
171     * indices), the second item in the returned version is the same as {@code get(ordering[1])}, etc.
172     * <br>
173     * Negative indices are considered reversed distances from the end of ordering, so -1 refers to the same index as
174     * {@code ordering[ordering.length - 1]}. If ordering is smaller than this IntVLA, only the indices up to the
175     * length of ordering will be modified. If ordering is larger than this IntVLA, only as many indices will be
176     * affected as this IntVLA's size, and reversed distances are measured from the end of this IntVLA instead of the
177     * end of ordering. Duplicate values in ordering will produce duplicate values in the returned IntVLA.
178     * <br>
179     * This method modifies this IntVLA in-place and also returns it for chaining.
180     *
181     * @param ordering an array or varargs of int indices, where the nth item in ordering changes the nth item in this
182     *                 IntVLA to have the value currently in this IntVLA at the index specified by the value in ordering
183     * @return this for chaining, after modifying it in-place
184     */
185    public IntVLA reorder (int... ordering) {
186        int ol;
187        if (ordering == null || (ol = Math.min(size, ordering.length)) == 0)
188            return this;
189        int[] items = this.items, alt = new int[ol];
190        for (int i = 0; i < ol; i++) {
191            alt[i] = items[(ordering[i] % ol + ol) % ol];
192        }
193        System.arraycopy(alt, 0, items, 0, ol);
194        return this;
195    }

197    public boolean contains (int value) {
198        int i = size - 1;
199        int[] items = this.items;
200        while (i >= 0)
201            if (items[i--] == value) return true;
202        return false;
203    }
204
205    public int indexOf (int value) {
206        int[] items = this.items;
207        for (int i = 0, n = size; i < n; i++)
208            if (items[i] == value) return i;
209        return -1;
210    }
211
212    public int lastIndexOf (int value) {
213        int[] items = this.items;
214        for (int i = size - 1; i >= 0; i--)
215            if (items[i] == value) return i;
216        return -1;
217    }
218
219    /**
220     * Removes the first occurrence of the requested value, and returns the index it was removed at (-1 if not found)
221     * @param value a value in this IntVLA to remove
222     * @return the index the value was found and removed at, or -1 if it was not present
223     */
224    public int removeValue (int value) {
225        int[] items = this.items;
226        for (int i = 0, n = size; i < n; i++) {
227            if (items[i] == value) {
228                removeIndex(i);
229                return i;
230            }
231        }
232        return -1;
233    }
234
235    /** Removes and returns the item at the specified index. */
236    public int removeIndex (int index) {
237        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
238        int[] items = this.items;
239        int value = items[index];
240        size--;
241        System.arraycopy(items, index + 1, items, index, size - index);
242        return value;
243    }
244
245    /** Removes the items between the specified indices, inclusive. */
246    public void removeRange (int start, int end) {
247        if (end >= size) throw new IndexOutOfBoundsException("end can't be >= size: " + end + " >= " + size);
248        if (start > end) throw new IndexOutOfBoundsException("start can't be > end: " + start + " > " + end);
249        int[] items = this.items;
250        int count = end - start + 1;
251        System.arraycopy(items, start + count, items, start, size - (start + count));
252        size -= count;
253    }
254
255    /** Removes from this array all of elements contained in the specified array.
256     * @return true if this array was modified. */
257    public boolean removeAll (IntVLA array) {
258        int size = this.size;
259        int startSize = size;
260        int[] items = this.items;
261        for (int i = 0, n = array.size; i < n; i++) {
262            int item = array.get(i);
263            for (int ii = 0; ii < size; ii++) {
264                if (item == items[ii]) {
265                    removeIndex(ii);
266                    size--;
267                    break;
268                }
269            }
270        }
271        return size != startSize;
272    }

275    /** Moves the item at the specified index to the first index and returns it. */
276    public int moveToFirst (int index) {
277        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
278        int[] items = this.items;
279        int value = items[index];
280        if(index == 0) return value;
281        System.arraycopy(items, 0, items, 1, index);
282        items[0] = value;
283        return value;
284    }
285
286    /** Moves the item at the specified index to the last index and returns it. */
287    public int moveToLast (int index) {
288        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
289        int[] items = this.items;
290        int value = items[index];
291        if(index == size - 1) return value;
292        System.arraycopy(items, index + 1, items, index, size - index - 1);
293        items[size - 1] = value;
294        return value;
295    }

317    /** Reduces the size of the backing array to the size of the actual items. This is useful to release memory when many items have
318     * been removed, or if it is known that more items will not be added.
319     * @return {@link #items} */
320    public int[] shrink () {
321        if (items.length != size) resize(size);
322        return items;
323    }
324
325    /** Increases the size of the backing array to accommodate the specified number of additional items. Useful before adding many
326     * items to avoid multiple backing array resizes.
327     * @return {@link #items} */
328    public int[] ensureCapacity (int additionalCapacity) {
329        int sizeNeeded = size + additionalCapacity;
330        if (sizeNeeded > items.length) resize(Math.max(8, sizeNeeded));
331        return items;
332    }

334    /** Sets the array size, leaving any values beyond the current size undefined.
335     * @return {@link #items} */
336    public int[] setSize (int newSize) {
337        if (newSize > items.length) resize(Math.max(8, newSize));
338        size = newSize;
339        return items;
340    }
341
342    protected int[] resize (int newSize) {
343        int[] newItems = new int[newSize];
344        int[] items = this.items;
345        System.arraycopy(items, 0, newItems, 0, Math.min(size, newItems.length));
346        this.items = newItems;
347        return newItems;
348    }
349
350    public void sort () {
351        Arrays.sort(items, 0, size);
352    }
353
354    public void reverse () {
355        int[] items = this.items;
356        for (int i = 0, lastIndex = size - 1, n = size / 2; i < n; i++) {
357            int ii = lastIndex - i;
358            int temp = items[i];
359            items[i] = items[ii];
360            items[ii] = temp;
361        }
362    }

364    /** Reduces the size of the array to the specified size. If the array is already smaller than the specified size, no action is
365     * taken. */
366    public void truncate (int newSize) {
367        if (size > newSize) size = newSize;
368    }

375    /**
376     * Shuffles this IntVLA in place using the given RNG.
377     * @param random an RNG used to generate the shuffled order
378     * @return this object, modified, after shuffling
379     */
380    public IntVLA shuffle(RNG random)
381    {
382        int n = size;
383        for (int i = 0; i < n; i++)
384        {
385            swap(i + random.nextInt(n - i), i);
386        }
387        return this;
388
389    }

415    @Override
416    public int hashCode () {
417        int[] items = this.items;
418        int h = 1;
419        for (int i = 0, n = size; i < n; i++)
420            h = h * 31 + items[i];
421        return h;
422    }
423    public int hashWisp () {
424        int[] data = this.items;
425        int result = 0x9E3779B9, a = 0x632BE5AB;
426        final int len = size;
427        for (int i = 0; i < len; i++) {
428            result += (a ^= 0x85157AF5 * data[i]);
429        }
430        return result * (a | 1) ^ (result >>> 11 | result << 21);
431    }

437    @Override
438        public boolean equals (Object object) {
439        if (object == this) return true;
440        if (!(object instanceof IntVLA)) return false;
441        IntVLA array = (IntVLA)object;
442        int n = size;
443        if (n != array.size) return false;
444        for (int i = 0; i < n; i++)
445            if (items[i] != array.items[i]) return false;
446        return true;

449    @Override
450        public String toString () {
451        if (size == 0) return "[]";
452        int[] items = this.items;
453        StringBuilder buffer = new StringBuilder(32);
454        buffer.append('[');
455        buffer.append(items[0]);
456        for (int i = 1; i < size; i++) {
457            buffer.append(", ");
458            buffer.append(items[i]);
459        }

476    public static IntVLA deserializeFromString(String data)
477    {
478        int amount = StringKit.count(data, ",");
479        if (amount <= 0) return new IntVLA();
480        IntVLA iv = new IntVLA(amount+1);
481        int dl = 1, idx = -dl, idx2;
482        for (int i = 0; i < amount; i++) {
483            iv.add(Integer.parseInt(StringKit.safeSubstring(data, idx+dl, idx = data.indexOf(",", idx+dl))));
484        }
485        if((idx2 = data.indexOf(",", idx+dl)) < 0)
486        {
487            iv.add(Integer.parseInt(StringKit.safeSubstring(data, idx+dl, data.length())));
488        }
489        else
490        {
491            iv.add(Integer.parseInt(StringKit.safeSubstring(data, idx+dl, idx2)));
492        }
493        return iv;
494    }
