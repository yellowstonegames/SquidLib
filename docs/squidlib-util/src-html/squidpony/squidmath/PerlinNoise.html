003/**
004 * This is Ken Perlin's third revision of his noise function. It is sometimes
005 * referred to as "Simplex Noise". Results are bound by (-1, 1) inclusive.
006 *
007 *
008 * It is significantly faster than his earlier versions. This particular version
009 * was originally from Stefan Gustavson. This is much preferred to the earlier
010 * versions of Perlin Noise due to the reasons noted in the articles:
011 * <ul>
012 * <li>http://www.itn.liu.se/~stegu/simplexnoise/simplexnoise.pdf</li>
013 * <li>http://webstaff.itn.liu.se/~stegu/TNM022-2005/perlinnoiselinks/ch02.pdf</li>
014 * </ul>
015 * But, Gustavson's paper is not without its own issues, particularly for 2D noise.
016 * More detail is noted here,
017 * http://stackoverflow.com/questions/18885440/why-does-simplex-noise-seem-to-have-more-artifacts-than-classic-perlin-noise#21568753
018 * and some changes have been made to 2D noise generation to reduce angular artifacts.
019 * Specifically for the 2D gradient table, code based on Gustavson's paper used 12
020 * points, with some duplicates, and not all on the unit circle. In this version,
021 * points are used on the unit circle starting at (1,0) and moving along the circle
022 * in increments of 1.61803398875 radians, that is, the golden ratio phi, getting the
023 * sin and cosine of 15 points after the starting point and storing them as constants.
024 * This definitely doesn't have a noticeable 45 degree angle artifact, though it does
025 * have, to a lesser extent, some other minor artifacts.
026 * <br>
027 * You can also consider {@link WhirlingNoise} as an alternative, which can be faster
028 * and also reduces the likelihood of angular artifacts. WhirlingNoise does not scale its
029 * input (it doesn't need to), so it won't produce the same results as PerlinNoise for the
030 * same inputs, but it will produce similar shape, density, and aesthetic quality of noise.
031 * @see WhirlingNoise A subclass that has a faster implementation and some different qualities.
032 */

411    /**
412     * 2D simplex noise.
413     * This doesn't use its parameters verbatim; xin and yin are both effectively divided by
414     * ({@link Math#E} * {@link Math#PI}), because without a step like that, any integer parameters would return 0 and
415     * only doubles with a decimal component would produce actual noise. This step allows integers to be passed in a
416     * arguments, and changes the cycle at which 0 is repeated to multiples of (E*PI).
417     *
418     * @param xin x input; works well if between 0.0 and 1.0, but anything is accepted
419     * @param yin y input; works well if between 0.0 and 1.0, but anything is accepted
420     * @return noise from -1.0 to 1.0, inclusive
421     */
422    public static double noise(double xin, double yin) {
423        xin *= epi;
424        yin *= epi;
425        double noise0, noise1, noise2; // from the three corners
426        // Skew the input space to determine which simplex cell we're in
427        double skew = (xin + yin) * F2; // Hairy factor for 2D
428        int i = (int) Math.floor(xin + skew);
429        int j = (int) Math.floor(yin + skew);
430        double t = (i + j) * G2;
431        double X0 = i - t; // Unskew the cell origin back to (x,y) space
432        double Y0 = j - t;
433        double x0 = xin - X0; // The x,y distances from the cell origin
434        double y0 = yin - Y0;
435        // For the 2D case, the simplex shape is an equilateral triangle.
436        // Determine which simplex we are in.
437        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
438        // coords
439        if (x0 > y0) {
440            i1 = 1;
441            j1 = 0;
442        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
443        else {
444            i1 = 0;
445            j1 = 1;
446        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
447        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
448        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
449        // where
450        // c = (3-sqrt(3))/6
451        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
452        // unskewed coords
453        double y1 = y0 - j1 + G2;
454        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
455        // unskewed coords
456        double y2 = y0 - 1.0 + 2.0 * G2;
457        // Work out the hashed gradient indices of the three simplex corners
458        int ii = i & 255;
459        int jj = j & 255;
460        int gi0 = perm[ii + perm[jj]];
461        int gi1 = perm[ii + i1 + perm[jj + j1]];
462        int gi2 = perm[ii + 1 + perm[jj + 1]];
463        // Calculate the contribution from the three corners
464        double t0 = 0.5 - x0 * x0 - y0 * y0;
465        if (t0 < 0) {
466            noise0 = 0.0;
467        } else {
468            t0 *= t0;
469            noise0 = t0 * t0 * dot(phiGrad2[gi0], x0, y0); // (x,y) of grad3 used
470            // for 2D gradient
471        }
472        double t1 = 0.5 - x1 * x1 - y1 * y1;
473        if (t1 < 0) {
474            noise1 = 0.0;
475        } else {
476            t1 *= t1;
477            noise1 = t1 * t1 * dot(phiGrad2[gi1], x1, y1);
478        }
479        double t2 = 0.5 - x2 * x2 - y2 * y2;
480        if (t2 < 0) {
481            noise2 = 0.0;
482        } else {
483            t2 *= t2;
484            noise2 = t2 * t2 * dot(phiGrad2[gi2], x2, y2);
485        }
486        // Add contributions from each corner to get the final noise value.
487        // The result is scaled to return values in the interval [-1,1].
488        return 70.0 * (noise0 + noise1 + noise2);
489    }

491    /**
492     * 3D simplex noise.
493     *
494     * @param xin X input
495     * @param yin Y input
496     * @param zin Z input
497     * @return noise from -1.0 to 1.0, inclusive
498     */
499    public static double noise(double xin, double yin, double zin) {
500        xin *= epi;
501        yin *= epi;
502        zin *= epi;
503        double n0, n1, n2, n3; // Noise contributions from the four corners
504        // Skew the input space to determine which simplex cell we're in
505        double s = (xin + yin + zin) * F3; // Very nice and simple skew
506        // factor for 3D
507        int i = (int) Math.floor(xin + s);
508        int j = (int) Math.floor(yin + s);
509        int k = (int) Math.floor(zin + s);
510        double t = (i + j + k) * G3;
511        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
512        double Y0 = j - t;
513        double Z0 = k - t;
514        double x0 = xin - X0; // The x,y,z distances from the cell origin
515        double y0 = yin - Y0;
516        double z0 = zin - Z0;
517        // For the 3D case, the simplex shape is a slightly irregular
518        // tetrahedron.
519        // Determine which simplex we are in.
520        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
521        // coords
522        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
523        // coords
524        if (x0 >= y0) {
525            if (y0 >= z0) {
526                i1 = 1;
527                j1 = 0;
528                k1 = 0;
529                i2 = 1;
530                j2 = 1;
531                k2 = 0;
532            } // X Y Z order
533            else if (x0 >= z0) {
534                i1 = 1;
535                j1 = 0;
536                k1 = 0;
537                i2 = 1;
538                j2 = 0;
539                k2 = 1;
540            } // X Z Y order
541            else {
542                i1 = 0;
543                j1 = 0;
544                k1 = 1;
545                i2 = 1;
546                j2 = 0;
547                k2 = 1;
548            } // Z X Y order
549        } else { // x0<y0
550            if (y0 < z0) {
551                i1 = 0;
552                j1 = 0;
553                k1 = 1;
554                i2 = 0;
555                j2 = 1;
556                k2 = 1;
557            } // Z Y X order
558            else if (x0 < z0) {
559                i1 = 0;
560                j1 = 1;
561                k1 = 0;
562                i2 = 0;
563                j2 = 1;
564                k2 = 1;
565            } // Y Z X order
566            else {
567                i1 = 0;
568                j1 = 1;
569                k1 = 0;
570                i2 = 1;
571                j2 = 1;
572                k2 = 0;
573            } // Y X Z order
574        }
575        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
576        // (x,y,z),
577        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
578        // (x,y,z), and
579        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
580        // (x,y,z), where
581        // c = 1/6.
582        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
583        // coords
584        double y1 = y0 - j1 + G3;
585        double z1 = z0 - k1 + G3;
586        double x2 = x0 - i2 + 2.0 * G3; // Offsets for third corner in
587        // (x,y,z) coords
588        double y2 = y0 - j2 + 2.0 * G3;
589        double z2 = z0 - k2 + 2.0 * G3;
590        double x3 = x0 - 1.0 + 3.0 * G3; // Offsets for last corner in
591        // (x,y,z) coords
592        double y3 = y0 - 1.0 + 3.0 * G3;
593        double z3 = z0 - 1.0 + 3.0 * G3;
594        // Work out the hashed gradient indices of the four simplex corners
595        int ii = i & 255;
596        int jj = j & 255;
597        int kk = k & 255;
598        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
599        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
600        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
601        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
602        // Calculate the contribution from the four corners
603        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
604        if (t0 < 0) {
605            n0 = 0.0;
606        } else {
607            t0 *= t0;
608            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
609        }
610        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
611        if (t1 < 0) {
612            n1 = 0.0;
613        } else {
614            t1 *= t1;
615            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
616        }
617        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
618        if (t2 < 0) {
619            n2 = 0.0;
620        } else {
621            t2 *= t2;
622            n2 = t2 * t2 * dot(grad3[gi2], x2, y2, z2);
623        }
624        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
625        if (t3 < 0) {
626            n3 = 0.0;
627        } else {
628            t3 *= t3;
629            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
630        }
631        // Add contributions from each corner to get the final noise value.
632        // The result is scaled to stay just inside [-1,1]
633        return 32.0 * (n0 + n1 + n2 + n3);
634    }

636    /**
637     * 4D simplex noise.
638     *
639     * @param x X position
640     * @param y Y position
641     * @param z Z position
642     * @param w Fourth-dimensional position. It is I, the Fourth-Dimensional Ziltoid the Omniscient!
643     * @return noise from -1.0 to 1.0, inclusive
644     */
645    public static double noise(double x, double y, double z, double w) {
646        x *= epi;
647        y *= epi;
648        z *= epi;
649        w *= epi;
650        // The skewing and unskewing factors are hairy again for the 4D case
651        double n0, n1, n2, n3, n4; // Noise contributions from the five
652        // corners
653        // Skew the (x,y,z,w) space to determine which cell of 24 simplices
654        // we're in
655        double s = (x + y + z + w) * F4; // Factor for 4D skewing
656        int i = (int) Math.floor(x + s);
657        int j = (int) Math.floor(y + s);
658        int k = (int) Math.floor(z + s);
659        int l = (int) Math.floor(w + s);
660        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
661        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
662        double Y0 = j - t;
663        double Z0 = k - t;
664        double W0 = l - t;
665        double x0 = x - X0; // The x,y,z,w distances from the cell origin
666        double y0 = y - Y0;
667        double z0 = z - Z0;
668        double w0 = w - W0;
669        // For the 4D case, the simplex is a 4D shape I won't even try to
670        // describe.
671        // To find out which of the 24 possible simplices we're in, we need
672        // to
673        // determine the magnitude ordering of x0, y0, z0 and w0.
674        // The method below is a good way of finding the ordering of x,y,z,w
675        // and
676        // then find the correct traversal order for the simplex were in.
677        // First, six pair-wise comparisons are performed between each
678        // possible pair
679        // of the four coordinates, and the results are used to add up binary
680        // bits
681        // for an integer index.
682        int c =
683                (x0 > y0 ? 32 : 0)
684              + (x0 > z0 ? 16 : 0)
685              + (y0 > z0 ? 8 : 0 )
686              + (x0 > w0 ? 4 : 0 )
687              + (y0 > w0 ? 2 : 0 )
688              + (z0 > w0 ? 1 : 0 );
689        int i1, j1, k1, l1; // The integer offsets for the second simplex
690        // corner
691        int i2, j2, k2, l2; // The integer offsets for the third simplex
692        // corner
693        int i3, j3, k3, l3; // The integer offsets for the fourth simplex
694        // corner
695        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
696        // order.
697        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
698        // y<w and x<w
699        // impossible. Only the 24 indices which have non-zero entries make
700        // any sense.
701        // We use a thresholding to set the coordinates in turn from the
702        // largest magnitude.
703        // The number 3 in the "simplex" array is at the position of the
704        // largest coordinate.
705        i1 = simplex[c][0] >= 3 ? 1 : 0;
706        j1 = simplex[c][1] >= 3 ? 1 : 0;
707        k1 = simplex[c][2] >= 3 ? 1 : 0;
708        l1 = simplex[c][3] >= 3 ? 1 : 0;
709        // The number 2 in the "simplex" array is at the second largest
710        // coordinate.
711        i2 = simplex[c][0] >= 2 ? 1 : 0;
712        j2 = simplex[c][1] >= 2 ? 1 : 0;
713        k2 = simplex[c][2] >= 2 ? 1 : 0;
714        l2 = simplex[c][3] >= 2 ? 1 : 0;
715        // The number 1 in the "simplex" array is at the second smallest
716        // coordinate.
717        i3 = simplex[c][0] >= 1 ? 1 : 0;
718        j3 = simplex[c][1] >= 1 ? 1 : 0;
719        k3 = simplex[c][2] >= 1 ? 1 : 0;
720        l3 = simplex[c][3] >= 1 ? 1 : 0;
721        // The fifth corner has all coordinate offsets = 1, so no need to
722        // look that up.
723        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w)
724        // coords
725        double y1 = y0 - j1 + G4;
726        double z1 = z0 - k1 + G4;
727        double w1 = w0 - l1 + G4;
728        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in
729        // (x,y,z,w) coords
730        double y2 = y0 - j2 + 2.0 * G4;
731        double z2 = z0 - k2 + 2.0 * G4;
732        double w2 = w0 - l2 + 2.0 * G4;
733        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in
734        // (x,y,z,w) coords
735        double y3 = y0 - j3 + 3.0 * G4;
736        double z3 = z0 - k3 + 3.0 * G4;
737        double w3 = w0 - l3 + 3.0 * G4;
738        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in
739        // (x,y,z,w) coords
740        double y4 = y0 - 1.0 + 4.0 * G4;
741        double z4 = z0 - 1.0 + 4.0 * G4;
742        double w4 = w0 - 1.0 + 4.0 * G4;
743        // Work out the hashed gradient indices of the five simplex corners
744        int ii = i & 255;
745        int jj = j & 255;
746        int kk = k & 255;
747        int ll = l & 255;
748        int gi0 = perm[ii + perm[jj + perm[kk + perm[ll]]]] & 31;
749        int gi1
750                = perm[ii + i1 + perm[jj + j1 + perm[kk + k1 + perm[ll + l1]]]] & 31;
751        int gi2
752                = perm[ii + i2 + perm[jj + j2 + perm[kk + k2 + perm[ll + l2]]]] & 31;
753        int gi3
754                = perm[ii + i3 + perm[jj + j3 + perm[kk + k3 + perm[ll + l3]]]] & 31;
755        int gi4
756                = perm[ii + 1 + perm[jj + 1 + perm[kk + 1 + perm[ll + 1]]]] & 31;
757        // Calculate the contribution from the five corners
758        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0;
759        if (t0 < 0) {
760            n0 = 0.0;
761        } else {
762            t0 *= t0;
763            n0 = t0 * t0 * dot(grad4[gi0], x0, y0, z0, w0);
764        }
765        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1;
766        if (t1 < 0) {
767            n1 = 0.0;
768        } else {
769            t1 *= t1;
770            n1 = t1 * t1 * dot(grad4[gi1], x1, y1, z1, w1);
771        }
772        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2;
773        if (t2 < 0) {
774            n2 = 0.0;
775        } else {
776            t2 *= t2;
777            n2 = t2 * t2 * dot(grad4[gi2], x2, y2, z2, w2);
778        }
779        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3;
780        if (t3 < 0) {
781            n3 = 0.0;
782        } else {
783            t3 *= t3;
784            n3 = t3 * t3 * dot(grad4[gi3], x3, y3, z3, w3);
785        }
786        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4;
787        if (t4 < 0) {
788            n4 = 0.0;
789        } else {
790            t4 *= t4;
791            n4 = t4 * t4 * dot(grad4[gi4], x4, y4, z4, w4);
792        }
793        // Sum up and scale the result to cover the range [-1,1]
794        return 27.0 * (n0 + n1 + n2 + n3 + n4);
795    }
