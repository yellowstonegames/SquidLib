008/**
009 * Like a kind of RNG, but fully deterministic in a way that depends on certain connected variables.
010 * Intended as a way to produce similar values when small changes occur in the connections, while potentially producing
011 * larger changes when the changes are more significant (unlike an RNG or hashing function, which can and should produce
012 * very different output given even slightly different seeds/input). This might be useful to produce procedural story
013 * data that is similar when most of the connected inputs are similar, or for terrain generation/population. This can
014 * produce ints and doubles, and does not produce a different output unless its input is changed (usually by altering a
015 * shared reference to {@code connections}).
016 * <br>
017 * Created by Tommy Ettinger on 5/18/2017.
018 */

036    public CosmicNumbering(long seed, double[] connections) {
037        if(connections == null || connections.length == 0)
038            this.connections = new double[1];
039        else
040            this.connections = connections;
041        len = this.connections.length;
042        upper = 1 << len;
043        scratch3 = new long[len * 3];
044        scratch = new double[upper];
045        seeds = new long[len];
046        seeds[0] = seed | 1L;
047        for (int i = 1; i < len; i++) {
048            seeds[i] = LinnormRNG.determine(seed + i) | 1L;
049        }
050        effect = 0x1.81p-62 * Math.pow(1.1875, len);
051    }

074    }

116    /**
117     * Gets a double determined by the current values in the connections, accessible via {@link #getConnections()}.
118     * Returns a value between -1.0 and 1.0 (exclusive on 1.0). Used as the basis for other methods in this class.
119     * @return a double between -1.0 and 1.0; will be the same value until/unless connections change
120     */
121    public final double getDoubleBase() {
122        for (int i = 0; i < len; i++) {
123            long seed = seeds[i];
124            scratch3[i * 3 + 1] = (scratch3[i * 3] = (scratch3[i * 3 + 2] = longFloor(connections[i])) * seed) + seed;
125        }
126        long working;
127        for (int i = 0; i < upper; i++) {
128            working = 0L;
129            for (int j = 0; j < len; j++) {
130                working += scratch3[j * 3 + (i >> j & 1)];
131            }
132            scratch[i] = determine(working) * effect;
133        }
134        for (int i = 0; i < len; ++i) {
135            for (int j = 0, t = upper >> i; j < t; j += 2) {
136                scratch[j >> 1] = cerp(scratch[j], scratch[j + 1], connections[i] - scratch3[i * 3 + 2]);
137            }
138        }
139        return NumberTools.sway(scratch[0]);
140    }

182    /**
183     * Gets an int determined by the current values in the connections, accessible via {@link #getConnections()}.
184     * Returns a value in the full range of ints, but is less likely to produce ints close to {@link Integer#MAX_VALUE}
185     * or {@link Integer#MIN_VALUE} (expect very few values in the bottom and top quarters of the range).
186     * @return an int which can be positive or negative; will be the same value until/unless connections change
187     */
188    public int getInt()
189    {
190        return (int)(0x80000000 * getDoubleBase());
191    }

194     * The same as {@link LinnormRNG#determine(long)}, except this assumes state has already been multiplied by
195     * 0x632BE59BD9B4E019L.
196     * @param state a long that should change in increments of 0x632BE59BD9B4E019L
197     * @return a pseudo-random permutation of state
198     */
199    public static long determine(long state)
200    {
201        return (state = ((state = ((state ^ 0x9E3779B97F4A7C15L) * 0xC6BC279692B5CC83L)) ^ state >>> 27) * 0xAEF17502108EF2D9L) ^ state >>> 25;
202    }
