001package squidpony.squidmath;
002
003import squidpony.annotation.Beta;
004
005import java.util.AbstractMap;
006import java.util.ArrayList;
007import java.util.Arrays;
008import java.util.Map;
009
010/**
011 * Mostly just a way to have an EnumMap-like class that can be serialized reasonably. Should be usable for anything an
012 * EnumMap is usable for. Serialization is possible with this, like EnumMap can be serialized in squidlib-extra, but
013 * this still needs some hack-y techniques.
014 * Created by Tommy Ettinger on 9/5/2017.
015 */
016@Beta
017public class EnumMapPlus<K extends Enum<K>, V> implements Map<K, V> {
018    public final K[] keys;
019    public final Class<K> keyType;
020    public final V[] vals;
021    public int size;
022    /**
023     * Creates an empty enum map with the specified key type.
024     *
025     * @param keyType the class object of the key type for this enum map
026     * @throws NullPointerException if <tt>keyType</tt> is null
027     */
028    @SuppressWarnings("unchecked")
029    public EnumMapPlus(Class<K> keyType) {
030        this.keyType = keyType;
031        keys = keyType.getEnumConstants();
032        vals = (V[]) new Object[keys.length];
033    }
034
035    /**
036     * Creates an enum map with the same key type as the specified enum
037     * map, initially containing the same mappings (if any).
038     *
039     * @param m the enum map from which to initialize this enum map
040     * @throws NullPointerException if <tt>m</tt> is null
041     */
042    public EnumMapPlus(EnumMapPlus<K, ? extends V> m) {
043        this(m.keyType);
044    }
045
046    @Override
047    public int size() {
048        return size;
049    }
050
051    @Override
052    public boolean isEmpty() {
053        return size <= 0;
054    }
055
056    public int validate(Object key) {
057        if(key != null && (key instanceof Enum) && ((Enum)key).getDeclaringClass() == keyType.getDeclaringClass())
058        {
059            int o = ((Enum)key).ordinal();
060            return (o >= 0 && o < keys.length) ? o : -1;
061        }
062        return -1;
063    }
064
065    @Override
066    public boolean containsKey(Object key) {
067        int v = validate(key);
068        return v != -1 && vals[v] != null;
069    }
070
071    @Override
072    public boolean containsValue(Object value) {
073        if(value == null)
074            return false;
075        int len = keys.length;
076        for (int i = 0; i < len; i++) {
077            if(value.equals(vals[i]))
078                return true;
079        }
080        return false;
081    }
082
083    @Override
084    public V get(Object key) {
085        int v = validate(key);
086        return (v != -1) ? vals[v] : null;
087    }
088
089    @Override
090    public V put(K key, V value) {
091        if(key == null || value == null)
092            return null;
093        int ord = key.ordinal();
094        V old = vals[ord];
095        if(old == null)
096            size++;
097        vals[ord] = value;
098        return old;
099    }
100    public V putAt(int place, V value) {
101        if(value == null || place < 0 || place >= keys.length)
102            return null;
103        V old = vals[place];
104        if(old == null)
105            size++;
106        vals[place] = value;
107        return old;
108    }
109
110    @Override
111    public V remove(Object key) {
112        int v = validate(key);
113        if(v != -1)
114        {
115            vals[v] = null;
116            size--;
117        }
118        return null;
119    }
120
121    @Override
122    public void putAll(Map<? extends K, ? extends V> m) {
123        for(Entry<? extends K, ? extends V> kv : m.entrySet())
124        {
125            put(kv.getKey(), kv.getValue());
126        }
127    }
128
129    @Override
130    public void clear() {
131        Arrays.fill(vals, null);
132        size = 0;
133    }
134
135    @Override
136    public OrderedSet<K> keySet() {
137        OrderedSet<K> k = new OrderedSet<>(size);
138        int len = keys.length;
139        for (int i = 0, s = 0; s < size && i < len; i++) {
140            if(vals[i] != null)
141            {
142                k.add(keys[i]);
143                s++;
144            }
145        }
146        return k;
147    }
148
149    @Override
150    public ArrayList<V> values() {
151        ArrayList<V> v = new ArrayList<>(size);
152        int len = keys.length;
153        V vl;
154        for (int i = 0, s = 0; s < size && i < len; i++) {
155            if((vl = vals[i]) != null)
156            {
157                v.add(vl);
158                s++;
159            }
160        }
161        return v;
162    }
163
164    @Override
165    public OrderedSet<Entry<K, V>> entrySet() {
166        OrderedSet<Entry<K, V>> kv = new OrderedSet<>(size);
167        int len = keys.length;
168        V v;
169        for (int i = 0, s = 0; s < size && i < len; i++) {
170            if((v = vals[i]) != null)
171            {
172                kv.add(new AbstractMap.SimpleEntry<>(keys[i], v));
173                s++;
174            }
175        }
176        return kv;
177    }
178
179    @Override
180    public boolean equals(Object o) {
181        if (this == o) return true;
182        if (o == null || getClass() != o.getClass()) return false;
183
184        EnumMapPlus<?, ?> that = (EnumMapPlus<?, ?>) o;
185
186        if (size != that.size) return false;
187        if (!keyType.equals(that.keyType)) return false;
188        final int len = vals.length;
189        V v;
190        for (int i = 0; i < len; i++) {
191            if((v = vals[i]) != null && !v.equals(that.vals[i]))
192                return false;
193        }
194        return true;
195    }
196
197    @Override
198    public int hashCode() {
199        int result = CrossHash.hash(keys);
200        result = 31 * result + keyType.hashCode();
201        result = 31 * result + CrossHash.hash(vals);
202        result = 31 * result + size;
203        return result;
204    }
205
206    @Override
207    public String toString() {
208        StringBuilder sb = new StringBuilder(size * 32);
209        int len = keys.length;
210        V v;
211        sb.append('{');
212        for (int i = 0, s = 0; s < size && i < len; i++) {
213            if((v = vals[i]) != null)
214            {
215                sb.append(keys[i]).append("=>").append(v);
216                if(++s < size)
217                    sb.append(", ");
218            }
219        }
220        return sb.append('}').toString();
221    }
222}
