008/**
009 * A variant on LapRNG that improves the quality at the expense of some speed. Running ZapRNG through PractRand, a tool
010 * for evaluating the statistical quality of PRNGs, found far, far fewer statistical failures than LapRNG, and also had
011 * them generally be less severe. The period is still relatively low, at (with a high degree of certainty) 2 to the 65,
012 * but the speed is still fairly good; slightly faster than ThunderRNG (an earlier version was within 1% difference, but
013 * subsequent changes sped up Zap by about 5-10%), faster than LightRNG all around, slower than LapRNG for all methods,
014 * and slower than FlapRNG on next() and nextInt() but not on nextLong(). Like LapRNG, there should be many possible
015 * sequences of length (2 to the 65) this can produce, depending on the relationship between the two longs used for
016 * state, determined at construction or when the seed is set.
017 * <br>
018 * Created by Tommy Ettinger on 6/7/2017.
019 */
020@Beta
021public class ZapRNG implements RandomnessSource, Serializable {
022    private static final long serialVersionUID = 1L;
023    /**
024     * Constructs a ZapRNG with a random internal state, using {@link Math#random()} three times to get enough bits.
025     */
026    public ZapRNG(){
027        this((int)((Math.random() * 2.0 - 1.0) * 0x80000000),
028                (int)((Math.random() * 2.0 - 1.0) * 0x80000000),
029                (int)((Math.random() * 2.0 - 1.0) * 0x80000000));
030    }

033    /**
034     * Given one long for a seed, this fills the two internal longs of state with different values, each produced by
035     * multiplying seed by a large constant and adding a different large constant. You can obtain the state that this
036     * ends up using with {@link #getState0()} and {@link #getState1()}.
037     * @param seed any long; will not be used verbatim for either internal state
038     */
039    public ZapRNG(final long seed) {
040        state0 = seed * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
041        state1 = seed * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
042    }

044     * Given one int for a seed, this fills the two internal longs of state with different values, each produced by
045     * multiplying seed by a large constant and adding a different large constant. You can obtain the state that this
046     * ends up using with {@link #getState0()} and {@link #getState1()}.
047     * @param seed any int; will not be used verbatim for either internal state
048     */
049    public ZapRNG(final int seed) {
050        state0 = seed * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
051        state1 = seed * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
052    }

054    /**
055     * The only constructor that does not modify its seeds in some way; you should ensure seed0 is large enough (at
056     * least 25 bits should be needed to represent it, so greater than 33554432 or less than -33554433). If seed0 is too
057     * small, then this will appear highly repetitive for many cycles before changing its output pattern.
058     * @param seed0 a long that should be greater than 33554432 or less than -33554433, preferably by a large amount
059     * @param seed1 any long
060     */
061    public ZapRNG(final long seed0, final long seed1) {
062        state0 = seed0;
063        state1 = seed1;
064    }

066    /**
067     * This constructor takes ints for seeds, but internally ZapRNG uses longs, so it multiplies each input by a large
068     * constant and adds another to get the state it actually will start with. You can obtain the state that this ends
069     * up using with {@link #getState0()} and {@link #getState1()}.
070     * @param seed0 any int, will not be used verbatim
071     * @param seed1 any int, will not be used verbatim
072     */
073    public ZapRNG(final int seed0, final int seed1) {
074        state0 = seed0 * 0xC6BC279692B5C483L + 0x8329C6EB9E6AD3E3L;
075        state1 = seed1 * 0x8329C6EB9E6AD3E3L - 0xC6BC279692B5C483L;
076    }

078    /**
079     * This constructor takes 3 ints for seeds, but internally ZapRNG uses 2 longs (one, state0, only considers its
080     * upper 48 bits significant and has closely correlated results if inputs don't differ on significant bits). To
081     * ensure the state works fairly well, this uses seed0 to affect the upper 48 bits of state0 (this also uses the
082     * product of seed1 and seed2), while employing all 3 seed parameters again to set state1 (but these can affect all
083     * 64 bits of state1). Note that only 96 bits of parameter are supplied here to set 128 bits of state, which is all
084     * right in this case because ZapRNG only has a period of 2 to the 65. Anything more than 65 bits of state isn't
085     * strictly necessary, though giving more should allow different, non-overlapping sequences to be produced.
086     * You can obtain the state that this ends up using with {@link #getState0()} and {@link #getState1()}.
087     * @param seed0 any int, will not be used verbatim
088     * @param seed1 any int, will not be used verbatim
089     * @param seed2 any int, will not be used verbatim
090     */
091    public ZapRNG(final int seed0, final int seed1, final int seed2) {
092        state0 = (seed0 * 0xBFL + seed1 * 0x1FL * seed2 << 24) ^ 0x8329C6EB9E6AD3E3L;
093        state1 = (seed1 * 0x8329C6EB9E6AD3E3L ^ seed2 - 0xC6BC279692B5C483L) + seed0 * 0x9E3779B97F4A7C15L;
094    }

096    /**
097     * This constructor gets three differently-calculated 32-bit hash codes from the given String or other
098     * CharSequence and gives them to the constructor that takes three ints, {@link #ZapRNG(int, int, int)}. You can
099     * pass a null seed and this will still work.
100     * @param seed any CharSequence, such as a String
101     */
102    public ZapRNG(final CharSequence seed)
103    {
104        this(CrossHash.Mist.alpha.hash(seed), CrossHash.Mist.beta.hash(seed), CrossHash.Mist.gamma.hash(seed));
105    }

109    /**
110     * Using this method, any algorithm that might use the built-in Java Random
111     * can interface with this randomness source.
112     *
113     * @param bits the number of bits to be returned
114     * @return the integer containing the appropriate number of bits
115     */
116    @Override
117    public final int next( final int bits ) {
118        return (int) (((state1 | 0xC6BC279692B5DBEFL) * (state1 += (state0 += 0x9E3779B97F4A7C15L) >>> 24)) >>> (64 - bits));
119    }

121    /**
122     * Gets a pseudo-random int, which can be positive or negative but is likely to be drawn from less possible options
123     * than the full range of {@link Integer#MIN_VALUE} to {@link Integer#MAX_VALUE}. Very fast, though.
124     * @return a pseudo-random 32-bit int
125     */
126    public final int nextInt()
127    {
128        return (int) ((state1 | 0xC6BC279692B5DBEFL) * (state1 += (state0 += 0x9E3779B97F4A7C15L) >>> 24));
129    }
130    /**
131     * Using this method, any algorithm that needs to efficiently generate more
132     * than 32 bits of random data can interface with this randomness source.
133     * This implementation is the primary one, and {@link #nextInt()} and {@link #next(int)} just use some portion of
134     * the bits this produces. It should be fairly fast on desktops, with good-not-great quality, unlike alternatives
135     * such as LapRNG, which is extremely fast but has low quality when inspected enough, and LightRNG, which the
136     * fastest of the "Made by an Expert" RandomnessSources and has superior quality but is somewhat slower.
137     * <p>
138     * Pseudo-random results may be between between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
139     *
140     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
141     */
142    @Override
143    public final long nextLong() {
144        //return (state1 += state0 ^ (state0 += 0xD43779B97F4A7C13L) >> 24);
145        return ((state1 | 0xC6BC279692B5DBEFL) * (state1 += (state0 += 0x9E3779B97F4A7C15L) >>> 24));
146
147        //return (state0 += (((state1 += 0xC6BC279692B5C483L) >>> 59) + 124) * 0x632AE59B79B4E319L);
148        //0x9E3779B97F4A7C15L
149        //0xBF58476D1CE4E5B9L
150        //0x632AE59B69B3C209L
151        //0xE32AF556CBA5E739L
152    }

154    /**
155     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
156     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
157     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
158     *
159     * @return a copy of this RandomnessSource
160     */
161    @Override
162    public RandomnessSource copy() {
163        return new ZapRNG(state0, state1);
164    }

240    /**
241     * Gets a seemingly-random value from one inputs as a long. If you expect to call this method repeatedly, consider
242     * using the technique {@code (state += ZapRNG.determine(state) + 0x9E3779B97F4A7C15L)}, which will
243     * update each call to use different values for state (it may be surprising that it is an expression
244     * that returns the value of state1 after assignment, so the parenthesized expression can be substituted for the
245     * method call on its own). You may want to experiment with using {@code ^=} or {@code -=} instead of {@code +=} in
246     * the given code, which may affect the results quite a bit.
247     * @param state any long
248     * @return any long, from the full range
249     */
250    public static long determine(final long state)
251    {
252        return ((state | 0xC6BC279692B5DBEFL) * (state + (state * 0x9E3779B97F4A7C15L ^ state << 28) >> 24));
253    }

255    /**
256     * Gets a seemingly-random value from two inputs as longs. If you expect to call this method repeatedly, consider
257     * using the technique {@code (state1 ^= ZapRNG.determine((state0 += 0x9E3779B97F4A7C15L), state1))}, which will
258     * update each call to use different values for state0 and state1 (it may be surprising that it is an expression
259     * that returns the value of state1 after assignment, so the parenthesized expression can be substituted for the
260     * method call on its own). You may want to experiment with using {@code +=} or {@code -=} instead of {@code ^=} in
261     * the given code, which will affect the results quite a bit. Unlike {@link #ZapRNG(long, long)}, this does not have
262     * any special requirements for state0, and both parameters should be free to be large or small.
263     * @param state0 any long
264     * @param state1 any long
265     * @return any long, from the full range
266     */
267    public static long determine(final long state0, final long state1)
268    {
269        return ((state1 | 0xC6BC279692B5DBEFL) * (state1 + (state0 * 0x9E3779B97F4A7C15L ^ state0 << 28) >> 24));
270    }
