001package squidpony.squidmath;
002
003import squidpony.annotation.Beta;
004
005import java.io.Serializable;
006import java.util.ArrayList;
007import java.util.SortedSet;
008
009/**
010 * A generic method of holding a probability table to determine weighted random
011 * outcomes.
012 *
013 * The weights do not need to add up to any particular value, they will be
014 * normalized when choosing a random entry.
015 *
016 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
017 *
018 * @param <T> The type of object to be held in the table
019 */
020@Beta
021public class ProbabilityTable<T> implements Serializable {
022    private static final long serialVersionUID = -1307656083434154736L;
023    /**
024     * The set of items that can be produced directly from {@link #random()} (without additional lookups).
025     */
026    public final Arrangement<T> table;
027    /**
028     * The list of items that can be produced indirectly from {@link #random()} (looking up values from inside
029     * the nested tables).
030     */
031    public final ArrayList<ProbabilityTable<T>> extraTable;
032    public final IntVLA weights;
033    public StatefulRandomness rng;
034    protected int total, normalTotal;
035
036    /**
037     * Creates a new probability table with a random seed.
038     */
039    public ProbabilityTable() {
040        this(new LightRNG());
041    }
042
043    /**
044     * Creates a new probability table with the provided source of randomness
045     * used. Gets one random long from rng to use as an internal identifier.
046     *
047     * @param rng the source of randomness
048     */
049    public ProbabilityTable(StatefulRandomness rng) {
050        this.rng = rng;
051        table = new Arrangement<>(64, 0.75f);
052        extraTable = new ArrayList<>(16);
053        weights = new IntVLA(64);
054        total = 0;
055        normalTotal = 0;
056    }
057
058    /**
059     * Creates a new probability table with the provided long seed used.
060     *
061     * @param seed the RNG seed as a long
062     */
063    public ProbabilityTable(long seed) {
064        this.rng = new LightRNG(seed);
065        table = new Arrangement<>(64, 0.75f);
066        extraTable = new ArrayList<>(16);
067        weights = new IntVLA(64);
068        total = 0;
069        normalTotal = 0;
070    }
071
072    /**
073     * Creates a new probability table with the provided String seed used.
074     *
075     * @param seed the RNG seed as a String
076     */
077    public ProbabilityTable(String seed) {
078        this(CrossHash.hash64(seed));
079    }
080
081    /**
082     * Returns an object randomly based on assigned weights.
083     *
084     * Returns null if no elements have been put in the table.
085     *
086     * @return the chosen object or null
087     */
088    public T random() {
089        if (table.isEmpty()) {
090            return null;
091        }
092        int index = (int) ((total * ((long)rng.next(31))) >>> 31), sz = table.size();
093        for (int i = 0; i < sz; i++) {
094            index -= weights.get(i);
095            if (index < 0)
096                return table.keyAt(i);
097        }
098        for (int i = 0; i < extraTable.size(); i++) {
099            index -= weights.get(sz + i);
100            if(index < 0)
101                return extraTable.get(i).random();
102        }
103        return null;//something went wrong, shouldn't have been able to get all the way through without finding an item
104    }
105
106    /**
107     * Adds the given item to the table.
108     *
109     * Weight must be greater than 0.
110     *
111     * @param item the object to be added
112     * @param weight the weight to be given to the added object
113     * @return this for chaining
114     */
115    public ProbabilityTable<T> add(T item, int weight) {
116        if(weight <= 0)
117            return this;
118        int i = table.getInt(item);
119        if (i < 0) {
120            weights.insert(table.size, Math.max(0, weight));
121            table.add(item);
122            int w = Math.max(0, weight);
123            total += w;
124            normalTotal += w;
125        } else {
126            int i2 = weights.get(i);
127            int w = Math.max(0, i2 + weight);
128            weights.set(i, w);
129            total += w - i2;
130            normalTotal += w - i2;
131        }
132        return this;
133    }
134
135    /**
136     * Given an OrderedMap of T element keys and Integer weight values, adds all T keys with their corresponding weights
137     * into this ProbabilityTable. You may want to use {@link OrderedMap#makeMap(Object, Object, Object...)} to produce
138     * the parameter, unless you already have one.
139     * @param itemsAndWeights an OrderedMap of T keys to Integer values, where a key will be an item this can retrieve
140     *                        and a value will be its weight
141     * @return this for chaining
142     */
143    public ProbabilityTable<T> addAll(OrderedMap<T, Integer> itemsAndWeights)
144    {
145        if(itemsAndWeights == null) return this;
146        int sz = itemsAndWeights.size;
147        for (int i = 0; i < sz; i++) {
148            add(itemsAndWeights.keyAt(i), itemsAndWeights.getAt(i));
149        }
150        return this;
151    }
152
153    /**
154     * Removes the possibility of generating the given T item, except by nested ProbabilityTable results.
155     * Returns true iff the item was removed.
156     * @param item the item to make less likely or impossible
157     * @return true if the probability changed or false if nothing changed
158     */
159    public boolean remove(T item)
160    {
161        return remove(item, weight(item));
162    }
163
164    /**
165     * Reduces the likelihood of generating the given T item by the given weight, which can reduce the chance below 0
166     * and thus remove the item entirely. Does not affect nested ProbabilityTables. The value for weight must be greater
167     * than 0, otherwise this simply returns false without doing anything. Returns true iff the probabilities changed.
168     * @param item the item to make less likely or impossible
169     * @param weight how much to reduce the item's weight by, as a positive non-zero int (greater values here subtract
170     *               more from the item's weight)
171     * @return true if the probability changed or false if nothing changed
172     */
173    public boolean remove(T item, int weight)
174    {
175        if(weight <= 0)
176            return false;
177        int idx = table.getInt(item);
178        if(idx < 0)
179            return false;
180        int o = weights.get(idx);
181        weights.incr(idx, -weight);
182        int w = weights.get(idx);
183        if(w <= 0)
184        {
185            table.removeAt(idx);
186            weights.removeIndex(idx);
187        }
188        w = Math.min(o, o - w);
189        total -= w;
190        normalTotal -= w;
191        return true;
192    }
193
194    /**
195     * Given an Iterable of T item keys to remove, this tries to remove each item in items, though it can't affect items
196     * in nested ProbabilityTables, and returns true if any probabilities were changed.
197     * @param items an Iterable of T items that will all be removed from the normal (non-nested) items in this
198     * @return true if the probabilities changed, or false otherwise
199     */
200    public boolean removeAll(Iterable<T> items)
201    {
202        boolean changed = false;
203        for(T t : items)
204        {
205            changed |= remove(t);
206        }
207        return changed;
208    }
209
210    /**
211     * Given an OrderedMap of T item keys and Integer weight values, reduces the weights in this ProbabilityTable for
212     * all T keys by their corresponding weights, removing them if the weight becomes 0 or less. You may want to use
213     * {@link OrderedMap#makeMap(Object, Object, Object...)} to produce the parameter, unless you already have one.
214     * Returns true iff the probabilities changed.
215     * @param itemsAndWeights an OrderedMap of T keys to Integer values, where a key will be an item that should be
216     *                        reduced in weight or removed and a value will be that item's weight
217     * @return true if the probabilities changed or false otherwise
218     */
219    public boolean removeAll(OrderedMap<T, Integer> itemsAndWeights)
220    {
221        if(itemsAndWeights == null) return false;
222        boolean changed = false;
223        int sz = itemsAndWeights.size;
224        for (int i = 0; i < sz; i++) {
225            changed |= remove(itemsAndWeights.keyAt(i), itemsAndWeights.getAt(i));
226        }
227        return changed;
228    }
229
230    /**
231     * Adds the given probability table as a possible set of results for this table.
232     * The table parameter should not be the same object as this ProbabilityTable, nor should it contain cycles
233     * that could reference this object from inside the values of table. This could cause serious issues that would
234     * eventually terminate in a StackOverflowError if the cycles randomly repeated for too long. Only the first case
235     * is checked for (if the contents of this and table are equivalent, it returns without doing anything; this also
236     * happens if table is empty or null).
237     *
238     * Weight must be greater than 0.
239     *
240     * @param table the ProbabilityTable to be added; should not be the same as this object (avoid cycles)
241     * @param weight the weight to be given to the added table
242     * @return this for chaining
243     */
244    public ProbabilityTable<T> add(ProbabilityTable<T> table, int weight) {
245        if(weight <= 0 || table == null || contentEquals(table) || table.total <= 0)
246            return this;
247        weights.add(Math.max(0, weight));
248        extraTable.add(table);
249        total += Math.max(0, weight);
250        return this;
251    }
252
253    /**
254     * Given an OrderedMap of ProbabilityTable keys and Integer weight values, adds all keys as nested tables with their
255     * corresponding weights into this ProbabilityTable. All ProbabilityTable keys should have the same T type as this
256     * ProbabilityTable. You may want to use {@link OrderedMap#makeMap(Object, Object, Object...)} to produce the
257     * parameter, unless you already have one.
258     *
259     * The same rules apply to this as apply to {@link #add(ProbabilityTable, int)}; that is, no key in itemsAndWeights
260     * can be the same object as this ProbabilityTable, nor should any key contain cycles that could reference this
261     * object from inside the values of a key. This could cause serious issues that would eventually terminate in a
262     * StackOverflowError if the cycles randomly repeated for too long. Only the first case is checked for (if the
263     * contents of this and a key are equivalent, it ignores that key; this also
264     * happens if a key is empty or null).
265
266     * @param itemsAndWeights an OrderedMap of T keys to Integer values, where a key will be an item this can retrieve
267     *                        and a value will be its weight
268     * @return this for chaining
269     */
270    public ProbabilityTable<T> addAllNested(OrderedMap<ProbabilityTable<T>, Integer> itemsAndWeights)
271    {
272        if(itemsAndWeights == null) return this;
273        int sz = itemsAndWeights.size;
274        for (int i = 0; i < sz; i++) {
275            add(itemsAndWeights.keyAt(i), itemsAndWeights.getAt(i));
276        }
277        return this;
278    }
279
280    /**
281     * Returns the weight of the item if the item is in the table. Returns zero
282     * if the item is not in the table.
283     *
284     * @param item the item searched for
285     * @return the weight of the item, or zero
286     */
287    public int weight(T item) {
288        int i = table.getInt(item);
289        return i < 0 ? 0 : weights.get(i);
290    }
291
292    /**
293     * Returns the weight of the extra table if present. Returns zero
294     * if the extra table is not present.
295     *
296     * @param item the extra ProbabilityTable to search for
297     * @return the weight of the ProbabilityTable, or zero
298     */
299    public int weight(ProbabilityTable<T> item) {
300        int i = extraTable.indexOf(item);
301        return i < 0 ? 0 : weights.get(i + table.size());
302    }
303
304    /**
305     * Provides a set of the items in this table, without reference to their
306     * weight. Includes nested ProbabilityTable values, but as is the case throughout
307     * this class, cyclical references to ProbabilityTable values that reference this
308     * table will result in significant issues (such as a {@link StackOverflowError}
309     * crashing your program).
310     *
311     * @return an OrderedSet of all items stored; iteration order should be predictable
312     */
313    public OrderedSet<T> items() {
314        OrderedSet<T> os = table.keysAsOrderedSet();
315        for (int i = 0; i < extraTable.size(); i++) {
316            os.addAll(extraTable.get(i).items());
317        }
318        return os;
319    }
320
321    /**
322     * Provides a set of the items in this table that are not in nested tables, without
323     * reference to their weight. These are the items that are simple to access, hence
324     * the name. If you want the items that are in both the top-level and nested tables,
325     * you can use {@link #items()}.
326     * @return a predictably-ordered set of the items in the top-level table
327     */
328    public SortedSet<T> simpleItems()
329    {
330        return table.keySet();
331    }
332
333    /**
334     * Provides a set of the nested ProbabilityTable values in this table, without reference
335     * to their weight. Does not include normal values (non-table); for that, use items().
336     *
337     * @return a "sorted" set of all nested tables stored, really sorted in insertion order
338     */
339    public ArrayList<ProbabilityTable<T>> tables() {
340        return extraTable;
341    }
342
343    /**
344     * Sets the current RNG to the given RNG. You may prefer using a StatefulRNG (typically passing one in the
345     * constructor, but you can pass one here too) and setting its state in other code, which does not require calling
346     * this method again when the StatefulRNG has its state set.
347     * @param random an RNG, typically with a seed you want control over; may be a StatefulRNG or some other subclass
348     */
349    public void setRandom(StatefulRNG random)
350    {
351        if(random != null && random.getRandomness() instanceof StatefulRandomness)
352            rng = (StatefulRandomness) random.getRandomness();
353    }
354
355    /**
356     * Gets the random number generator (a StatefulRandomness) this uses.
357     * @return the StatefulRandomness used by this class, which is often (but not always) a LightRNG
358     */
359    public StatefulRandomness getRandom()
360    {
361        return rng;
362    }
363
364    /**
365     * Copies this ProbabilityTable so nothing in the copy is shared with the original, except for the T items (which
366     * might not be possible to copy). The RNG is also copied.
367     * @return a copy of this ProbabilityTable; no references should be shared except for T items
368     */
369    public ProbabilityTable<T> copy()
370    {
371        ProbabilityTable<T> n = new ProbabilityTable<>(rng.copy());
372        n.weights.addAll(weights);
373        n.table.putAll(table);
374        for (int i = 0; i < extraTable.size(); i++) {
375            n.extraTable.add(extraTable.get(i).copy());
376        }
377        n.total = total;
378        n.normalTotal = normalTotal;
379        return n;
380    }
381
382    @Override
383    public boolean equals(Object o) {
384        if (this == o) return true;
385        if (o == null || getClass() != o.getClass()) return false;
386
387        ProbabilityTable<?> that = (ProbabilityTable<?>) o;
388
389        if (!table.equals(that.table)) return false;
390        if (!extraTable.equals(that.extraTable)) return false;
391        if (!weights.equals(that.weights)) return false;
392        return rng != null ? rng.equals(that.rng) : that.rng == null;
393    }
394
395    public boolean contentEquals(ProbabilityTable<T> o) {
396        if (this == o) return true;
397        if (o == null) return false;
398
399        if (!table.equals(o.table)) return false;
400        if (!extraTable.equals(o.extraTable)) return false;
401        return weights.equals(o.weights);
402    }
403
404    @Override
405    public int hashCode() {
406        int result = table.hashCode();
407        result = 31 * result + extraTable.hashCode();
408        result = 31 * result + weights.hashWisp();
409        result = 31 * result + (rng != null ? rng.hashCode() : 0);
410        return result;
411    }
