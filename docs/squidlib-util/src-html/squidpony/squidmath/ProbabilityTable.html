001package squidpony.squidmath;
002
003import squidpony.annotation.Beta;
004
005import java.io.Serializable;
006import java.util.SortedSet;
007
008/**
009 * A generic method of holding a probability table to determine weighted random
010 * outcomes.
011 *
012 * The weights do not need to add up to any particular value, they will be
013 * normalized when choosing a random entry.
014 *
015 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
016 *
017 * @param <T> The type of object to be held in the table
018 */
019@Beta
020public class ProbabilityTable<T> implements Serializable {
021    private static final long serialVersionUID = -1307656083434154736L;
022    public final Arrangement<T> table;
023    public final IntVLA weights;
024    protected RNG rng;
025    protected int total;
026
027    /**
028     * Creates a new probability table.
029     */
030    public ProbabilityTable() {
031        this(new StatefulRNG());
032    }
033
034    /**
035     * Creates a new probability table with the provided source of randomness
036     * used.
037     *
038     * @param rng the source of randomness
039     */
040    public ProbabilityTable(RNG rng) {
041        this.rng = rng;
042        table = new Arrangement<>(64, 0.75f);
043        weights = new IntVLA(64);
044        total = 0;
045    }
046
047    /**
048     * Creates a new probability table with the provided long seed used.
049     *
050     * @param seed the RNG seed as a long
051     */
052    public ProbabilityTable(long seed) {
053        this.rng = new StatefulRNG(seed);
054        table = new Arrangement<>(64, 0.75f);
055        weights = new IntVLA(64);
056        total = 0;
057    }
058
059    /**
060     * Creates a new probability table with the provided String seed used.
061     *
062     * @param seed the RNG seed as a String
063     */
064    public ProbabilityTable(String seed) {
065        this.rng = new StatefulRNG(CrossHash.Lightning.hash64(seed));
066        table = new Arrangement<>(64, 0.75f);
067        weights = new IntVLA(64);
068        total = 0;
069    }
070
071    /**
072     * Returns an object randomly based on assigned weights.
073     *
074     * Returns null if no elements have been put in the table.
075     *
076     * @return the chosen object or null
077     */
078    public T random() {
079        if (table.isEmpty()) {
080            return null;
081        }
082        int index = rng.nextInt(total);
083        for (int i = 0; i < table.size(); i++) {
084            index -= weights.get(i);
085            if (index < 0) {
086                return table.keyAt(i);
087            }
088        }
089        return null;//something went wrong, shouldn't have been able to get all the way through without finding an item
090    }
091
092    /**
093     * Adds the given item to the table.
094     *
095     * Weight must be greater than 0.
096     *
097     * @param item the object to be added
098     * @param weight the weight to be given to the added object
099     * @return this for chaining
100     */
101    public ProbabilityTable<T> add(T item, int weight) {
102        int i = table.getInt(item);
103        if (i < 0) {
104            table.add(item);
105            weights.add(Math.max(0, weight));
106            total += Math.max(0, weight);
107        } else {
108            i = weights.get(i);
109            table.add(item);
110            weights.add(Math.max(0, i + weight));
111            total += Math.max(0, i + weight) - i;
112        }
113        return this;
114    }
115
116    /**
117     * Returns the weight of the item if the item is in the table. Returns zero
118     * of the item is not in the table.
119     *
120     * @param item the item searched for
121     * @return the weight of the item, or zero
122     */
123    public int weight(T item) {
124        int i = table.getInt(item);
125        return i < 0 ? 0 : weights.get(i);
126    }
127
128    /**
129     * Provides a set of the items in this table, without reference to their
130     * weight.
131     *
132     * @return a "sorted" set of all items stored, really sorted in insertion order
133     */
134    public SortedSet<T> items() {
135        return table.keySet();
136    }
137
138    /**
139     * Sets the current RNG to the given RNG. You may prefer using a StatefulRNG (typically passing one in the
140     * constructor, but you can pass one here too) and setting its state in other code, which does not require calling
141     * this method again when the StatefulRNG has its state set.
142     * @param random an RNG, typically with a seed you want control over; may be a StatefulRNG or some other subclass
143     */
144    public void setRandom(RNG random)
145    {
146        rng = random;
147    }
148}
