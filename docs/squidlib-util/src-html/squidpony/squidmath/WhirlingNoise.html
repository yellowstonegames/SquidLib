008/**
009 * Another experimental noise class. Extends PerlinNoise and should have similar quality, but can be faster and has less
010 * periodic results. This is still considered experimental because the exact output may change in future versions, along
011 * with the scale (potentially) and the parameters it takes. In general, {@link #noise(double, double)} and
012 * {@link #noise(double, double, double)} should have similar appearance to {@link PerlinNoise#noise(double, double)}
013 * and {@link PerlinNoise#noise(double, double, double)}, but are not forced to a zoomed-in scale like PerlinNoise makes
014 * its results, are less likely to repeat sections of noise, and are also somewhat faster (a 20% speedup can be expected
015 * over PerlinNoise using those two methods). Sound good?
016 * <br>
017 * Created by Tommy Ettinger on 12/14/2016.
018 */
019@Beta
020public class WhirlingNoise extends PerlinNoise implements Noise.Noise2D, Noise.Noise3D, Noise.Noise4D {

161            final float len = 1f / (float)Math.sqrt(x * x + y * y + z * z), len3 = len * root2, len4 = len * root3;

252     * 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks because

345     * 2D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
346     * different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method. Roughly
347     * 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks because
348     * it uses a pseudo-random function (curiously, {@link LightRNG#determine(long)}, which has rather good distribution
349     * and is fast) instead of a number chosen by hash from a single 256-element array.
350     *
351     * @param xin X input; works well if between 0.0 and 1.0, but anything is accepted
352     * @param yin Y input; works well if between 0.0 and 1.0, but anything is accepted
353     * @return noise from -1.0 to 1.0, inclusive
354     */
355    public static double noise(final double xin, final double yin, final int seed) {
356        //xin *= epi;
357        //yin *= epi;
358        double noise0, noise1, noise2; // from the three corners
359        // Skew the input space to figure out which simplex cell we're in
360        double skew = (xin + yin) * F2; // Hairy factor for 2D
361        int i = fastFloor(xin + skew);
362        int j = fastFloor(yin + skew);
363        double t = (i + j) * G2;
364        double X0 = i - t; // Unskew the cell origin back to (x,y) space
365        double Y0 = j - t;
366        double x0 = xin - X0; // The x,y distances from the cell origin
367        double y0 = yin - Y0;
368        // For the 2D case, the simplex shape is an equilateral triangle.
369        // determine which simplex we are in.
370        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
371        // coords
372        if (x0 > y0) {
373            i1 = 1;
374            j1 = 0;
375        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
376        else {
377            i1 = 0;
378            j1 = 1;
379        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
380        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
381        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
382        // where
383        // c = (3-sqrt(3))/6
384        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
385        // unskewed coords
386        double y1 = y0 - j1 + G2;
387        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
388        // unskewed coords
389        double y2 = y0 - 1.0 + 2.0 * G2;
390        // Work out the hashed gradient indices of the three simplex corners
391        /*
392        int ii = i & 255;
393        int jj = j & 255;
394        int gi0 = perm[ii + perm[jj]] & 15;
395        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
396        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
397        */
398        /*
399        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
400                i + i1 + ((j + j1) * 0x9E3779B9),
401                i + 1 + ((j + 1) * 0x9E3779B9),
402                seed);
403        int gi0 = hash & 15;
404        int gi1 = (hash >>>= 4) & 15;
405        int gi2 = (hash >>> 4) & 15;
406        */
407        int gi0 = determineBounded(seed + i + determine(j), 16);
408        int gi1 = determineBounded(seed + i + i1 + determine(j + j1), 16);
409        int gi2 = determineBounded(seed + i + 1 + determine(j + 1), 16);
410
411        // Calculate the contribution from the three corners
412        double t0 = 0.5 - x0 * x0 - y0 * y0;
413        if (t0 < 0) {
414            noise0 = 0.0;
415        } else {
416            t0 *= t0;
417            noise0 = t0 * t0 * dot(phiGrad2[gi0], x0, y0);
418            // for 2D gradient
419        }
420        double t1 = 0.5 - x1 * x1 - y1 * y1;
421        if (t1 < 0) {
422            noise1 = 0.0;
423        } else {
424            t1 *= t1;
425            noise1 = t1 * t1 * dot(phiGrad2[gi1], x1, y1);
426        }
427        double t2 = 0.5 - x2 * x2 - y2 * y2;
428        if (t2 < 0) {
429            noise2 = 0.0;
430        } else {
431            t2 *= t2;
432            noise2 = t2 * t2 * dot(phiGrad2[gi2], x2, y2);
433        }
434        // Add contributions from each corner to get the final noise value.
435        // The result is scaled to return values in the interval [-1,1].
436        return 70.0 * (noise0 + noise1 + noise2);
437    }

859        //int hash = (int) rawNoise(i, j, k, seed);
860        //int gi0 = (hash >>>= 4) % 12, gi1 = (hash >>>= 4) % 12, gi2 = (hash >>>= 4) % 12, gi3 = (hash >>>= 4) % 12;
861        // Calculate the contribution from the four corners
862        float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
863        if (t0 < 0) {
864            n0 = 0f;
865        } else {
866            t0 *= t0;
867            n0 = t0 * t0 * dotf(grad3f[gi0], x0, y0, z0);
868        }
869        float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
870        if (t1 < 0) {
871            n1 = 0f;
872        } else {
873            t1 *= t1;
874            n1 = t1 * t1 * dotf(grad3f[gi1], x1, y1, z1);
875        }
876        float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
877        if (t2 < 0) {
878            n2 = 0f;
879        } else {
880            t2 *= t2;
881            n2 = t2 * t2 * dotf(grad3f[gi2], x2, y2, z2);
882        }
883        float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
884        if (t3 < 0) {
885            n3 = 0f;
886        } else {
887            t3 *= t3;
888            n3 = t3 * t3 * dotf(grad3f[gi3], x3, y3, z3);
889        }
890        // Add contributions from each corner to get the final noise value.
891        // The result is scaled to stay just inside [-1,1]
892        return 32f * (n0 + n1 + n2 + n3);
893    }

911    /**
912     * 4D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
913     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double, double)} and this
914     * method. Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in
915     * chunks because it uses a pseudo-random function (curiously, {@link LightRNG#determine(long)}, which has rather
916     * good distribution and is fast) instead of a number chosen by hash from a single 256-element array.
917     * @param x X input
918     * @param y Y input
919     * @param z Z input
920     * @param w W input (fourth-dimensional)
921     * @param seed any int; will be used to completely alter the noise
922     * @return noise from -1.0 to 1.0, inclusive
923     */
924    public static double noise(final double x, final double y, final double z, final double w, final int seed)
925    {
926        // The skewing and unskewing factors are hairy again for the 4D case
927
928        // Skew the (x,y,z,w) space to figure out which cell of 24 simplices
929        // we're in
930        double s = (x + y + z + w) * F4; // Factor for 4D skewing
931        int i = fastFloor(x + s);
932        int j = fastFloor(y + s);
933        int k = fastFloor(z + s);
934        int l = fastFloor(w + s);
935        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
936        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
937        double Y0 = j - t;
938        double Z0 = k - t;
939        double W0 = l - t;
940        double x0 = x - X0; // The x,y,z,w distances from the cell origin
941        double y0 = y - Y0;
942        double z0 = z - Z0;
943        double w0 = w - W0;
944        // For the 4D case, the simplex is a 4D shape I won't even try to
945        // describe.
946        // To find out which of the 24 possible simplices we're in, we need
947        // to figure out the magnitude ordering of x0, y0, z0 and w0.
948        // The method below is a good way of finding the ordering of x,y,z,w
949        // and
950        // then find the correct traversal order for the simplex were in.
951        // First, six pair-wise comparisons are performed between each
952        // possible pair
953        // of the four coordinates, and the results are used to add up binary
954        // bits
955        // for an integer index.
956        int c = (x0 > y0 ? 32 : 0) | (x0 > z0 ? 16 : 0) | (y0 > z0 ? 8 : 0) |
957                (x0 > w0 ? 4 : 0) | (y0 > w0 ? 2 : 0) | (z0 > w0 ? 1 : 0);
958
959        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
960        // order.
961        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
962        // y<w and x<w
963        // impossible. Only the 24 indices which have non-zero entries make
964        // any sense.
965        // We use a thresholding to set the coordinates in turn from the
966        // largest magnitude.
967        // The number 3 in the "simplex" array is at the position of the
968        // largest coordinate.
969
970        // The integer offsets for the second simplex corner
971        int i1 = simplex[c][0] >= 3 ? 1 : 0;
972        int j1 = simplex[c][1] >= 3 ? 1 : 0;
973        int k1 = simplex[c][2] >= 3 ? 1 : 0;
974        int l1 = simplex[c][3] >= 3 ? 1 : 0;
975        // The number 2 in the "simplex" array is at the second largest
976        // coordinate.
977
978        // The integer offsets for the third simplex corner
979        int i2 = simplex[c][0] >= 2 ? 1 : 0;
980        int j2 = simplex[c][1] >= 2 ? 1 : 0;
981        int k2 = simplex[c][2] >= 2 ? 1 : 0;
982        int l2 = simplex[c][3] >= 2 ? 1 : 0;
983        // The number 1 in the "simplex" array is at the second smallest
984        // coordinate.
985
986        // The integer offsets for the fourth simplex corner
987        int i3 = simplex[c][0] >= 1 ? 1 : 0;
988        int j3 = simplex[c][1] >= 1 ? 1 : 0;
989        int k3 = simplex[c][2] >= 1 ? 1 : 0;
990        int l3 = simplex[c][3] >= 1 ? 1 : 0;
991        // The fifth corner has all coordinate offsets = 1, so no need to
992        // look that up.
993        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w) coords
994        double y1 = y0 - j1 + G4;
995        double z1 = z0 - k1 + G4;
996        double w1 = w0 - l1 + G4;
997        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in (x,y,z,w) coords
998        double y2 = y0 - j2 + 2.0 * G4;
999        double z2 = z0 - k2 + 2.0 * G4;
1000        double w2 = w0 - l2 + 2.0 * G4;
1001        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in (x,y,z,w) coords
1002        double y3 = y0 - j3 + 3.0 * G4;
1003        double z3 = z0 - k3 + 3.0 * G4;
1004        double w3 = w0 - l3 + 3.0 * G4;
1005        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in (x,y,z,w) coords
1006        double y4 = y0 - 1.0 + 4.0 * G4;
1007        double z4 = z0 - 1.0 + 4.0 * G4;
1008        double w4 = w0 - 1.0 + 4.0 * G4;
1009
1010        int gi0 = determineBounded(seed + i + determine(j + determine(k + determine(l))), 32);
1011        int gi1 = determineBounded(seed + i + i1 + determine(j + j1 + determine(k + k1 + determine(l + l1))), 32);
1012        int gi2 = determineBounded(seed + i + i2 + determine(j + j2 + determine(k + k2 + determine(l + l2))), 32);
1013        int gi3 = determineBounded(seed + i + i3 + determine(j + j3 + determine(k + k3 + determine(l + l3))), 32);
1014        int gi4 = determineBounded(seed + i + 1 + determine(j + 1 + determine(k + 1 + determine(l + 1))), 32);
1015
1016        // Noise contributions from the five corners are n0 to n4
1017
1018        // Calculate the contribution from the five corners
1019        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0, n0;
1020        if (t0 < 0) {
1021            n0 = 0.0;
1022        } else {
1023            t0 *= t0;
1024            n0 = t0 * t0 * dot(grad4f[gi0], x0, y0, z0, w0);
1025        }
1026        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1, n1;
1027        if (t1 < 0) {
1028            n1 = 0.0;
1029        } else {
1030            t1 *= t1;
1031            n1 = t1 * t1 * dot(grad4f[gi1], x1, y1, z1, w1);
1032        }
1033        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2,  n2;
1034        if (t2 < 0) {
1035            n2 = 0.0;
1036        } else {
1037            t2 *= t2;
1038            n2 = t2 * t2 * dot(grad4f[gi2], x2, y2, z2, w2);
1039        }
1040        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3, n3;
1041        if (t3 < 0) {
1042            n3 = 0.0;
1043        } else {
1044            t3 *= t3;
1045            n3 = t3 * t3 * dot(grad4f[gi3], x3, y3, z3, w3);
1046        }
1047        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4, n4;
1048        if (t4 < 0) {
1049            n4 = 0.0;
1050        } else {
1051            t4 *= t4;
1052            n4 = t4 * t4 * dot(grad4f[gi4], x4, y4, z4, w4);
1053        }
1054        // Sum up and scale the result to cover the range [-1,1]
1055        return 27.0 * (n0 + n1 + n2 + n3 + n4);
1056    }
