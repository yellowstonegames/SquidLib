008/**
009 * Another experimental noise class. Extends PerlinNoise and should have similar quality, but can be faster and has less
010 * periodic results. This is still considered experimental because the exact output may change in future versions, along
011 * with the scale (potentially) and the parameters it takes. In general, {@link #noise(double, double)} and
012 * {@link #noise(double, double, double)} should have similar appearance to {@link PerlinNoise#noise(double, double)}
013 * and {@link PerlinNoise#noise(double, double, double)}, but are not forced to a zoomed-in scale like PerlinNoise makes
014 * its results, are less likely to repeat sections of noise, and are also somewhat faster (a 20% speedup can be expected
015 * over PerlinNoise using those two methods). Sound good? This also performs minimal (if any) arithmetic on 64-bit long
016 * numbers, instead using double and int, a style that is ideal for GWT (this is true of PerlinNoise too).
017 * <br>
018 * Created by Tommy Ettinger on 12/14/2016.
019 */
020@Beta
021public class WhirlingNoise extends PerlinNoise {
022    private static int fastFloor(double t) {
023        return t >= 0 ? (int) t : (int) t - 1;
024    }
025    private static int fastFloor(float t) {
026        return t >= 0 ? (int) t : (int) t - 1;
027    }
028    protected static final float root3 = 1.7320508f, root5 = 2.236068f,
029            F2f = 0.5f * (root3 - 1f),
030            G2f = (3f - root3) * 0.16666667f,
031            F3f = 0.33333334f,
032            G3f = 0.16666667f,
033            F4f = (root5 - 1f) * 0.25f,
034            G4f = (5f - root5) * 0.05f,
035            unit1_4f =  0.70710678118f, unit1_8f = 0.38268343236f, unit3_8f = 0.92387953251f;
036
037    protected static final float[][] grad2f = {
038            {1f, 0f}, {-1f, 0f}, {0f, 1f}, {0f, -1f},
039            {unit3_8f, unit1_8f}, {unit3_8f, -unit1_8f}, {-unit3_8f, unit1_8f}, {-unit3_8f, -unit1_8f},
040            {unit1_4f, unit1_4f}, {unit1_4f, -unit1_4f}, {-unit1_4f, unit1_4f}, {-unit1_4f, -unit1_4f},
041            {unit1_8f, unit3_8f}, {unit1_8f, -unit3_8f}, {-unit1_8f, unit3_8f}, {-unit1_8f, -unit3_8f}};
042
043    protected static float dotf(final float g[], final float x, final float y) {
044        return g[0] * x + g[1] * y;
045    }
046
047    protected static float dotf(final int g[], final float x, final float y, final float z) {
048        return g[0] * x + g[1] * y + g[2] * z;
049    }

072    /**
073     * 2D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
074     * different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method. Roughly
075     * 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks because
076     * it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized for the
077     * limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
078     *
079     * @param xin x input; works well if between 0.0 and 1.0, but anything is accepted
080     * @param yin y input; works well if between 0.0 and 1.0, but anything is accepted
081     * @return noise from -1.0 to 1.0, inclusive
082     */
083    public static double noise(double xin, double yin) {
084        //xin *= epi;
085        //yin *= epi;
086        double noise0, noise1, noise2; // from the three corners
087        // Skew the input space to determine which simplex cell we're in
088        double skew = (xin + yin) * F2; // Hairy factor for 2D
089        int i = fastFloor(xin + skew);
090        int j = fastFloor(yin + skew);
091        double t = (i + j) * G2;
092        double X0 = i - t; // Unskew the cell origin back to (x,y) space
093        double Y0 = j - t;
094        double x0 = xin - X0; // The x,y distances from the cell origin
095        double y0 = yin - Y0;
096        // For the 2D case, the simplex shape is an equilateral triangle.
097        // Determine which simplex we are in.
098        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
099        // coords
100        if (x0 > y0) {
101            i1 = 1;
102            j1 = 0;
103        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
104        else {
105            i1 = 0;
106            j1 = 1;
107        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
108        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
109        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
110        // where
111        // c = (3-sqrt(3))/6
112        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
113        // unskewed coords
114        double y1 = y0 - j1 + G2;
115        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
116        // unskewed coords
117        double y2 = y0 - 1.0 + 2.0 * G2;
118        // Work out the hashed gradient indices of the three simplex corners
119        /*
120        int ii = i & 255;
121        int jj = j & 255;
122        int gi0 = perm[ii + perm[jj]] & 15;
123        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
124        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
125        */
126        /*
127        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
128                i + i1 + ((j + j1) * 0x9E3779B9),
129                i + 1 + ((j + 1) * 0x9E3779B9),
130                seed);
131        int gi0 = hash & 15;
132        int gi1 = (hash >>>= 4) & 15;
133        int gi2 = (hash >>> 4) & 15;
134        */
135        int gi0 = determine(i + determine(j)) & 15;
136        int gi1 = determine(i + i1 + determine(j + j1)) & 15;
137        int gi2 = determine(i + 1 + determine(j + 1)) & 15;
138
139        // Calculate the contribution from the three corners
140        double t0 = 0.5 - x0 * x0 - y0 * y0;
141        if (t0 < 0) {
142            noise0 = 0.0;
143        } else {
144            t0 *= t0;
145            noise0 = t0 * t0 * dot(grad2[gi0], x0, y0);
146            // for 2D gradient
147        }
148        double t1 = 0.5 - x1 * x1 - y1 * y1;
149        if (t1 < 0) {
150            noise1 = 0.0;
151        } else {
152            t1 *= t1;
153            noise1 = t1 * t1 * dot(grad2[gi1], x1, y1);
154        }
155        double t2 = 0.5 - x2 * x2 - y2 * y2;
156        if (t2 < 0) {
157            noise2 = 0.0;
158        } else {
159            t2 *= t2;
160            noise2 = t2 * t2 * dot(grad2[gi2], x2, y2);
161        }
162        // Add contributions from each corner to get the final noise value.
163        // The result is scaled to return values in the interval [-1,1].
164        return 70.0 * (noise0 + noise1 + noise2);
165    }

179        //yin *= epi;
180        float noise0, noise1, noise2; // from the three corners
181        float xin = (float)x, yin = (float)y;
182        // Skew the input space to determine which simplex cell we're in
183        float skew = (xin + yin) * F2f; // Hairy factor for 2D
184        int i = fastFloor(xin + skew);
185        int j = fastFloor(yin + skew);

187        float X0 = i - t; // Unskew the cell origin back to (x,y) space

190        float y0 = yin - Y0;
191        // For the 2D case, the simplex shape is an equilateral triangle.

207        float x1 = x0 - i1 + G2f; // Offsets for middle corner in (x,y)
208        // unskewed coords
209        float y1 = y0 - j1 + G2f;
210        float x2 = x0 - 1f + 2f * G2f; // Offsets for last corner in (x,y)
211        // unskewed coords
212        float y2 = y0 - 1f + 2f * G2f;
213        // Work out the hashed gradient indices of the three simplex corners
214        /*
215        int ii = i & 255;
216        int jj = j & 255;
217        int gi0 = perm[ii + perm[jj]] & 15;

232        int gi2 = PintRNG.determine(i + 1 + PintRNG.determine(j + 1)) & 15;

240            noise0 = t0 * t0 * dotf(grad2f[gi0], x0, y0);

242        }

244        if (t1 < 0) {
245            noise1 = 0f;
246        } else {
247            t1 *= t1;
248            noise1 = t1 * t1 * dotf(grad2f[gi1], x1, y1);
249        }

251        if (t2 < 0) {
252            noise2 = 0f;
253        } else {
254            t2 *= t2;
255            noise2 = t2 * t2 * dotf(grad2f[gi2], x2, y2);
256        }
257        // Add contributions from each corner to get the final noise value.
258        // The result is scaled to return values in the interval [-1,1].
259        return 70f * (noise0 + noise1 + noise2);
260    }

262    /**
263     * 3D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
264     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double)} and this method.
265     * Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks
266     * because it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized for the
267     * limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
268     * @param xin X input
269     * @param yin Y input
270     * @param zin Z input
271     * @return noise from -1.0 to 1.0, inclusive
272     */
273    public static double noise(double xin, double yin, double zin) {
274        //xin *= epi;
275        //yin *= epi;
276        //zin *= epi;
277        double n0, n1, n2, n3; // Noise contributions from the four corners
278        // Skew the input space to determine which simplex cell we're in
279        double s = (xin + yin + zin) * F3; // Very nice and simple skew
280        // factor for 3D
281        int i = fastFloor(xin + s);
282        int j = fastFloor(yin + s);
283        int k = fastFloor(zin + s);
284        double t = (i + j + k) * G3;
285        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
286        double Y0 = j - t;
287        double Z0 = k - t;
288        double x0 = xin - X0; // The x,y,z distances from the cell origin
289        double y0 = yin - Y0;
290        double z0 = zin - Z0;
291        // For the 3D case, the simplex shape is a slightly irregular
292        // tetrahedron.
293        // Determine which simplex we are in.
294        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
295        // coords
296        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
297        // coords
298        if (x0 >= y0) {
299            if (y0 >= z0) {
300                i1 = 1;
301                j1 = 0;
302                k1 = 0;
303                i2 = 1;
304                j2 = 1;
305                k2 = 0;
306            } // X Y Z order
307            else if (x0 >= z0) {
308                i1 = 1;
309                j1 = 0;
310                k1 = 0;
311                i2 = 1;
312                j2 = 0;
313                k2 = 1;
314            } // X Z Y order
315            else {
316                i1 = 0;
317                j1 = 0;
318                k1 = 1;
319                i2 = 1;
320                j2 = 0;
321                k2 = 1;
322            } // Z X Y order
323        } else { // x0<y0
324            if (y0 < z0) {
325                i1 = 0;
326                j1 = 0;
327                k1 = 1;
328                i2 = 0;
329                j2 = 1;
330                k2 = 1;
331            } // Z Y X order
332            else if (x0 < z0) {
333                i1 = 0;
334                j1 = 1;
335                k1 = 0;
336                i2 = 0;
337                j2 = 1;
338                k2 = 1;
339            } // Y Z X order
340            else {
341                i1 = 0;
342                j1 = 1;
343                k1 = 0;
344                i2 = 1;
345                j2 = 1;
346                k2 = 0;
347            } // Y X Z order
348        }
349        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
350        // (x,y,z),
351        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
352        // (x,y,z), and
353        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
354        // (x,y,z), where
355        // c = 1/6.
356        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
357        // coords
358        double y1 = y0 - j1 + G3;
359        double z1 = z0 - k1 + G3;
360        double x2 = x0 - i2 + F3; // Offsets for third corner in
361        // (x,y,z) coords
362        double y2 = y0 - j2 + F3;
363        double z2 = z0 - k2 + F3;
364        double x3 = x0 - 0.5; // Offsets for last corner in
365        // (x,y,z) coords
366        double y3 = y0 - 0.5;
367        double z3 = z0 - 0.5;
368        // Work out the hashed gradient indices of the four simplex corners
369
370        /*
371        int ii = i & 255;
372        int jj = j & 255;
373        int kk = k & 255;
374
375        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
376        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
377        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
378        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
379        */
380
381        int gi0 = determineBounded(i + determine(j + determine(k)), 12);
382        int gi1 = determineBounded(i + i1 + determine(j + j1 + determine(k + k1)), 12);
383        int gi2 = determineBounded(i + i2 + determine(j + j2 + determine(k + k2)), 12);
384        int gi3 = determineBounded(i + 1 + determine(j + 1 + determine(k + 1)), 12);
385
386        /*
387        int hash = (int) rawNoise(i + ((j + k * 0x632BE5AB) * 0x9E3779B9),
388                i + i1 + ((j + j1 + (k + k1) * 0x632BE5AB) * 0x9E3779B9),
389                i + i2 + ((j + j2 + (k + k2) * 0x632BE5AB) * 0x9E3779B9),
390                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),
391                seed);
392        int gi0 = (hash >>>= 4) % 12;
393        int gi1 = (hash >>>= 4) % 12;
394        int gi2 = (hash >>>= 4) % 12;
395        int gi3 = (hash >>> 4) % 12;
396        */
397
398        //int hash = (int) rawNoise(i, j, k, seed);
399        //int gi0 = (hash >>>= 4) % 12, gi1 = (hash >>>= 4) % 12, gi2 = (hash >>>= 4) % 12, gi3 = (hash >>>= 4) % 12;
400        // Calculate the contribution from the four corners
401        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
402        if (t0 < 0) {
403            n0 = 0.0;
404        } else {
405            t0 *= t0;
406            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
407        }
408        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
409        if (t1 < 0) {
410            n1 = 0.0;
411        } else {
412            t1 *= t1;
413            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
414        }
415        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
416        if (t2 < 0) {
417            n2 = 0.0;
418        } else {
419            t2 *= t2;
420            n2 = t2 * t2 * dot(grad3[gi2], x2, y2, z2);
421        }
422        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
423        if (t3 < 0) {
424            n3 = 0.0;
425        } else {
426            t3 *= t3;
427            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
428        }
429        // Add contributions from each corner to get the final noise value.
430        // The result is scaled to stay just inside [-1,1]
431        return 32.0 * (n0 + n1 + n2 + n3);
432    }

526        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
527        // (x,y,z), where
528        // c = 1/6.
529        float x1 = x0 - i1 + G3f; // Offsets for second corner in (x,y,z)
530        // coords
531        float y1 = y0 - j1 + G3f;
532        float z1 = z0 - k1 + G3f;
533        float x2 = x0 - i2 + F3f; // Offsets for third corner in
534        // (x,y,z) coords
535        float y2 = y0 - j2 + F3f;
536        float z2 = z0 - k2 + F3f;
537        float x3 = x0 - 0.5f; // Offsets for last corner in
538        // (x,y,z) coords
539        float y3 = y0 - 0.5f;
540        float z3 = z0 - 0.5f;
541        // Work out the hashed gradient indices of the four simplex corners
542
543        /*
544        int ii = i & 255;
545        int jj = j & 255;
546        int kk = k & 255;
547
548        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
549        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
550        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
551        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
552        */
553
554        int gi0 = PintRNG.determineBounded(i + PintRNG.determine(j + PintRNG.determine(k)), 12);
555        int gi1 = PintRNG.determineBounded(i + i1 + PintRNG.determine(j + j1 + PintRNG.determine(k + k1)), 12);
556        int gi2 = PintRNG.determineBounded(i + i2 + PintRNG.determine(j + j2 + PintRNG.determine(k + k2)), 12);
557        int gi3 = PintRNG.determineBounded(i + 1 + PintRNG.determine(j + 1 + PintRNG.determine(k + 1)), 12);
558
559        /*
560        int hash = (int) rawNoise(i + ((j + k * 0x632BE5AB) * 0x9E3779B9),
561                i + i1 + ((j + j1 + (k + k1) * 0x632BE5AB) * 0x9E3779B9),
562                i + i2 + ((j + j2 + (k + k2) * 0x632BE5AB) * 0x9E3779B9),
563                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),
564                seed);
565        int gi0 = (hash >>>= 4) % 12;
566        int gi1 = (hash >>>= 4) % 12;
567        int gi2 = (hash >>>= 4) % 12;
568        int gi3 = (hash >>> 4) % 12;
569        */
570
571        //int hash = (int) rawNoise(i, j, k, seed);
572        //int gi0 = (hash >>>= 4) % 12, gi1 = (hash >>>= 4) % 12, gi2 = (hash >>>= 4) % 12, gi3 = (hash >>>= 4) % 12;
573        // Calculate the contribution from the four corners
574        float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
575        if (t0 < 0) {
576            n0 = 0f;
577        } else {
578            t0 *= t0;
579            n0 = t0 * t0 * dotf(grad3[gi0], x0, y0, z0);
580        }
581        float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
582        if (t1 < 0) {
583            n1 = 0f;
584        } else {
585            t1 *= t1;
586            n1 = t1 * t1 * dotf(grad3[gi1], x1, y1, z1);
587        }
588        float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
589        if (t2 < 0) {
590            n2 = 0f;
591        } else {
592            t2 *= t2;
593            n2 = t2 * t2 * dotf(grad3[gi2], x2, y2, z2);
594        }
595        float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
596        if (t3 < 0) {
597            n3 = 0f;
598        } else {
599            t3 *= t3;
600            n3 = t3 * t3 * dotf(grad3[gi3], x3, y3, z3);
601        }
602        // Add contributions from each corner to get the final noise value.
603        // The result is scaled to stay just inside [-1,1]
604        return 32f * (n0 + n1 + n2 + n3);
605    }

607    /**
608     * 4D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
609     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double, double)} and this
610     * method. Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in
611     * chunks because it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized
612     * for the limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
613     * @param x X input
614     * @param y Y input
615     * @param z Z input
616     * @param w W input (fourth-dimensional)
617     * @return noise from -1.0 to 1.0, inclusive
618     */
619    public static double noise(double x, double y, double z, double w) {
620        // The skewing and unskewing factors are hairy again for the 4D case
621
622        // Skew the (x,y,z,w) space to determine which cell of 24 simplices
623        // we're in
624        double s = (x + y + z + w) * F4; // Factor for 4D skewing
625        int i = fastFloor(x + s);
626        int j = fastFloor(y + s);
627        int k = fastFloor(z + s);
628        int l = fastFloor(w + s);
629        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
630        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
631        double Y0 = j - t;
632        double Z0 = k - t;
633        double W0 = l - t;
634        double x0 = x - X0; // The x,y,z,w distances from the cell origin
635        double y0 = y - Y0;
636        double z0 = z - Z0;
637        double w0 = w - W0;
638        // For the 4D case, the simplex is a 4D shape I won't even try to
639        // describe.
640        // To find out which of the 24 possible simplices we're in, we need
641        // to
642        // determine the magnitude ordering of x0, y0, z0 and w0.
643        // The method below is a good way of finding the ordering of x,y,z,w
644        // and
645        // then find the correct traversal order for the simplex were in.
646        // First, six pair-wise comparisons are performed between each
647        // possible pair
648        // of the four coordinates, and the results are used to add up binary
649        // bits
650        // for an integer index.
651        int c = (x0 > y0 ? 32 : 0) | (x0 > z0 ? 16 : 0) | (y0 > z0 ? 8 : 0) |
652                (x0 > w0 ? 4 : 0) | (y0 > w0 ? 2 : 0) | (z0 > w0 ? 1 : 0);
653
654        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
655        // order.
656        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
657        // y<w and x<w
658        // impossible. Only the 24 indices which have non-zero entries make
659        // any sense.
660        // We use a thresholding to set the coordinates in turn from the
661        // largest magnitude.
662        // The number 3 in the "simplex" array is at the position of the
663        // largest coordinate.
664
665        // The integer offsets for the second simplex corner
666        int i1 = simplex[c][0] >= 3 ? 1 : 0;
667        int j1 = simplex[c][1] >= 3 ? 1 : 0;
668        int k1 = simplex[c][2] >= 3 ? 1 : 0;
669        int l1 = simplex[c][3] >= 3 ? 1 : 0;
670        // The number 2 in the "simplex" array is at the second largest
671        // coordinate.
672
673        // The integer offsets for the third simplex corner
674        int i2 = simplex[c][0] >= 2 ? 1 : 0;
675        int j2 = simplex[c][1] >= 2 ? 1 : 0;
676        int k2 = simplex[c][2] >= 2 ? 1 : 0;
677        int l2 = simplex[c][3] >= 2 ? 1 : 0;
678        // The number 1 in the "simplex" array is at the second smallest
679        // coordinate.
680
681        // The integer offsets for the fourth simplex corner
682        int i3 = simplex[c][0] >= 1 ? 1 : 0;
683        int j3 = simplex[c][1] >= 1 ? 1 : 0;
684        int k3 = simplex[c][2] >= 1 ? 1 : 0;
685        int l3 = simplex[c][3] >= 1 ? 1 : 0;
686        // The fifth corner has all coordinate offsets = 1, so no need to
687        // look that up.
688        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w) coords
689        double y1 = y0 - j1 + G4;
690        double z1 = z0 - k1 + G4;
691        double w1 = w0 - l1 + G4;
692        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in (x,y,z,w) coords
693        double y2 = y0 - j2 + 2.0 * G4;
694        double z2 = z0 - k2 + 2.0 * G4;
695        double w2 = w0 - l2 + 2.0 * G4;
696        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in (x,y,z,w) coords
697        double y3 = y0 - j3 + 3.0 * G4;
698        double z3 = z0 - k3 + 3.0 * G4;
699        double w3 = w0 - l3 + 3.0 * G4;
700        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in (x,y,z,w) coords
701        double y4 = y0 - 1.0 + 4.0 * G4;
702        double z4 = z0 - 1.0 + 4.0 * G4;
703        double w4 = w0 - 1.0 + 4.0 * G4;
704
705        int gi0 = determine(i + determine(j + determine(k + determine(l)))) & 31;
706        int gi1 = determine(i + i1 + determine(j + j1 + determine(k + k1 + determine(l + l1)))) & 31;
707        int gi2 = determine(i + i2 + determine(j + j2 + determine(k + k2 + determine(l + l2)))) & 31;
708        int gi3 = determine(i + i3 + determine(j + j3 + determine(k + k3 + determine(l + l3)))) & 31;
709        int gi4 = determine(i + 1 + determine(j + 1 + determine(k + 1 + determine(l + 1)))) & 31;
710
711        // Noise contributions from the five corners are n0 to n4
712
713        // Calculate the contribution from the five corners
714        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0, n0;
715        if (t0 < 0) {
716            n0 = 0.0;
717        } else {
718            t0 *= t0;
719            n0 = t0 * t0 * dot(grad4[gi0], x0, y0, z0, w0);
720        }
721        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1, n1;
722        if (t1 < 0) {
723            n1 = 0.0;
724        } else {
725            t1 *= t1;
726            n1 = t1 * t1 * dot(grad4[gi1], x1, y1, z1, w1);
727        }
728        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2,  n2;
729        if (t2 < 0) {
730            n2 = 0.0;
731        } else {
732            t2 *= t2;
733            n2 = t2 * t2 * dot(grad4[gi2], x2, y2, z2, w2);
734        }
735        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3, n3;
736        if (t3 < 0) {
737            n3 = 0.0;
738        } else {
739            t3 *= t3;
740            n3 = t3 * t3 * dot(grad4[gi3], x3, y3, z3, w3);
741        }
742        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4, n4;
743        if (t4 < 0) {
744            n4 = 0.0;
745        } else {
746            t4 *= t4;
747            n4 = t4 * t4 * dot(grad4[gi4], x4, y4, z4, w4);
748        }
749        // Sum up and scale the result to cover the range [-1,1]
750        return 27.0 * (n0 + n1 + n2 + n3 + n4);
751    }
