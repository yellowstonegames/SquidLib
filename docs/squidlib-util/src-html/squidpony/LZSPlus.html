035/**
036 * LZ-String compression, taking Strings and compressing them to other Strings, optionally with some encryption.
037 * This role was performed by the <a href="https://github.com/tommyettinger/BlazingChain">BlazingChain library</a>,
038 * which was a dependency of squidlib-extra, but recent developments have allowed all dependencies to be removed other
039 * than SquidLib, while also probably reducing memory usage by a fair amount. The actual implementation of this class is
040 * very unusual, with the LZ-String encoding part derived from
041 * <a href="https://github.com/rufushuang/lz-string4java">rufushuang's lz-string4java</a> (which is a port of
042 * <a href="https://github.com/pieroxy/lz-string">pieroxy's lz-string</a>), while the encryption-like part (which is not
043 * very strong) was added in SquidLib. The unusual part is how instead of relying on a HashMap of String keys to boxed
044 * Integer values as BlazingChain did, this uses an UnorderedSet of char arrays, using a custom IHasher to consider some
045 * of the array a key to hash and some only to read in as a value (producing an int like the Integer from before). This
046 * allows quite a bit of RAM to be conserved on larger texts. This version includes the fixes present in version 1.4.4.2
047 * of BlazingChain, which allow decompression for very large texts to work.
048 * <br>
049 * Created by Tommy Ettinger on 7/13/2017.
050 */
051public final class LZSPlus {
052
053    /**
054     * Compresses the given text using LZ-String compression; does not encrypt the result.
055     * @param uncompressedStr text to compress
056     * @return a compressed version of the given text
057     */
058    public static String compress(String uncompressedStr) {
059        return compress(uncompressedStr, null);
060    }
061
062    private static final CrossHash.IHasher stringWithIntHasher = new CrossHash.IHasher() {
063        @Override
064        public int hash(Object data) {
065            return (data instanceof char[]) ? CrossHash.hash((char[]) data, 0, ((char[]) data).length - 2) : data.hashCode();
066        }
067
068        @Override
069        public boolean areEqual(Object left, Object right) {
070            if (left == right) return true;
071            if (left instanceof char[] && right instanceof char[]) {
072                char[] l = (char[]) left, r = (char[]) right;
073                int len = l.length;
074                if (len != r.length)
075                    return false;
076                for (int i = len - 2; i >= 0; i--) {
077                    if(l[i] != r[i])
078                        return false;
079                }
080                return true;
081            }
082            return Objects.equals(left, right);
083        }
084    };
085    private static char[] combine(final char[] left, final char[] right)
086    {
087        final char[] next = new char[left.length + right.length - 2];
088        System.arraycopy(left, 0, next, 0, left.length - 2);
089        System.arraycopy(right, 0, next, left.length - 2, right.length - 2);
090        return next;
091    }
092    /**
093     * Compresses the given text using LZ-String compression and encrypts (somewhat) the compressed result so it can't
094     * be read back without the same keys as a long array. Shorter long arrays give less security to encryption, though
095     * there isn't much security to begin with. You can produce a decent-quality array for this purpose with
096     * {@link Garbler#makeKeyArray(int, String)}; the size parameter could reasonably be anywhere from 2 to 32. If the
097     * keys array is null or empty, this only compresses and does not perform an additional encryption step.
098     * @param uncompressedStr text to compress and optionally encrypt
099     * @param keys the long array that will be used to encrypt the output, and will be required to decrypt the result; may be null
100     * @return a compressed and optionally encrypted version of the given text
101     */
102    public static String compress(String uncompressedStr, long[] keys) {
103        if (uncompressedStr == null) return null;
104        if (uncompressedStr.isEmpty()) return "";
105        int i, value;
106        UnorderedSet<char[]> context_dictionary = new UnorderedSet<>(128, 0.7f, stringWithIntHasher);
107        UnorderedSet<char[]> context_dictionaryToCreate = new UnorderedSet<>(128, 0.7f, stringWithIntHasher);
108        char[] context_c;
109        char[] context_wc;
110        char[] context_w = new char[2];
111        char[] tmp;
112        int context_enlargeIn = 2; // Compensate for the first entry which should not count
113        int context_dictSize = 3;
114        int context_numBits = 2;
115        int len = uncompressedStr.length();
116        StringBuilder context_data = new StringBuilder(len >>> 1);
117        int context_data_val = 0;
118        int context_data_position = 0;
119        int ii;
120
121        for (ii = 0; ii < len; ii++) {
122            context_c = new char[]{uncompressedStr.charAt(ii), 0, 0};
123            if (!context_dictionary.contains(context_c)) {
124                context_c[1] = (char) (context_dictSize >>> 16);
125                context_c[2] = (char)(context_dictSize++ & 0xFFFF);
126                context_dictionary.add(context_c);
127                context_dictionaryToCreate.add(context_c);
128            }
129
130            context_wc = combine(context_w, context_c);
131            if (context_dictionary.contains(context_wc)) {
132                context_w = context_wc;
133            } else {
134                if (context_dictionaryToCreate.contains(context_w)) {
135                    if ((value = context_w[0]) < 256) {
136                        for (i = 0; i < context_numBits; i++) {
137                            context_data_val = (context_data_val << 1);
138                            if (context_data_position == 14) {
139                                context_data_position = 0;
140                                context_data.append((char) (context_data_val + 32));
141                                context_data_val = 0;
142                            } else {
143                                context_data_position++;
144                            }
145                        }
146                        for (i = 0; i < 8; i++) {
147                            context_data_val = (context_data_val << 1) | (value & 1);
148                            if (context_data_position == 14) {
149                                context_data_position = 0;
150                                context_data.append((char) (context_data_val + 32));
151                                context_data_val = 0;
152                            } else {
153                                context_data_position++;
154                            }
155                            value >>= 1;
156                        }
157                    } else {
158                        value = 1;
159                        for (i = 0; i < context_numBits; i++) {
160                            context_data_val = (context_data_val << 1) | value;
161                            if (context_data_position == 14) {
162                                context_data_position = 0;
163                                context_data.append((char) (context_data_val + 32));
164                                context_data_val = 0;
165                            } else {
166                                context_data_position++;
167                            }
168                            value = 0;
169                        }
170                        value = context_w[0];
171                        for (i = 0; i < 16; i++) {
172                            context_data_val = (context_data_val << 1) | (value & 1);
173                            if (context_data_position == 14) {
174                                context_data_position = 0;
175                                context_data.append((char) (context_data_val + 32));
176                                context_data_val = 0;
177                            } else {
178                                context_data_position++;
179                            }
180                            value >>= 1;
181                        }
182                    }
183                    context_enlargeIn--;
184                    if (context_enlargeIn == 0) {
185                        context_enlargeIn = 1 << context_numBits++;
186                    }
187                    context_dictionaryToCreate.remove(context_w);
188                } else {
189                    tmp = context_dictionary.get(context_w);
190                    if(tmp == null)
191                        value = -1;
192                    else {
193                        value = tmp[tmp.length - 2] << 16 | tmp[tmp.length - 1];
194                    }
195                    for (i = 0; i < context_numBits; i++) {
196                        context_data_val = (context_data_val << 1) | (value & 1);
197                        if (context_data_position == 14) {
198                            context_data_position = 0;
199                            context_data.append((char) (context_data_val + 32));
200                            context_data_val = 0;
201                        } else {
202                            context_data_position++;
203                        }
204                        value >>= 1;
205                    }
206
207                }
208                context_enlargeIn--;
209                if (context_enlargeIn == 0) {
210                    context_enlargeIn = 1 << context_numBits++;
211                }
212                // Add wc to the dictionary.
213                context_wc[context_wc.length - 2] = (char) (context_dictSize >>> 16);
214                context_wc[context_wc.length - 1] = (char)(context_dictSize++ & 0xFFFF);
215                context_dictionary.add(context_wc);
216                context_w = context_c;
217            }
218        }
219
220        // Output the code for w.
221        if (context_w.length > 2) {
222            if (context_dictionaryToCreate.contains(context_w)) {
223                if (context_w[0] < 256) {
224                    for (i = 0; i < context_numBits; i++) {
225                        context_data_val = (context_data_val << 1);
226                        if (context_data_position == 14) {
227                            context_data_position = 0;
228                            context_data.append((char) (context_data_val + 32));
229                            context_data_val = 0;
230                        } else {
231                            context_data_position++;
232                        }
233                    }
234                    value = context_w[0];
235                    for (i = 0; i < 8; i++) {
236                        context_data_val = (context_data_val << 1) | (value & 1);
237                        if (context_data_position == 14) {
238                            context_data_position = 0;
239                            context_data.append((char) (context_data_val + 32));
240                            context_data_val = 0;
241                        } else {
242                            context_data_position++;
243                        }
244                        value >>= 1;
245                    }
246                } else {
247                    value = 1;
248                    for (i = 0; i < context_numBits; i++) {
249                        context_data_val = (context_data_val << 1) | value;
250                        if (context_data_position == 14) {
251                            context_data_position = 0;
252                            context_data.append((char) (context_data_val + 32));
253                            context_data_val = 0;
254                        } else {
255                            context_data_position++;
256                        }
257                        value = 0;
258                    }
259                    value = context_w[0];
260                    for (i = 0; i < 16; i++) {
261                        context_data_val = (context_data_val << 1) | (value & 1);
262                        if (context_data_position == 14) {
263                            context_data_position = 0;
264                            context_data.append((char) (context_data_val + 32));
265                            context_data_val = 0;
266                        } else {
267                            context_data_position++;
268                        }
269                        value >>= 1;
270                    }
271                }
272
273                context_dictionaryToCreate.remove(context_w);
274            } else {
275                tmp = context_dictionary.get(context_w);
276                if(tmp == null)
277                    value = -1;
278                else {
279                    value = tmp[tmp.length - 2] << 16 | tmp[tmp.length - 1];
280                }
281                for (i = 0; i < context_numBits; i++) {
282                    context_data_val = (context_data_val << 1) | (value & 1);
283                    if (context_data_position == 14) {
284                        context_data_position = 0;
285                        context_data.append((char) (context_data_val + 32));
286                        context_data_val = 0;
287                    } else {
288                        context_data_position++;
289                    }
290                    value >>= 1;
291                }
292
293            }
294        }
295
296        // Mark the end of the stream
297        value = 2;
298        for (i = 0; i < context_numBits; i++) {
299            context_data_val = (context_data_val << 1) | (value & 1);
300            if (context_data_position == 14) {
301                context_data_position = 0;
302                context_data.append((char) (context_data_val + 32));
303                context_data_val = 0;
304            } else {
305                context_data_position++;
306            }
307            value >>= 1;
308        }
309
310        // Flush the last char
311        while (true) {
312            context_data_val = (context_data_val << 1);
313            if (context_data_position == 14) {
314                context_data.append((char) (context_data_val + 32));
315                break;
316            } else
317                context_data_position++;
318        }
319        if(keys != null && keys.length > 0)
320        {
321            int mainLength = context_data.length(), smallLength = mainLength >> 2, rem = mainLength & 3;
322            char[] raw = new char[mainLength];
323            context_data.getChars(0, mainLength, raw, 0);
324            for (int j = 0; j < keys.length; j++) {
325                long z = keys[j], inc = 0x9E3779BAL * z + 0x632BE5ABL, r;
326                for (int k = 0, n = 0; k < smallLength; k++) {
327                    r = splitMix64(z += inc);
328                    raw[n] ^= (r & 0x7fff);
329                    raw[n++] += 32;
330                    raw[n] ^= (r >>> 16 & 0x7fff);
331                    raw[n++] += 32;
332                    raw[n] ^= (r >>> 32 & 0x7fff);
333                    raw[n++] += 32;
334                    raw[n] ^= (r >>> 49);
335                    raw[n++] += 32;
336                }
337                z += inc;
338                switch(rem)
339                {
340                    case 3:
341                        raw[mainLength-3] ^= splitMix64(z) >>> 49;
342                    case 2:
343                        raw[mainLength-2] ^= splitMix64(z) >>> 16 & 0x7fff;
344                    case 1:
345                        raw[mainLength-1] ^= splitMix64(z) & 0x7fff;
346                }
347            }
348            return String.valueOf(raw);
349        }
350        else
351            return context_data.toString();
352    }
353    /**
354     * Decompresses text that was compressed with LZ-String compression; does not reverse decryption so it can only
355     * decompress Strings produced by {@link #compress(String)}, or {@link #compress(String, long[])} with an empty or
356     * null keys parameter.
357     * @param compressed text that was compressed by {@link #compress(String)}
358     * @return the original text, decompressed from the given text
359     */
360
361    public static String decompress(String compressed) {
362        return decompress(compressed, null);
363    }
364
365    /**
366     * Decompresses text that was compressed with LZ-String compression, reversing any encryption if the keys long array
367     * matches the long array passed to {@link #compress(String, long[])} (keys can be null if no array was passed).
368     * @param compressed text that was compressed by {@link #compress(String, long[])}
369     * @param keys the long array that was used to encrypt the output, and must match to decrypt the result; may be null
370     * @return the original text, decompressed and decrypted from compressed
371     */
372    public static String decompress(String compressed, long[] keys) {
373        if (compressed == null)
374            return null;
375        if (compressed.isEmpty())
376            return "";
377        final char[] comp = compressed.toCharArray();
378        int length = comp.length;
379        if(keys != null && keys.length > 0)
380        {
381            int smallLength = length >> 2, rem = length & 3;
382            for (int j = keys.length - 1; j >= 0; j--) {
383                long z = keys[j], inc = 0x9E3779BAL * z + 0x632BE5ABL, r;
384                z += (1+smallLength) * inc;
385                switch (rem)
386                {
387                    case 3:
388                        comp[length-3] ^= splitMix64(z) >>> 49;
389                    case 2:
390                        comp[length-2] ^= splitMix64(z) >>> 16 & 0x7fff;
391                    case 1:
392                        comp[length-1] ^= splitMix64(z) & 0x7fff;
393                }
394                for (int k = 0, n = (length & -4) - 1; k < smallLength ; k++) {
395                    r = splitMix64(z -= inc);
396                    comp[n] -= 32;
397                    comp[n--] ^= (r >>> 49);
398                    comp[n] -= 32;
399                    comp[n--] ^= (r >>> 32 & 0x7fff);
400                    comp[n] -= 32;
401                    comp[n--] ^= (r >>> 16 & 0x7fff);
402                    comp[n] -= 32;
403                    comp[n--] ^= (r & 0x7fff);
404
405                }
406            }
407        }
408
409
410        ArrayList<String> dictionary = new ArrayList<>();
411        int enlargeIn = 4, dictSize = 4, numBits = 3, position = 16384, index = 1, resb, maxpower, power;
412        String entry, w, c;
413        ArrayList<String> result = new ArrayList<>();
414        char bits, val = (char) (comp[0] - 32);
415
416        for (char i = 0; i < 3; i++) {
417            dictionary.add(i, String.valueOf(i));
418        }
419
420        bits = 0;
421        maxpower = 2;
422        power = 0;
423        while (power != maxpower) {
424            resb = val & position;
425            position >>= 1;
426            if (position == 0) {
427                position = 16384;
428                val = (char) (comp[index++] - 32);
429            }
430            bits |= (resb > 0 ? 1 : 0) << power++;
431        }
432
433        switch (bits) {
434            case 0:
435                bits = 0;
436                maxpower = 8;
437                power = 0;
438                while (power != maxpower) {
439                    resb = val & position;
440                    position >>= 1;
441                    if (position == 0) {
442                        position = 16384;
443                        val = (char) (comp[index++] - 32);
444                    }
445                    bits |= (resb > 0 ? 1 : 0) << power++;
446                }
447                c = String.valueOf(bits);
448                break;
449            case 1:
450                bits = 0;
451                maxpower = 16;
452                power = 0;
453                while (power != maxpower) {
454                    resb = val & position;
455                    position >>= 1;
456                    if (position == 0) {
457                        position = 16384;
458                        val = (char) (comp[index++] - 32);
459                    }
460                    bits |= (resb > 0 ? 1 : 0) << power++;
461                }
462                c = String.valueOf(bits);
463                break;
464            default:
465                return "";
466        }
467        dictionary.add(c);
468        w = c;
469        result.add(w);
470        while (true) {
471            if (index > length) {
472                return "";
473            }
474
475            int cc = 0;
476            maxpower = numBits;
477            power = 0;
478            while (power != maxpower) {
479                resb = val & position;
480                position >>= 1;
481                if (position == 0) {
482                    position = 16384;
483                    val = (char) (comp[index++] - 32);
484                }
485                cc |= (resb > 0 ? 1 : 0) << power++;
486            }
487            
488            switch (cc) {
489                case 0:
490                    bits = 0;
491                    maxpower = 8;
492                    power = 0;
493                    while (power != maxpower) {
494                        resb = val & position;
495                        position >>= 1;
496                        if (position == 0) {
497                            position = 16384;
498                            val = (char) (comp[index++] - 32);
499                        }
500                        bits |= (resb > 0 ? 1 : 0) << power++;
501                    }
502
503                    dictionary.add(String.valueOf(bits));
504                    cc = dictSize++;
505                    enlargeIn--;
506                    break;
507                case 1:
508                    bits = 0;
509                    maxpower = 16;
510                    power = 0;
511                    while (power != maxpower) {
512                        resb = val & position;
513                        position >>= 1;
514                        if (position == 0) {
515                            position = 16384;
516                            val = (char) (comp[index++] - 32);
517                        }
518                        bits |= (resb > 0 ? 1 : 0) << power++;
519                    }
520                    dictionary.add(String.valueOf(bits));
521                    cc = dictSize++;
522                    enlargeIn--;
523                    break;
524                case 2:
525                    StringBuilder sb = new StringBuilder(result.size());
526                    for (int i = 0; i < result.size(); i++) {
527                        sb.append(result.get(i));
528                    }
529                    return sb.toString();
530            }
531
532            if (enlargeIn == 0) {
533                enlargeIn = 1 << numBits;
534                numBits++;
535            }
536
537            if (cc < dictionary.size() && dictionary.get(cc) != null) {
538                entry = dictionary.get(cc);
539            } else {
540                if (cc == dictSize) {
541                    entry = w + w.charAt(0);
542                } else {
543                    return "";
544                }
545            }
546            result.add(entry);
547
548            // Add w+entry[0] to the dictionary.
549            dictionary.add(w + entry.charAt(0));
550            dictSize++;
551            enlargeIn--;
552
553            w = entry;
554
555            if (enlargeIn == 0) {
556                enlargeIn = 1 << numBits;
557                numBits++;
558            }
559
560        }
561
562    }

564    /**
565     * Get a long from this with {@code splitMix64(z += 0x9E3779B97F4A7C15L)}, where z is a long to use as state.
566     * 0x9E3779B97F4A7C15L can be changed for any odd long if the same number is used across calls.
567     * @param state long, must be changed with each call; {@code splitMix64(z += 0x9E3779B97F4A7C15L)} is recommended
568     * @return a pseudo-random long
569     */
570    private static long splitMix64(long state) {
571        state = ((state >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
572        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
573        return state ^ (state >>> 31);
574    }
