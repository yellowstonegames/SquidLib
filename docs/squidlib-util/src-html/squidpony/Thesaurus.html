066    /**
067     * Allows this Thesaurus to find the exact words in synonyms and, when requested, replace each occurrence with a
068     * different word from the same Collection. Each word in synonyms should have the same part of speech, so "demon"
069     * and "devils" should not be in the same list of synonyms (singular noun and plural noun), but "demon" and "devil"
070     * could be (each is a singular noun). The Strings in synonyms should all be lower-case, since case is picked up
071     * from the text as it is being replaced and not from the words themselves. Proper nouns should normally not be used
072     * as synonyms, since this could get very confusing if it changed occurrences of "Germany" to "France" at random and
073     * a character's name, like "Dorothy", to "Anne", "Emily", "Cynthia", etc. in the middle of a section about Dorothy.
074     * The word matching pattern this uses only matches all-letter words, not words that contain hyphens, apostrophes,
075     * or other punctuation.
076     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
077     * @return this for chaining
078     */
079    public Thesaurus addSynonyms(Collection<String> synonyms)
080    {
081        if(synonyms.isEmpty())
082            return this;
083        long prevState = rng.getState();
084        rng.setState(CrossHash.hash64(synonyms));
085        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng);
086        for(String syn : synonyms)
087        {
088            mappings.put(syn, shuffler);
089        }
090        rng.setState(prevState);
091        return this;
092    }

094    /**
095     * Allows this Thesaurus to replace a specific keyword, typically containing multiple backtick characters ('`') so
096     * it can't be confused with a "real word," with one of the words in synonyms (chosen in shuffled order). The
097     * backtick is the only punctuation character that this class' word matcher considers part of a word, both for this
098     * reason and because it is rarely used in English text.
099     * @param keyword a word (typically containing backticks, '`') that will be replaced by a word from synonyms
100     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
101     * @return this for chaining
102     */
103    public Thesaurus addCategory(String keyword, Collection<String> synonyms)
104    {
105        if(synonyms.isEmpty())
106            return this;
107        long prevState = rng.getState();
108        rng.setState(CrossHash.hash64(synonyms));
109        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng);
110        mappings.put(keyword, shuffler);
111        rng.setState(prevState);
112        return this;
113    }

248    /**
249     * Given a String, StringBuilder, or other CharSequence that should contain words this knows synonyms for, this
250     * replaces each occurrence of such a known word with one of its synonyms, leaving unknown words untouched. Words
251     * that were learned together as synonyms with addSynonyms() will be replaced in such a way that an individual
252     * replacement word should not occur too close to a previous occurrence of the same word; that is, replacing the
253     * text "You fiend! You demon! You despoiler of creation; devil made flesh!", where "fiend", "demon", and "devil"
254     * are all synonyms, would never produce a string that contained "fiend" as the replacement for all three of those.
255     * @param text a CharSequence, such as a String, that contains words in the source language
256     * @return a String of the translated text.
257     */
258    public String process(CharSequence text)
259    {
260        Replacer rep = wordMatch.replacer(new SynonymSubstitution());
261        return rep.replace(text);
262    }

264    public String lookup(String word)
265    {
266        if(word.isEmpty())
267            return word;
268        String word2 = word.toLowerCase();
269        if(mappings.containsKey(word2))
270        {
271            String nx = mappings.get(word2).next();
272            if(nx.isEmpty())
273                return nx;
274            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
275                return nx.toUpperCase();
276            if(Category.Lu.contains(word.charAt(0)))
277            {
278                return Character.toUpperCase(nx.charAt(0)) + nx.substring(1, nx.length());
279            }
280            return nx;
281        }
282        else if(languages.containsKey(word2))
283        {
284            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
285                return languages.get(word2).word(rng, false, rng.between(2, 4)).toUpperCase();
286            if(Category.Lu.contains(word.charAt(0)))
287            {
288                return languages.get(word2).word(rng, true, rng.between(2, 4));
289            }
290            return languages.get(word2).word(rng, false, rng.between(2, 4));
291
292        }
293        return word;
294    }

304    private class RandomLanguageSubstitution implements Substitution
305    {
306        @Override
307        public void appendSubstitution(MatchResult match, TextBuffer dest) {
308            FakeLanguageGen lang = FakeLanguageGen.randomLanguage(rng.nextLong());
309            randomLanguages.add(lang);
310            if(match.isCaptured(1))
311            {
312                lang = FakeLanguageGen.randomLanguage(rng.nextLong());
313                randomLanguages.add(lang);
314                do {
315                    latestGenerated = randomLanguages.get(0).word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)))
316                            + "-" + randomLanguages.get(1).word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)));
317                }while (latestGenerated.length() <= 5 || latestGenerated.length() >= 17);
318                dest.append(latestGenerated);
319            }
320            else
321            {
322                do{
323                    latestGenerated = lang.word(rng, true, Math.min(rng.between(2, 5), rng.between(1, 5)));
324                }while (latestGenerated.length() <= 2 || latestGenerated.length() >= 11);
325                dest.append(latestGenerated);
326            }
327        }
328    }

356    /**
357     * Generates a random possible name for a nation, such as "Iond-Gouccief Alliance" or "The Last Drayo Commonwealth".
358     * Needs {@link #addKnownCategories()} to be called on this Thesaurus first. May use accented characters, as in
359     * "Thùdshù-Hyóttiálb Hegemony" or "The Glorious Chô Empire"; if you want to strip these out and replace accented
360     * chars with their un-accented counterparts, you can use {@link FakeLanguageGen#removeAccents(CharSequence)}, which
361     * returns a CharSequence that can be converted to String if needed. Shortly after calling this method, but before
362     * calling it again, you can retrieve the generated random languages, if any were used while making nation names, by
363     * getting the FakeLanguageGen elements of this class' {@link #randomLanguages} field. Using one of these
364     * FakeLanguageGen objects, you can produce many more words with a similar style to the nation name, like "Drayo" in
365     * "The Last Drayo Commonwealth". If more than one language was used in the nation name, as in "Thùdshù-Hyóttiálb
366     * Hegemony", you will have two languages in randomLanguages, so here "Thùdshù" would be generated by the first
367     * language, and "Hyóttiálb" by the second language. Calling this method replaces the current contents of
368     * randomLanguages, so if you want to use those languages, get them while you can. This also assigns the
369     * {@link #latestGenerated} field to contain the part of the nation name without any larger titles; in the case of
370     * "The Glorious Chô Empire", the latestGenerated field would be assigned "Chô" at the same time the longer name
371     * would be returned. This field will be reassigned if this method is called again.
372     *
373     * @return a random name for a nation or a loose equivalent to a nation, as a String
374     */
375    public String makeNationName()
376    {
377        String working = process(rng.getRandomElement(nationTerms));
378        int frustration = 0;
379        while (frustration++ < 8 && similarFinder.matches(working))
380            working = process(rng.getRandomElement(nationTerms));
381        randomLanguages.clear();
382        RandomLanguageSubstitution sub = new RandomLanguageSubstitution();
383        Replacer replacer = Pattern.compile("@(-@)?").replacer(sub);
384        return replacer.replace(working);
385    }

390     * Empire", if the given language can produce them; if you want to strip these out and replace accented chars
391     * with their un-accented counterparts, you can use {@link FakeLanguageGen#removeAccents(CharSequence)}, which
392     * returns a CharSequence that can be converted to String if needed, or simply get an accent-less language by

395     * "The Glorious Chô Empire", the latestGenerated field would be assigned "Chô" at the same time the longer name

408        String working = process(rng.getRandomElement(nationTerms));
