001package squidpony.squidmath;
002
003import squidpony.ArrayTools;
004import squidpony.annotation.Beta;
005
006import java.util.Iterator;
007import java.util.SortedSet;
008
009/**
010 * An ordered bidirectional map-like data structure, with unique A keys and unique B keys updated together like a map
011 * that can be queried by A keys, B keys, or int indices. Does not implement any interfaces that you would expect for a
012 * data structure, because almost every method it has needs to specify whether it applies to A or B items, but you can
013 * get collections that implement SortedSet of its A or B keys.
014 * <br>
015 * Called K2 because it has 2 key sets; other collections can have other keys or have values, like K2V1.
016 * Created by Tommy Ettinger on 10/25/2016.
017 */
018@Beta
019public class K2<A, B>
020{
021    public Arrangement<A> keysA;
022    public Arrangement<B> keysB;
023
024    /**
025     * Constructs an empty K2.
026     */
027    public K2()
028    {
029        this(Arrangement.DEFAULT_INITIAL_SIZE, Arrangement.DEFAULT_LOAD_FACTOR);
030    }
031
032    /**
033     * Constructs a K2 with the expected number of indices to hold (the number of A and number of B items is always
034     * the same, and this will be more efficient if expected is greater than that number).
035     * @param expected
036     */
037    public K2(int expected)
038    {
039        this(expected, Arrangement.DEFAULT_LOAD_FACTOR);
040    }
041
042    /**
043     * Constructs a K2 with the expected number of indices to hold (the number of A and number of B items is always
044     * the same, and this will be more efficient if expected is greater than that number) and the load factor to use,
045     * between 0.1f and 0.8f usually (using load factors higher than 0.8f can cause problems).
046     * @param expected the amount of indices (the count of A items is the same as the count of B items) this should hold
047     * @param f the load factor, probably between 0.1f and 0.8f
048     */
049    public K2(int expected, float f)
050    {
051        keysA = new Arrangement<>(expected, f);
052        keysB = new Arrangement<>(expected, f);
053    }
054
055    /**
056     * Constructs a K2 with the expected number of indices to hold (the number of A and number of B items are always
057     * equal, and this will be more efficient if expected is greater than that number), the load factor to use,
058     * between 0.1f and 0.8f usually (using load factors higher than 0.8f can cause problems), and two IHasher
059     * implementations, such as {@link CrossHash#generalHasher}, that will be used to hash and compare for equality with
060     * A keys and B keys, respectively. Specifying an IHasher is usually needed if your keys are arrays (there are
061     * existing implementations for 1D arrays of all primitive types, CharSequence, and Object in CrossHash), or if you
062     * want hashing by identity and reference equality (which would use {@link CrossHash#identityHasher}, and might be
063     * useful if keys are mutable). Other options are possible with custom IHashers, like hashing Strings but ignoring,
064     * or only considering, a certain character for the hash and equality checks.
065     * @param expected the amount of indices (the count of A items is the same as the count of B items) this should hold
066     * @param f the load factor, probably between 0.1f and 0.8f
067     * @param hasherA an implementation of CrossHash.IHasher meant for A keys
068     * @param hasherB an implementation of CrossHash.IHasher meant for B keys
069     */
070    public K2(int expected, float f, CrossHash.IHasher hasherA, CrossHash.IHasher hasherB)
071    {
072        keysA = new Arrangement<>(expected, f, hasherA);
073        keysB = new Arrangement<>(expected, f, hasherB);
074    }
075
076    /**
077     * Constructs a K2 from a pair of Iterables that will be processed in pairs, adding a unique A from aKeys if and
078     * only if it can also add a unique B from bKeys, otherwise skipping that pair.
079     * @param aKeys an Iterable of A that should all be unique
080     * @param bKeys an Iterable of B that should all be unique
081     */
082    public K2(Iterable<? extends A> aKeys, Iterable<? extends B> bKeys)
083    {
084        keysA = new Arrangement<>();
085        keysB = new Arrangement<>();
086        if(aKeys != null && bKeys != null)
087            putAll(aKeys, bKeys);
088    }
089
090    public K2(K2<? extends A, ? extends B> other)
091    {
092        if(other == null)
093        {
094            keysA = new Arrangement<>();
095            keysB = new Arrangement<>();
096        }
097        else
098        {
099            keysA = new Arrangement<>(other.keysA);
100            keysB = new Arrangement<>(other.keysB);
101        }
102    }
103
104    public K2(Arrangement<A> aItems, Arrangement<B> bItems)
105    {
106        if(aItems == null || bItems == null)
107        {
108            keysA = new Arrangement<>();
109            keysB = new Arrangement<>();
110        }
111        else
112        {
113            int amt = Math.min(aItems.size, bItems.size);
114            keysA = aItems.take(amt);
115            keysB = bItems.take(amt);
116        }
117    }
118    /**
119     * Returns true if this contains the A, key, in its collection of A items.
120     * @param key the A to check the presence of
121     * @return true if key is present in this; false otherwise
122     */
123    public boolean containsA(A key) { return keysA.containsKey(key); }
124    /**
125     * Returns true if this contains the B, key, in its collection of B items.
126     * @param key the B to check the presence of
127     * @return true if key is present in this; false otherwise
128     */
129    public boolean containsB(B key) { return keysB.containsKey(key); }
130
131    /**
132     * Returns true if index is between 0 (inclusive) and {@link #size()} (exclusive), or false otherwise.
133     * @param index the index to check
134     * @return true if index is a valid index in the ordering of this K2
135     */
136    public boolean containsIndex(int index) { return keysA.containsValue(index); }

138    /**
139     * Given an A object key, finds the position in the ordering that A has, or -1 if key is not present.
140     * Unlike {@link java.util.List#indexOf(Object)}, this runs in constant time.
141     * @param key the A to find the position of
142     * @return the int index of key in the ordering, or -1 if it is not present
143     */
144    public int indexOfA(Object key)
145    {
146        return keysA.getInt(key);
147    }
148    /**
149     * Given a B object key, finds the position in the ordering that B has, or -1 if key is not present.
150     * Unlike {@link java.util.List#indexOf(Object)}, this runs in constant time.
151     * @param key the B to find the position of
152     * @return the int index of key in the ordering, or -1 if it is not present
153     */
154    public int indexOfB(Object key)
155    {
156        return keysB.getInt(key);
157    }
158
159    /**
160     * Given an int index, finds the associated A key (using index as a point in the ordering).
161     * @param index an int index into this K2
162     * @return the A object with index for its position in the ordering, or null if index was invalid
163     */
164    public A getAAt(int index)
165    {
166        return keysA.keyAt(index);
167    }
168    /**
169     * Given an int index, finds the associated B key (using index as a point in the ordering).
170     * @param index an int index into this K2
171     * @return the B object with index for its position in the ordering, or null if index was invalid
172     */
173    public B getBAt(int index)
174    {
175        return keysB.keyAt(index);
176    }

178    /**
179     * Given an A object, finds the associated B object (it will be at the same point in the ordering).
180     * @param key an A object to use as a key
181     * @return the B object associated with key, or null if key was not present
182     */
183    public B getBFromA(Object key)
184    {
185        return keysB.keyAt(keysA.getInt(key));
186    }
187
188    /**
189     * Given a B object, finds the associated A object (it will be at the same point in the ordering).
190     * @param key a B object to use as a key
191     * @return the A object associated with key, or null if key was not present
192     */
193    public A getAFromB(Object key)
194    {
195        return keysA.keyAt(keysB.getInt(key));
196    }

199     * Gets a random A from this K2 using the given IRNG.
200     * @param random generates a random index to get an A with
201     * @return a randomly chosen A, or null if this is empty
202     */
203    public A randomA(IRNG random)
204    {
205        return keysA.randomKey(random);
206    }

208    /**
209     * Gets a random B from this K2 using the given IRNG.
210     * @param random generates a random index to get a B with
211     * @return a randomly chosen B, or null if this is empty
212     */
213    public B randomB(IRNG random)
214    {
215        return keysB.randomKey(random);
216    }
217    /**
218     * Changes an existing A key, {@code past}, to another A key, {@code future}, if past exists in this K2
219     * and future does not yet exist in this K2. This will retain past's point in the ordering for future, so
220     * the associated other key(s) will still be associated in the same way.
221     * @param past an A key, that must exist in this K2's A keys, and will be changed
222     * @param future an A key, that cannot currently exist in this K2's A keys, but will if this succeeds
223     * @return this for chaining
224     */
225    public K2<A, B> alterA(A past, A future)
226    {
227        if(keysA.containsKey(past) && !keysA.containsKey(future))
228            keysA.alter(past, future);
229        return this;
230    }
231
232    /**
233     * Changes an existing B key, {@code past}, to another B key, {@code future}, if past exists in this K2
234     * and future does not yet exist in this K2. This will retain past's point in the ordering for future, so
235     * the associated other key(s) will still be associated in the same way.
236     * @param past a B key, that must exist in this K2's B keys, and will be changed
237     * @param future a B key, that cannot currently exist in this K2's B keys, but will if this succeeds
238     * @return this for chaining
239     */
240    public K2<A, B> alterB(B past, B future)
241    {
242        if(keysB.containsKey(past) && !keysB.containsKey(future))
243            keysB.alter(past, future);
244        return this;
245    }
246
247    /**
248     * Changes the A key at {@code index} to another A key, {@code future}, if index is valid and future does not
249     * yet exist in this K2. The position in the ordering for future will be the same as index, and the same
250     * as the key this replaced, if this succeeds, so the other key(s) at that position will still be associated in
251     * the same way.
252     * @param index a position in the ordering to change; must be at least 0 and less than {@link #size()}
253     * @param future an A key, that cannot currently exist in this K2's A keys, but will if this succeeds
254     * @return this for chaining
255     */
256    public K2<A, B> alterAAt(int index, A future)
257    {
258        if(!keysA.containsKey(future) && index >= 0 && index < keysA.size)
259            keysA.alter(keysA.keyAt(index), future);
260        return this;
261    }
262
263
264    /**
265     * Changes the B key at {@code index} to another B key, {@code future}, if index is valid and future does not
266     * yet exist in this K2. The position in the ordering for future will be the same as index, and the same
267     * as the key this replaced, if this succeeds, so the other key(s) at that position will still be associated in
268     * the same way.
269     * @param index a position in the ordering to change; must be at least 0 and less than {@link #size()}
270     * @param future a B key, that cannot currently exist in this K2's B keys, but will if this succeeds
271     * @return this for chaining
272     */
273    public K2<A, B> alterBAt(int index, B future)
274    {
275        if(!keysB.containsKey(future) && index >= 0 && index < keysB.size)
276            keysB.alter(keysB.keyAt(index), future);
277        return this;
278    }
279    /**
280     * Adds an A key and a B key at the same point in the ordering (the end) to this K2. Neither parameter can be
281     * present in this collection before this is called. If you want to change or update an existing key, use
282     * {@link #alterA(Object, Object)} or {@link #alterB(Object, Object)}.
283     * @param a an A key to add; cannot already be present
284     * @param b a B key to add; cannot already be present
285     * @return true if this collection changed as a result of this call
286     */
287    public boolean put(A a, B b)
288    {
289        if(!keysA.containsKey(a) && !keysB.containsKey(b))
290        {
291            keysA.add(a);
292            keysB.add(b);
293            return true;
294        }
295        return false;
296    }
297
298    /**
299     * Puts all unique A and B keys in {@code aKeys} and {@code bKeys} into this K2 at the end. If an A in aKeys or a B
300     * in bKeys is already present when this would add one, this will not put the A and B keys at that point in the
301     * iteration order, and will place the next unique A and B it finds in the arguments at that position instead.
302     * @param aKeys an Iterable or Collection of A keys to add; should all be unique (like a Set)
303     * @param bKeys an Iterable or Collection of B keys to add; should all be unique (like a Set)
304     * @return true if this collection changed as a result of this call
305     */
306    public boolean putAll(Iterable<? extends A> aKeys, Iterable<? extends B> bKeys)
307    {
308        if(aKeys == null || bKeys == null) return false;
309        Iterator<? extends A> aIt = aKeys.iterator();
310        Iterator<? extends B> bIt = bKeys.iterator();
311        boolean changed = false;
312        while (aIt.hasNext() && bIt.hasNext())
313        {
314            changed = put(aIt.next(), bIt.next()) || changed;
315        }
316        return changed;
317    }
318    /**
319     * Puts all unique A and B keys in {@code other} into this K2, respecting other's ordering. If an A or a B in other
320     * is already present when this would add one, this will not put the A and B keys at that point in the iteration
321     * order, and will place the next unique A and B it finds in the arguments at that position instead.
322     * @param other another K2 collection with the same A and B types
323     * @return true if this collection changed as a result of this call
324     */
325    public boolean putAll(K2<? extends A, ? extends B> other)
326    {
327        if(other == null) return false;
328        boolean changed = false;
329        int sz = other.size();
330        for (int i = 0; i < sz; i++) {
331            changed = put(other.getAAt(i), other.getBAt(i)) || changed;
332        }
333        return changed;
334    }
335
336    /**
337     * Adds an A key and a B key at the given index in the ordering to this K2. Neither a nor b can be
338     * present in this collection before this is called. If you want to change or update an existing key, use
339     * {@link #alterA(Object, Object)} or {@link #alterB(Object, Object)}. The index this is given should be at
340     * least 0 and no greater than {@link #size()}.
341     * @param index the point in the ordering to place a and b into; later entries will be shifted forward
342     * @param a an A key to add; cannot already be present
343     * @param b a B key to add; cannot already be present
344     * @return true if this collection changed as a result of this call
345     */
346    public boolean putAt(int index, A a, B b)
347    {
348        if(!keysA.containsKey(a) && !keysB.containsKey(b))
349        {
350            keysA.addAt(index, a);
351            keysB.addAt(index, b);
352            return true;
353        }
354        return false;
355    }

357    /**
358     * Removes a given A key, if {@code removing} exists in this K2's A keys, and also removes any keys
359     * associated with its point in the ordering.
360     * @param removing the A key to remove
361     * @return this for chaining
362     */
363    public K2<A, B> removeA(A removing)
364    {
365        keysB.removeAt(keysA.removeInt(removing));
366        return this;
367    }
368    /**
369     * Removes a given B key, if {@code removing} exists in this K2's B keys, and also removes any keys

371     * @param removing the B key to remove
372     * @return this for chaining
373     */
374    public K2<A, B> removeB(B removing)
375    {
376        keysA.removeAt(keysB.removeInt(removing));
377        return this;
378    }
379    /**
380     * Removes a given point in the ordering, if {@code index} is at least 0 and less than {@link #size()}.
381     * @param index the position in the ordering to remove
382     * @return this for chaining
383     */
384    public K2<A, B> removeAt(int index)
385    {
386        keysA.removeAt(index);
387        keysB.removeAt(index);
388        return this;
389    }

391    /**
392     * Reorders this K2 using {@code ordering}, which have the same length as this K2's {@link #size()}
393     * and can be generated with {@link ArrayTools#range(int)} (which, if applied, would produce no
394     * change to the current ordering), {@link IRNG#randomOrdering(int)} (which gives a random ordering, and if
395     * applied immediately would be the same as calling {@link #shuffle(IRNG)}), or made in some other way. If you
396     * already have an ordering and want to make a different ordering that can undo the change, you can use
397     * {@link ArrayTools#invertOrdering(int[])} called on the original ordering.
398     * @param ordering an int array or vararg that should contain each int from 0 to {@link #size()} (or less)
399     * @return this for chaining
400     */
401    public K2<A, B> reorder(int... ordering)
402    {
403        keysA.reorder(ordering);
404        keysB.reorder(ordering);
405        return this;
406    }
407
408    /**
409     * Generates a random ordering with rng and applies the same ordering to all kinds of keys this has; they will
410     * maintain their current association to other keys but their ordering/indices will change.
411     * @param rng an IRNG to produce the random ordering this will use
412     * @return this for chaining
413     */
414    public K2<A, B> shuffle(IRNG rng)
415    {
416        int[] ordering = rng.randomOrdering(keysA.size);
417        keysA.reorder(ordering);
418        keysB.reorder(ordering);
419        return this;
420    }
