024/** A resizable, ordered or unordered variable-length int array. Avoids boxing that occurs with ArrayList of Integer.
025 * If unordered, this class avoids a memory copy when removing elements (the last element is moved to the removed
026 * element's position).
027 * <br>
028 * Was called IntArray in libGDX; to avoid confusion with the fixed-length primitive array type, VLA (variable-length
029 * array) was chosen as a different name.
030 * Copied from LibGDX by Tommy Ettinger on 10/1/2015.
031 * @author Nathan Sweet */
032public class IntVLA implements Serializable, Cloneable {
033    private static final long serialVersionUID = -2948161891082748626L;
034
035    public int[] items;
036    public int size;
037
038    /** Creates an ordered array with a capacity of 16. */
039    public IntVLA() {
040        this(16);
041    }
042
043    /** Creates an ordered array with the specified capacity. */
044    public IntVLA(int capacity) {
045        items = new int[capacity];
046    }
047
048    /** Creates a new array containing the elements in the specific array. The new array will be ordered if the specific array is
049     * ordered. The capacity is set to the number of elements, so any subsequent elements added will cause the backing array to be
050     * grown. */
051    public IntVLA(IntVLA array) {
052        size = array.size;
053        items = new int[size];
054        System.arraycopy(array.items, 0, items, 0, size);
055    }
056
057    /** Creates a new ordered array containing the elements in the specified array. The capacity is set to the number of elements,
058     * so any subsequent elements added will cause the backing array to be grown. */
059    public IntVLA(int[] array) {
060        this(array, 0, array.length);
061    }
062
063    /** Creates a new array containing the elements in the specified array. The capacity is set to the number of elements, so any
064     * subsequent elements added will cause the backing array to be grown.
065     * @param array the int array to copy from
066     * @param startIndex the first index in array to copy from
067     * @param count the number of ints to copy from array into this IntVLA
068     */
069    public IntVLA(int[] array, int startIndex, int count) {
070        this(count);
071        size = count;
072        System.arraycopy(array, startIndex, items, 0, count);
073    }

075    public void add (int value) {
076        int[] items = this.items;
077        if (size == items.length) items = resize(size << 1 | 8);
078        items[size++] = value;
079    }

081    public void addAll (IntVLA array) {
082        addAll(array, 0, array.size);
083    }
084
085    public void addAll (IntVLA array, int offset, int length) {
086        if (offset + length > array.size)
087            throw new IllegalArgumentException("offset + length must be <= size: " + offset + " + " + length + " <= " + array.size);
088        addAll(array.items, offset, length);
089    }
090
091    public void addAll (int... array) {
092        addAll(array, 0, array.length);
093    }
094
095    public void addAll (int[] array, int offset, int length) {
096        int[] items = this.items;
097        int sizeNeeded = size + length;
098        if (sizeNeeded > items.length) items = resize(sizeNeeded << 1 | 8);
099        System.arraycopy(array, offset, items, size, length);
100        size += length;
101    }
102
103    public void addRange (int start, int end) {
104        int[] items = this.items;
105        int sizeNeeded = size + end - start;
106        if (sizeNeeded > items.length) items = resize(sizeNeeded << 1 | 8);
107        for(int r = start, i = size; r < end; r++, i++)
108        {
109            items[i] = r;
110        }
111        size += end - start;
112    }
113
114    public void addFractionRange (int start, int end, int fraction) {
115        int[] items = this.items;
116        int sizeNeeded = size + (end - start) / fraction + 2;
117        if (sizeNeeded > items.length) items = resize(sizeNeeded << 1 | 8);
118        for(int r = start, i = size; r < end; r = fraction * ((r / fraction) + 1), i++, size++)
119        {
120            items[i] = r;
121        }
122    }

124    public int get (int index) {
125        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
126        return items[index];
127    }

129    public void set (int index, int value) {
130        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
131        items[index] = value;
132    }

134    /**
135     * Adds value to the item in the IntVLA at index. Calling it "add" would overlap with the collection method.
136     * @param index
137     * @param value
138     */
139    public void incr (int index, int value) {
140        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
141        items[index] += value;
142    }
143
144    public void mul (int index, int value) {
145        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
146        items[index] *= value;
147    }
148
149    public void insert (int index, int value) {
150        if (index > size) throw new IndexOutOfBoundsException("index can't be > size: " + index + " > " + size);
151        int[] items = this.items;
152        if (size == items.length) items = resize(size << 1 | 8);
153        System.arraycopy(items, index, items, index + 1, size - index);
154        size++;
155        items[index] = value;
156    }
157
158    public void swap (int first, int second) {
159        if (first >= size) throw new IndexOutOfBoundsException("first can't be >= size: " + first + " >= " + size);
160        if (second >= size) throw new IndexOutOfBoundsException("second can't be >= size: " + second + " >= " + size);
161        int[] items = this.items;
162        int firstValue = items[first];
163        items[first] = items[second];
164        items[second] = firstValue;
165    }

167    /**
168     * Given an array or varargs of replacement indices for the values of this IntVLA, reorders this so the first item
169     * in the returned version is the same as {@code get(ordering[0])} (with some care taken for negative or too-large
170     * indices), the second item in the returned version is the same as {@code get(ordering[1])}, etc.
171     * <br>
172     * Negative indices are considered reversed distances from the end of ordering, so -1 refers to the same index as
173     * {@code ordering[ordering.length - 1]}. If ordering is smaller than this IntVLA, only the indices up to the
174     * length of ordering will be modified. If ordering is larger than this IntVLA, only as many indices will be
175     * affected as this IntVLA's size, and reversed distances are measured from the end of this IntVLA instead of the
176     * end of ordering. Duplicate values in ordering will produce duplicate values in the returned IntVLA.
177     * <br>
178     * This method modifies this IntVLA in-place and also returns it for chaining.
179     *
180     * @param ordering an array or varargs of int indices, where the nth item in ordering changes the nth item in this
181     *                 IntVLA to have the value currently in this IntVLA at the index specified by the value in ordering
182     * @return this for chaining, after modifying it in-place
183     */
184    public IntVLA reorder (int... ordering) {
185        int ol;
186        if (ordering == null || (ol = Math.min(size, ordering.length)) == 0)
187            return this;
188        int[] items = this.items, alt = new int[ol];
189        for (int i = 0; i < ol; i++) {
190            alt[i] = items[(ordering[i] % ol + ol) % ol];
191        }
192        System.arraycopy(alt, 0, items, 0, ol);
193        return this;
194    }

196    public boolean contains (int value) {
197        int i = size - 1;
198        int[] items = this.items;
199        while (i >= 0)
200            if (items[i--] == value) return true;
201        return false;
202    }

204    /**
205     * Tries to find the first occurrence of {@code value} in this IntVLA, and returns the index that value appears at
206     * if it is present, or -1 if it is not present.
207     * @param value a value to search for in this
208     * @return the first index of value, if found, or -1 otherwise
209     */
210    public int indexOf (int value) {
211        int[] items = this.items;
212        for (int i = 0, n = size; i < n; i++)
213            if (items[i] == value) return i;
214        return -1;
215    }
216
217    public int lastIndexOf (int value) {
218        int[] items = this.items;
219        for (int i = size - 1; i >= 0; i--)
220            if (items[i] == value) return i;
221        return -1;
222    }
223
224    /**
225     * Removes the first occurrence of the requested value, and returns the index it was removed at (-1 if not found)
226     * @param value a value in this IntVLA to remove
227     * @return the index the value was found and removed at, or -1 if it was not present
228     */
229    public int removeValue (int value) {
230        int[] items = this.items;
231        for (int i = 0, n = size; i < n; i++) {
232            if (items[i] == value) {
233                removeIndex(i);
234                return i;
235            }
236        }
237        return -1;
238    }
239
240    /** Removes and returns the item at the specified index. */
241    public int removeIndex (int index) {
242        if (index >= size) throw new IndexOutOfBoundsException("index can't be >= size: " + index + " >= " + size);
243        int[] items = this.items;
244        int value = items[index];
245        size--;
246        System.arraycopy(items, index + 1, items, index, size - index);
247        return value;
248    }
249
250    /** Removes the items between the specified indices, inclusive. */
251    public void removeRange (int start, int end) {
252        if (end >= size) throw new IndexOutOfBoundsException("end can't be >= size: " + end + " >= " + size);
253        if (start > end) throw new IndexOutOfBoundsException("start can't be > end: " + start + " > " + end);
254        int[] items = this.items;
255        int count = end - start + 1;
256        System.arraycopy(items, start + count, items, start, size - (start + count));
257        size -= count;
258    }
259
260    /** Removes from this array all of elements contained in the specified array.
261     * @return true if this array was modified. */
262    public boolean removeAll (IntVLA array) {
263        int size = this.size;
264        int startSize = size;
265        int[] items = this.items;
266        for (int i = 0, n = array.size; i < n; i++) {
267            int item = array.get(i);
268            for (int ii = 0; ii < size; ii++) {
269                if (item == items[ii]) {
270                    removeIndex(ii);
271                    size--;
272                    break;
273                }
274            }
275        }
276        return size != startSize;
277    }

280    /** Moves the specified value to the first index and returns its previous index. If value is not present, this
281     * returns -1.
282     * @param value an int value that should already be in this IntVLA
283     * @return the previous index of value, or -1 if it was not present
284     */
285    public int moveToFirst (int value) {
286        int[] items = this.items;
287        int index = indexOf(value);
288        if(index <= 0) return index;
289        System.arraycopy(items, 0, items, 1, index);
290        items[0] = value;
291        return index;
292    }
293
294    /** Moves the specified value to the last index and returns its previous index. If value is not present, this
295     * returns -1.
296     * @param value an int value that should already be in this IntVLA
297     * @return the previous index of value, or -1 if it was not present
298     */
299    public int moveToLast (int value) {
300        int[] items = this.items;
301        int index = indexOf(value);
302        if(index == size - 1 || index == -1) return index;
303        System.arraycopy(items, index + 1, items, index, size - index - 1);
304        items[size - 1] = value;
305        return index;
306    }

328    /** Reduces the size of the backing array to the size of the actual items. This is useful to release memory when many items have
329     * been removed, or if it is known that more items will not be added.
330     * @return {@link #items} */
331    public int[] shrink () {
332        if (items.length != size) resize(size);
333        return items;
334    }
335
336    /** Increases the size of the backing array to accommodate the specified number of additional items. Useful before adding many
337     * items to avoid multiple backing array resizes.
338     * @return {@link #items} */
339    public int[] ensureCapacity (int additionalCapacity) {
340        int sizeNeeded = size + additionalCapacity;
341        if (sizeNeeded > items.length) resize(Math.max(8, sizeNeeded));
342        return items;
343    }
344
345    /** Sets the array size, leaving any values beyond the current size undefined.
346     * @return {@link #items} */
347    public int[] setSize (int newSize) {
348        if (newSize > items.length) resize(Math.max(8, newSize));
349        size = newSize;
350        return items;
351    }
352
353    protected int[] resize (int newSize) {
354        int[] newItems = new int[newSize];
355        int[] items = this.items;
356        System.arraycopy(items, 0, newItems, 0, Math.min(size, newItems.length));
357        this.items = newItems;
358        return newItems;
359    }
360
361    public void sort () {
362        Arrays.sort(items, 0, size);
363    }
364
365    public void reverse () {
366        int[] items = this.items;
367        for (int i = 0, lastIndex = size - 1, n = size / 2; i < n; i++) {
368            int ii = lastIndex - i;
369            int temp = items[i];
370            items[i] = items[ii];
371            items[ii] = temp;
372        }
373    }

375    /** Reduces the size of the array to the specified size. If the array is already smaller than the specified size, no action is
376     * taken. */
377    public void truncate (int newSize) {
378        if (size > newSize) size = newSize;
379    }

386    /**
387     * Shuffles this IntVLA in place using the given IRNG.
388     * @param random an IRNG, such as an RNG, used to generate the shuffled order
389     * @return this object, modified, after shuffling
390     */
391    public IntVLA shuffle(IRNG random)
392    {
393        int n = size;
394        for (int i = 0; i < n; i++)
395        {
396            swap(i + random.nextInt(n - i), i);
397        }
398        return this;
399    }

425    @Override
426    public int hashCode () {
427        int[] items = this.items;
428        int h = 1;
429        for (int i = 0, n = size; i < n; i++)
430            h = h * 31 + items[i];
431        return h;
432    }
433    public int hashWisp () {
434        int[] data = this.items;
435        long result = 0x9E3779B97F4A7C94L, a = 0x632BE59BD9B4E019L;
436        final int len = size;
437        for (int i = 0; i < len; i++) {
438            result += (a ^= 0x8329C6EB9E6AD3E3L * data[i]);
439        }
440        return (int)(result * (a | 1L) ^ (result >>> 27 | result << 37));
441    }

454    @Override
455        public boolean equals (Object object) {
456        if (object == this) return true;
457        if (!(object instanceof IntVLA)) return false;
458        IntVLA array = (IntVLA)object;
459        int n = size;
460        if (n != array.size) return false;

462            if (items[i] != array.items[i]) return false;

466    @Override
467        public String toString () {
468        if (size == 0) return "[]";
469        int[] items = this.items;
470        StringBuilder buffer = new StringBuilder(32);
471        buffer.append('[');
472        buffer.append(items[0]);
473        for (int i = 1; i < size; i++) {
474            buffer.append(", ");
475            buffer.append(items[i]);
476        }
477        buffer.append(']');
478        return buffer.toString();

493    public static IntVLA deserializeFromString(String data)
494    {
495        int amount = StringKit.count(data, ",");
496        if (amount <= 0) return new IntVLA();
497        IntVLA iv = new IntVLA(amount+1);
498        int dl = 1, idx = -dl, idx2;
499        for (int i = 0; i < amount; i++) {
500            iv.add(StringKit.intFromDec(data, idx+dl, idx = data.indexOf(",", idx+dl)));
501        }
502        if((idx2 = data.indexOf(",", idx+dl)) < 0)
503        {
504            iv.add(StringKit.intFromDec(data, idx+dl, data.length()));
505        }
506        else
507        {
508            iv.add(StringKit.intFromDec(data, idx+dl, idx2));
509        }
510        return iv;
511    }
