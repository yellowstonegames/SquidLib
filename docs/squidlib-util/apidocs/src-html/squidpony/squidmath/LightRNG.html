015/**
016 * This is a SplittableRandom-style generator, meant to have a tiny state
017 * that permits storing many different generators with low overhead.
018 * It should be rather fast, though no guarantees can be made on all hardware.
019 * <br>
020 * Benchmarking on a Windows laptop with an i7-4700MQ processor running OpenJDK 8
021 * reports generation of 64-bit random long output as 17.8x faster than generating
022 * an equivalent number of random longs with java.util.Random, and generation of
023 * 32-bit random int output as 9.8x faster. Specifically, generating 1 billion longs
024 * took about 1.28 nanoseconds per long (1.277 seconds for the whole group) with
025 * LightRNG, while java.util.Random (which is meant to produce int, to be fair) took
026 * about 22.8 nanoseconds per long (22.797 seconds for the whole group). XorRNG
027 * appears to be occasionally faster on int output than LightRNG, but it isn't clear
028 * why or what causes that (JIT or GC internals, possibly). XorRNG is slightly
029 * slower at generating 64-bit random data, including long and double, but not by
030 * a significant degree (a multiplier between 0.9 and 1.2 times). The only deciding
031 * factor then is state size, where LightRNG is as small as possible for any JVM
032 * object with even a single field: 16 bytes (on a 64-bit JVM; 8-byte objects with
033 * 4 bytes or less of non-static members may be possible on 32-bit JVMs but I can't
034 * find anything confirming that guess).
035 * <br>
036 * So yes, this should be very fast, and with only a single long used per LightRNG,
037 * it is about as memory-efficient as these generators get.
038 * <br>
039 * Written in 2015 by Sebastiano Vigna (vigna@acm.org)
040 * @author Sebastiano Vigna
041 * @author Tommy Ettinger
042 */
043public final class LightRNG implements RandomnessSource, StatefulRandomness, SkippingRandomness, Serializable
044{

068    /**
069     * Gets a pseudo-random int with at most the specified count of bits; for example, if bits is 3 this can return any
070     * int between 0 and 2 to the 3 (that is, 8), exclusive on the upper end. That would mean 7 could be returned, but
071     * no higher ints, and 0 could be returned, but no lower ints.
072     * 
073     * The algorithm used here changed on March 8, 2018 when LightRNG was remade the default generator in SquidLib.
074     * The older method is available as {@link #compatibleNext(int)}, but its use is discouraged; it's slightly slower 
075     * for no good reason.
076     * @param bits the number of bits to be returned; if 0 or less, or if 32 or greater, can return any 32-bit int
077     * @return a pseudo-random int that uses at most the specified amount of bits
078     */
079    @Override
080    public final int next( int bits ) {
081        long z = state += 0x9E3779B97F4A7C15L;
082        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
083        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
084        return (int)(z ^ (z >>> 31)) >>> (32 - bits);
085    }
086    /**
087     * Gets a pseudo-random int with at most the specified count of bits; for example, if bits is 3 this can return any
088     * int between 0 and 2 to the 3 (that is, 8), exclusive on the upper end. That would mean 7 could be returned, but
089     * no higher ints, and 0 could be returned, but no lower ints.
090     *
091     * The algorithm used here is the older version of {@link #next(int)} before some things changed on March 8 2018.
092     * Using this method is discouraged unless you need to reproduce values exactly; it's slightly slower for no good
093     * reason. Calling {@code next(32)} and {@code compatibleNext(32)} should have identical results, but other values
094     * for bits will probably be different.
095     * @param bits the number of bits to be returned; if 0 or less, or if 32 or greater, can return any 32-bit int
096     * @return a pseudo-random int that uses at most the specified amount of bits
097     */
098    public final int compatibleNext( int bits ) {
099        long z = state += 0x9E3779B97F4A7C15L;
100        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
101        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
102        return (int)( (z ^ (z >>> 31)) & ( 1L << bits ) - 1 );
103    }

139    /**
140     * Returns a random non-negative integer between 0 (inclusive) and the given bound (exclusive),
141     * or 0 if the bound is 0. The bound can be negative, which will produce 0 or a negative result.
142     * Uses an aggressively optimized technique that has some bias, but mostly for values of
143     * bound over 1 billion. This method uses the same technique as {@link RNG#nextIntHasty(int)},
144     * and like that method will always advance state exactly once (equivalent to one call to
145     * {@link #nextLong()}).
146     * <br>
147     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
148     *
149     * @param bound the outer bound (exclusive), can be negative or positive
150     * @return the found number
151     */
152    public int nextInt( final int bound ) {
153        long z = state += 0x9E3779B97F4A7C15L;
154        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
155        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
156        return (int) ((bound * ((z ^ (z >>> 31)) & 0x7FFFFFFFL)) >> 31);
157    }

159    /**
160     * Like {@link #compatibleNext(int)}, but for compatibility with {@link #nextInt(int)}.
161     * Exclusive on the upper bound.  The lower bound is 0.
162     * @param bound the upper bound; should be positive
163     * @return a random int less than n and at least equal to 0
164     */
165    public int compatibleNextInt( final int bound ) {
166        if (bound <= 0) return 0;
167        int threshold = (0x7fffffff - bound + 1) % bound;
168        for (; ; ) {
169            int bits = (int) (nextLong() & 0x7fffffff);
170            if (bits >= threshold)
171                return bits % bound;
172        }
173    }
174    /**
175     * Inclusive lower, exclusive upper. Although you should usually use a higher value for upper than for lower, you
176     * can use a lower "upper" than "lower" and this will still work, producing an int between the two bounds.
177     * @param lower the lower bound, inclusive, can be positive or negative
178     * @param upper the upper bound, exclusive, can be positive or negative, usually should be greater than lower
179     * @return a random int between lower (inclusive) and upper (exclusive)
180     */
181    public int nextInt( final int lower, final int upper ) {
182        return lower + nextInt(upper - lower);
183    }

185    /**
186     * Inclusive lower, exclusive upper.
187     * @param lower the lower bound, inclusive, can be positive or negative
188     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
189     * @return a random int between lower (inclusive) and upper (exclusive)
190     */
191    public int compatibleNextInt( final int lower, final int upper ) {
192        if ( upper - lower <= 0 ) throw new IllegalArgumentException("Upper bound must be greater than lower bound");
193        return lower + compatibleNextInt(upper - lower);
194    }

196    /**
197     * Exclusive on the outer bound; the inner bound is 0. The bound may be negative, which will produce a non-positive
198     * result.
199     * @param bound the outer exclusive bound; may be positive or negative
200     * @return a random long between 0 (inclusive) and bound (exclusive)
201     */
202    public long nextLong(long bound) {
203        long rand = nextLong();
204        final long randLow = rand & 0xFFFFFFFFL;
205        final long boundLow = bound & 0xFFFFFFFFL;
206        rand >>>= 32;
207        bound >>= 32;
208        final long z = (randLow * boundLow >> 32);
209        long t = rand * boundLow + z;
210        final long tLow = t & 0xFFFFFFFFL;
211        t >>>= 32;
212        return rand * bound + t + (tLow + randLow * bound >> 32) - (z >> 63) - (bound >> 63);
213    }

215    /**
216     * Inclusive inner, exclusive outer; both inner and outer can be positive or negative.
217     * @param inner the inner bound, inclusive, can be positive or negative
218     * @param outer the outer bound, exclusive, can be positive or negative and may be greater than or less than inner
219     * @return a random long that may be equal to inner and will otherwise be between inner and outer
220     */
221    public long nextLong(final long inner, final long outer) {
222        return inner + nextLong(outer - inner);
223    }

226    /**
227     * Exclusive on the upper bound. The lower bound is 0. Unlike {@link #nextInt(int)} or {@link #nextLong(long)}, this
228     * may sometimes advance the state more than once, depending on what numbers are produced internally and the bound.
229     * {@link #nextLong(long)} is preferred because it is much faster and reliably advances the state only once. Because
230     * this method uses rejection sampling, getting multiple random longs to "smooth the odds" when the bound is such
231     * that it can't fairly distribute one random long across all possible outcomes, it may be more "fair" than
232     * {@link #nextLong(long)}, though it could potentially consume more of the period faster if pathologically bad
233     * bounds were used very often, and if enough of the period is gone then statistical flaws may become detectable.
234     * @param bound the upper bound; if this isn't positive, this method always returns 0
235     * @return a random long less than n and at least equal to 0
236     */
237    public long compatibleNextLong(final long bound) {
238        if ( bound <= 0 ) return 0;
239        long threshold = (0x7fffffffffffffffL - bound + 1) % bound;
240        for (;;) {
241            long bits = nextLong() & 0x7fffffffffffffffL;
242            if (bits >= threshold)
243                return bits % bound;
244        }
245    }

250     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
251     * @return a random long at least equal to lower and less than upper

254        if ( upper - lower <= 0 )  throw new IllegalArgumentException("Upper bound must be greater than lower bound");

299    /**
300     * Given a byte array as a parameter, this will fill the array with random bytes (modifying it
301     * in-place). Calls nextLong() {@code Math.ceil(bytes.length / 8.0)} times.
302     * @param bytes a byte array that will have its contents overwritten with random bytes.
303     */
304    public void nextBytes( final byte[] bytes ) {
305        int i = bytes.length, n;
306        while( i != 0 ) {
307            n = Math.min( i, 8 );
308            for ( long bits = nextLong(); n-- != 0; bits >>>= 8 ) bytes[ --i ] = (byte)bits;
309        }
310    }

338    /**
339     * Advances or rolls back the LightRNG's state without actually generating each number. Skips forward
340     * or backward a number of steps specified by advance, where a step is equal to one call to nextLong(),
341     * and returns the random number produced at that step (you can get the state with {@link #getState()}).
342     *
343     * @param advance Number of future generations to skip over; can be negative to backtrack, 0 gets the most recent generated number
344     * @return the random long generated after skipping advance numbers
345     */
346    @Override
347    public long skip(long advance) {
348        long z = (state += 0x9E3779B97F4A7C15L * advance);
349        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
350        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
351        return z ^ (z >>> 31);
352    }

383        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
384        return state ^ (state >>> 31);

393     * Returns a random float that is deterministic based on state; if state is the same on two calls to this, this will
394     * return the same float. This is expected to be called with a changing variable, e.g. {@code determine(++state)},
395     * where the increment for state should be odd but otherwise doesn't really matter. This multiplies state by
396     * {@code 0x9E3779B97F4A7C15L} within this method, so using a small increment won't be much different from using a

398     * only 2 to the 30 possible floats between 0 and 1.
399     * @param state a variable that should be different every time you want a different random result;

403     */
