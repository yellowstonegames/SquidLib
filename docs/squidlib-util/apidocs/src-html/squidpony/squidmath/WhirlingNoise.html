010/**
011 * Another experimental noise class. Extends PerlinNoise and should have similar quality, but can be faster and has less
012 * periodic results. This is still considered experimental because the exact output may change in future versions, along
013 * with the scale (potentially) and the parameters it takes. In general, {@link #noise(double, double)} and
014 * {@link #noise(double, double, double)} should have similar appearance to {@link PerlinNoise#noise(double, double)},
015 * {@link PerlinNoise#noise(double, double, double)}, and {@link PerlinNoise#noise(double, double, double, double)}, but
016 * are not forced to a zoomed-in scale like PerlinNoise makes its results, have a lower chance of forming shapes with
017 * multiple octaves that appear "non-natural" (such as long straight lines, axis-aligned shapes, and hard angles), and
018 * the 3D and 4D noise variants tend to be faster due to a more efficient method for hashing points. WhirlingNoise is
019 * preferred to other related {@link Noise.Noise3D} or {@link Noise.Noise4D} implementations for most world map
020 * generation (usually using {@link squidpony.squidgrid.mapping.WorldMapGenerator}) because it is less likely to produce
021 * artificial-seeming shapes, which are out-of-place in a world map.
022 * <br>
023 * Created by Tommy Ettinger on 12/14/2016. The technique for point hashing here is based closely on
024 * <a href="http://citeseerx.ist.psu.edu/viewdoc/download?doi=10.1.1.140.3594&rep=rep1&type=pdf">this paper</a>,
025 * with credit to Andrew Kensler, Aaron Knoll and Peter Shirley.
026 */

482    /**
483     * 2D simplex noise returning a float; extremely similar to {@link #noise(double, double)}, but this may be slightly
484     * faster or slightly slower. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result
485     * will be different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method.
486     * This also cannot take a seed, while {@link #noise(double, double, long)} can.
487     * @param x x input; works well if between 0.0 and 1.0, but anything is accepted
488     * @param y y input; works well if between 0.0 and 1.0, but anything is accepted
489     * @return noise from -1.0 to 1.0, inclusive
490     */
491    public static float noiseAlt(double x, double y) {
492        //xin *= SCALE;
493        //yin *= SCALE;
494        float noise0, noise1, noise2; // from the three corners
495        float xin = (float)x, yin = (float)y;
496        // Skew the input space to figure out which simplex cell we're in
497        float skew = (xin + yin) * FastNoise.F2f; // Hairy factor for 2D
498        int i = fastFloor(xin + skew);
499        int j = fastFloor(yin + skew);
500        float t = (i + j) * FastNoise.G2f;
501        float X0 = i - t; // Unskew the cell origin back to (x,y) space
502        float Y0 = j - t;
503        float x0 = xin - X0; // The x,y distances from the cell origin
504        float y0 = yin - Y0;
505        // For the 2D case, the simplex shape is an equilateral triangle.
506        // determine which simplex we are in.
507        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
508        // coords
509        if (x0 > y0) {
510            i1 = 1;
511            j1 = 0;
512        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
513        else {
514            i1 = 0;
515            j1 = 1;
516        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
517        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
518        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
519        // where: c = (3-sqrt(3))/6
520        float x1 = x0 - i1 + FastNoise.G2f; // Offsets for middle corner in (x,y)
521        // unskewed coords
522        float y1 = y0 - j1 + FastNoise.G2f;
523        float x2 = x0 - 1f + FastNoise.H2f; // Offsets for last corner in (x,y)
524        // unskewed coords
525        float y2 = y0 - 1f + FastNoise.H2f;
526        // Work out the hashed gradient indices of the three simplex corners
527//        int gi0 = determine256(i + determine(j));
528//        int gi1 = determine256(i + i1 + determine(j + j1));
529//        int gi2 = determine256(i + 1 + determine(j + 1));
530        final int gi0 = perm_x[i & 255] ^ perm_y[j & 255];
531        final int gi1 = perm_x[i + i1 & 255] ^ perm_y[j + j1 & 255];
532        final int gi2 = perm_x[i + 1 & 255] ^ perm_y[j + 1 & 255];
533
534        // Calculate the contribution from the three corners
535        float t0 = 0.75f - x0 * x0 - y0 * y0;
536        if (t0 < 0) {
537            noise0 = 0.0f;
538        } else {
539            t0 *= t0;
540            noise0 = t0 * t0 * FastNoise.dotf(FastNoise.phiGrad2f[gi0], x0, y0);
541            // for 2D gradient
542        }
543        float t1 = 0.75f - x1 * x1 - y1 * y1;
544        if (t1 < 0) {
545            noise1 = 0.0f;
546        } else {
547            t1 *= t1;
548            noise1 = t1 * t1 * FastNoise.dotf(FastNoise.phiGrad2f[gi1], x1, y1);
549        }
550        float t2 = 0.75f - x2 * x2 - y2 * y2;
551        if (t2 < 0) {
552            noise2 = 0.0f;
553        } else {
554            t2 *= t2;
555            noise2 = t2 * t2 * FastNoise.dotf(FastNoise.phiGrad2f[gi2], x2, y2);
556        }
557        // Add contributions from each corner to get the final noise value.
558        // The result is scaled to return values in the interval [-1,1].
559        return 9.125f * (noise0 + noise1 + noise2);
560    }

720            n1 = 0f;
721        } else {
722            t1 *= t1;
723            n1 = t1 * t1 * dotf(grad3f[gi1], x1, y1, z1);
724        }
725        float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
726        if (t2 < 0) {
727            n2 = 0f;
728        } else {
729            t2 *= t2;
730            n2 = t2 * t2 * dotf(grad3f[gi2], x2, y2, z2);
731        }
732        float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
733        if (t3 < 0) {
734            n3 = 0f;
735        } else {
736            t3 *= t3;
737            n3 = t3 * t3 * dotf(grad3f[gi3], x3, y3, z3);
738        }
739        // Add contributions from each corner to get the final noise value.
740        // The result is scaled to stay just inside [-1,1]
741        return  31.5f * (n0 + n1 + n2 + n3);
742    }
