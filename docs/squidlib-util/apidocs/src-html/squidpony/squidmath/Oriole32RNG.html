001package squidpony.squidmath;
002
003import squidpony.StringKit;
004
005import java.io.Serializable;
006
007/**
008 * A modification of Blackman and Vigna's xoroshiro128+ generator using two 32-bit ints of state instead of two 64-bit
009 * longs and also incorporating a large-increment counter (Weyl sequence) that is added to the rotated xoroshiro output;
010 * this is tied with {@link Lathe32RNG} for the fastest generator on GWT I have found that also passes the full 32TB
011 * battery of PractRand's statistical tests. ThrustAlt32RNG is faster in some browsers and also passes tests, but
012 * cannot produce all outputs, and its period is too small for serious usage. In statistical testing, xoroshiro always
013 * fails some binary matrix rank tests, but smaller-state versions fail other tests as well. The changes Oriole makes
014 * apply only to the output of xoroshiro, not its well-tested state transition for the "xoroshiro state" part of this
015 * generator, and these changes eliminate all statistical failures on 32TB of tested data, avoiding the failures the
016 * small-state variant of xoroshiro suffers on BinaryRank, BCFN, DC6, and FPF. It avoids multiplication, like xoroshiro
017 * and much of the xorshift family of generators, and any arithmetic it performs is safe for GWT. Oriole takes advantage
018 * of xoroshiro's flaws being mostly confined to its output's lower bits by rotating the output of xoroshiro (the weaker
019 * 32-bit-state version) and adding a "Weyl sequence," essentially a large-increment counter that overflows and wraps
020 * frequently, to the output. Although the upper bits of xoroshiro are not free of artifacts either, they are harder to
021 * find issues with (see
022 * <a href="http://www.pcg-random.org/posts/xoroshiro-fails-truncated.html">this article by PCG-Random's author</a> for
023 * more detail). It is unclear if the changes made here would improve the larger-state version, but they probably would
024 * help to some extent with at least the binary rank failures. The period is also improved by incorporating the Weyl
025 * sequence, up to 0xFFFFFFFFFFFFFFFF00000000 .
026 * <br>
027 * The name comes from the insectivorous orange songbird called an oriole, which as I have found from the one that
028 * visits my backyard, reacts quickly and is always looking for bugs to remove. It also sounds like "xoro."
029 * <br>
030 * <a href="http://xoroshiro.di.unimi.it/xoroshiro128plus.c">Original version here for xorshiro128+</a>; this version
031 * uses <a href="https://groups.google.com/d/msg/prng/Ll-KDIbpO8k/bfHK4FlUCwAJ">different constants</a> by the same
032 * author, Sebastiano Vigna.
033 * <br>
034 * Written in 2016 by David Blackman and Sebastiano Vigna (vigna@acm.org)
035 * Ported and modified in 2018 by Tommy Ettinger
036 * @author Sebastiano Vigna
037 * @author David Blackman
038 * @author Tommy Ettinger (if there's a flaw, use SquidLib's or Sarong's issues and don't bother Vigna or Blackman, it's probably a mistake in SquidLib's implementation)
039 * @see Lathe32RNG A related generator that implements StatefulRandomness and has very similar speed
040 */
041public final class Oriole32RNG implements RandomnessSource, Serializable {
042
043    private static final long serialVersionUID = 1L;
044
045    private int stateA, stateB, stateC;
046
047    /**
048     * Creates a new generator seeded using three calls to Math.random().
049     */
050    public Oriole32RNG() {
051        setState((int)((Math.random() * 2.0 - 1.0) * 0x80000000), (int)((Math.random() * 2.0 - 1.0) * 0x80000000),
052                (int)((Math.random() * 2.0 - 1.0) * 0x80000000));
053    }
054    /**
055     * Constructs this Oriole32RNG by dispersing the bits of seed using {@link #setSeed(int)} across the two parts of state
056     * this has.
057     * @param seed a long that won't be used exactly, but will affect both components of state
058     */
059    public Oriole32RNG(final int seed) {
060        setSeed(seed);
061    }
062    /**
063     * Constructs this Oriole32RNG by calling {@link #setState(int, int, int)} on stateA and stateB as given but
064     * producing stateC via {@code stateA ^ stateB}; see that method for the specific details (stateA and stateB are
065     * kept as-is unless they are both 0).
066     * @param stateA the number to use as the first part of the state; this will be 1 instead if both seeds are 0
067     * @param stateB the number to use as the second part of the state
068     */
069    public Oriole32RNG(final int stateA, final int stateB) {
070        setState(stateA, stateB, stateA ^ stateB);
071    }
072
073    /**
074     * Constructs this Oriole32RNG by calling {@link #setState(int, int, int)} on the arguments as given; see that
075     * method for the specific details (stateA and stateB are kept as-is unless they are both 0).
076     * @param stateA the number to use as the first part of the state; this will be 1 instead if both seeds are 0
077     * @param stateB the number to use as the second part of the state
078     * @param stateC the number to use as the counter part of the state (third part)
079     */
080    public Oriole32RNG(final int stateA, final int stateB, final int stateC) {
081        setState(stateA, stateB, stateC);
082    }
083
084    @Override
085    public final int next(int bits) {
086        final int s0 = stateA;
087        int s1 = stateB;
088        final int result = s0 + s1;
089        s1 ^= s0;
090        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
091        stateB = (s1 << 28 | s1 >>> 4); // c
092        return (result << 29 | result >>> 3) + (stateC += 0x632BE5AB) >>> (32 - bits);
093    }
094
095    /**
096     * Can return any int, positive or negative, of any size permissible in a 32-bit signed integer.
097     * @return any int, all 32 bits are random
098     */
099    public final int nextInt() {
100        final int s0 = stateA;
101        int s1 = stateB;
102        final int result = s0 + s1;
103        s1 ^= s0;
104        stateA = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
105        stateB = (s1 << 28 | s1 >>> 4); // c
106        return (result << 29 | result >>> 3) + (stateC += 0x632BE5AB);
107    }
108
109    @Override
110    public final long nextLong() {
111        final int s0 = stateA;
112        int s1 = stateB;
113        final int high = s0 + s1;
114        s1 ^= s0;
115        final int s00 = (s0 << 13 | s0 >>> 19) ^ s1 ^ (s1 << 5); // a, b
116        s1 = (s1 << 28 | s1 >>> 4); // c
117        final int low = s00 + s1;
118        s1 ^= s00;
119        stateA = (s00 << 13 | s00 >>> 19) ^ s1 ^ (s1 << 5); // a, b
120        stateB = (s1 << 28 | s1 >>> 4); // c
121        return (long)((high << 29 | high >>> 3) + (stateC + 0x632BE5AB)) << 32 ^ ((low << 29 | low >>> 3) + (stateC += 0xC657CB56));
122    }
123
124    /**
125     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
126     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
127     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
128     *
129     * @return a copy of this RandomnessSource
130     */
131    @Override
132    public Oriole32RNG copy() {
133        return new Oriole32RNG(stateA, stateB, stateC);
134    }
135
136    /**
137     * Sets the state of this generator using one int, running it through Zog32RNG's algorithm three times to get 
138     * three ints.
139     * @param seed the int to use to assign this generator's state
140     */
141    public void setSeed(final int seed) {
142        int z = seed + 0xC74EAD55, a = seed ^ z;
143        a ^= a >>> 14;
144        z = (z ^ z >>> 10) * 0xA5CB3;
145        a ^= a >>> 15;
146        stateA = (z ^ z >>> 20) + (a ^= a << 13);
147        z = seed + 0x8E9D5AAA;
148        a ^= a >>> 14;
149        z = (z ^ z >>> 10) * 0xA5CB3;
150        a ^= a >>> 15;
151        stateB = (z ^ z >>> 20) + (a ^ a << 13);
152        if((stateA | stateB) == 0)
153            stateA = 1;
154        z = seed - 0xC74EAD55;
155        a ^= a >>> 14;
156        z = (z ^ z >>> 10) * 0xA5CB3;
157        a ^= a >>> 15;
158        stateC = (z ^ z >>> 20) + (a ^ a << 13);
159    }
160
161    public int getStateA()
162    {
163        return stateA;
164    }
165    public void setStateA(int stateA)
166    {
167        this.stateA = (stateA | stateB) == 0 ? 1 : stateA;
168    }
169    public int getStateB()
170    {
171        return stateB;
172    }
173    public void setStateB(int stateB)
174    {
175        this.stateB = stateB;
176        if((stateB | stateA) == 0) stateA = 1;
177    }
178    public int getStateC()
179    {
180        return stateC;
181    }
182    public void setStateC(int stateC)
183    {
184        this.stateC = stateC;
185    }
186
187    /**
188     * Sets the current internal state of this Oriole32RNG with three ints, where stateA and stateB can each be any int
189     * unless they are both 0, and stateC can be any int without restrictions.
190     * @param stateA any int except 0 (0 will be treated as 1 instead)
191     * @param stateB any int
192     */
193    public void setState(int stateA, int stateB, int stateC)
194    {
195        this.stateA = (stateA | stateB) == 0 ? 1 : stateA;
196        this.stateB = stateB;
197        this.stateC = stateC;
198    }
199    @Override
200    public String toString() {
201        return "Oriole32RNG with stateA 0x" + StringKit.hex(stateA) + ", stateB 0x" + StringKit.hex(stateB) + ", and stateC 0x" + StringKit.hex(stateC);
202    }
203
204    @Override
205    public boolean equals(Object o) {
206        if (this == o) return true;
207        if (o == null || getClass() != o.getClass()) return false;
208
209        Oriole32RNG oriole32RNG = (Oriole32RNG) o;
210
211        return stateA == oriole32RNG.stateA && stateB == oriole32RNG.stateB && stateC == oriole32RNG.stateC;
212    }
213
214    @Override
215    public int hashCode() {
216        return 31 * stateA + stateB;
217    }
218}
