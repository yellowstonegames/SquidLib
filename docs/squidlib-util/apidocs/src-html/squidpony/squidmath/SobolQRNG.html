030/**
031 * Implementation of a Sobol sequence as a Quasi-Random Number Generator.
032 * <p>
033 * A Sobol sequence is a low-discrepancy sequence with the property that for all values of N,
034 * its subsequence (x1, ... xN) has a low discrepancy. It can be used to generate quasi-random
035 * points in a space S, which are equi-distributed. This is not a true random number generator,
036 * and is not even a pseudo-random number generator; the sequence generated from identical
037 * starting points with identical dimensions will be exactly the same. Calling this class'
038 * nextVector, nextIntVector, and nextLongVector methods all increment the position in the
039 * sequence, and do not use separate sequences for separate types.
040 * <p>
041 * The implementation already comes with support for up to 1000 dimensions with direction numbers
042 * calculated from <a href="http://web.maths.unsw.edu.au/~fkuo/sobol/">Stephen Joe and Frances Kuo</a>.
043 * <p>
044 * The generator supports two modes:
045 * <ul>
046 *   <li>sequential generation of points: {@link #nextVector()}, {@link #nextIntVector()},
047 *   {@link #nextLongVector()}, and the bounded variants on each of those</li>
048 *   <li>random access to the i-th point in the sequence: {@link #skipTo(int)}</li>
049 * </ul>
050 *
051 * For reference,
052 * <a href="http://en.wikipedia.org/wiki/Sobol_sequence">Sobol sequence (Wikipedia)</a>
053 * and the data, <a href="http://web.maths.unsw.edu.au/~fkuo/sobol/">Sobol sequence direction numbers</a>
054 *
055 * Created by Tommy Ettinger on 5/2/2015 based off Apache Commons Math 4.
056 */
057public class SobolQRNG implements RandomnessSource {
058

110    /**
111     * Construct a new Sobol sequence generator for the given space dimension.
112     * You should call {@link #skipTo(int)} with a fairly large number (over 1000) to ensure the results aren't
113     * too obviously non-random. If you skipTo(1), all doubles in that result will be 0.5, and if you skipTo(0),
114     * all will be 0 (this class starts at index 1 instead of 0 for that reason). This is true for all dimensions.
115     *
116     * @param dimension the space dimension
117     * @throws ArithmeticException if the space dimension is outside the allowed range of [1, 1000]
118     */
119    public SobolQRNG(final int dimension) throws ArithmeticException {
120        if (dimension < 1 || dimension > MAX_DIMENSION_GWT) {
121            throw new ArithmeticException("Dimension " + dimension + "is outside the GWT-compatible range of" +
122                    "[1, 16]; did you want the GWT-incompatible two-argument constructor?");
123        }
124
125        this.dimension = dimension;
126
127        // init data structures
128        direction = new long[dimension][BITS + 1];
129        x = new long[dimension];
130
131        // special case: dimension 1 -> use unit initialization
132        for (int i = 1; i <= BITS; i++) {
133            direction[0][i] = 1L << (BITS - i);
134        }
135        for (int d = 0; d < dimension-1; d++) {
136            initDirectionVector(RESOURCE_PRELOAD[d]);
137        }
138
139    }

142     * Construct a new Sobol sequence generator for the given space dimension.
143     * You should call {@link #skipTo(int)} with a fairly large number (over 1000) to ensure the results aren't
144     * too obviously non-random. If you skipTo(1), all doubles in that result will be 0.5, and if you skipTo(0),
145     * all will be 0 (this class starts at index 1 instead of 0 for that reason). This is true for all dimensions.
146     *
147     * @param dimension the space dimension
148     * @param ignored not used, except to differentiate this from the GWT-compatible constructor
149     * @throws ArithmeticException if the space dimension is outside the allowed range of [1, 1000]
150     */
151    @GwtIncompatible /* Because of getResourceAsStream */
152    public SobolQRNG(final int dimension, boolean ignored) throws ArithmeticException {
153        if (dimension < 1 || dimension > MAX_DIMENSION) {
154            throw new ArithmeticException("Dimension " + dimension + "is outside the allowed range of [1, 1000]");
155        }
156
157        // initialize the other dimensions with direction numbers from a resource
158        final InputStream is = getClass().getResourceAsStream(RESOURCE_NAME);
159        if (is == null) {
160            throw new ArithmeticException("Could not load embedded resource 'new-joe-kuo-6.1000'");
161        }
162
163        this.dimension = dimension;
164
165        // init data structures
166        direction = new long[dimension][BITS + 1];
167        x = new long[dimension];
168
169        try {
170            initFromStream(is);
171        } catch (IOException e) {
172            // the internal resource file could not be read -> should not happen
173            throw new InternalError("Resource is unreadable, abort!\n" + e.getMessage());
174        } catch (ArithmeticException e) {
175            // the internal resource file could not be parsed -> should not happen
176            throw new InternalError("Resource is unreadable, abort!\n" + e.getMessage());
177        } finally {
178            try {
179                is.close();
180            } catch (IOException e) { // NOPMD
181                // ignore
182            }
183        }
184    }

194     * The columns are:

198     *  <li>a: the number representing the coefficients</li>

200     * </ul>
201     * Example:
202     * <pre>
203     * d       s       a       m_i
204     * 2       1       0       1
205     * 3       2       1       1 3
206     * </pre>
207     * <p>
208     * The input stream <i>must</i> be an ASCII text containing one valid direction vector per line.
209     *
210     * @param dimension the space dimension
211     * @param is the stream to read the direction vectors from
212     * @throws ArithmeticException if the space dimension is outside the range [1, max], where
213     *   max refers to the maximum dimension found in the input stream
214     * @throws IOException if an error occurs while reading from the input stream
215     * /
216    public SobolQRNG(final int dimension, final InputStream is)
217            throws ArithmeticException, IOException {

220            throw new ArithmeticException("The given dimension, " + dimension + ", is too low (must be greater than 0)");
221        }
222
223        this.dimension = dimension;
224
225        // init data structures
226        direction = new long[dimension][BITS + 1];
227        x = new long[dimension];
228
229        // initialize the other dimensions with direction numbers from the stream
230        int lastDimension = initFromStream(is);
231        if (lastDimension < dimension) {
232            throw new ArithmeticException("Dimension " + dimension + "is outside the allowed range of [1, 1000]");

236    /**
237     * Load the direction vector for each dimension from the given stream.
238     * <p>
239     * The input stream <i>must</i> be an ASCII text containing one
240     * valid direction vector per line.
241     *
242     * @param is the input stream to read the direction vector from
243     * @return the last dimension that has been read from the input stream
244     * @throws IOException if the stream could not be read
245     * @throws ArithmeticException if the content could not be parsed successfully
246     */
247    @GwtIncompatible
248    private int initFromStream(final InputStream is) throws ArithmeticException, IOException {
249
250        // special case: dimension 1 -> use unit initialization
251        for (int i = 1; i <= BITS; i++) {
252            direction[0][i] = 1L << (BITS - i);
253        }
254
255        final Charset charset = Charset.forName(FILE_CHARSET);
256        final BufferedReader reader = new BufferedReader(new InputStreamReader(is, charset));
257        int dim = -1;
258
259        try {
260            // ignore first line
261            reader.readLine();
262
263            int lineNumber = 2;
264            int index = 1;
265            String line;
266            while ( (line = reader.readLine()) != null) {
267                StringTokenizer st = new StringTokenizer(line, " ");
268                try {
269                    dim = Integer.parseInt(st.nextToken());
270                    if (dim >= 2 && dim <= dimension) { // we have found the right dimension
271                        final int s = Integer.parseInt(st.nextToken());
272                        final int a = Integer.parseInt(st.nextToken());
273                        final int[] m = new int[s + 1];
274                        for (int i = 1; i <= s; i++) {
275                            m[i] = Integer.parseInt(st.nextToken());
276                        }
277                        initDirectionVector(index++, a, m);
278                    }
279
280                    if (dim > dimension) {
281                        return dim;
282                    }
283                } catch (NoSuchElementException e) {
284                    throw new ArithmeticException("Resource error at line " + lineNumber + ":\n  " + line);
285                } catch (NumberFormatException e) {
286                    throw new ArithmeticException("Resource error at line " + lineNumber + ":\n  " + line);
287                }
288                lineNumber++;
289            }
290        } finally {
291            reader.close();
292        }
293
294        return dim;
295    }

297    /**
298     * Calculate the direction numbers from the given polynomial.
299     *
300     * @param d the dimension, zero-based
301     * @param a the coefficients of the primitive polynomial
302     * @param m the initial direction numbers
303     */
304    private void initDirectionVector(final int d, final int a, final int[] m) {
305        final int s = m.length - 1;
306        for (int i = 1; i <= s; i++) {
307            direction[d][i] = (long) m[i] << (BITS - i);
308        }
309        for (int i = s + 1; i <= BITS; i++) {
310            direction[d][i] = direction[d][i - s] ^ (direction[d][i - s] >> s);
311            for (int k = 1; k <= s - 1; k++) {
312                direction[d][i] ^= ((a >> (s - 1 - k)) & 1) * direction[d][i - k];
313            }
314        }
315    }
316    /**
317     * Calculate the direction numbers from the given polynomial.
318     *
319     * @param m the initial direction numbers
320     */
321    private void initDirectionVector(final int[] m) {
322        final int s = m.length - 2, d = m[0] - 1, a = m[1];
323        for (int i = 1; i <= s; i++) {
324            direction[d][i] = (long) m[i+1] << (BITS - i);
325        }
326        for (int i = s + 1; i <= BITS; i++) {
327            direction[d][i] = direction[d][i - s] ^ (direction[d][i - s] >> s);
328            for (int k = 1; k <= s - 1; k++) {
329                direction[d][i] ^= ((a >> (s - 1 - k)) & 1) * direction[d][i - k];
330            }
331        }
332    }

334    /** Generate a random vector.
335     * @return a random vector as an array of double in the range [0.0, 1.0).
336     */
337    public double[] nextVector() {
338        final double[] v = new double[dimension];
339        if (count == 0) {
340            count++;
341            return v;
342        }
343
344        // find the index c of the rightmost 0
345        int c = 1 + Integer.numberOfTrailingZeros(count);
346
347        for (int i = 0; i < dimension; i++) {
348            x[i] ^= direction[i][c];
349            v[i] = (double) x[i] / SCALE;
350        }
351        count++;
352        return v;
353    }
354    /** Generate a random vector.
355     * @return a random vector as an array of double in the range [0.0, 1.0).
356     */
357    public double[] fillVector(double[] toFill) {
358        if (count == 0 || toFill == null) {
359            count++;
360            return toFill;
361        }
362
363        // find the index c of the rightmost 0
364        int c = 1 + Integer.numberOfTrailingZeros(count);
365
366        for (int i = 0; i < dimension && i < toFill.length; i++) {
367            x[i] ^= direction[i][c];
368            toFill[i] = (double) x[i] / SCALE;
369        }
370        count++;
371        return toFill;
372    }
373    /** Generate a random vector.
374     * @return a random vector as an array of double in the range [0.0, 1.0).
375     */
376    public Coord nextCoord(int xLimit, int yLimit) {
377        if (count == 0) {
378            count++;
379            return Coord.get(0, 0);
380        }
381
382        // find the index c of the rightmost 0
383        int cx = 0, cy = 0, c = 1 + Integer.numberOfTrailingZeros(count);
384
385        if(dimension <= 0)
386            return Coord.get(0, 0);
387        x[0] ^= direction[0][c];
388        cx = (int)((x[0] >>> 20) % xLimit);
389
390        if(dimension == 1)
391        {
392            x[0] ^= direction[0][1 + Integer.numberOfTrailingZeros(++count)];
393            cy = (int) ((x[0] >>> 20) % yLimit);
394        }
395        else {
396            x[1] ^= direction[1][c];
397            cy = (int) ((x[1] >>> 20) % yLimit);
398            // ensure the next number stays on track for other dimensions
399            if(dimension > 2)
400            {
401                for (int i = 2; i < dimension; i++) {
402                    x[i] ^= direction[i][c];
403                }
404            }
405        }
406        count++;
407        return Coord.get(cx, cy);
408    }

453     * @param max the maximum exclusive value for the elements of the desired vector; minimum is 0 inclusive.
454     * @return a random vector as an array of long (only 52 bits are actually used for the result, plus sign bit).
455     */
456    public long[] nextLongVector(final long max) {
457        final long[] v = new long[dimension];
458
459        if (count == 0) {
460            count++;
461            return v;
462        }
463
464        // find the index c of the rightmost 0
465        int c = 1 + Integer.numberOfTrailingZeros(count);
466
467        for (int i = 0; i < dimension; i++) {
468            x[i] ^= direction[i][c];
469            //suboptimal, but this isn't meant for quality of randomness, actually the opposite.

471        }
472        count++;
473        return v;

520    /**
521     * Skip to the i-th point in the Sobol sequence.
522     * <p>
523     * This operation can be performed in O(1).
524     * If index is somehow negative, this uses its absolute value instead of throwing an exception.
525     * If index is 0, the result will always be entirely 0.
526     * You should skipTo a number greater than 1000 if you want random-seeming individual numbers in each vector.
527     *
528     * @param index the index in the sequence to skip to
529     * @return the i-th point in the Sobol sequence
530     */
531    public double[] skipTo(final int index) {
532        if (index == 0) {
533            // reset x vector
534            Arrays.fill(x, 0);
535        } else {
536            final int i = (index > 0) ? (index - 1) : (-index - 1);
537            final long grayCode = i ^ (i >> 1); // compute the gray code of i = i XOR floor(i / 2)
538            for (int j = 0; j < dimension; j++) {
539                long result = 0;
540                for (int k = 1; k <= BITS; k++) {
541                    final long shift = grayCode >> (k - 1);
542                    if (shift == 0) {
543                        // stop, as all remaining bits will be zero
544                        break;
545                    }
546                    // the k-th bit of i
547                    final long ik = shift & 1;
548                    result ^= ik * direction[j][k];
549                }
550                x[j] = result;
551            }
552        }
553        count = (index >= 0) ? index : -index;
554        return nextVector();
555    }

567    /**
568     *
569     * @param bits the number of bits to be returned
570     * @return a quasi-random int with at most the specified number of bits
571     */
572    @Override
573    public int next(int bits) {
574        return nextIntVector()[0] >>> (32 - bits);
575    }

577    /**
578     * Using this method, any algorithm that needs to efficiently generate more
579     * than 32 bits of random data can interface with this randomness source.
580     * <p/>
581     * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
582     *
583     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
584     */
585    @Override
586    public long nextLong() {
587        if (dimension > 1) {
588            long[] l = nextLongVector();
589            return (l[0] << 32) ^ l[1];
590        }
591        return ((long) nextIntVector()[0] << 32) ^ nextIntVector()[0];
592    }
593    public double nextDouble() {
594        return nextVector()[0];
595    }
596
597    public double nextDouble(double max) {
598        return nextVector(max)[0];
599    }
600    /**
601     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
602     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
603     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
604     * @return a copy of this RandomnessSource
605     */
606    @Override
607    public SobolQRNG copy() {
608        SobolQRNG next = new SobolQRNG(dimension);
609        next.count = count;
610        next.skipTo(count);
611        return next;
612    }

621        if (this == o) return true;

630    @Override
631    public int hashCode() {
632        int result = 31 * dimension + count | 0; // bitwise OR with 0 is a GWT thing
633        result = 31 * result + CrossHash.hash(direction) | 0;
634        return  31 * result + CrossHash.hash(x) | 0;
635    }
