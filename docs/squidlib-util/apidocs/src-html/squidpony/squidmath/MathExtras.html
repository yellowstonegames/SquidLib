018import java.math.BigInteger;

020/**
021 * Mathematical operations not provided by {@link Math java.lang.Math}.
022 * <br>
023 * Originally part of the <a href="http://maths.uncommons.org/">Uncommon Maths software package</a> as Maths.
024 * @author Daniel Dyer

026public final class MathExtras
027{
028    // The biggest factorial that can be calculated using 64-bit signed longs.
029    private static final int MAX_LONG_FACTORIAL = 20;
030
031    // Cache BigInteger factorial values because they are expensive to generate.
032    private static final int CACHE_SIZE = 256;
033    static final long[] factorialsStart = new long[]{
034            0, 1, 2, 6, 24, 120, 720, 5040, 40320, 362880, 3628800, 39916800, 479001600,
035            6227020800L, 87178291200L, 1307674368000L, 20922789888000L, 355687428096000L,
036            6402373705728000L, 121645100408832000L, 2432902008176640000L
037    };
038    private static final OrderedMap<Integer, BigInteger> BIG_FACTORIALS
039        = new OrderedMap<Integer, BigInteger>();
040
041
042    private MathExtras()
043    {
044        // Prevent instantiation.
045    }
046
047
048    /**
049     * Calculates the factorial of n where n is a number in the
050     * range 0 - 20.  Zero factorial is equal to 1.  For values of
051     * n greater than 20 you must use {@link #bigFactorial(int)}.
052     * @param n The factorial to calculate.
053     * @return The factorial of n.
054     * @see #bigFactorial(int)
055     */
056    public static long factorial(int n)
057    {
058        if (n < 0 || n > MAX_LONG_FACTORIAL)
059        {
060            throw new IllegalArgumentException("Argument must be in the range 0 - 20.");
061        }
062        /*
063        long factorial = 1;
064        for (int i = n; i > 1; i--)
065        {
066            factorial *= i;
067        }*/
068        return factorialsStart[n];
069    }
070
071
072    /**
073     * Calculates the factorial of n where n is a positive integer.
074     * Zero factorial is equal to 1.  For values of n up to 20, consider
075     * using {@link #factorial(int)} instead since it uses a faster
076     * implementation.
077     * @param n The factorial to calculate.
078     * @return The factorial of n.
079     * @see #factorial(int)
080     */
081    public static BigInteger bigFactorial(int n)
082    {
083        if (n < 0)
084        {
085            throw new IllegalArgumentException("Argument must greater than or equal to zero.");
086        }
087
088        BigInteger factorial = null;
089        if (n < CACHE_SIZE) // Check for a cached value.
090        {
091            factorial = BIG_FACTORIALS.get(n);
092        }
093
094        if (factorial == null)
095        {
096            factorial = BigInteger.ONE;
097            for (int i = n; i > 1; i--)
098            {
099                factorial = factorial.multiply(BigInteger.valueOf(i));
100            }
101            if (n < CACHE_SIZE) // Cache value.
102            {
103                if(!BIG_FACTORIALS.containsKey(n))
104                    BIG_FACTORIALS.put(n, factorial);
105            }
106        }
107
108        return factorial;
109    }

112    /**
113     * Calculate the first argument raised to the power of the second.
114     * This method only supports non-negative powers.
115     * @param value The number to be raised.
116     * @param power The exponent (must be positive).
117     * @return {@code value} raised to {@code power}.
118     */
119    public static long raiseToPower(int value, int power)
120    {
121        if (power < 0)
122        {
123            throw new IllegalArgumentException("This method does not support negative powers.");
124        }
125        long result = 1;
126        for (int i = 0; i < power; i++)
127        {
128            result *= value;
129        }
130        return result;
131    }

134    /**
135     * Calculate logarithms for arbitrary bases.
136     * @param base The base for the logarithm.
137     * @param arg The value to calculate the logarithm for.
138     * @return The log of {@code arg} in the specified {@code base}.
139     */
140    public static double log(double base, double arg)
141    {
142        // Use natural logarithms and change the base.
143        return Math.log(arg) / Math.log(base);
144    }

147    /**
148     * Checks that two values are approximately equal (plus or minus a specified tolerance).
149     * @param value1 The first value to compare.
150     * @param value2 The second value to compare.
151     * @param tolerance How much (in percentage terms, as a percentage of the first value)
152     * the values are allowed to differ and still be considered equal.  Expressed as a value
153     * between 0 and 1.
154     * @return true if the values are approximately equal, false otherwise.
155     */
156    public static boolean approxEquals(double value1,
157                                       double value2,
158                                       double tolerance)
159    {
160        if (tolerance < 0 || tolerance > 1)
161        {
162            throw new IllegalArgumentException("Tolerance must be between 0 and 1.");
163        }
164        return Math.abs(value1 - value2) <= value1 * tolerance;
165    }

168    /**
169     * If the specified value is not greater than or equal to the specified minimum and
170     * less than or equal to the specified maximum, adjust it so that it is.
171     * @param value The value to check.
172     * @param min The minimum permitted value.
173     * @param max The maximum permitted value.
174     * @return {@code value} if it is between the specified limits, {@code min} if the value
175     * is too low, or {@code max} if the value is too high.
176     */
177    public static int clamp(int value, int min, int max)
178    {
179        return Math.min(Math.max(value, min), max);
180    }

226    /**
227     * Like the modulo operator {@code %}, but the result will always match the sign of {@code d} instead of {@code op}.
228     * @param op the dividend; negative values are permitted and wrap instead of producing negative results
229     * @param d the divisor; if this is negative then the result will be negative, otherwise it will be positive
230     * @return the remainder of the division of op by d, with a sign matching d
231     */
232    public static double remainder(final double op, final double d) {
233        return (op % d + d) % d;
234    }

236    /**
237     * Determines the greatest common divisor of a pair of natural numbers
238     * using the Euclidean algorithm.  This method only works with natural
239     * numbers.  If negative integers are passed in, the absolute values will
240     * be used.  The return value is always positive.
241     * @param a The first value.
242     * @param b The second value.
243     * @return The greatest common divisor.
244     */
245    public static long greatestCommonDivisor(long a, long b)
246    {
247        a = Math.abs(a);
248        b = Math.abs(b);
249        while (b != 0)
250        {
251            long temp = b;
252            b = a % b;
253            a = temp;
254        }
255        return a;
256    }
