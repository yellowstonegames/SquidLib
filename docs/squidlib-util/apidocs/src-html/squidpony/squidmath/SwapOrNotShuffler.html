005/**
006 * Gets a sequence of distinct pseudo-random ints (typically used as indices) from 0 to some bound, without storing all
007 * of the sequence in memory. Uses a Swap-Or-Not shuffle with many rounds on a non-power-of-two domain (0 inclusive to
008 * bound exclusive), as described in <a href="https://arxiv.org/abs/1208.1176">this paper by Viet Tung Hoang, Ben
009 * Morris, and Phillip Rogaway</a>. The API is very simple; you construct a SwapOrNotShuffler by specifying how many
010 * items it can shuffle, and you can optionally use a seed (it will be random if you don't specify a seed). Call
011 * {@link #next()} on a SwapOrNotShuffler to get the next distinct int in the shuffled ordering; next() will return -1
012 * if there are no more distinct ints (if {@link #bound} items have already been returned). You can go back to the
013 * previous item with {@link #previous()}, which similarly returns -1 if it can't go earlier in the sequence. You can
014 * restart the sequence with {@link #restart()} to use the same sequence over again, or {@link #restart(int)} to use a
015 * different seed (the bound is fixed).
016 * <br>
017 * This class is extremely similar to {@link LowStorageShuffler}; both classes are optimized for usage on GWT but
018 * SwapOrNotShuffler is meant to have higher quality in general at the expense of speed. LowStorageShuffler performs
019 * better when the bound is small, though this performance fluctuates based on how much lower the bound is than the
020 * next-highest power of four. There's also {@link ShuffledIntSequence}, which extends LowStorageShuffler and uses
021 * different behavior so it "re-shuffles" the results when all results have been produced, and
022 * {@link SNShuffledIntSequence}, which extends this class but is otherwise like ShuffledIntSequence. In most cases,
023 * LowStorageShuffler is a better choice than this class, especially for large bounds.
024 * <br>
025 * Don't use this for cryptographic purposes; even though this uses a lot of rounds, it may still not be enough, the
026 * "function" is probably as insecure as it gets, and in this version there are only 2 to the 32 possible keys, which
027 * would make a brute-force attack trivial. It seems good enough for generating shuffles for domains with unusual small
028 * sizes when the purpose is game-related, and not anything too serious. While the Swap-or-Not Shuffle algorithm is
029 * capable of strong security guarantees, this implementation emphasizes speed and does not offer any hope of security
030 * against a competent attacker.
031 * <br>
032 * Created by Tommy Ettinger on 10/1/2018.
033 * @author Viet Tung Hoang, Ben Morris, and Phillip Rogaway
034 * @author Tommy Ettinger
035 */
036public class SwapOrNotShuffler implements Serializable {
037    private static final long serialVersionUID = 1L;
038    public final int bound;
039    protected final int ROUNDS;
040    protected int index;
041    protected final int[] keys;
042    protected int func;
043    /**
044     * Constructs a SwapOrNotShuffler with a random seed and a bound of 10.
045     */
046    public SwapOrNotShuffler(){
047        this(10);
048    }
049    /**
050     * Constructs a SwapOrNotShuffler with the given exclusive upper bound and a random seed.
051     * @param bound how many distinct ints this can return
052     */
053    public SwapOrNotShuffler(int bound)
054    {
055        this(bound, (int)((Math.random() - 0.5) * 0x1.0p32));
056    }
057
058    /**
059     * Constructs a SwapOrNotShuffler with the given exclusive upper bound and int seed.
060     * @param bound how many distinct ints this can return
061     * @param seed any int; will be used to get several seeds used internally
062     */
063    public SwapOrNotShuffler(int bound, int seed)
064    {
065        // initialize our state
066        this.bound = Math.max(1, bound);
067        ROUNDS = this.bound + 4;
068        keys = new int[ROUNDS];
069        restart(seed);
070    }
071
072    /**
073     * Gets the next distinct int in the sequence, or -1 if all distinct ints have been returned that are non-negative
074     * and less than {@link #bound}.
075     * @return the next item in the sequence, or -1 if the sequence has been exhausted
076     */
077    public int next()
078    {
079        if (index < bound)
080        {
081            return encode(index++);
082        }
083        // end of shuffled list if we got here.
084        return -1;
085    }
086    /**
087     * Gets the previous returned int from the sequence (as yielded by {@link #next()}), or -1 if next() has never been
088     * called (or the SwapOrNotShuffler has reached the beginning from repeated calls to this method).
089     * @return the previously-given item in the sequence, or -1 if this can't go earlier
090     */
091    public int previous()
092    {
093        if (index > 0) {
094            return encode(--index);
095        }
096        // end of shuffled list if we got here.
097        return -1;
098    }
099
100    /**
101     * Starts the same sequence over from the beginning.
102     */
103    public void restart()
104    {
105        index = 0;
106    }
107
108    /**
109     * Starts the sequence over, but can change the seed (completely changing the sequence). If {@code seed} is the same
110     * as the seed given in the constructor, this will use the same sequence, acting like {@link #restart()}.
111     * @param seed any int; will be used to get several seeds used internally
112     */
113    public void restart(int seed)
114    {
115        index = 0;
116        int z;
117        func = ~seed;
118        for (int i = 0; i < ROUNDS; i++) {
119            z = (seed += 0xC13FA9A9); // 2 to the 32 divided by the plastic constant (an irrational number)
120            // 32-bit no-multiply version of Paperweight algorithm
121            z ^= z >>> 13;
122            z = (z << 19) - z;
123            z ^= z >>> 12;
124            z = (z << 17) - z;
125            z ^= z >>> 14;
126            z = (z << 13) - z;
127            func += (keys[i] = (int)((bound * ((z ^ z >>> 15) & 0xFFFFFFFFL)) >> 32)) ^ z; // can we avoid this multiply?
128            // func may be out of range for an int on GWT; this is OK because it always has bitwise
129            // ops used on it before anything else. Bitwise ops fix the range to a valid int.
130        }
131//        z = (z << 9) + (z << 8 | z >>> 24) ^ 0x9E3779B9;
132//        z = (z << 27) + (z << 20 | z >>> 12);
133//        func = (z << 19) + (z << 14 | z >>> 18);
134    }

152    /**
153     * @param index the index to cipher; must be between 0 and {@link #bound}, inclusive
154     * @return the ciphered index, which will be less than {@link #bound}
155     */
156    public int encode(int index)
157    {
158        int f = func ^ func >>> 13;
159        for (int i = 0; i < ROUNDS; i++) { 
160            int key = keys[i] - index;
161            key += (key >> 31 & bound);
162            // (f ^ func >>> (f += i)) is a very special piece of code.
163            // it takes the current value of f and XORs it with a shifted version of func; the shift is unusual.
164            // to do the shift, it adds the current round counter i to f, and then uses the bottom 5 bits as the shift.
165            if(((f ^ func >>> (f += i)) + Math.max(index, key) & 1) == 0) index = key;
166        }
167        return index;
168    }
