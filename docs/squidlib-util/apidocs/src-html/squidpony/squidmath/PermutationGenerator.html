026/**
027 * Permutation generator for generating all permutations for all sets up to
028 * 20 elements in size.  While 20 may seem a low limit, bear in mind that
029 * the number of permutations of a set of size n is n!  For a set of 21
030 * items, the number of permutations is bigger than can be stored in Java's
031 * 64-bit long integer data type.  Therefore it seems unlikely that you
032 * could ever generate, let alone process, all of the permutations in a
033 * reasonable time frame.  For this reason the implementation is optimised for
034 * sets of size 20 or less (this affords better performance by allowing primitive
035 * numeric types to be used for calculations rather than
036 * {@link java.math.BigInteger}).
037 * <br>
038 * Originally part of the <a href="http://maths.uncommons.org/">Uncommon Maths software package</a>.
039 * @param <T> The type of element that the permutation will consist of.
040 * @author Daniel Dyer (modified from the original version written by Michael
041 * Gilleland of Merriam Park Software -
042 * <a href="http://www.merriampark.com/perm.htm">http://www.merriampark.com/perm.htm</a>).
043 * @see CombinationGenerator
044 */
045public class PermutationGenerator<T> implements Iterable<List<T>>, Serializable
046{

055    /**
056     * Permutation generator that generates all possible orderings of
057     * the elements in the specified set.
058     * @param elements The elements to permute; will be modified, so this should be copied beforehand
059     */
060    public PermutationGenerator(T[] elements)
061    {
062        if (elements.length > 20)
063        {
064            throw new IllegalArgumentException("Size must be less than or equal to 20.");
065        }
066        this.elements = elements;
067        permutationIndices = new int[elements.length];
068        totalPermutations = MathExtras.factorial(elements.length);
069        reset();
070    }

161    /**
162     * Generate the next permutation and return an array containing
163     * the elements in the appropriate order.  This overloaded method
164     * allows the caller to provide an array that will be used and returned.
165     * The purpose of this is to improve performance when iterating over
166     * permutations. This method allows a single array instance to be reused.
167     * @param destination Provides an array to use to create the
168     * permutation.  The specified array must be the same length as a
169     * permutation.  This is the array that will be returned, once
170     * it has been filled with the elements in the appropriate order.
171     * @return The next permutation as an array.
172     */
173    public T[] nextPermutationAsArray(T[] destination)
174    {
175        if (destination.length != elements.length)
176        {
177            throw new IllegalArgumentException("Destination array must be the same length as permutations.");
178        }
179        generateNextPermutationIndices();
180        // Generate actual permutation.
181        for (int i = 0; i < permutationIndices.length; i++)
182        {
183            destination[i] = elements[permutationIndices[i]];
184        }
185        return destination;
186    }

202    /**
203     * Generate the next permutation and return a list containing
204     * the elements in the appropriate order.  This overloaded method
205     * allows the caller to provide a list that will be used and returned.
206     * The purpose of this is to improve performance when iterating over
207     * permutations.  If the {@link #nextPermutationAsList()} method is
208     * used it will create a new list every time.  When iterating over
209     * permutations this will result in lots of short-lived objects that
210     * have to be garbage collected.  This method allows a single list
211     * instance to be reused in such circumstances.
212     * @param destination Provides a list to use to create the
213     * permutation.  This is the list that will be returned, once
214     * it has been filled with the elements in the appropriate order.
215     * @return The next permutation as a list.
216     */
217    public List<T> nextPermutationAsList(List<T> destination)
218    {
219        generateNextPermutationIndices();
220        // Generate actual permutation.
221        destination.clear();
222        for (int i : permutationIndices)
223        {
224            destination.add(elements[i]);
225        }
226        return destination;
227    }

229    /**
230     * Generate the indices into the elements array for the next permutation.  The
231     * algorithm is from Kenneth H. Rosen, Discrete Mathematics and its Applications,
232     * 2nd edition (NY: McGraw-Hill, 1991), p. 284)
233     */
234    private void generateNextPermutationIndices()
235    {
236        if (remainingPermutations == 0)
237        {
238            throw new IllegalStateException("There are no permutations remaining.  " +
239                                            "Generator must be reset to continue using.");
240        }
241        else if (remainingPermutations < totalPermutations)
242        {
243            // Find largest index j with permutationIndices[j] < permutationIndices[j + 1]
244            int j = permutationIndices.length - 2;
245            while (permutationIndices[j] > permutationIndices[j + 1])
246            {
247                j--;
248            }
249
250            // Find index k such that permutationIndices[k] is smallest integer greater than
251            // permutationIndices[j] to the right of permutationIndices[j].
252            int k = permutationIndices.length - 1;
253            while (permutationIndices[j] > permutationIndices[k])
254            {
255                k--;
256            }
257
258            // Interchange permutation indices.
259            int temp = permutationIndices[k];
260            permutationIndices[k] = permutationIndices[j];
261            permutationIndices[j] = temp;
262
263            // Put tail end of permutation after jth position in increasing order.
264            int r = permutationIndices.length - 1;
265            int s = j + 1;
266
267            while (r > s)
268            {
269                temp = permutationIndices[s];
270                permutationIndices[s] = permutationIndices[r];
271                permutationIndices[r] = temp;
272                r--;
273                s++;
274            }
275        }
276        --remainingPermutations;
277    }

279    private int[] getPermutationShift(T[] perm) {
280        int[] sh = new int[perm.length];
281        boolean[] taken = new boolean[perm.length];
282
283        for (int i = 0; i < perm.length - 1; i++) {
284            int ctr = -1;
285            for (int j = 0; j < perm.length; j++) {
286                if (!taken[j])
287                    ctr++;
288                if (perm[j] == elements[i]) {
289                    taken[j] = true;
290                    sh[i] = ctr;
291                    break;
292                }
293            }
294        }
295        return sh;
296    }

298    private int[] getPermutationShift(List<T> perm) {
299        int length = perm.size();
300        int[] sh = new int[length];
301        boolean[] taken = new boolean[length];
302
303        for (int i = 0; i < length - 1; i++) {
304            int ctr = -1;
305            for (int j = 0; j < length; j++) {
306                if (!taken[j])
307                    ctr++;
308                if (perm.get(j) == elements[i]) {
309                    taken[j] = true;
310                    sh[i] = ctr;
311                    break;
312                }
313            }
314        }
315        return sh;
316    }

318    /**
319     * Given an array of T that constitutes a permutation of the elements this was constructed with, finds the specific
320     * index of the permutation given a factoradic numbering scheme (not used by the rest of this class, except the
321     * decodePermutation() method). The index can be passed to decodePermutation to reproduce the permutation passed to
322     * this, or modified and then passed to decodePermutation(). Determines equality by identity, not by .equals(), so
323     * that equivalent values that have different references/identities can appear in the permuted elements.
324     * <br>
325     * Credit goes to user Joren on StackOverflow, http://stackoverflow.com/a/1506337
326     * @param perm an array of T that must be a valid permutation of this object's elements
327     * @return an encoded number that can be used to reconstruct the permutation when passed to decodePermutation()
328     */
329    public long encodePermutation(T[] perm)
330    {
331        long e = 0;
332        if(perm == null || perm.length != elements.length)
333            return e;
334        int[] shift = getPermutationShift(perm);
335        for (int i = 1; i < shift.length; i++) {
336            e += shift[i] * MathExtras.factorialsStart[i];
337        }
338        return e;
339    }

364    private int[] factoradicDecode(long e)
365    {
366        int[] sequence = new int[elements.length];
367        int base = 2;
368
369        for (int k = 1; k < elements.length; k++)
370        {
371            sequence[elements.length - 1 - k] = (int)(e % base);
372            e /= base;
373
374            base++;
375        }
376        return sequence;
377    }

379    /**
380     * Given a long between 0 and the total number of permutations possible (see getTotalPermutations() for how to access
381     * this) and an array of T with the same length as the elements this was constructed with, fills the array with the
382     * permutation described by the long as a special (factoradic) index into the possible permutations. You can get an
383     * index for a specific permutation with encodePermutation() or by generating a random number between 0 and
384     * getTotalPermutations(), if you want it randomly.
385     * <br>
386     * Credit goes to user Joren on StackOverflow, http://stackoverflow.com/a/1506337
387     * @param encoded the index encoded as a long
388     * @param destination an array of T that must have equivalent length to the elements this was constructed with
389     * @return the looked-up permutation, which is the same value destination will be assigned
390     */
391    public T[] decodePermutation(long encoded, T[] destination)
392    {
393        if(destination == null)
394            return null;
395        encoded %= totalPermutations;
396        int[] sequence = factoradicDecode(encoded);
397
398        boolean[] set = new boolean[elements.length];
399
400        for (int i = 0; i < elements.length; i++)
401        {
402            int s = sequence[i];
403            int remainingPosition = 0;
404            int index;
405
406            // Find the s'th position in the permuted list that has not been set yet.
407            for (index = 0; index < elements.length; index++)
408            {
409                if (!set[index])
410                {
411                    if (remainingPosition == s)
412                        break;
413
414                    remainingPosition++;
415                }
416            }
417
418            destination[index] = elements[i];
419            set[index] = true;
420        }
421        return destination;
422    }

436        ArrayList<T> list = new ArrayList<T>(elements.length);
437        Collections.addAll(list, elements);

526    private static int[] getPermutationShift(int[] perm) {
527        int[] sh = new int[perm.length];
528        boolean[] taken = new boolean[perm.length];
529
530        for (int i = 0; i < perm.length - 1; i++) {
531            int ctr = -1;
532            for (int j = 0; j < perm.length; j++) {
533                if (!taken[j])
534                    ctr++;
535                if (perm[j] == i) {
536                    taken[j] = true;
537                    sh[i] = ctr;
538                    break;
539                }
540            }
541        }
542        return sh;

544    /**
545     * Given an array of int that constitutes a permutation of indices, where no element in perm is repeated and all
546     * ints are less than perm.length, finds the specific index of the permutation given a factoradic numbering scheme
547     * (not used by the rest of this class, except the decodePermutation() method). The index can be passed to
548     * decodePermutation to reproduce the index permutation passed to this, or modified and then passed to
549     * decodePermutation().
550     * <br>
551     * If perm is more than 20 items in length, you should use {@link #encodeBigPermutation} instead.
552     * <br>
553     * Credit goes to user Joren on StackOverflow, http://stackoverflow.com/a/1506337
554     * @param perm an array of int that is a permutation of the range from 0 (inclusive) to perm.length (exclusive, must be no more than 20)
555     * @return an encoded number that can be used to reconstruct the permutation when passed to decodePermutation()
556     */
557    public static long encodePermutation(int[] perm)
558    {
559        long e = 0;
560        if(perm == null || perm.length <= 0)
561            return e;
562        int[] shift = getPermutationShift(perm);
563        for (int i = 1; i < shift.length; i++) {
564            e += shift[i] * MathExtras.factorialsStart[i];
565        }
566        return e;
567    }

569    /**
570     * Given an array of int that constitutes a permutation of indices, where no element in perm is repeated and all
571     * ints are less than perm.length, finds the specific index of the permutation given a factoradic numbering scheme
572     * (not used by the rest of this class, except the decodePermutation() method). The index can be passed to
573     * decodePermutation to reproduce the index permutation passed to this, or modified and then passed to
574     * decodePermutation().
575     * <br>
576     * If perm is 20 items or less in length, you can use {@link #encodePermutation} instead to get a 64-bit encoding.
577     * <br>
578     * Credit goes to user Joren on StackOverflow, http://stackoverflow.com/a/1506337
579     * @param perm an array of int that is a permutation of the range from 0 (inclusive) to perm.length (exclusive)
580     * @return an encoded number that can be used to reconstruct the permutation when passed to decodePermutation()
581     */
582    public static BigInteger encodeBigPermutation(int[] perm)
583    {
584        BigInteger e = BigInteger.ZERO;
585        if(perm == null || perm.length <= 0)
586            return e;
587        int[] shift = getPermutationShift(perm);
588        for (int i = 1; i < shift.length; i++) {
589            e = e.add(MathExtras.bigFactorial(i).multiply(BigInteger.valueOf(shift[i])));
590        }
591        return e;
592    }

595    private static int[] factoradicDecode(long e, int count)
596    {
597        int[] sequence = new int[count];
598        int base = 2;
599
600        for (int k = 1; k < count; k++)
601        {
602            sequence[count - 1 - k] = (int)(e % base);
603            e /= base;
604
605            base++;
606        }
607        return sequence;
608    }

610    private static int[] factoradicDecode(BigInteger e, int count)
611    {
612        int[] sequence = new int[count];
613        BigInteger base = BigInteger.valueOf(2);
614
615        for (int k = 1; k < count; k++)
616        {
617            sequence[count - 1 - k] = e.mod(base).intValue();
618            e = e.divide(base);
619
620            base = base.add(BigInteger.ONE);
621        }
622        return sequence;
623    }

673     * Given a long between 0 and the total number of permutations possible (see getBigTotalPermutations() for how to
674     * access this) and an int count of how many indices to find a permutation of, returns an array with the permutation
675     * of the indices described by the long as a special (factoradic) index into the possible permutations. You can get
676     * an index for a specific permutation with encodeBigPermutation() or by generating a random number between 0 and
677     * getBigTotalPermutations(), if you want it randomly.
678     * <br>
679     * Credit goes to user Joren on StackOverflow, http://stackoverflow.com/a/1506337
680     * @param encoded the index encoded as a BigInteger
681     * @param count a positive int that will be the size of the returned array
682     * @return the looked-up permutation as an int array with length equal to count

684    public static int[] decodePermutation(BigInteger encoded, int count)
685    {
686        if(count <= 0)
687            return new int[0];
688        BigInteger enc = encoded.mod(MathExtras.bigFactorial(count));
689        int[] sequence = factoradicDecode(enc, count), destination = new int[count];
690        //char[] list = new char[] { 'a', 'b', 'c', 'd', 'e' }; //change for elements
691
692        //char[] permuted = new char[n]; //change for destination
693        boolean[] set = new boolean[count];
694
695        for (int i = 0; i < count; i++)
696        {
697            int s = sequence[i];
698            int remainingPosition = 0;
699            int index;
700
701            // Find the s'th position in the permuted list that has not been set yet.
702            for (index = 0; index < count; index++)
703            {
704                if (!set[index])
705                {
706                    if (remainingPosition == s)
707                        break;
708
709                    remainingPosition++;
710                }
711            }
712
713            destination[index] = i;
714            set[index] = true;
715        }
716        return destination;
717    }

719    /**
720     * Given a long between 0 and the total number of permutations possible (see getTotalPermutations() for how to access
721     * this) and an int count of how many indices to find a permutation of, returns an array with the permutation
722     * of the indices described by the long as a special (factoradic) index into the possible permutations. You can get
723     * an index for a specific permutation with encodePermutation() or by generating a random number between 0 and
724     * getTotalPermutations(), if you want it randomly. This variant adds an int to each item in the returned array,
725     * which may be useful if generating indices that don't start at 0.
726     * <br>
727     * Credit goes to user Joren on StackOverflow, http://stackoverflow.com/a/1506337
728     * @param encoded the index encoded as a long
729     * @param count an int between 1 and 20, inclusive, that will be the size of the returned array
730     * @param add an int to add to each item of the permutation
731     * @return the looked-up permutation as an int array with length equal to count
732     */
733    public static int[] decodePermutation(long encoded, int count, int add)
734    {
735        int[] p = decodePermutation(encoded, count);
736        for (int i = 0; i < p.length; i++) {
737            p[i] += add;
738        }
739        return p;
