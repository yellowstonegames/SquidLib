017package squidpony.squidai.astar;
018
019/** 
020 * A connection between two nodes of the {@link Graph}. The connection has a non-negative cost that often represents
021 * time or distance. However, the cost can be anything you want, for instance a combination of time, distance, and other
022 * factors. Possible such factors could be how dangerous or unpleasant it seems to go between two nodes, how much it
023 * costs to make payments to use toll roads, whether a specific path crosses an international border, and so on.
024 * 
025 * @param <N> Type of node
026 * 
027 * @author davebaol */
028public interface Connection<N> {
029
030        /** Returns the non-negative cost of this connection */
031        float getCost();
032
033        /** Returns the node that this connection came from */
034        N getFromNode();
035
036        /** Returns the node that this connection leads to */
037        N getToNode();
038
039}
