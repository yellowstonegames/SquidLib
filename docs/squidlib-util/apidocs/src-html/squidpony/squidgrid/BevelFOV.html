009/**
010 * Use FOV instead, probably with type SHADOW; this has the same results as shadowcasting FOV but is slower, currently.
011 * This will probably be rewritten in a very different way.
012 * <br>
013 * A different kind of FOV implementation that pre-processes the map to better analyze the locations of corners.
014 * Based loosely on Adam Milazzo's FOV, http://www.adammil.net/blog/v125_Roguelike_Vision_Algorithms.html , which in
015 * turn is based on the diamond walls technique (on the same page).
016 */
017@Beta
018public class BevelFOV extends FOV {

046     * radius.
047     *
048     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()

050     * @param starty        the vertical component of the starting location
051     * @return the computed light grid

055        return calculateFOV(resistanceMap, startx, starty, Integer.MAX_VALUE);

078    /**
079     * Calculates the Field Of View for the provided map from the given x, y
080     * coordinates. Returns a light map where the values represent a percentage
081     * of fully lit.
082     * <p>
083     * The starting point for the calculation is considered to be at the center
084     * of the origin cell. Radius determinations are determined by the provided
085     * RadiusStrategy.
086     *
087     * @param resistanceMap   the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
088     * @param startX          the horizontal component of the starting location
089     * @param startY          the vertical component of the starting location
090     * @param radius          the distance the light will extend to
091     * @param radiusTechnique provides a means to calculate the radius as desired
092     * @return the computed light grid
093     */
094    @Override
095    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius, Radius radiusTechnique) {
096        double rad = Math.max(1, radius);
097        double decay = 1.0 / rad;
098
099        int width = resistanceMap.length;
100        int height = resistanceMap[0].length;
101
102        if (light == null)
103            light = new double[width][height];
104        else {
105            if (light.length != width || light[0].length != height)
106                light = new double[width][height];
107            else {
108                ArrayTools.fill(light, 0.0);
109            }
110        }
111        light[startX][startY] = 1;//make the starting space full power
112        maybeCache(resistanceMap);
113        for (Direction d : Direction.DIAGONALS) {
114            shadowCast(1, 1.0, 0.0, 0, d.deltaX, d.deltaY, 0, rad, startX, startY, decay, radiusTechnique);
115            shadowCast(1, 1.0, 0.0, d.deltaX, 0, 0, d.deltaY, rad, startX, startY, decay, radiusTechnique);
116        }
117        return light;
118    }

120    /**
121     * Calculates the Field Of View for the provided map from the given x, y
122     * coordinates. Returns a light map where the values represent a percentage
123     * of fully lit.
124     * <p>
125     * The starting point for the calculation is considered to be at the center
126     * of the origin cell. Radius determinations are determined by the provided
127     * RadiusStrategy. A conical section of FOV is lit by this method if
128     * span is greater than 0.
129     *
130     * @param resistanceMap   the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
131     * @param startX          the horizontal component of the starting location
132     * @param startY          the vertical component of the starting location
133     * @param radius          the distance the light will extend to
134     * @param radiusTechnique provides a means to calculate the radius as desired
135     * @param angle           the angle in degrees that will be the center of the FOV cone, 0 points right
136     * @param span            the angle in degrees that measures the full arc contained in the FOV cone
137     * @return the computed light grid
138     */
139    @Override
140    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius, Radius radiusTechnique, double angle, double span) {
141        double rad = Math.max(1, radius);
142
143        double decay = 1.0 / rad;
144
145        double angle2 = Math.toRadians((angle > 360.0 || angle < 0.0)
146                ? GwtCompatibility.IEEEremainder(angle + 720.0, 360.0) : angle);
147        double span2 = Math.toRadians(span);
148        int width = resistanceMap.length;
149        int height = resistanceMap[0].length;
150
151        if (light == null)
152            light = new double[width][height];
153        else {
154            if (light.length != width || light[0].length != height)
155                light = new double[width][height];
156            else {
157                ArrayTools.fill(light, 0.0);
158            }
159        }
160        light[startX][startY] = 1;//make the starting space full power
161        maybeCache(resistanceMap);
162
163        int ctr = 0;
164        boolean started = false;
165        for (Direction d : ccw) {
166            ctr %= 4;
167            ++ctr;
168            if (angle <= Math.PI / 2.0 * ctr + span / 2.0)
169                started = true;
170            if (started) {
171                if (ctr < 4 && angle < Math.PI / 2.0 * (ctr - 1) - span / 2.0)
172                    break;
173                light = shadowCastLimited(1, 1.0, 0.0, 0, d.deltaX, d.deltaY, 0, rad, startX, startY, decay, radiusTechnique, angle2, span2);
174                light = shadowCastLimited(1, 1.0, 0.0, d.deltaX, 0, 0, d.deltaY, rad, startX, startY, decay, radiusTechnique, angle2, span2);
175            }
176        }
177        return light;
178    }

180    /**
181     * Calculates what cells are visible from (startX,startY) using the given resistanceMap; this can be given to
182     * mixVisibleFOVs() to limit extra light sources to those visible from the starting point.
183     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
184     * @param startX the center of the LOS map; typically the player's x-position
185     * @param startY the center of the LOS map; typically the player's y-position
186     * @return an LOS map with the given starting point
187     */
188    public double[][] calculateLOSMap(double[][] resistanceMap, int startX, int startY)
189    {
190        if(resistanceMap == null || resistanceMap.length == 0)
191            return new double[0][0];
192        return calculateFOV(resistanceMap, startX, startY, resistanceMap.length + resistanceMap[0].length, Radius.SQUARE);
193    }

195    private void maybeCache(final double[][] resistanceMap)
196    {
197        if(resMapCache != resistanceMap)
198        {
199            storeMap(resistanceMap);
200        }
201    }
202
203    /**
204     * Pre-calculates some time-intensive calculations that generally only need to be done once per map. You should pass
205     * the same exact value for resistanceMap to {@link #calculateFOV(double[][], int, int)} or its overloads if you
206     * want to avoid re-calculating this map analysis. If an element of resistanceMap changes, you should call this
207     * method again before you call calculateFOV, though it can wait until you've made all the changes you want to
208     * resistanceMap's elements. This class does not have a way to tell when the array has been modified in user code,
209     * so you should take care if you change resistanceMap to call this method.
210     * @param resistanceMap a 2D double array, as produced by {@link squidpony.squidgrid.mapping.DungeonUtility#generateResistances(char[][])}
211     */
212    public void storeMap(final double[][] resistanceMap)
213    {
214        if(resistanceMap == null || resistanceMap.length <= 0 || resistanceMap[0].length <= 0)
215            throw new IllegalArgumentException("resistanceMap cannot be null or empty");
216        if(resMapCache == null
217                || resMapCache.length != resistanceMap.length
218                || resMapCache[0].length != resistanceMap[0].length)
219        {
220            resMapCache = resistanceMap;
221            if(map == null)
222                map = new GreasedRegion(resistanceMap, 1.0, Double.POSITIVE_INFINITY, 3).removeCorners();
223            else
224                map.refill(resistanceMap, 1.0, Double.POSITIVE_INFINITY, 3).removeCorners();
225        }
226        else
227        {
228            for (int x = 0; x < resistanceMap.length; x++) {
229                for (int y = 0; y < resistanceMap[x].length; y++) {
230                    if(resMapCache[x][y] != resistanceMap[x][y])
231                    {
232                        resMapCache = resistanceMap;
233                        if(map == null)
234                            map = new GreasedRegion(resistanceMap, 1.0, Double.POSITIVE_INFINITY, 3).removeCorners();
235                        else
236                            map.refill(resistanceMap, 1.0, Double.POSITIVE_INFINITY, 3).removeCorners();
237                        return;
238                    }
239                }
240            }
241        }
242    }

244    private double[][] shadowCast(int row, double start, double end, int xx, int xy, int yx, int yy,
245                                         double radius, int startx, int starty, double decay, Radius radiusStrategy) {
246        double newStart = 0;
247        if (start < end) {
248            return light;
249        }
250        int width = light.length;
251        int height = light[0].length;
252
253        boolean blocked = false;
254        for (int distance = row; distance <= radius && distance < (width + height) && !blocked; distance++) {
255            int deltaY = -distance;
256            for (int deltaX = -distance; deltaX <= 0; deltaX++) {
257                int currentX = startx + deltaX * xx + deltaY * xy;
258                int currentY = starty + deltaX * yx + deltaY * yy;
259                double leftSlope = (deltaX - 0.5f) / (deltaY + 0.5f);
260                double rightSlope = (deltaX + 0.5f) / (deltaY - 0.5f);
261
262                if (!(currentX >= 0 && currentY >= 0 && currentX < width && currentY < height) || start < rightSlope) {
263                    continue;
264                } else if (end > leftSlope) {
265                    break;
266                }
267                double deltaRadius = radiusStrategy.radius(deltaX, deltaY);
268                //check if it's within the lightable area and light if needed
269                if (deltaRadius <= radius) {
270                    light[currentX][currentY] = 1 - decay * deltaRadius;
271                }
272
273                if (blocked) { //previous cell was a blocking one
274                    if (map.contains(currentX * 3 + 1 + xx + xy, currentY * 3 + 1 + yx + yy)) { // hit a wall
275                        newStart = rightSlope;
276                    } else if(!map.contains(currentX * 3 + 1, currentY * 3 + 1)){
277                        blocked = false;
278                        start = newStart;
279                    }
280                } else {
281                    if (distance < radius && map.contains(currentX * 3 + 1, currentY * 3 + 1))
282                    { // hit a wall within sight line
283                        blocked = true;
284                        light = shadowCast(distance + 1, start, leftSlope, xx, xy, yx, yy, radius, startx, starty, decay, radiusStrategy);
285                        newStart = rightSlope;
286                    }
287                }
288            }
289        }
290        return light;
291    }
292    private double[][] shadowCastLimited(int row, double start, double end, int xx, int xy, int yx, int yy,
293                                                double radius, int startx, int starty, double decay,
294                                                Radius radiusStrategy, double angle, double span) {
295        double newStart = 0;
296        if (start < end) {
297            return light;
298        }
299        int width = light.length;
300        int height = light[0].length;
301
302        boolean blocked = false;
303        for (int distance = row; distance <= radius && distance < (width + height) && !blocked; distance++) {
304            int deltaY = -distance;
305            for (int deltaX = -distance; deltaX <= 0; deltaX++) {
306                int currentX = startx + deltaX * xx + deltaY * xy;
307                int currentY = starty + deltaX * yx + deltaY * yy;
308                double leftSlope = (deltaX - 0.5f) / (deltaY + 0.5f);
309                double rightSlope = (deltaX + 0.5f) / (deltaY - 0.5f);
310
311                if (!(currentX >= 0 && currentY >= 0 && currentX < width && currentY < height) || start < rightSlope) {
312                    continue;
313                } else if (end > leftSlope) {
314                    break;
315                }
316                double newAngle = NumberTools.atan2(currentY - starty, currentX - startx) + Math.PI * 2;
317                if (Math.abs(GwtCompatibility.IEEEremainder(angle - newAngle + Math.PI * 8, Math.PI * 2)) > span / 2.0)
318                    continue;
319
320                double deltaRadius = radiusStrategy.radius(deltaX, deltaY);
321                //check if it's within the lightable area and light if needed
322                if (deltaRadius <= radius) {
323                    light[currentX][currentY] = 1 - decay * deltaRadius;
324                }
325
326                if (blocked) { //previous cell was a blocking one
327                    if (map.contains(currentX * 3 + 1 + xx + xy, currentY * 3 + 1 + yx + yy)) {//hit a wall
328                        newStart = rightSlope;
329                    } else {
330                        blocked = false;
331                        start = newStart;
332                    }
333                } else {
334                    if (map.contains(currentX * 3 + 1 + xx + xy, currentY * 3 + 1 + yx + yy) && distance < radius) {//hit a wall within sight line
335                        blocked = true;
336                        light = shadowCastLimited(distance + 1, start, leftSlope, xx, xy, yx, yy, radius, startx, starty, decay, radiusStrategy, angle, span);
337                        newStart = rightSlope;
338                    }
339                }
340            }
341        }
342        return light;
343    }
