012/**
013 * A subsection of a (typically modern-day or sci-fi) area map that can be placed by ModularMapGenerator.
014 * Created by Tommy Ettinger on 4/4/2016.
015 */
016public class MapModule implements Comparable<MapModule>, Serializable {
017    private static final long serialVersionUID = -1273406898212937188L;
018
019    /**
020     * The contents of this section of map.
021     */
022    public char[][] map;
023    /**
024     * The room/cave/corridor/wall status for each cell of this section of map.
025     */
026    public int[][] environment;
027    /**
028     * Stores Coords just outside the contents of the MapModule, where doors are allowed to connect into this.
029     * Uses Coord positions that are relative to this MapModule's map field, not whatever this is being placed into.
030     */
031    public Coord[] validDoors;
032    /**
033     * The minimum point on the bounding rectangle of the room, including walls.
034     */
035    public Coord min;
036    /**
037     * The maximum point on the bounding rectangle of the room, including walls.
038     */
039    public Coord max;
040
041    public ArrayList<Coord> leftDoors, rightDoors, topDoors, bottomDoors;

043    public int category;

045    private static final char[] validPacking = new char[]{'.', ',', '"', '^', '<', '>'},
046            doors = new char[]{'+', '/'};
047    public MapModule()
048    {
049        this(DungeonUtility.wallWrap(ArrayTools.fill('.', 8, 8)));
050    }

052    /**
053     * Constructs a MapModule given only a 2D char array as the contents of this section of map. The actual MapModule
054     * will use doors in the 2D char array as '+' or '/' if present. Otherwise, the valid locations for doors will be
055     * any outer wall adjacent to a floor ('.'), shallow water (','), grass ('"'), trap  ('^'), or staircase (less than
056     * or greater than signs). The max and min Coords of the bounding rectangle, including one layer of outer walls,
057     * will also be calculated. The map you pass to this does need to have outer walls present in it already.
058     * @param map the 2D char array that contains the contents of this section of map
059     */
060    public MapModule(char[][] map)
061    {
062        if(map == null || map.length <= 0)
063            throw new UnsupportedOperationException("Given map cannot be empty in MapModule");
064        CoordPacker.init();
065        this.map = ArrayTools.copy(map);
066        environment = ArrayTools.fill(DungeonUtility.ROOM_FLOOR, this.map.length, this.map[0].length);
067        for (int x = 0; x < map.length; x++) {
068            for (int y = 0; y < map[0].length; y++) {
069                if(this.map[x][y] == '#')
070                    environment[x][y] = DungeonUtility.ROOM_WALL;
071            }
072        }
073        short[] pk = CoordPacker.fringe(
074                CoordPacker.pack(this.map, validPacking),
075                1, this.map.length, this.map[0].length, false, true);
076        Coord[] tmp = CoordPacker.bounds(pk);
077        min = tmp[0];
078        max = tmp[1];
079        category = categorize(Math.max(max.x, max.y));
080        short[] drs = CoordPacker.pack(this.map, doors);
081        if(drs.length >= 2)
082            validDoors = CoordPacker.allPacked(drs);
083        else {
084            validDoors = CoordPacker.fractionPacked(pk, 5);//CoordPacker.allPacked(pk);
085            //for(Coord dr : validDoors)
086            //    this.map[dr.x][dr.y] = '+';
087        }
088        initSides();
089    }
090    /**
091     * Constructs a MapModule given only a short array of packed data (as produced by CoordPacker and consumed or produced
092     * by several other classes) that when unpacked will yield the contents of this section of map. The actual MapModule
093     * will use a slightly larger 2D array than the given width and height to ensure walls can be drawn around the floors,
094     * and the valid locations for doors will be any outer wall adjacent to an "on" coordinate in packed. The max and min
095     * Coords of the bounding rectangle, including one layer of outer walls, will also be calculated. Notably, the packed
096     * data you pass to this does not need to have a gap between floors and the edge of the map to make walls.
097     * @param packed the short array, as packed data from CoordPacker, that contains the contents of this section of map
098     */
099    public MapModule(short[] packed, int width, int height)
100    {
101        if(map == null || map.length <= 0)
102            throw new UnsupportedOperationException("Given map cannot be empty in MapModule");
103        CoordPacker.init();
104        map = ArrayTools.copy(CoordPacker.unpackChar(packed, width, height, '.', '#'));
105        environment = ArrayTools.fill(DungeonUtility.ROOM_FLOOR, this.map.length, this.map[0].length);
106        for (int x = 0; x < map.length; x++) {
107            for (int y = 0; y < map[0].length; y++) {
108                if(map[x][y] == '#')
109                    environment[x][y] = DungeonUtility.ROOM_WALL;
110            }
111        }
112        short[] pk = CoordPacker.fringe(
113                CoordPacker.pack(map, validPacking),
114                1, map.length, map[0].length, false, true);
115        Coord[] tmp = CoordPacker.bounds(pk);
116        min = tmp[0];
117        max = tmp[1];
118        category = categorize(Math.max(max.x, max.y));
119        short[] drs = CoordPacker.pack(map, doors);
120        if(drs.length >= 2)
121            validDoors = CoordPacker.allPacked(drs);
122        else {
123            validDoors = CoordPacker.fractionPacked(pk, 5);//CoordPacker.allPacked(pk);
124            //for(Coord dr : validDoors)
125            //    this.map[dr.x][dr.y] = '+';
126        }
127        initSides();
128    }

130    /**
131     * Constructs a MapModule from the given arguments without modifying them, copying map without changing its size,
132     * copying validDoors, and using the same min and max (which are immutable, so they can be reused).
133     * @param map the 2D char array that contains the contents of this section of map; will be copied exactly
134     * @param validDoors a Coord array that stores viable locations to place doors in map; will be cloned
135     * @param min the minimum Coord of this MapModule's bounding rectangle
136     * @param max the maximum Coord of this MapModule's bounding rectangle
137     */
138    public MapModule(char[][] map, Coord[] validDoors, Coord min, Coord max)
139    {
140        CoordPacker.init();
141        this.map = ArrayTools.copy(map);
142        environment = ArrayTools.fill(DungeonUtility.ROOM_FLOOR, this.map.length, this.map[0].length);
143        for (int x = 0; x < map.length; x++) {
144            for (int y = 0; y < map[0].length; y++) {
145                if(this.map[x][y] == '#')
146                    environment[x][y] = DungeonUtility.ROOM_WALL;
147            }
148        }
149        this.validDoors = GwtCompatibility.cloneCoords(validDoors);
150        this.min = min;
151        this.max = max;
152        category = categorize(Math.max(max.x, max.y));
153        ArrayList<Coord> doors2 = new ArrayList<>(16);
154        for (int x = 0; x < map.length; x++) {
155            for (int y = 0; y < map[x].length; y++) {
156                if(map[x][y] == '+' || map[x][y] == '/')
157                    doors2.add(Coord.get(x, y));
158            }
159        }
160        if(!doors2.isEmpty()) this.validDoors = doors2.toArray(new Coord[doors2.size()]);
161        initSides();
162    }

164    /**
165     * Copies another MapModule and uses it to construct a new one.
166     * @param other an already-constructed MapModule that this will copy
167     */
168    public MapModule(MapModule other)
169    {
170        this(other.map, other.validDoors, other.min, other.max);
171    }

173    /**
174     * Rotates a copy of this MapModule by the given number of 90-degree turns. Describing the turns as clockwise or
175     * counter-clockwise depends on whether the y-axis "points up" or "points down." If higher values for y are toward the
176     * bottom of the screen (the default for when 2D arrays are printed), a turn of 1 is clockwise 90 degrees, but if the
177     * opposite is true and higher y is toward the top, then a turn of 1 is counter-clockwise 90 degrees.
178     * @param turns the number of 90 degree turns to adjust this by
179     * @return a new MapModule (copied from this one) that has been rotated by the given amount
180     */
181    public MapModule rotate(int turns)
182    {
183        turns %= 4;
184        char[][] map2;
185        Coord[] doors2;
186        Coord min2, max2;
187        int xSize = map.length - 1, ySize = map[0].length - 1;
188        switch (turns)
189        {
190            case 1:
191                map2 = new char[map[0].length][map.length];
192                for (int i = 0; i < map.length; i++) {
193                    for (int j = 0; j < map[0].length; j++) {
194                        map2[ySize - j][i] = map[i][j];
195                    }
196                }
197                doors2 = new Coord[validDoors.length];
198                for (int i = 0; i < validDoors.length; i++) {
199                    doors2[i] = Coord.get(ySize - validDoors[i].y, validDoors[i].x);
200                }
201                min2 = Coord.get(ySize - max.y, min.x);
202                max2 = Coord.get(ySize - min.y, max.x);
203                return new MapModule(map2, doors2, min2, max2);
204            case 2:
205                map2 = new char[map.length][map[0].length];
206                for (int i = 0; i < map.length; i++) {
207                    for (int j = 0; j < map[0].length; j++) {
208                        map2[xSize - i][ySize - j] = map[i][j];
209                    }
210                }
211                doors2 = new Coord[validDoors.length];
212                for (int i = 0; i < validDoors.length; i++) {
213                    doors2[i] = Coord.get(xSize - validDoors[i].x, ySize - validDoors[i].y);
214                }
215                min2 = Coord.get(xSize - max.x, ySize - max.y);
216                max2 = Coord.get(xSize - min.x, ySize - min.y);
217                return new MapModule(map2, doors2, min2, max2);
218            case 3:
219                map2 = new char[map[0].length][map.length];
220                for (int i = 0; i < map.length; i++) {
221                    for (int j = 0; j < map[0].length; j++) {
222                        map2[j][xSize - i] = map[i][j];
223                    }
224                }
225                doors2 = new Coord[validDoors.length];
226                for (int i = 0; i < validDoors.length; i++) {
227                    doors2[i] = Coord.get(validDoors[i].y, xSize - validDoors[i].x);
228                }
229                min2 = Coord.get(min.y, xSize - max.x);
230                max2 = Coord.get(max.y, xSize - min.x);
231                return new MapModule(map2, doors2, min2, max2);
232            default:
233                return new MapModule(map, validDoors, min, max);
234        }
235    }

237    public MapModule flip(boolean flipLeftRight, boolean flipUpDown)
238    {
239        if(!flipLeftRight && !flipUpDown)
240            return new MapModule(map, validDoors, min, max);
241        char[][] map2 = new char[map.length][map[0].length];
242        Coord[] doors2 = new Coord[validDoors.length];
243        Coord min2, max2;
244        int xSize = map.length - 1, ySize = map[0].length - 1;
245        if(flipLeftRight && flipUpDown)
246        {
247            for (int i = 0; i < map.length; i++) {
248                for (int j = 0; j < map[0].length; j++) {
249                    map2[xSize - i][ySize - j] = map[i][j];
250                }
251            }
252            for (int i = 0; i < validDoors.length; i++) {
253                doors2[i] = Coord.get(xSize - validDoors[i].x, ySize - validDoors[i].y);
254            }
255            min2 = Coord.get(xSize - max.x, ySize - max.y);
256            max2 = Coord.get(xSize - min.x, xSize - min.y);
257        }
258        else if(flipLeftRight)
259        {
260            for (int i = 0; i < map.length; i++) {
261                System.arraycopy(map[i], 0, map2[xSize - i], 0, map[0].length);
262            }
263            for (int i = 0; i < validDoors.length; i++) {
264                doors2[i] = Coord.get(xSize - validDoors[i].x, validDoors[i].y);
265            }
266            min2 = Coord.get(xSize - max.x, min.y);
267            max2 = Coord.get(xSize - min.x, max.y);
268        }
269        else
270        {
271            for (int i = 0; i < map.length; i++) {
272                for (int j = 0; j < map[0].length; j++) {
273                    map2[i][ySize - j] = map[i][j];
274                }
275            }
276            for (int i = 0; i < validDoors.length; i++) {
277                doors2[i] = Coord.get(validDoors[i].x, ySize - validDoors[i].y);
278            }
279            min2 = Coord.get(min.x, ySize - max.y);
280            max2 = Coord.get(max.x, xSize - min.y);
281        }
282        return new MapModule(map2, doors2, min2, max2);
283    }

285    private static int categorize(int n)
286    {
287        int highest = Integer.highestOneBit(n);
288        return Math.max(4, (highest == NumberTools.lowestOneBit(n)) ? highest : highest << 1);
289    }
290    private void initSides()
291    {
292        leftDoors = new ArrayList<>(8);
293        rightDoors = new ArrayList<>(8);
294        topDoors = new ArrayList<>(8);
295        bottomDoors = new ArrayList<>(8);
296        for(Coord dr : validDoors)
297        {
298            if(dr.x * max.y < dr.y * max.x && dr.y * max.x < (max.x - dr.x) * max.y)
299                leftDoors.add(dr);
300            else if(dr.x * max.y> dr.y * max.x && dr.y * max.x > (max.x - dr.x) * max.y)
301                rightDoors.add(dr);
302            else if(dr.x * max.y > dr.y * max.x && dr.y * max.x < (max.x - dr.x) * max.y)
303                topDoors.add(dr);
304            else if(dr.x * max.y < dr.y * max.x && dr.y * max.x > (max.x - dr.x) * max.y)
305                bottomDoors.add(dr);
306        }
307    }

309    @Override
310    public int compareTo(MapModule o) {
311        if(o == null) return 1;
312        return category - o.category;
313    }
