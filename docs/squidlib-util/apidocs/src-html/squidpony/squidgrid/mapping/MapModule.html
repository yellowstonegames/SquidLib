003import squidpony.ArrayTools;
004import squidpony.squidmath.Coord;
005import squidpony.squidmath.GreasedRegion;

011/**
012 * A subsection of a (typically modern-day or sci-fi) area map that can be placed by ModularMapGenerator.
013 * Created by Tommy Ettinger on 4/4/2016.
014 */
015public class MapModule implements Comparable<MapModule>, Serializable {
016    private static final long serialVersionUID = -1273406898212937188L;
017
018    /**
019     * The contents of this section of map.
020     */
021    public char[][] map;
022    /**
023     * The room/cave/corridor/wall status for each cell of this section of map.
024     */
025    public int[][] environment;
026    /**
027     * Stores Coords just outside the contents of the MapModule, where doors are allowed to connect into this.
028     * Uses Coord positions that are relative to this MapModule's map field, not whatever this is being placed into.
029     */
030    public GreasedRegion validDoors;
031    /**
032     * The minimum point on the bounding rectangle of the room, including walls.
033     */
034    public Coord min;
035    /**
036     * The maximum point on the bounding rectangle of the room, including walls.
037     */
038    public Coord max;
039
040    public ArrayList<Coord> leftDoors, rightDoors, topDoors, bottomDoors;
041
042    public int category;
043
044    private static final char[]
045            validPacking = new char[]{'.', ',', '"', '^', '<', '>'},
046            doors = new char[]{'+', '/'};
047    public MapModule()
048    {
049        this(DungeonUtility.wallWrap(ArrayTools.fill('.', 8, 8)));
050    }
051
052    /**
053     * Constructs a MapModule given only a 2D char array as the contents of this section of map. The actual MapModule
054     * will use doors in the 2D char array as '+' or '/' if present. Otherwise, the valid locations for doors will be
055     * any outer wall adjacent to a floor ('.'), shallow water (','), grass ('"'), trap  ('^'), or staircase (less than
056     * or greater than signs). The max and min Coords of the bounding rectangle, including one layer of outer walls,
057     * will also be calculated. The map you pass to this does need to have outer walls present in it already.
058     * @param map the 2D char array that contains the contents of this section of map
059     */
060    public MapModule(char[][] map)
061    {
062        if(map == null || map.length <= 0)
063            throw new UnsupportedOperationException("Given map cannot be empty in MapModule");
064        this.map = ArrayTools.copy(map);
065        environment = ArrayTools.fill(DungeonUtility.ROOM_FLOOR, this.map.length, this.map[0].length);
066        for (int x = 0; x < map.length; x++) {
067            for (int y = 0; y < map[0].length; y++) {
068                if(this.map[x][y] == '#')
069                    environment[x][y] = DungeonUtility.ROOM_WALL;
070            }
071        }
072        GreasedRegion pack = new GreasedRegion(this.map, validPacking).fringe();
073//        short[] pk = CoordPacker.fringe(
074//                CoordPacker.pack(this.map, validPacking),
075//                1, this.map.length, this.map[0].length, false, true);
076//        Coord[] tmp = CoordPacker.bounds(pk);
077        min = Coord.get(pack.xBound(true), pack.yBound(true));
078        max = Coord.get(pack.xBound(false), pack.yBound(false));
079        category = categorize(Math.max(max.x, max.y));
080        validDoors = new GreasedRegion(this.map, doors);
081        if(validDoors.size() < 2)
082        {
083            validDoors.remake(pack).quasiRandomRegion(0.2);
084        }
085        initSides();
086    }
087
088    /**
089     * Constructs a MapModule from the given arguments without modifying them, copying map without changing its size,
090     * copying validDoors, and using the same min and max (which are immutable, so they can be reused).
091     * @param map the 2D char array that contains the contents of this section of map; will be copied exactly
092     * @param validDoors a Coord array that stores viable locations to place doors in map; will be cloned
093     * @param min the minimum Coord of this MapModule's bounding rectangle
094     * @param max the maximum Coord of this MapModule's bounding rectangle
095     */
096    public MapModule(char[][] map, GreasedRegion validDoors, Coord min, Coord max)
097    {
098        this.map = ArrayTools.copy(map);
099        environment = ArrayTools.fill(DungeonUtility.ROOM_FLOOR, this.map.length, this.map[0].length);
100        for (int x = 0; x < map.length; x++) {
101            for (int y = 0; y < map[0].length; y++) {
102                if(this.map[x][y] == '#')
103                    environment[x][y] = DungeonUtility.ROOM_WALL;
104            }
105        }
106        this.min = min;
107        this.max = max;
108        category = categorize(Math.max(max.x, max.y));
109        
110        this.validDoors = new GreasedRegion(map, doors);
111        if(this.validDoors.isEmpty()) this.validDoors.remake(validDoors);
112        
113        initSides();
114    }
115
116    /**
117     * Copies another MapModule and uses it to construct a new one.
118     * @param other an already-constructed MapModule that this will copy
119     */
120    public MapModule(MapModule other)
121    {
122        this(other.map, other.validDoors, other.min, other.max);
123    }
124
125    /**
126     * Rotates a copy of this MapModule by the given number of 90-degree turns. Describing the turns as clockwise or
127     * counter-clockwise depends on whether the y-axis "points up" or "points down." If higher values for y are toward the
128     * bottom of the screen (the default for when 2D arrays are printed), a turn of 1 is clockwise 90 degrees, but if the
129     * opposite is true and higher y is toward the top, then a turn of 1 is counter-clockwise 90 degrees.
130     * @param turns the number of 90 degree turns to adjust this by
131     * @return a new MapModule (copied from this one) that has been rotated by the given amount
132     */
133    public MapModule rotate(int turns)
134    {
135        turns &= 3;
136        char[][] map2;
137        Coord min2, max2;
138        GreasedRegion doors2;
139        int xSize = map.length - 1, ySize = map[0].length - 1;
140        switch (turns)
141        {
142            case 1:
143                map2 = new char[map[0].length][map.length];
144                for (int i = 0; i < map.length; i++) {
145                    for (int j = 0; j < map[0].length; j++) {
146                        map2[ySize - j][i] = map[i][j];
147                    }
148                }
149                doors2 = new GreasedRegion(validDoors.height, validDoors.width);
150                for (int i = 0; i < validDoors.size(); i++) {
151                    Coord n = validDoors.nth(i);
152                    doors2.insert(ySize - n.y, n.x);
153                }
154                min2 = Coord.get(ySize - max.y, min.x);
155                max2 = Coord.get(ySize - min.y, max.x);
156                return new MapModule(map2, doors2, min2, max2);
157            case 2:
158                map2 = new char[map.length][map[0].length];
159                for (int i = 0; i < map.length; i++) {
160                    for (int j = 0; j < map[0].length; j++) {
161                        map2[xSize - i][ySize - j] = map[i][j];
162                    }
163                }
164                doors2 = new GreasedRegion(validDoors.width, validDoors.height);
165                for (int i = 0; i < validDoors.size(); i++) {
166                    Coord n = validDoors.nth(i);
167                    doors2.insert(xSize - n.x, ySize - n.y);
168                }
169                min2 = Coord.get(xSize - max.x, ySize - max.y);
170                max2 = Coord.get(xSize - min.x, ySize - min.y);
171                return new MapModule(map2, doors2, min2, max2);
172            case 3:
173                map2 = new char[map[0].length][map.length];
174                for (int i = 0; i < map.length; i++) {
175                    for (int j = 0; j < map[0].length; j++) {
176                        map2[j][xSize - i] = map[i][j];
177                    }
178                }
179                doors2 = new GreasedRegion(validDoors.height, validDoors.width);
180                for (int i = 0; i < validDoors.size(); i++) {
181                    Coord n = validDoors.nth(i);
182                    doors2.insert(n.y, xSize - n.x);
183                }
184                min2 = Coord.get(min.y, xSize - max.x);
185                max2 = Coord.get(max.y, xSize - min.x);
186                return new MapModule(map2, doors2, min2, max2);
187            default:
188                return new MapModule(map, validDoors, min, max);
189        }
190    }
191
192    public MapModule flip(boolean flipLeftRight, boolean flipUpDown)
193    {
194        if(!flipLeftRight && !flipUpDown)
195            return new MapModule(map, validDoors, min, max);
196        char[][] map2 = new char[map.length][map[0].length];
197        GreasedRegion doors2 = new GreasedRegion(map.length, map[0].length);
198        Coord min2, max2;
199        int xSize = map.length - 1, ySize = map[0].length - 1;
200        if(flipLeftRight && flipUpDown)
201        {
202            for (int i = 0; i < map.length; i++) {
203                for (int j = 0; j < map[0].length; j++) {
204                    map2[xSize - i][ySize - j] = map[i][j];
205                }
206            }
207            for (int i = 0; i < validDoors.size(); i++) {
208                Coord n = validDoors.nth(i);
209                doors2.insert(xSize - n.x, ySize - n.y);
210            }
211            min2 = Coord.get(xSize - max.x, ySize - max.y);
212            max2 = Coord.get(xSize - min.x, xSize - min.y);
213        }
214        else if(flipLeftRight)
215        {
216            for (int i = 0; i < map.length; i++) {
217                System.arraycopy(map[i], 0, map2[xSize - i], 0, map[0].length);
218            }
219            for (int i = 0; i < validDoors.size(); i++) {
220                Coord n = validDoors.nth(i);
221                doors2.insert(xSize - n.x, n.y);
222            }
223            min2 = Coord.get(xSize - max.x, min.y);
224            max2 = Coord.get(xSize - min.x, max.y);
225        }
226        else
227        {
228            for (int i = 0; i < map.length; i++) {
229                for (int j = 0; j < map[0].length; j++) {
230                    map2[i][ySize - j] = map[i][j];
231                }
232            }
233            for (int i = 0; i < validDoors.size(); i++) {
234                Coord n = validDoors.nth(i);
235                doors2.insert(n.x, ySize - n.y);
236            }
237            min2 = Coord.get(min.x, ySize - max.y);
238            max2 = Coord.get(max.x, xSize - min.y);
239        }
240        return new MapModule(map2, doors2, min2, max2);
241    }
242
243    private static int categorize(int n)
244    {
245        int highest = Integer.highestOneBit(n);
246        return Math.max(4, (highest == NumberTools.lowestOneBit(n)) ? highest : highest << 1);
247    }
248    private void initSides()
249    {
250        leftDoors = new ArrayList<>(8);
251        rightDoors = new ArrayList<>(8);
252        topDoors = new ArrayList<>(8);
253        bottomDoors = new ArrayList<>(8);
254        for(Coord dr : validDoors)
255        {
256            if(dr.x * max.y < dr.y * max.x && dr.y * max.x < (max.x - dr.x) * max.y)
257                leftDoors.add(dr);
258            else if(dr.x * max.y > dr.y * max.x && dr.y * max.x > (max.x - dr.x) * max.y)
259                rightDoors.add(dr);
260            else if(dr.x * max.y > dr.y * max.x && dr.y * max.x < (max.x - dr.x) * max.y)
261                topDoors.add(dr);
262            else if(dr.x * max.y < dr.y * max.x && dr.y * max.x > (max.x - dr.x) * max.y)
263                bottomDoors.add(dr);
264        }
265    }
266
267    @Override
268    public int compareTo(MapModule o) {
269        if(o == null) return 1;
270        return category - o.category;
271    }
