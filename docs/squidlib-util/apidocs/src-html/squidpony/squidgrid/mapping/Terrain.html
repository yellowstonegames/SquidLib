001package squidpony.squidgrid.mapping;
002
003import squidpony.annotation.Beta;
004
005/**
006 * A base class for cell level features common to roguelike dungeons.
007 * 
008 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
009 */
010@Beta
011public class Terrain {
012
013    public static final Terrain FLOOR = new Terrain('.', 0),
014            WALL = new Terrain('#', 1),
015            CLOSED_DOOR = new Terrain('+', 2),
016            OPEN_DOOR = new Terrain('/', 2),
017            EMPTY_SPACE = new Terrain('_', 30),
018            LIQUID = new Terrain('~', 26),
019            ENTRANCE = new Terrain('<', 34),
020            EXIT = new Terrain('>', 38);
021    private final char symbol;
022    private final int color;
023
024    private Terrain(char symbol, int colorIndex) {
025        this.symbol = symbol;
026        color = colorIndex;
027    }
028
029    /**
030     * Returns the character representation for this terrain. This is meant to
031     * be for display purposes and should not be used to check for equality
032     * since multiple different terrains may return the same symbol.
033     *
034     * @return the representation of this terrain as a char
035     */
036    public char symbol() {
037        return symbol;
038    }
039
040    /**
041     * Returns the color for this terrain as an index into some unspecified palette.
042     *
043     * @return the color for this terrain as an index into some unspecified palette
044     */
045    public int colorIndex() {
046        return color;
047    }
048
049    @Override
050    public boolean equals(Object o) {
051        if (this == o) return true;
052        if (o == null || getClass() != o.getClass()) return false;
053
054        Terrain terrain = (Terrain) o;
055        
056        return symbol == terrain.symbol && color == terrain.color;
057    }
058
059    @Override
060    public int hashCode() {
061        return  31 * (int) symbol + color;
062    }
063}
