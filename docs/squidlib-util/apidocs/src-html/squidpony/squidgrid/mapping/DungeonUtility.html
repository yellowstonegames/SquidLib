074    /**
075     * Finds a random Coord where the x and y match up to a [x][y] location on map that has '.' as a value.
076     * Uses this class' rng field for pseudo-random number generation. May fail if there are very few floor tiles;
077     * use {@link #randomCell(short[])} given {@link #packedFloors(char[][])} to get a floor cell guaranteed if there is
078     * at least one, or better still, make a {@link GreasedRegion#GreasedRegion(char[][], char)} and get a single Coord
079     * from it with {@link GreasedRegion#singleRandom(IRNG)}. You can reuse a GreasedRegion with modifications, like
080     * removing cells you already picked, but using packedFloors() needs to make new short arrays each time. 
081     *
082     * @param map a char[][] that should contain a '.' floor tile
083     * @return a Coord that corresponds to a '.' in map, or null if a '.' cannot be found or if map is too small
084     */
085    public Coord randomFloor(char[][] map) {
086        int width = map.length;
087        int height = map[0].length;
088        if (width < 3 || height < 3)
089            return null;
090        int x = rng.nextInt(width - 2) + 1, y = rng.nextInt(height - 2) + 1;
091        for (int i = 0; i < 20; i++) {
092            if (map[x][y] == '.') {
093                return Coord.get(x, y);
094            } else {
095                x = rng.nextInt(width - 2) + 1;
096                y = rng.nextInt(height - 2) + 1;
097            }
098        }
099        x = 1;
100        y = 1;
101        if (map[x][y] == '.')
102            return Coord.get(x, y);
103
104        while (map[x][y] != '.') {
105            x += 1;
106            if (x >= width - 1) {
107                x = 1;
108                y += 1;
109            }
110            if (y >= height - 1)
111                return null;
112        }
113        return Coord.get(x, y);
114    }

148     * Finds a random Coord where the x and y match up to a [x][y] location on map that has the same value as the
149     * parameter tile. Uses this class' rng field for pseudo-random number generation.
150     *
151     * @param map  a char[][] that should contain the desired tile
152     * @param tile the char to search for
153     * @return a Coord that corresponds to a map element equal to tile, or null if tile cannot be found or if map is too small.
154     */
155    public Coord randomMatchingTile(char[][] map, char tile) {
156        int width = map.length;
157        int height = map[0].length;
158        if (width < 3 || height < 3)
159            return null;
160        int x = rng.nextInt(width - 2) + 1, y = rng.nextInt(height - 2) + 1;
161        for (int i = 0; i < 30; i++) {
162            if (map[x][y] == tile) {
163                return Coord.get(x, y);
164            } else {
165                x = rng.nextInt(width - 2) + 1;
166                y = rng.nextInt(height - 2) + 1;
167            }
168        }
169        x = 1;
170        y = 1;
171        if (map[x][y] == tile)
172            return Coord.get(x, y);
173
174        while (map[x][y] != tile) {
175            x += 1;
176            if (x >= width - 1) {
177                x = 1;
178                y += 1;
179            }
180            if (y >= height - 1)
181                return null;
182        }
183        return Coord.get(x, y);
184    }

186    /**
187     * Gets a random Coord that is adjacent to start, validating whether the position can exist on the given map.
188     * Adjacency defaults to four-way cardinal directions unless eightWay is true, in which case it uses Chebyshev.
189     * This can step into walls, and should NOT be used for movement.  It is meant for things like sound that can
190     * exist in walls, or for assigning decor to floors or walls that are adjacent to floors.
191     *
192     * @param map      a char[][] map that this will only use for its width and height; contents are ignored
193     * @param start    the starting position
194     * @param eightWay true to choose a random orthogonal or diagonal direction; false to only choose from orthogonal
195     * @return a Coord that is adjacent to start on the map, or null if start is off the map or the map is very small
196     */
197    public Coord randomStep(char[][] map, Coord start, boolean eightWay) {
198        int width = map.length;
199        int height = map[0].length;
200        if (width < 3 || height < 3 || start.x < 0 || start.y < 0 || start.x > width - 1 || start.y > height - 1)
201            return null;
202        Coord stepped = Coord.get(start.x, start.y);
203
204        if (eightWay) {
205            int mv = rng.nextInt(9);
206            return Coord.get(Math.min(Math.max(0, stepped.x + (mv % 3) - 1), height - 1),
207                    Math.min(Math.max(0, stepped.y + (mv / 3) - 1), height - 1));
208        } else {
209            int mv = rng.nextInt(5);
210            switch (mv) {
211                case 0:
212                    return Coord.get(Math.min(Math.max(0, stepped.x - 1), height - 1),
213                            stepped.y);
214                case 1:
215                    return Coord.get(Math.min(Math.max(0, stepped.x + 1), height - 1),
216                            stepped.y);
217                case 2:
218                    return Coord.get(stepped.x,
219                            Math.min(Math.max(0, stepped.y - 1), height - 1));
220                case 3:
221                    return Coord.get(stepped.x,
222                            Math.min(Math.max(0, stepped.y + 1), height - 1));
223                default:
224                    return stepped;
225            }
226        }
227    }

229    /**
230     * Finds a random Coord where the x and y match up to a [x][y] location on map that has '.' as a value,
231     * and a square of cells extending in the positive x and y directions with a side length of size must also have
232     * '.' as their values.
233     * Uses this class' rng field for pseudo-random number generation.
234     *
235     * @param map  a char[][] that should contain at least one floor represented by '.'
236     * @param size the side length of a square that must be completely filled with floors for this to return it
237     * @return a Coord that corresponds to a '.' in map, or null if a '.' cannot be found or if map is too small.
238     */
239    public Coord randomFloorLarge(char[][] map, int size) {
240        int width = map.length;
241        int height = map[0].length;
242        if (width < size + 2 || height < size + 2)
243            return null;
244        int x = rng.nextInt(width - size), y = rng.nextInt(height - size);
245        CELL:
246        for (int i = 0; i < 20; i++, x = rng.nextInt(width - size), y = rng.nextInt(height - size)) {
247            if (map[x][y] == '.') {
248                for (int j = 0; j < size; j++) {
249                    for (int k = 0; k < size; k++) {
250                        if (map[x + j][y + k] != '.')
251                            continue CELL;
252                    }
253                }
254                return Coord.get(x, y);
255            }
256        }
257        x = 1;
258        y = 1;
259
260        SLOW:
261        while (true) {
262            x += 1;
263            if (x >= width - size) {
264                x = 1;
265                y += 1;
266            }
267            if (y >= height - size)
268                return null;
269            if (map[x][y] == '.') {
270                for (int j = 0; j < size; j++) {
271                    for (int k = 0; k < size; k++) {
272                        if (map[x + j][y + k] != '.')
273                            continue SLOW;
274                    }
275                }
276                return Coord.get(x, y);
277            }
278        }
279    }

320    /**
321     * Takes a char[][] dungeon map that uses '#' to represent walls, and returns a new char[][] that uses unicode box
322     * drawing characters to draw straight, continuous lines for walls, filling regions between walls (that were
323     * filled with more walls before) with space characters, ' '. If keepSingleHashes is true, then '#' will be used if
324     * a wall has no orthogonal wall neighbors; if it is false, then a horizontal line will be used for stand-alone
325     * wall cells. If the lines "point the wrong way," such as having multiple horizontally adjacent vertical lines
326     * where there should be horizontal lines, call transposeLines() on the returned map, which will keep the dimensions
327     * of the map the same and only change the line chars. You will also need to call transposeLines if you call
328     * hashesToLines on a map that already has "correct" line-drawing characters, which means hashesToLines should only
329     * be called on maps that use '#' for walls. If you have a jumbled map that contains two or more of the following:
330     * "correct" line-drawing characters, "incorrect" line-drawing characters, and '#' characters for walls, you can
331     * reset by calling linesToHashes() and then potentially calling hashesToLines() again.
332     *
333     * @param map              a 2D char array indexed with x,y that uses '#' for walls
334     * @param keepSingleHashes true if walls that are not orthogonally adjacent to other walls should stay as '#'
335     * @return a copy of the map passed as an argument with box-drawing characters replacing '#' walls
336     */
337    public static char[][] hashesToLines(char[][] map, boolean keepSingleHashes) {
338        int width = map.length + 2;
339        int height = map[0].length + 2;
340
341        char[][] dungeon = new char[width][height];
342        for (int i = 1; i < width - 1; i++) {
343            System.arraycopy(map[i - 1], 0, dungeon[i], 1, height - 2);
344        }
345        for (int i = 0; i < width; i++) {
346            dungeon[i][0] = '\1';
347            dungeon[i][height - 1] = '\1';
348        }
349        for (int i = 0; i < height; i++) {
350            dungeon[0][i] = '\1';
351            dungeon[width - 1][i] = '\1';
352        }
353        for (int x = 1; x < width - 1; x++) {
354            for (int y = 1; y < height - 1; y++) {
355                if (map[x - 1][y - 1] == '#') {
356                    int q = 0;
357                    q |= (y <= 1 || map[x - 1][y - 2] == '#' || map[x - 1][y - 2] == '+' || map[x - 1][y - 2] == '/') ? 1 : 0;
358                    q |= (x >= width - 2 || map[x][y - 1] == '#' || map[x][y - 1] == '+' || map[x][y - 1] == '/') ? 2 : 0;
359                    q |= (y >= height - 2 || map[x - 1][y] == '#' || map[x - 1][y] == '+' || map[x - 1][y] == '/') ? 4 : 0;
360                    q |= (x <= 1 || map[x - 2][y - 1] == '#' || map[x - 2][y - 1] == '+' || map[x - 2][y - 1] == '/') ? 8 : 0;
361
362                    q |= (y <= 1 || x >= width - 2 || map[x][y - 2] == '#' || map[x][y - 2] == '+' || map[x][y - 2] == '/') ? 16 : 0;
363                    q |= (y >= height - 2 || x >= width - 2 || map[x][y] == '#' || map[x][y] == '+' || map[x][y] == '/') ? 32 : 0;
364                    q |= (y >= height - 2 || x <= 1 || map[x - 2][y] == '#' || map[x - 2][y] == '+' || map[x - 2][y] == '/') ? 64 : 0;
365                    q |= (y <= 1 || x <= 1 || map[x - 2][y - 2] == '#' || map[x - 2][y - 2] == '+' || map[x - 2][y - 2] == '/') ? 128 : 0;
366                    if (!keepSingleHashes && wallLookup[q] == '#') {
367                        dungeon[x][y] = '─';
368                    } else {
369                        dungeon[x][y] = wallLookup[q];
370                    }
371                }
372            }
373        }
374        char[][] portion = new char[width - 2][height - 2];
375        for (int i = 1; i < width - 1; i++) {
376            for (int j = 1; j < height - 1; j++) {
377                switch (dungeon[i][j]) {
378                    case '\1':
379                        portion[i - 1][j - 1] = ' ';
380                        break;
381                    default: // ┼┌┘
382                        portion[i - 1][j - 1] = dungeon[i][j];
383                }
384            }
385        }
386        return portion;
387    }

389    /**
390     * Reverses most of the effects of hashesToLines(). The only things that will not be reversed are the placement of
391     * space characters in unreachable wall-cells-behind-wall-cells, which remain as spaces. This is useful if you
392     * have a modified map that contains wall characters of conflicting varieties, as described in hashesToLines().
393     *
394     * @param map a 2D char array indexed with x,y that uses box-drawing characters for walls
395     * @return a copy of the map passed as an argument with '#' replacing box-drawing characters for walls
396     */
397    public static char[][] linesToHashes(char[][] map) {
398
399        int width = map.length;
400        int height = map[0].length;
401        char[][] portion = new char[width][height];
402        for (int i = 0; i < width; i++) {
403            for (int j = 0; j < height; j++) {
404                switch (map[i][j]) {
405                    case '\1':
406                    case '├':
407                    case '┤':
408                    case '┴':
409                    case '┬':
410                    case '┌':
411                    case '┐':
412                    case '└':
413                    case '┘':
414                    case '│':
415                    case '─':
416                    case '┼':
417                        portion[i][j] = '#';
418                        break;
419                    default:
420                        portion[i][j] = map[i][j];
421                }
422            }
423        }
424        return portion;
425    }

481    /**
482     * When a map is generated by DungeonGenerator with addDoors enabled, different chars are used for vertical and
483     * horizontal doors ('+' for vertical and '/' for horizontal).  This makes all doors '+', which is useful if you
484     * want '/' to be used for a different purpose and/or to distinguish open and closed doors.
485     *
486     * @param map a char[][] that may have both '+' and '/' for doors
487     * @return a char[][] that only uses '+' for all doors
488     */
489    public static char[][] closeDoors(char[][] map) {
490
491        int width = map.length;
492        int height = map[0].length;
493        char[][] portion = new char[width][height];
494        for (int i = 0; i < width; i++) {

1696    public static ArrayList<Coord> pointPath(int width, int height, IRNG rng) {
1697        if (width <= 2 || height <= 2)
1698            throw new IllegalArgumentException("width and height must be greater than 2");
1699        CoordPacker.init();
1700        long columnAlterations = (rng.nextLong() & 0xFFFFFFFFFFFFL);
1701        float columnBase = width / (Long.bitCount(columnAlterations) + 48.0f);
1702        long rowAlterations = (rng.nextLong() & 0xFFFFFFFFFFFFL);
1703        float rowBase = height / (Long.bitCount(rowAlterations) + 48.0f);
1704
1705        int[] columns = new int[16], rows = new int[16];
1706        int csum = 0, rsum = 0;
1707        long b = 7;
1708        for (int i = 0; i < 16; i++, b <<= 3) {
1709            columns[i] = csum + (int) (columnBase * 0.5f * (3 + Long.bitCount(columnAlterations & b)));
1710            csum += (int) (columnBase * (3 + Long.bitCount(columnAlterations & b)));
1711            rows[i] = rsum + (int) (rowBase * 0.5f * (3 + Long.bitCount(rowAlterations & b)));
1712            rsum += (int) (rowBase * (3 + Long.bitCount(rowAlterations & b)));
1713        }
1714        int cs = width - csum;
1715        int rs = height - rsum;
1716        int cs2 = cs, rs2 = rs, cs3 = cs, rs3 = rs;
1717        for (int i = 0; i <= 7; i++) {
1718            cs2 = cs2 * i / 7;
1719            rs2 = rs2 * i / 7;
1720            columns[i] -= cs2;
1721            rows[i] -= rs2;
1722        }
1723        for (int i = 15; i >= 8; i--) {
1724            cs3 = cs3 * (i - 8) >> 3;
1725            rs3 = rs3 * (i - 8) >> 3;
1726            columns[i] += cs3;
1727            rows[i] += rs3;
1728        }
1729
1730        ArrayList<Coord> points = new ArrayList<>(80);
1731        int m = rng.nextInt(64);
1732        Coord temp = CoordPacker.mooreToCoord(m), next;
1733        temp = Coord.get(columns[temp.x], rows[temp.y]);
1734        for (int i = 0, r; i < 256; r = rng.between(4, 12), i += r, m += r) {
1735            next = CoordPacker.mooreToCoord(m);
1736            next = Coord.get(columns[next.x], rows[next.y]);
1737            points.addAll(OrthoLine.line(temp, next));
1738            temp = next;
1739        }
1740        points.add(points.get(0));
1741        return points;

1746     * {@link #pointPath(int, int, IRNG)} with the given IRNG), then finding chars in blocking that are on that path and
1747     * replacing them with replacement. Modifies map in-place (!) and returns an ArrayList of Coord points that will
