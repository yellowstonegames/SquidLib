006/**
007 * Represents the eight grid directions and the deltaX, deltaY values associated
008 * with those directions.
009 *
010 * The grid referenced has x positive to the right and y positive downwards on
011 * screen.
012 *
013 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
014 */
015public enum Direction {
016
017    UP(0, -1), DOWN(0, 1), LEFT(-1, 0), RIGHT(1, 0), UP_LEFT(-1, -1), UP_RIGHT(1, -1), DOWN_LEFT(-1, 1), DOWN_RIGHT(1, 1), NONE(0, 0);
018    /**
019     * An array which holds only the four cardinal directions.
020     */
021    public static final Direction[] CARDINALS = {UP, DOWN, LEFT, RIGHT};
022    /**
023     * An array which holds only the four cardinal directions in clockwise order.
024     */
025    public static final Direction[] CARDINALS_CLOCKWISE = {UP, RIGHT, DOWN, LEFT};
026    /**
027     * An array which holds only the four cardinal directions in counter-clockwise order.
028     */
029    public static final Direction[] CARDINALS_COUNTERCLOCKWISE = {UP, LEFT, DOWN, RIGHT};
030    /**
031     * An array which holds only the four diagonal directions.
032     */
033    public static final Direction[] DIAGONALS = {UP_LEFT, UP_RIGHT, DOWN_LEFT, DOWN_RIGHT};
034    /**
035     * An array which holds all eight OUTWARDS directions.
036     */
037    public static final Direction[] OUTWARDS = {UP, DOWN, LEFT, RIGHT, UP_LEFT, UP_RIGHT, DOWN_LEFT, DOWN_RIGHT};
038    /**
039     * An array which holds all eight OUTWARDS directions in clockwise order.
040     */
041    public static final Direction[] CLOCKWISE = {UP, UP_RIGHT, RIGHT, DOWN_RIGHT, DOWN, DOWN_LEFT, LEFT, UP_LEFT};
042    /**
043     * An array which holds all eight OUTWARDS directions in counter-clockwise order.
044     */
045    public static final Direction[] COUNTERCLOCKWISE = {UP, UP_LEFT, LEFT, DOWN_LEFT, DOWN, DOWN_RIGHT, RIGHT, UP_RIGHT};
046    /**
047     * The x coordinate difference for this direction.
048     */
049    public final int deltaX;
050    /**
051     * The y coordinate difference for this direction.
052     */
053    public final int deltaY;
054
055    /**
056     * Returns the direction that most closely matches the input.
057     *
058     * This can be used to get the primary magnitude intercardinal direction
059     * from an origin point to an event point, such as a mouse click on a grid.
060     *
061     * If the point given is exactly on a boundary between directions then the
062     * direction clockwise is returned.
063     *
064     * @param x
065     * @param y
066     * @return
067     */
068    public static Direction getDirection(int x, int y) {
069        if (x == 0 && y == 0) {
070            return NONE;
071        }
072
073        double degree = Math.toDegrees(NumberTools.atan2(y, x));
074        degree += 450;//rotate to all positive and 0 is up
075        degree %= 360;//normalize
076        if (degree < 22.5) {
077            return UP;
078        } else if (degree < 67.5) {
079            return UP_RIGHT;
080        } else if (degree < 112.5) {
081            return RIGHT;
082        } else if (degree < 157.5) {
083            return DOWN_RIGHT;
084        } else if (degree < 202.5) {
085            return DOWN;
086        } else if (degree < 247.5) {
087            return DOWN_LEFT;
088        } else if (degree < 292.5) {
089            return LEFT;
090        } else if (degree < 337.5) {
091            return UP_LEFT;
092        } else {
093            return UP;
094        }
095    }
096
097    /**
098     * Gets an estimate at the correct direction that a position lies in given the distance towards it on the x and y
099     * axes. If x and y are both between -1 and 1 inclusive, this will always be accurate, and should be faster than
100     * {@link #getDirection(int, int)} by avoiding trigonometry or any other math on doubles. If at least one of x or y
101     * is 0, then this will also be accurate and will produce either a cardinal direction or NONE if both are 0. If x
102     * and y are both non-zero, this will always produce a diagonal, even if a cardinal direction should be more
103     * accurate; this behavior may sometimes be desirable to detect when some position is even slightly off from a true
104     * cardinal direction.
105     * @param x the relative x position to find the direction towards
106     * @param y the relative y position to find the direction towards
107     * @return the Direction that x,y lies in, roughly; will always be accurate for arguments between -1 and 1 inclusive
108     */
109    public static Direction getRoughDirection(int x, int y)
110    {
111        x = x == 0 ? 0 : (x >> 31 | 1); // signum with less converting to/from float
112        y = y == 0 ? 0 : (y >> 31 | 1); // signum with less converting to/from float
113        switch (x)
114        {
115            case -1:
116                switch (y)
117                {
118                    case 1: return DOWN_LEFT;
119                    case -1: return UP_LEFT;
120                    default: return LEFT;
121                }
122            case 1:
123                switch (y)
124                {
125                    case 1: return DOWN_RIGHT;
126                    case -1: return UP_RIGHT;
127                    default: return RIGHT;
128                }
129            default:
130                switch (y)
131                {
132                    case 1: return DOWN;
133                    case -1: return UP;
134                    default: return NONE;
135                }
136        }
137    }
138
139    /**
140     * Returns the direction that most closely matches the input.
141     *
142     * This can be used to get the primary magnitude cardinal direction from an
143     * origin point to an event point, such as a mouse click on a grid.
144     *
145     * If the point given is directly diagonal then the direction clockwise is
146     * returned.
147     *
148     * @param x
149     * @param y
150     * @return
151     */
152    public static Direction getCardinalDirection(int x, int y) {
153        if (x == 0 && y == 0) {
154            return NONE;
155        }
156
157        int absx = Math.abs(x);
158
159        if (y > absx) {
160            return UP;
161        }
162
163        int absy = Math.abs(y);
164
165        if (absy > absx) {
166            return DOWN;
167        }
168
169        if (x > 0) {
170            if (-y == x) {//on diagonal
171                return DOWN;
172            }
173            return RIGHT;
174        }
175
176        if (y == x) {//on diagonal
177            return UP;
178        }
179        return LEFT;
180
181    }
182
183        /**
184         * @param from
185         *            The starting point.
186         * @param to
187         *            The desired point to reach.
188         * @return The direction to follow to go from {@code from} to {@code to}. It
189         *         can be cardinal or diagonal.
190         */
191        public static Direction toGoTo(Coord from, Coord to) {
192                return getDirection(to.x - from.x, to.y - from.y);
193        }
194
195    /**
196     * Returns the Direction one step clockwise including diagonals.
197     *
198     * If considering only Cardinal directions, calling this twice will get the
199     * next clockwise cardinal direction.
200     *
201     * @return
202     */
203    public Direction clockwise() {
204        switch (this) {
205            case UP:
206                return UP_RIGHT;
207            case DOWN:
208                return DOWN_LEFT;
209            case LEFT:
210                return UP_LEFT;
211            case RIGHT:
212                return DOWN_RIGHT;
213            case UP_LEFT:
214                return UP;
215            case UP_RIGHT:
216                return RIGHT;
217            case DOWN_LEFT:
218                return LEFT;
219            case DOWN_RIGHT:
220                return DOWN;
221            case NONE:
222            default:
223                return NONE;
224        }
225    }
226
227    /**
228     * Returns the Direction one step counterclockwise including diagonals.
229     *
230     * If considering only Cardinal directions, calling this twice will get the
231     * next counterclockwise cardinal direction.
232     *
233     * @return
234     */
235    public Direction counterClockwise() {
236        switch (this) {
237            case UP:
238                return UP_LEFT;
239            case DOWN:
240                return DOWN_RIGHT;
241            case LEFT:
242                return DOWN_LEFT;
243            case RIGHT:
244                return UP_RIGHT;
245            case UP_LEFT:
246                return LEFT;
247            case UP_RIGHT:
248                return UP;
249            case DOWN_LEFT:
250                return DOWN;
251            case DOWN_RIGHT:
252                return RIGHT;
253            case NONE:
254            default:
255                return NONE;
256        }
257    }
258
259    /**
260     * Returns the direction directly opposite of this one.
261     *
262     * @return
263     */
264    public Direction opposite() {
265        switch (this) {
266            case UP:
267                return DOWN;
268            case DOWN:
269                return UP;
270            case LEFT:
271                return RIGHT;
272            case RIGHT:
273                return LEFT;
274            case UP_LEFT:
275                return DOWN_RIGHT;
276            case UP_RIGHT:
277                return DOWN_LEFT;
278            case DOWN_LEFT:
279                return UP_RIGHT;
280            case DOWN_RIGHT:
281                return UP_LEFT;
282            case NONE:
283            default:
284                return NONE;
285        }
286    }

288    /**
289     * @return Whether this is a diagonal move.
290     */
291    public boolean isDiagonal() {
292        return (deltaX & deltaY) != 0;
293    }
294
295    /**
296     * @return Whether this is a cardinal-direction move.
297     */
298    public boolean isCardinal() {
299        return (deltaX + deltaY & 1) != 0;
300    }

302        /**
303         * @return {@code true} if {@code this} has an upward component.
304         */
305        public boolean hasUp() {
306                switch (this) {
307                case UP:
308                case UP_LEFT:
309                case UP_RIGHT:
310                        return true;
311                case DOWN:
312                case DOWN_LEFT:
313                case DOWN_RIGHT:
314                case LEFT:
315                case NONE:
316                case RIGHT:
317                        return false;
318                }
319                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
320        }
321
322        /**
323         * @return {@code true} if {@code this} has a downward component.
324         */
325        public boolean hasDown() {
326                switch (this) {
327                case DOWN:
328                case DOWN_LEFT:
329                case DOWN_RIGHT:
330                        return true;
331                case LEFT:
332                case NONE:
333                case RIGHT:
334                case UP:
335                case UP_LEFT:
336                case UP_RIGHT:
337                        return false;
338                }
339                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
340        }
341
342        /**
343         * @return {@code true} if {@code this} has a left component.
344         */
345        public boolean hasLeft() {
346                switch (this) {
347                case DOWN_LEFT:
348                case LEFT:
349                case UP_LEFT:
350                        return true;
351                case DOWN:
352                case DOWN_RIGHT:
353                case NONE:
354                case RIGHT:
355                case UP:
356                case UP_RIGHT:
357                        return false;
358                }
359                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
360        }
361
362        /**
363         * @return {@code true} if {@code this} has a right component.
364         */
365        public boolean hasRight() {
366                switch (this) {
367                case RIGHT:
368                case DOWN_RIGHT:
369                case UP_RIGHT:
370                        return true;
371                case DOWN:
372                case NONE:
373                case UP:
374                case DOWN_LEFT:
375                case LEFT:
376                case UP_LEFT:
377                        return false;
378                }
379                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
380        }

382    Direction(int x, int y) {
383        deltaX = x;
384        deltaY = y;
385    }
