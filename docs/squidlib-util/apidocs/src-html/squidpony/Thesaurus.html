095    /**
096     * Allows this Thesaurus to find the exact words in synonyms and, when requested, replace each occurrence with a
097     * different word from the same Collection. Each word in synonyms should have the same part of speech, so "demon"
098     * and "devils" should not be in the same list of synonyms (singular noun and plural noun), but "demon" and "devil"
099     * could be (each is a singular noun). The Strings in synonyms should all be lower-case, since case is picked up
100     * from the text as it is being replaced and not from the words themselves. Proper nouns should normally not be used
101     * as synonyms, since this could get very confusing if it changed occurrences of "Germany" to "France" at random and
102     * a character's name, like "Dorothy", to "Anne", "Emily", "Cynthia", etc. in the middle of a section about Dorothy.
103     * The word matching pattern this uses only matches all-letter words, not words that contain hyphens, apostrophes,
104     * or other punctuation.
105     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
106     * @return this for chaining
107     */
108    public Thesaurus addSynonyms(Collection<String> synonyms)
109    {
110        if(synonyms.isEmpty())
111            return this;
112        //long prevState = rng.getState();
113        //rng.setState(prevState ^ CrossHash.hash64(synonyms));
114        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng, true);
115        for(String syn : synonyms)
116        {
117            mappings.put(syn, shuffler);
118        }
119        //rng.setState(prevState);
120        return this;
121    }

301    private StringBuilder modify(CharSequence text)
302    {
303        Matcher m;
304        StringBuilder sb = new StringBuilder(text);
305        Replacer.StringBuilderBuffer tb, working = Replacer.wrap(sb);
306        StringBuilder tmp;
307        boolean found;
308        FakeLanguageGen.Alteration alt;
309        for (int a = 0; a < alterations.size(); a++) {
310            alt = alterations.get(a);
311            tmp = working.sb;
312            tb = Replacer.wrap(new StringBuilder(tmp.length()));
313            m = alt.replacer.getPattern().matcher(tmp);
314
315            found = false;
316            while (true) {
317                if (rng.nextDouble() < alt.chance) {
318                    if (!Replacer.replaceStep(m, alt.replacer.getSubstitution(), tb))
319                        break;
320                    found = true;
321                } else {
322                    if (!m.find())
323                        break;
324                    found = true;
325                    m.getGroup(MatchResult.PREFIX, tb);
326                    m.getGroup(MatchResult.MATCH, tb);
327                    m.setTarget(m, MatchResult.SUFFIX);
328                }
329            }
330            if (found) {
331                m.getGroup(MatchResult.TARGET, tb);
332                working = tb;
333            }
334        }
335        return working.sb;
336
337    }
338    /**
339     * Given a String, StringBuilder, or other CharSequence that should contain words this knows synonyms for, this
340     * replaces each occurrence of such a known word with one of its synonyms, leaving unknown words untouched. Words
341     * that were learned together as synonyms with addSynonyms() will be replaced in such a way that an individual
342     * replacement word should not occur too close to a previous occurrence of the same word; that is, replacing the
343     * text "You fiend! You demon! You despoiler of creation; devil made flesh!", where "fiend", "demon", and "devil"
344     * are all synonyms, would never produce a string that contained "fiend" as the replacement for all three of those.
345     * @param text a CharSequence, such as a String, that contains words in the source language
346     * @return a String of the translated text.
347     */
348    public String process(CharSequence text)
349    {
350        Replacer rep = wordMatch.replacer(new SynonymSubstitution());
351        if(alterations.isEmpty())
352            return rep.replace(text);
353        else
354            return modify(rep.replace(text)).toString();
355    }

357    public String lookup(String word)
358    {
359        if(word.isEmpty())
360            return word;
361        String word2 = word.toLowerCase();
362        if(mappings.containsKey(word2))
363        {
364            String nx = mappings.get(word2).next();
365            if(nx.isEmpty())
366                return nx;
367            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
368                return nx.toUpperCase();
369            if(Category.Lu.contains(word.charAt(0)))
370            {
371                return Character.toUpperCase(nx.charAt(0)) + nx.substring(1);
372            }
373            return nx;
374        }
375        else if(languages.containsKey(word2))
376        {
377            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
378                return languages.get(word2).word(rng, false, rng.between(2, 4)).toUpperCase();
379            if(Category.Lu.contains(word.charAt(0)))
380            {
381                return languages.get(word2).word(rng, true, rng.between(2, 4));
382            }
383            return languages.get(word2).word(rng, false, rng.between(2, 4));
384        }
385        return word;
386    }

400    private void writeLookup(TextBuffer dest, StringBuilder word) {
401        if(word.length() <= 0)
402            return;
403        if(mappings.containsKey(word))
404        {
405            String nx = mappings.get(word).next();
406            if(nx.isEmpty())
407                return;
408            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
409            {
410                dest.append(nx.toUpperCase());
411                return;
412            }
413            if(Category.Lu.contains(word.charAt(0)))
414            {
415                dest.append(Character.toUpperCase(nx.charAt(0)));
416                dest.append(nx.substring(1));
417                return;
418            }
419            dest.append(nx);
420            return;
421        }
422        else if(languages.containsKey(word))
423        {
424            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
425            {
426                dest.append(languages.get(word).word(rng, false, rng.between(2, 4)).toUpperCase());
427            }
