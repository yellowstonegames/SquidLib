081    private static long seed = 9000;
082    private static int iseed = 9000;
083
084    public long doFNV()
085    {
086        LongPeriodRNG rng = new LongPeriodRNG(seed);
087
088        for (int i = 0; i < 1000000; i++) {
089            rng.nextLong();
090            seed += CrossHash.hash64(rng.state);
091        }
092        return seed;
093    }
094
095    @Benchmark
096    @BenchmarkMode(Mode.AverageTime)
097    @OutputTimeUnit(TimeUnit.MILLISECONDS)
098    public void measureFNV() throws InterruptedException {
099        seed = 9000;
100        doFNV();
101    }

258     * Note the units are different, scores are consistent with each other.
259     *
260     * You can run this test:
261     *
262     * a) Via the command line from the squidlib-performance module's root folder:
263     *    $ mvn clean install
264     *    $ java -jar target/benchmarks.jar HashBenchmark -wi 3 -i 3 -f 1

266     *    (we requested 5 warmup/measurement iterations, single fork)
267     *
268     * b) Via the Java API:

275                .include(HashBenchmark.class.getSimpleName())
276                .timeout(TimeValue.seconds(60))
277                .warmupIterations(8)
278                .measurementIterations(8)
279                .forks(1)
280                .build();

282        new Runner(opt).run();
