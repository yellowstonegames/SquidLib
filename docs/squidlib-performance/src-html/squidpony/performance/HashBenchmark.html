122    private static long seed = 9000;
123    private static int iseed = 9000;

125    public long doFNV()
126    {
127        final LongPeriodRNG rng = new LongPeriodRNG(seed);
128
129        for (int i = 0; i < 1000000; i++) {
130            rng.nextLong();
131            seed += CrossHash.hash64(rng.state);
132        }
133        return seed;
134    }

413    /*
414     * ============================== HOW TO RUN THIS TEST: ====================================
415     *
416     * You are expected to see the different run modes for the same benchmark.
417     * Note the units are different, scores are consistent with each other.
418     *
419     * You can run this test:
420     *
421     * a) Via the command line from the squidlib-performance module's root folder:
422     *    $ mvn clean install
423     *    $ java -jar target/benchmarks.jar HashBenchmark -wi 8 -i 8 -f 1
424     *
425     *    (we requested 8 warmup/measurement iterations, single fork)
426     *
427     * b) Via the Java API:
428     *    (see the JMH homepage for possible caveats when running from IDE:
429     *      http://openjdk.java.net/projects/code-tools/jmh/)
430     */
431
432    public static void main(String[] args) throws RunnerException {
433        Options opt = new OptionsBuilder()
434                .include(HashBenchmark.class.getSimpleName())
435                .timeout(TimeValue.seconds(60))
436                .warmupIterations(8)
437                .measurementIterations(8)
438                .forks(1)
439                .build();
440
441        new Runner(opt).run();
442    }
