439    class GridGraph implements IndexedGraph<Coord>
440    {
441        public ObjectIntMap<Coord> points = new ObjectIntMap<>(DIMENSION * DIMENSION);
442
443        public GridGraph(Coord[] pts)
444        {
445            for (int i = 0; i < pts.length; i++) {
446                points.put(pts[i], i);
447            }
448        }
449        @Override
450        public int getIndex(Coord node) {
451            return points.get(node, -1);
452        }
453
454        @Override
455        public int getNodeCount() {
456            return points.size;
457        }
458
459        @Override
460        public Array<Connection<Coord>> getConnections(Coord fromNode) {
461            Array<Connection<Coord>> conn = new Array<>(false, 8);
462            if(map[fromNode.x][fromNode.y] != '.')
463                return conn;
464            Coord t;
465            for (int i = 0; i < 8; i++) {
466                t = fromNode.translate(Direction.OUTWARDS[i]);
467                if (t.isWithin(DIMENSION, DIMENSION) && map[t.x][t.y] == '.')
468                    conn.add(new DefaultConnection<>(fromNode, t));
469            }
470            return conn;
471        }
472    }

474    public long doPathGDXAStar()
475    {
476        IndexedAStarPathFinder<Coord> astar = new IndexedAStarPathFinder<Coord>(new GridGraph(floors.asCoords()));
477        GraphPath<Coord> dgp = new DefaultGraphPath<Coord>(PATH_LENGTH);
478        Heuristic<Coord> heu = new Heuristic<Coord>() {
479            @Override
480            public float estimate(Coord node, Coord endNode) {
481                return Math.abs(node.x - endNode.x) + Math.abs(node.y - endNode.y);
482            }
483        };
484        Coord r;
485        long scanned = 0;
486        DungeonUtility utility = new DungeonUtility(new StatefulRNG(new LightRNG(0x1337BEEFDEAL)));
487        for (int x = 1; x < DIMENSION - 1; x++) {
488            for (int y = 1; y < DIMENSION - 1; y++) {
489                if (map[x][y] == '#')
490                    continue;
491                // this should ensure no blatant correlation between R and W
492                utility.rng.setState((x << 22) | (y << 16) | (x * y));
493                r = floors.singleRandom(utility.rng);
494                dgp.clear();
495                if(astar.searchNodePath(r, Coord.get(x, y), heu, dgp))
496                    scanned+= dgp.getCount();
497            }
498        }
499        return scanned;
500    }
