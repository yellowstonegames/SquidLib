083    public long doScanDijkstra()
084    {
085        DijkstraMap dijkstra = new DijkstraMap(
086                map, DijkstraMap.Measurement.CHEBYSHEV, new StatefulRNG(0x1337BEEF));
087        dijkstra.setBlockingRequirement(0);
088
089        long scanned = 0;
090        for (int x = 1; x < DIMENSION - 1; x++) {
091            for (int y = 1; y < DIMENSION - 1; y++) {
092                if (map[x][y] == '#')
093                    continue;
094                dijkstra.setGoal(x, y);
095                dijkstra.scan(null);
096                dijkstra.clearGoals();
097                dijkstra.resetMap();
098                scanned++;
099            }
100        }
101        return scanned;
102    }

133    public long doPathDijkstra()
134    {
135        DijkstraMap dijkstra = new DijkstraMap(
136                map, DijkstraMap.Measurement.CHEBYSHEV, new StatefulRNG(new LightRNG(0x1337BEEF)));
137        dijkstra.setBlockingRequirement(0);
138        Coord r;
139        long scanned = 0;
140        DungeonUtility utility = new DungeonUtility(new StatefulRNG(new LightRNG(0x1337BEEFDEAL)));
141        for (int x = 1; x < DIMENSION - 1; x++) {
142            for (int y = 1; y < DIMENSION - 1; y++) {
143                if (map[x][y] == '#')
144                    continue;
145                // this should ensure no blatant correlation between R and W
146                utility.rng.setState((x << 22) | (y << 16) | (x * y));
147                ((StatefulRNG) dijkstra.rng).setState((x << 20) | (y << 14) | (x * y));
148                r = floors.singleRandom(utility.rng);
149                dijkstra.findPath(PATH_LENGTH, null, null, r, Coord.get(x, y));
150                dijkstra.clearGoals();
151                dijkstra.resetMap();
152                scanned += dijkstra.path.size();
153            }
154        }
155        return scanned;
156    }

280    class GridGraph implements IndexedGraph<Coord>
281    {
282        public ObjectIntMap<Coord> points = new ObjectIntMap<>(DIMENSION * DIMENSION);
283
284        public GridGraph(Coord[] pts)
285        {
286            for (int i = 0; i < pts.length; i++) {
287                points.put(pts[i], i);
288            }
289        }
290        @Override
291        public int getIndex(Coord node) {
292            return points.get(node, -1);
293        }
294
295        @Override
296        public int getNodeCount() {
297            return points.size;
298        }
299
300        @Override
301        public Array<Connection<Coord>> getConnections(Coord fromNode) {
302            Array<Connection<Coord>> conn = new Array<>(false, 8);
303            if(map[fromNode.x][fromNode.y] != '.')
304                return conn;
305            Coord t;
306            for (int i = 0; i < 8; i++) {
307                t = fromNode.translate(Direction.OUTWARDS[i]);
308                if (t.isWithin(DIMENSION, DIMENSION) && map[t.x][t.y] == '.')
309                    conn.add(new DefaultConnection<>(fromNode, t));
310            }
311            return conn;
312        }
313    }

315    public long doPathGDXAStar()
316    {
317        IndexedAStarPathFinder<Coord> astar = new IndexedAStarPathFinder<Coord>(new GridGraph(floors.asCoords()));
318        GraphPath<Coord> dgp = new DefaultGraphPath<Coord>(PATH_LENGTH);
319        Heuristic<Coord> heu = new Heuristic<Coord>() {
320            @Override
321            public float estimate(Coord node, Coord endNode) {
322                return Math.abs(node.x - endNode.x) + Math.abs(node.y - endNode.y);
323            }
324        };
325        Coord r;
326        long scanned = 0;
327        DungeonUtility utility = new DungeonUtility(new StatefulRNG(new LightRNG(0x1337BEEFDEAL)));
328        for (int x = 1; x < DIMENSION - 1; x++) {
329            for (int y = 1; y < DIMENSION - 1; y++) {
330                if (map[x][y] == '#')
331                    continue;
332                // this should ensure no blatant correlation between R and W
333                utility.rng.setState((x << 22) | (y << 16) | (x * y));
334                r = floors.singleRandom(utility.rng);
335                dgp.clear();
336                if(astar.searchNodePath(r, Coord.get(x, y), heu, dgp))
337                    scanned+= dgp.getCount();
338            }
339        }
340        return scanned;
341    }

351    {
352        IndexedAStarPathFinder<Coord> astar = new IndexedAStarPathFinder<Coord>(new GridGraph(floors.asCoords()));
353        GraphPath<Coord> dgp = new DefaultGraphPath<Coord>(9);
354        Heuristic<Coord> heu = new Heuristic<Coord>() {
355            @Override
356            public float estimate(Coord node, Coord endNode) {
357                return Math.abs(node.x - endNode.x) + Math.abs(node.y - endNode.y);
358            }
359        };
360        Coord r;
361        long scanned = 0;
362        for (int x = 1; x < DIMENSION - 1; x++) {
363            for (int y = 1; y < DIMENSION - 1; y++) {
364                if (map[x][y] == '#')
365                    continue;
366                r = nearbyMap[x][y];
367                dgp.clear();
368                if(astar.searchNodePath(r, Coord.get(x, y), heu, dgp))
369                    scanned++;
370            }
371        }
372        return scanned;
373    }
