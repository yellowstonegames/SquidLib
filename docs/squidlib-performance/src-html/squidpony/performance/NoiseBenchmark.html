080    public double doPerlin2D()
081    {
082        for (double x = 0.0, a = 0.0; x < 64000000.0; x++) {
083            seed += PerlinNoise.noise(a += 0.0625, a);
084        }
085        return seed;
086    }

093        System.out.println(doPerlin2D());
094    }

316    /*
317     * ============================== HOW TO RUN THIS TEST: ====================================
318     *
319     * You are expected to see the different run modes for the same benchmark.
320     * Note the units are different, scores are consistent with each other.
321     *
322     * You can run this test:
323     *
324     * a) Via the command line from the squidlib-performance module's root folder:
325     *    $ mvn clean install
326     *    $ java -jar target/benchmarks.jar NoiseBenchmark -wi 4 -i 4 -f 1
327     *
328     *    (we requested 5 warmup/measurement iterations, single fork)
329     *
330     * b) Via the Java API:
331     *    (see the JMH homepage for possible caveats when running from IDE:
332     *      http://openjdk.java.net/projects/code-tools/jmh/)
333     */
334
335    public static void main(String[] args) throws RunnerException {
336        Options opt = new OptionsBuilder()
337                .include(NoiseBenchmark.class.getSimpleName())
338                .timeout(TimeValue.seconds(30))
339                .warmupIterations(4)
340                .measurementIterations(4)
341                .forks(1)
342                .build();
343
344        new Runner(opt).run();
345    }
