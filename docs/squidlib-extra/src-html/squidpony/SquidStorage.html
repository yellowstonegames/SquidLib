001package squidpony;
002
003import blazing.chain.LZSEncoding;
004import com.badlogic.gdx.Gdx;
005import com.badlogic.gdx.Preferences;
006import com.badlogic.gdx.utils.JsonWriter;
007import squidpony.annotation.Beta;
008
009import java.util.Map;
010
011/**
012 * Helps games store information in libGDX's Preferences class as Strings, then get it back out.
013 * Created by Tommy Ettinger on 9/16/2016.
014 */
015@Beta
016public class SquidStorage {
017    public final Preferences preferences;
018    public final String storageName;
019    public final DataConverter json;
020    protected StringStringMap contents;
021    public boolean compress = true;
022
023    /**
024     * Please don't use this constructor if possible; it simply calls {@link #SquidStorage(String)} with the constant
025     * String "nameless". This could easily overlap with other files/sections in Preferences, so you should always
026     * prefer giving a String argument to the constructor, typically the name of the game.
027     * @see #SquidStorage(String) the recommended constructor to use
028     */
029    public SquidStorage()
030    {
031        this("nameless");
032    }
033
034    /**
035     * Creates a SquidStorage with the given fileName to save using Preferences from libGDX. The name should generally
036     * be the name of this game or application, and must be a valid name for a file (so no slashes, backslashes, colons,
037     * semicolons, or commas for certain, and other non-alphanumeric characters are also probably invalid). You should
038     * not assume anything is present in the Preferences storage unless you have put it there, and this applies doubly
039     * to games or applications other than your own; you should avoid values for fileName that might overlap with
040     * another game's Preferences values.
041     * <br>
042     * To organize saved data into sub-sections, you specify logical units (like different players' saved games) with a
043     * String outerName when you call {@link #store(String)}, and can further distinguish data under the outerName when
044     * you call {@link #put(String, Object)} to put each individual item into the saved storage with its own innerName.
045     * <br>
046     * Calling this also sets up custom serializers for several important types in SquidLib; char[][], OrderedMap,
047     * IntDoubleOrderedMap, FakeLanguageGen, GreasedRegion, and notably Pattern from RegExodus all have smaller
048     * serialized representations than the default. OrderedMap allows non-String keys, which gets around a limitation in
049     * JSON maps normally, and both FakeLanguageGen and Pattern are amazingly smaller with the custom representation.
050     * The custom char[][] representation is about half the normal size by omitting commas after each char.
051     * @param fileName the valid file name to create or open from Preferences; typically the name of the game/app.
052     */
053    public SquidStorage(final String fileName)
054    {
055        storageName = fileName;
056        preferences = Gdx.app.getPreferences(storageName);
057        json = new DataConverter(JsonWriter.OutputType.minimal);
058
059        contents = new StringStringMap(16, 0.2f);
060    }
061
062    /**
063     * Prepares to store the Object {@code o} to be retrieved with {@code innerName} in the current group of objects.
064     * Does not write to a permanent location until {@link #store(String)} is called. The innerName used to store an
065     * object is required to get it back again, and can also be used to remove it before storing (or storing again).
066     * @param innerName one of the two Strings needed to retrieve this later
067     * @param o the Object to prepare to store
068     * @return this for chaining
069     */
070    public SquidStorage put(String innerName, Object o)
071    {
072        contents.put(innerName, json.toJson(o));
073        return this;
074    }
075
076    /**
077     * Actually stores all objects that had previously been prepared with {@link #put(String, Object)}, with
078     * {@code outerName} used as a key to retrieve any object in the current group. Flushes the preferences, making the
079     * changes permanent (until overwritten), but does not change the current group (you may want to call this method
080     * again with additional items in the current group, and that would simply involve calling put() again). If you want
081     * to clear the current group, use {@link #clear()}. If you want to remove just one object from the current group,
082     * use {@link #remove(String)}.
083     * @param outerName one of the two Strings needed to retrieve any of the objects in the current group
084     * @return this for chaining
085     */
086    public SquidStorage store(String outerName)
087    {
088        if(compress)
089            preferences.putString(outerName, LZSEncoding.compressToUTF16(json.toJson(contents, StringStringMap.class)));
090        else
091            preferences.putString(outerName, json.toJson(contents, StringStringMap.class));
092        preferences.flush();
093        return this;
094    }
095
096    /**
097     * Gets a String representation of the data that would be saved when {@link #store(String)} is called. This can be
098     * useful for finding particularly problematic objects that require unnecessary space when serialized.
099     * @return a String that previews what would be stored permanently when {@link #store(String)} is called
100     */
101    public String show()
102    {
103        if(compress)
104            return LZSEncoding.compressToUTF16(json.toJson(contents, StringStringMap.class));
105        else
106            return json.toJson(contents, StringStringMap.class);
107    }
108
109    /**
110     * Clears the current group of objects; recommended if you intend to store under multiple outerName keys.
111     * @return this for chaining
112     */
113    public SquidStorage clear()
114    {
115        contents.clear();
116        return this;
117    }
118
119    /**
120     * Removes one object from the current group by the {@code innerName} it was prepared with using
121     * {@link #put(String, Object)}. This does not affect already-stored objects unless {@link #store(String)} is called
122     * after this, in which case the new version of the current group, without the object this removed, is stored.
123     * @param innerName the String key used to put an object in the current group with {@link #put(String, Object)}
124     * @return this for chaining
125     */
126    public SquidStorage remove(String innerName)
127    {
128        contents.remove(innerName);
129        return this;
130    }
131
132    /**
133     * Gets an object from the storage by the given {@code outerName} key from {@link #store(String)} and
134     * {@code innerName} key from {@link #put(String, Object)}, and uses the class given by {@code type} for the
135     * returned value, assuming it matches the object that was originally put with those keys. If no such object is
136     * present, returns null. Results are undefined if {@code type} doesn't match the actual class of the stored object.
137     * @param outerName the key used to store the group of objects with {@link #store(String)}
138     * @param innerName the key used to store the specific object with {@link #put(String, Object)}
139     * @param type the class of the value; for a class like RNG, use {@code RNG.class}, but changed to fit
140     * @param <T> the type of the value to retrieve; if type was {@code RNG.class}, this would be {@code RNG}
141     * @return the retrieved value if successful, or null otherwise
142     */
143    @SuppressWarnings("unchecked")
144    public <T> T get(String outerName, String innerName, Class<T> type)
145    {
146        StringStringMap om;
147        String got;
148        if(compress)
149            got = LZSEncoding.decompressFromUTF16(preferences.getString(outerName));
150        else
151            got = preferences.getString(outerName);
152        if(got == null) return null;
153        om = json.fromJson(StringStringMap.class, got);
154        if(om == null) return null;
155        return json.fromJson(type, om.get(innerName));
156    }
157
158    /**
159     * Gets the approximate size of the currently-stored preferences. This assumes UTF-16 storage, which is the case for
160     * GWT's LocalStorage. Since GWT is restricted to the size the browser permits for LocalStorage, and this limit can
161     * be rather small (about 5 MB, sometimes more but not reliably), this method is especially useful there, but it may
162     * yield inaccurate sizes on other platforms that save Preferences data differently.
163     * @return the size, in bytes, of the already-stored preferences
164     */
165    public int preferencesSize()
166    {
167        Map<String, ?> p = preferences.get();
168        int byteSize = 0;
169        for(String k : p.keySet())
170        {
171            byteSize += k.length();
172            byteSize += preferences.getString(k, "").length();
173        }
174        return byteSize * 2;
175    }
176
177}
