001package squidpony.store.util;
002
003import squidpony.squidmath.CrossHash;
004import squidpony.squidmath.LightRNG;
005import squidpony.squidmath.PermutedRNG;
006

009 */
010public final class Garbler {
011
012    private Garbler()
013    {
014    }
015    public static String garble(final String text) {
016        return garble(text, "HOWARD PHILLIPS LOVECRAFT, EVERYBODY!");
017    }
018
019    public static String garble(final String text, final String keyText)
020    {
021        return garble(text, PermutedRNG.determine(CrossHash.Wisp.hash64(keyText) ^ 0x9E3779B97F4A7C15L));
022
023    }
024    public static String garble(final String text, final long key)
025    {
026        final char[] cs = text.toCharArray();
027        final int len = cs.length;
028        LightRNG rng = new LightRNG(key);
029        long wiggle;
030        for (int i = len - 1; i > 0; i--) {
031            wiggle = rng.nextLong();
032            int r = (int) (((i+1L) * (wiggle & 0x7FFFFFFFL)) >>> 31);
033            char c = cs[r];
034            cs[r] = cs[i];
035            cs[i] = (char) (c ^ (-33 & wiggle >>> 49 + 2 * (wiggle >>> 61)));
036        }
037        wiggle = rng.nextLong();
038        cs[0] ^= (-33 & wiggle >>> 49 + 2 * (wiggle >>> 61));
039        return String.valueOf(cs);
040    }
041
042    public static String degarble(final String garbled) {
043        return degarble(garbled, "HOWARD PHILLIPS LOVECRAFT, EVERYBODY!");
044    }
045
046    public static String degarble(final String garbled, final String keyText)
047    {
048        return degarble(garbled, PermutedRNG.determine(CrossHash.Wisp.hash64(keyText) ^ 0x9E3779B97F4A7C15L));
049    }
050
051    public static String degarble(final String garbled, final long key)
052    {
053        final char[] cs = garbled.toCharArray();
054        final int len = cs.length - 1;
055        LightRNG rng = new LightRNG(key);
056        long wiggle = rng.skip(len+1L);
057        cs[0] ^= (-33 & wiggle >>> 49 + 2 * (wiggle >>> 61));
058        for (int i = 1; i <= len; i++) {
059            wiggle = rng.skip(-1L);
060            int r = (int) (((i+1L) * (wiggle & 0x7FFFFFFFL)) >>> 31);
061            if(i == r)
062            {
063                cs[i] ^= (-33 & wiggle >>> 49 + 2 * (wiggle >>> 61));
064            }
065            else {
066                char c = cs[r];
067                cs[r] = (char) (cs[i] ^ (-33 & wiggle >>> 49 + 2 * (wiggle >>> 61)));
068                cs[i] = c;
069            }
070        }
071        return String.valueOf(cs);
072    }
073}
