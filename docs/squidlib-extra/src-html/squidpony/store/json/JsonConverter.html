016/**
017 * Augmented version of LibGDX's Json class that knows how to handle various data types common in SquidLib.
018 * This includes OrderedMap, which notably allows non-String keys (LibGDX's default Map serializer requires keys to be
019 * Strings), but does not currently allow the IHasher to be set (which only should affect OrderedMaps with array keys).
020 * It also makes significantly shorter serialized output for 2D char arrays, GreasedRegion and FakeLanguageGen objects,
021 * and various collections (IntDoubleOrderedMap, IntVLA, Arrangement, K2, and K2V1 at least).
022 * Created by Tommy Ettinger on 1/9/2017.
023 */
024public class JsonConverter extends Json {
025    /**
026     * Creates a new JsonConverter using "minimal" output type, so it omits double quotes whenever possible but gives
027     * up compatibility with most other JSON readers. Give the constructor
028     * {@link JsonWriter.OutputType#json} if you need full compatibility.
029     */
030    public JsonConverter() {
031        super();
032        initialize(this);
033    }
034
035    /**
036     * Creates a new JsonConverter with the given OutputType; typically minimal is fine, but compatibility may require
037     * you to use json; the javascript type is a sort of middle ground.
038     * @param outputType a JsonWriter.OutputType enum value that determines what syntax can be omitted from the output
039     */
040    public JsonConverter(JsonWriter.OutputType outputType) {
041        super(outputType);
042        initialize(this);
043    }

095        json.setSerializer(Pattern.class, new Serializer<Pattern>() {
096            @Override
097            public void write(Json json, Pattern object, Class knownType) {
098                if(object == null)
099                {
100                    json.writeValue(null);
101                    return;
102                }
103                json.writeValue((Object) object.serializeToString(), String.class);
104            }
105
106            @Override
107            public Pattern read(Json json, JsonValue jsonData, Class type) {
108                if(jsonData == null || jsonData.isNull()) return null;
109                String data = jsonData.asString();
110                if(data == null || data.length() < 2) return null;
111                return Pattern.deserializeFromString(data);
112            }
113        });

156        json.setSerializer(IntDoubleOrderedMap.class, new Serializer<IntDoubleOrderedMap>() {
157            @Override
158            public void write(Json json, IntDoubleOrderedMap object, Class knownType) {
159                if(object == null)
160                {
161                    json.writeValue(null);
162                    return;
163                }
164                json.writeObjectStart();
165                json.writeArrayStart("k");
166                IntDoubleOrderedMap.KeyIterator ki = object.keySet().iterator();
167                while (ki.hasNext())
168                    json.writeValue(ki.nextInt());
169                json.writeArrayEnd();
170                json.writeArrayStart("v");
171                IntDoubleOrderedMap.DoubleIterator vi = object.values().iterator();
172                while (vi.hasNext())
173                    json.writeValue(vi.nextDouble());
174                json.writeArrayEnd();
175                json.writeValue("f", object.f);
176                json.writeObjectEnd();
177            }
178
179            @Override
180            public IntDoubleOrderedMap read(Json json, JsonValue jsonData, Class type) {
181                if(jsonData == null || jsonData.isNull()) return null;
182                return new IntDoubleOrderedMap(jsonData.get("k").asIntArray(), jsonData.get("v").asDoubleArray(), jsonData.getFloat("f"));
183            }

210        json.setSerializer(OrderedMap.class, new Serializer<OrderedMap>() {
211            @Override
212            public void write(Json json, OrderedMap object, Class knownType) {
213                if(object == null)
214                {
215                    json.writeValue(null);
216                    return;
217                }
218                json.writeObjectStart();
219                json.writeValue("f", object.f);
220                if(!object.isEmpty()) {
221                    json.writeValue("k", object.firstKey(), null);
222                    json.writeValue("v", object.getAt(0), null);
223                    int sz = object.size();
224                    Object[] r = new Object[(sz - 1) * 2];
225                    for (int i = 1, p = 0; i < sz; i++) {
226                        r[p++] = object.keyAt(i);
227                        r[p++] = object.getAt(i);
228                    }
229                    json.writeValue("r", r, Object[].class, Object.class);
230                }
231                json.writeObjectEnd();
232            }
233
234            @Override
235            @SuppressWarnings("unchecked")
236            public OrderedMap read(Json json, JsonValue jsonData, Class type) {
237                if(jsonData == null || jsonData.isNull()) return null;
238                float f = json.readValue("f", float.class, jsonData);
239                Object k = json.readValue("k", null, INVALID, jsonData);
240                Object v = json.readValue("v", null, INVALID, jsonData);
241                Object[] r = json.readValue("r", Object[].class, jsonData);
242                if(k == INVALID)
243                    return new OrderedMap(0, f);
244                return Maker.makeOM(f, k, v, r);
245                //return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
246                //        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
247            }
248        });

250        json.setSerializer(EnumMap.class, new Serializer<EnumMap>() {
251            @Override
252            public void write(Json json, EnumMap object, Class knownType) {
253                if(object == null)
254                {
255                    json.writeValue(null);
256                    return;
257                }
258                json.writeObjectStart();
259                if(!object.isEmpty()) {
260                    Iterator it = object.entrySet().iterator();
261                    Map.Entry en = (Map.Entry)it.next();
262                    json.writeValue("e", en.getKey(), Enum.class);
263                    json.writeValue("v", en.getValue(), null);
264                    int sz = object.size();
265                    Object[] r = new Object[(sz - 1) * 2];
266                    for (int i = 1, p = 0; i < sz; i++) {
267                        if(!it.hasNext())
268                            break;
269                        en = (Map.Entry)it.next();
270                        r[p++] = en.getKey();
271                        r[p++] = en.getValue();
272                    }
273                    json.writeValue("r", r, Object[].class, Object.class);
274                }
275                json.writeObjectEnd();
276            }
277
278            @Override
279            @SuppressWarnings("unchecked")
280            public EnumMap read(Json json, JsonValue jsonData, Class type) {
281                if(jsonData == null || jsonData.isNull() || jsonData.size == 0) return null;
282                return new EnumMap(Maker.makeOM(0.75f,
283                        json.readValue("e", null, jsonData),
284                        json.readValue("v", null, jsonData),
285                        json.readValue("r", Object[].class, jsonData)));
286                //return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
287                //        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
288            }
289        });

360        json.setSerializer(char[][].class, new Serializer<char[][]>() {
361            @Override
362            public void write(Json json, char[][] object, Class knownType) {
363                if(object == null)
364                {
365                    json.writeValue(null);
366                    return;
367                }
368                int sz = object.length;
369                json.writeArrayStart();
370                for (int i = 0; i < sz; i++) {
371                    json.writeValue(String.valueOf(object[i]));
372                }
373                json.writeArrayEnd();
374            }
375
376            @Override
377            public char[][] read(Json json, JsonValue jsonData, Class type) {
378                if(jsonData == null || jsonData.isNull())
379                    return null;
380                int sz = jsonData.size;
381                char[][] data = new char[sz][];
382                JsonValue c = jsonData.child();
383                for (int i = 0; i < sz && c != null; i++, c = c.next()) {
384                    data[i] = c.asString().toCharArray();
385                }
386                return data;
387            }
