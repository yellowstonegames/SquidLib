018/**
019 * Augmented version of LibGDX's Json class that knows how to handle various data types common in SquidLib.
020 * This includes OrderedMap, which notably allows non-String keys (LibGDX's default Map serializer requires keys to be
021 * Strings), but does not currently allow the IHasher to be set (which only should affect OrderedMaps with array keys).
022 * It also makes significantly shorter serialized output for 2D char arrays, GreasedRegion and FakeLanguageGen objects,
023 * and various collections (IntDoubleOrderedMap, IntVLA, Arrangement, K2, and K2V1 at least).
024 * Created by Tommy Ettinger on 1/9/2017.
025 */
026public class JsonConverter extends Json {
027    /**
028     * Creates a new JsonConverter using "minimal" output type, so it omits double quotes whenever possible but gives
029     * up compatibility with most other JSON readers. Give the constructor
030     * {@link JsonWriter.OutputType#json} if you need full compatibility.
031     */
032    public JsonConverter() {
033        super();
034        initialize(this);
035    }
036
037    /**
038     * Creates a new JsonConverter with the given OutputType; typically minimal is fine, but compatibility may require
039     * you to use json; the javascript type is a sort of middle ground.
040     * @param outputType a JsonWriter.OutputType enum value that determines what syntax can be omitted from the output
041     */
042    public JsonConverter(JsonWriter.OutputType outputType) {
043        super(outputType);
044        initialize(this);
045    }

111        json.setSerializer(Pattern.class, new Serializer<Pattern>() {
112            @Override
113            public void write(Json json, Pattern object, Class knownType) {
114                if(object == null)
115                {
116                    json.writeValue(null);
117                    return;
118                }
119                json.writeValue((Object) object.serializeToString(), String.class);
120            }
121
122            @Override
123            public Pattern read(Json json, JsonValue jsonData, Class type) {
124                if(jsonData == null || jsonData.isNull()) return null;
125                String data = jsonData.asString();
126                if(data == null || data.length() < 2) return null;
127                return Pattern.deserializeFromString(data);
128            }

173            @Override
174            public void write(Json json, IntDoubleOrderedMap object, Class knownType) {
175                if(object == null)
176                {
177                    json.writeValue(null);
178                    return;
179                }
180                json.writeObjectStart();
181                json.writeArrayStart("k");
182                IntDoubleOrderedMap.KeyIterator ki = object.keySet().iterator();
183                while (ki.hasNext())
184                    json.writeValue(ki.nextInt());
185                json.writeArrayEnd();
186                json.writeArrayStart("v");
187                IntDoubleOrderedMap.DoubleIterator vi = object.values().iterator();
188                while (vi.hasNext())
189                    json.writeValue(vi.nextDouble());
190                json.writeArrayEnd();
191                json.writeValue("f", object.f);
192                json.writeObjectEnd();
193            }

226        json.setSerializer(OrderedMap.class, new Serializer<OrderedMap>() {
227            @Override
228            public void write(Json json, OrderedMap object, Class knownType) {
229                if(object == null)
230                {
231                    json.writeValue(null);
232                    return;
233                }
234                json.writeObjectStart();
235                json.writeValue("f", object.f);
236                if(!object.isEmpty()) {
237                    json.writeValue("k", object.firstKey(), null);
238                    json.writeValue("v", object.getAt(0), null);
239                    int sz = object.size();
240                    Object[] r = new Object[(sz - 1) * 2];
241                    for (int i = 1, p = 0; i < sz; i++) {
242                        r[p++] = object.keyAt(i);
243                        r[p++] = object.getAt(i);
244                    }
245                    json.writeValue("r", r, Object[].class, Object.class);
246                }
247                json.writeObjectEnd();
248            }
249
250            @Override
251            @SuppressWarnings("unchecked")
252            public OrderedMap read(Json json, JsonValue jsonData, Class type) {
253                if(jsonData == null || jsonData.isNull()) return null;
254                float f = json.readValue("f", float.class, jsonData);
255                Object k = json.readValue("k", null, INVALID, jsonData);
256                Object v = json.readValue("v", null, INVALID, jsonData);
257                Object[] r = json.readValue("r", Object[].class, jsonData);
258                if(k == INVALID)
259                    return new OrderedMap(0, f);
260                return Maker.makeOM(f, k, v, r);
261                //return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
262                //        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
263            }

265        json.setSerializer(EnumOrderedMap.class, new Serializer<EnumOrderedMap>() {
266            @Override
267            public void write(Json json, EnumOrderedMap object, Class knownType) {
268                if(object == null)
269                {
270                    json.writeValue(null);
271                    return;
272                }
273                json.writeObjectStart();
274                if(!object.isEmpty()) {
275                    json.writeValue("c", object.firstKey().getClass().getName());
276                    json.writeValue("k", object.firstKey(), null);
277                    json.writeValue("v", object.getAt(0), null);
278                    int sz = object.size();
279                    Object[] r = new Object[(sz - 1) * 2];
280                    for (int i = 1, p = 0; i < sz; i++) {
281                        r[p++] = object.keyAt(i);
282                        r[p++] = object.getAt(i);
283                    }
284                    json.writeValue("r", r, Object[].class, Object.class);
285                }
286                else
287                    json.writeValue("c", "default");
288                json.writeObjectEnd();
289            }
290
291            @Override
292            @SuppressWarnings("unchecked")
293            public EnumOrderedMap read(Json json, JsonValue jsonData, Class type) {
294                if(jsonData == null || jsonData.isNull()) return null;
295                String c = json.readValue("c", String.class, "", jsonData);
296                if("default".equals(c))
297                    return new EnumOrderedMap();
298                try {
299                    Class<? extends Enum<?>> cl = ClassReflection.forName(c);
300                    if(!ClassReflection.isEnum(cl))
301                        return null;
302                    Enum<?> k = json.readValue("k", cl, jsonData);
303                    Object v = json.readValue("v", null, INVALID, jsonData);
304                    Object[] r = json.readValue("r", Object[].class, jsonData);
305                    if(v == INVALID)
306                        return new EnumOrderedMap();
307                    return Maker.makeEOM(k, v, r);
308                } catch (ReflectionException e) {
309                    return null;
310                }
311                //return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
312                //        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
313            }
314        });

361        json.setSerializer(EnumMap.class, new Serializer<EnumMap>() {
362            @Override
363            public void write(Json json, EnumMap object, Class knownType) {
364                if(object == null)
365                {
366                    json.writeValue(null);
367                    return;
368                }
369                json.writeObjectStart();
370                if(!object.isEmpty()) {
371                    Iterator it = object.entrySet().iterator();
372                    Map.Entry en = (Map.Entry)it.next();
373                    json.writeValue("e", en.getKey(), Enum.class);
374                    json.writeValue("v", en.getValue(), null);
375                    int sz = object.size();
376                    Object[] r = new Object[(sz - 1) * 2];
377                    for (int i = 1, p = 0; i < sz; i++) {
378                        if(!it.hasNext())
379                            break;
380                        en = (Map.Entry)it.next();
381                        r[p++] = en.getKey();
382                        r[p++] = en.getValue();
383                    }
384                    json.writeValue("r", r, Object[].class, Object.class);
385                }
386                json.writeObjectEnd();
387            }

471        json.setSerializer(char[][].class, new Serializer<char[][]>() {
472            @Override
473            public void write(Json json, char[][] object, Class knownType) {
474                if(object == null)
475                {
476                    json.writeValue(null);
477                    return;
478                }
479                int sz = object.length;
480                json.writeArrayStart();
481                for (int i = 0; i < sz; i++) {
482                    json.writeValue(String.valueOf(object[i]));
483                }
484                json.writeArrayEnd();
485            }
486
487            @Override
488            public char[][] read(Json json, JsonValue jsonData, Class type) {
489                if(jsonData == null || jsonData.isNull())
490                    return null;
491                int sz = jsonData.size;
492                char[][] data = new char[sz][];
493                JsonValue c = jsonData.child();
494                for (int i = 0; i < sz && c != null; i++, c = c.next()) {
495                    data[i] = c.asString().toCharArray();
496                }
497                return data;
498            }
