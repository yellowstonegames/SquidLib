011/**
012 * An ordered bidirectional map-like data structure, with "bundles" of unique E (for Element) keys and unique S (for
013 * Single) keys updated together like a map that can be queried by E key bundles, S keys, or int indices, as well as
014 * like a multimap that can be queried by lone E keys. A bundle can be specified as a Collection of E values, an array
015 * of E values, or in two parts as either of the above given to a BundleBiMap method along with an int array that can
016 * represent variations on E keys (e.g. some quantity for an E value that permits different amounts to be passed with
017 * the unique E values to getS and have it yield a different S for a, say, 1 Foo and 1 Bar vs. 2 Foo and 3 Bar). This
018 * allows an intended purpose for this class as a way to describe crafting recipes that may need various amounts of an
019 * item to be mixed in, and since the quantities are optional, it can still be used to group multiple E keys with S
020 * keys. The multimap property allows you to use an E element key to get all of the S keys that are associated with a
021 * bundle that contains that E key (you can also get the ordering indices for those S keys, which you can use to get the
022 * full bundle if you want).
023 * <br>
024 * Does not implement any interfaces that you would expect for a data structure, because almost every method it has
025 * needs to specify whether it applies to E or S items, or otherwise doesn't fit a normal Collection-like interface's
026 * requirements.
027 * <br>
028 * Closely related to Arrangement, K2, and K2V1 in implementation.
029 * <br>
030 * Created by Tommy Ettinger on 4/26/2017.
031 */
032@Beta
033public class BundleBiMap<E, S>
034{
035    private Arrangement<E> elements;
036    private K2<int[][], S> bm;
037    private ArrayList<IntVLA> mm;

047    /**
048     * Constructs a BundleBiMap with the expected number of indices to hold (the number of bundles and number of S items
049     * is always the same, and this will be more efficient if expected is greater than that number).
050     * @param expected how many bundle-to-single pairings this is expected to hold
051     */
052    public BundleBiMap(int expected)
053    {
054        this(expected, Arrangement.DEFAULT_LOAD_FACTOR);
055    }

057    /**
058     * Constructs a BundleBiMap with the expected number of indices to hold (the number of bundles and number of S items
059     * is always the same, and this will be more efficient if expected is greater than that number) and the load factor
060     * to use, between 0.1f and 0.8f usually (using load factors higher than 0.8f can cause problems).
061     * @param expected the amount of indices (the count of bundles is the same as the count of S items) this should hold
062     * @param f the load factor, probably between 0.1f and 0.8f
063     */
064    public BundleBiMap(int expected, float f)
065    {
066        elements = new Arrangement<>(expected, f);
067        bm = new K2<>(expected, f, CrossHash.int2DHasher, CrossHash.defaultHasher);
068        mm = new ArrayList<>(expected * 4);
069    }

072    /**
073     * Constructs a BundleBiMap using another BundleBiMap to copy.
074     * @param other the other BundleBiMap to copy
075     */
076    public BundleBiMap(BundleBiMap<? extends E, ? extends S> other)
077    {
078        if(other == null)
079        {
080            elements = new Arrangement<>(64, 0.75f);
081            bm = new K2<>(16, 0.75f, CrossHash.int2DHasher, CrossHash.defaultHasher);
082            mm = new ArrayList<>(64);
083        }
084        else
085        {
086            elements = new Arrangement<>(other.elements);
087            bm = new K2<>(other.bm);
088            mm = new ArrayList<>(other.mm);
089        }
090    }

172    /**
173     * Adds a bundle of E keys and a S key at the same point in the ordering (the end) to this BundleBiMap. Neither
174     * parameter can be present in this collection before this is called.
175     * @param e an array of E keys to add; the array cannot already be present, nor can it be null
176     * @param s e S key to add; cannot already be present
177     * @return true if this collection changed as e result of this call
178     */
179    public boolean put(E[] e, S s)
180    {
181        if(e == null) return false;
182        int len = elements.size;
183        elements.putAll(e);
184        for (int i = len; i < elements.size; i++) {
185            mm.add(new IntVLA(e.length));
186        }
187
188        elements.putAll(e);
189        int[][] bundle = new int[][]{elements.getArray(e)};
190        if(!bm.put(bundle, s))
191            return false;
192        for (int i = 0; i < e.length; i++) {
193            mm.get(bundle[0][i]).add(bm.size()-1);
194        }
195        return true;
196    }

200     * present in this collection before this is called.

217            return false;
218        for (int i = 0; i < e.length; i++) {
219            mm.get(bundle[0][i]).add(bm.size()-1);

222    }

274    /**
275     * Puts all unique E and S keys in {@code aKeys} and {@code bKeys} into this K2 at the end. If an E in aKeys or a S
276     * in bKeys is already present when this would add one, this will not put the E and S keys at that point in the
277     * iteration order, and will place the next unique E and S it finds in the arguments at that position instead.
278     * @param aKeys an Iterable or Collection of E keys to add; should all be unique (like a Set)
279     * @param bKeys an Iterable or Collection of S keys to add; should all be unique (like a Set)
280     * @return true if this collection changed as a result of this call
281     */
282    public boolean putAll(Iterable<? extends Collection<? extends E>> aKeys, Iterable<? extends S> bKeys)
283    {
284        if(aKeys == null || bKeys == null) return false;
285        Iterator<? extends Collection<? extends E>> aIt = aKeys.iterator();
286        Iterator<? extends S> bIt = bKeys.iterator();
287        boolean changed = false;
288        while (aIt.hasNext() && bIt.hasNext())
289        {
290            changed = put(aIt.next(), bIt.next()) || changed;
291        }
292        return changed;
293    }
294    /**
295     * Puts all unique E and S keys in {@code other} into this K2, respecting other's ordering. If an E or a S in other
296     * is already present when this would add one, this will not put the E and S keys at that point in the iteration
297     * order, and will place the next unique E and S it finds in the arguments at that position instead.
298     * @param other another K2 collection with the same E and S types
299     * @return true if this collection changed as a result of this call
300     */
301    public boolean putAll(BundleBiMap<? extends E, ? extends S> other)
302    {
303        if(other == null) return false;
304        boolean changed = false;
305        int sz = other.size();
306        for (int i = 0; i < sz; i++) {
307            int[][] bundle = other.bm.getAAt(i);
308            if(bundle == null || bundle.length == 0) continue;
309            if(bundle.length == 1)
310                changed |= put(elements.keysAt(bundle[0]), other.bm.getBAt(i));
311            else
312                changed |= put(elements.keysAt(bundle[0]), bundle[1], other.bm.getBAt(i));
313        }
314        return changed;
315    }

548     * {@link #getBundleVariationAt(int)} to get the possible variation at that position, or {@link #getCodeAt(int)} to
549     * get the coded bundle at that position.
550     * @return an OrderedSet of all coded 2D int array bundles that contain the given element, or null if E is not in any bundles
551     */
552    public int[] getManyIndices(E element)
553    {
554        if(element == null) return null;
555        int pos;
556        if((pos = elements.getInt(element)) < 0) return null;
557        return mm.get(pos).toArray();
558    }

560    /**
561     * Reorders this BundleBiMap using {@code ordering}, which has the same length as this object's {@link #size()}
562     * and can be generated with {@link ArrayTools#range(int)} (which, if applied, would produce no
563     * change to the current ordering), {@link IRNG#randomOrdering(int)} (which gives a random ordering, and if
564     * applied immediately would be the same as calling {@link #shuffle(IRNG)}), or made in some other way. If you
565     * already have an ordering and want to make a different ordering that can undo the change, you can use
566     * {@link ArrayTools#invertOrdering(int[])} called on the original ordering. The effects of this method, if called
567     * with an ordering that has repeat occurrences of an int or contains ints that are larger than its size should
568     * permit, are undefined other than the vague definition of "probably bad, unless you like crashes."
569     * @param ordering an int array or vararg that must contain each int from 0 to {@link #size()}
570     * @return this for chaining
571     */
572    public BundleBiMap<E, S> reorder(int... ordering)
573    {
574        if(ordering == null || ordering.length != bm.size()) return this;
575        bm.reorder(ordering);
576        int len = mm.size();
577        for (int i = 0; i < len; i++) {
578            IntVLA iv = mm.get(i);
579            for (int j = 0; j < iv.size; j++) {
580                iv.set(i, ordering[iv.get(i)]);
581            }
582        }
583        return this;
584    }
585
586    /**
587     * Generates a random ordering with rng and applies the same ordering to all kinds of keys this has; they will
588     * maintain their current association to other keys but their ordering/indices will change.
589     * @param rng an IRNG to produce the random ordering this will use
590     * @return this for chaining
591     */
592    public BundleBiMap<E, S> shuffle(IRNG rng)
593    {
594        return reorder(rng.randomOrdering(bm.size()));
595    }
